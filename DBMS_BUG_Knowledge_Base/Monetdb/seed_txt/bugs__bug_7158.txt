DROP AGGREGATE IF EXISTS python_aggregate;
CREATE AGGREGATE python_aggregate(val INTEGER) RETURNS INTEGER LANGUAGE PYTHON { try: unique = numpy.unique(aggr_group) x = numpy.zeros(shape=(unique.size)) for i in range(0, unique.size): x[i] = numpy.sum(val[aggr_group==unique[i]]) except NameError: x = numpy.sum(val) return(x) };
DROP TABLE IF EXISTS test_empty;
CREATE TABLE test_empty (x INTEGER NOT NULL, y VARCHAR(50));
SELECT python_aggregate(x) FROM test_empty;
