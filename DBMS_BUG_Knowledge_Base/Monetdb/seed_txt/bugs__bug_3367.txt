CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(100), age INT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO users (id, name, age) VALUES (1, 'Alice', 30), (2, 'Bob', 25), (3, 'Charlie', NULL);
WITH ranked_users AS (SELECT id, name, age, ROW_NUMBER() OVER(ORDER BY users.age DESC) AS r FROM users) SELECT name, age FROM ranked_users WHERE r <= 2 ORDER BY r;
CREATE TABLE orders (order_id INT PRIMARY KEY, user_id INT, amount DECIMAL(10, 2), FOREIGN KEY (user_id) REFERENCES users(id));
