CREATE SCHEMA test_db;
CREATE TABLE test_db.users (id INT PRIMARY KEY, name VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE test_db.orders (order_id INT PRIMARY KEY, user_id INT, amount DECIMAL(10, 2), FOREIGN KEY (user_id) REFERENCES test_db.users(id));
SELECT u.name, o.amount FROM test_db.users u LEFT JOIN test_db.orders o ON u.id = o.user_id WHERE o.amount IS NULL;
