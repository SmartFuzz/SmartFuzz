CREATE TABLE "sys"."bigaggr" ("a1" INTEGER, "a2" INTEGER, "a3" INTEGER DEFAULT 0);
INSERT INTO "sys"."bigaggr" ("a1", "a2", "a3") VALUES (1, 2, NULL), (3, 4, 5), (NULL, NULL, 6);
CREATE TABLE "sys"."bigaggr_agg" AS SELECT a1, SUM(a2) AS total_a2 FROM "sys"."bigaggr" GROUP BY a1;
SELECT COUNT(*) FROM "sys"."bigaggr_agg" WHERE total_a2 IS NOT NULL;
CREATE TABLE "sys"."bigaggr_join" AS SELECT b.a1, b.a2 FROM "sys"."bigaggr" b JOIN "sys"."bigaggr_agg" a ON b.a1 = a.a1;
