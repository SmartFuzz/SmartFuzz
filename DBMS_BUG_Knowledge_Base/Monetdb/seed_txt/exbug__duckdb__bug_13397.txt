CREATE TABLE test_table (id INTEGER, value INTEGER, description VARCHAR);
INSERT INTO test_table (id, value, description) VALUES (1, 0, 'zero'), (2, 1, 'one'), (3, 2, 'two');
SELECT COUNT(*) AS count_zeros FROM (SELECT CASE WHEN value = 0 THEN 1 ELSE 0 END AS zero_count FROM test_table) AS subquery;
SELECT COUNT(*) AS count_ones FROM (SELECT CASE WHEN value = 1 THEN 1 ELSE 0 END AS one_count FROM test_table WHERE value IS NOT NULL) AS subquery;
SELECT COUNT(*) AS count_zeros FROM (SELECT COALESCE(CASE WHEN value = 0 THEN 1 ELSE 0 END, 0) AS zero_count FROM (VALUES (NULL), (0), (1), (2)) AS temp(value)) AS subquery;
