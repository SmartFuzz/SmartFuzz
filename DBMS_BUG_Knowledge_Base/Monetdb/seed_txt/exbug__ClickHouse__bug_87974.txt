DROP TABLE IF EXISTS test_left;
DROP TABLE IF EXISTS test_right;
CREATE TABLE test_left (a BIGINT, b VARCHAR(255), c VARCHAR(255), d INT) USING MEMORY;
CREATE TABLE test_right (a BIGINT, b VARCHAR(255), c VARCHAR(255), d INT) USING MEMORY;
INSERT INTO test_left SELECT MOD(number, 10000), CAST(MOD(number, 10000) AS VARCHAR), CAST(MOD(number, 10000) AS VARCHAR), MOD(number, 5) FROM (SELECT ROW_NUMBER() OVER () AS number FROM sysibm.sysdummy1 LIMIT 100000);
INSERT INTO test_right SELECT MOD(number, 10), CAST(MOD(number, 10) AS VARCHAR), CAST(MOD(number, 10) AS VARCHAR), NULL FROM (SELECT ROW_NUMBER() OVER () AS number FROM sysibm.sysdummy1 LIMIT 10000);
