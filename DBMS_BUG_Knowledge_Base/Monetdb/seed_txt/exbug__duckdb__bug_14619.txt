SELECT id, ARRAY(SELECT ix FROM UNNEST(values) WITH ORDINALITY AS t(x, ix)) AS transformed_ix FROM test_table;
SELECT id, MAX(CASE WHEN ix IS NOT NULL THEN ix END) AS reduced_ix FROM (SELECT UNNEST(values) WITH ORDINALITY AS t(x, ix) FROM test_table) GROUP BY id;
