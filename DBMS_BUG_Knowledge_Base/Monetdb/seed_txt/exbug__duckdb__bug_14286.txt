CREATE TABLE test_data (id INTEGER, name VARCHAR, category VARCHAR);
INSERT INTO test_data VALUES (1, '大熊猫', '熊'), (2, '大熊猫', '猫'), (3, '小熊猫', '猫');
SELECT category, ARRAY_AGG(DISTINCT name) AS a_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
SELECT category, ARRAY_AGG(name) AS b_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
SELECT category, ARRAY_AGG(ROW(id, name, category)) AS c_list FROM test_data GROUP BY category;
