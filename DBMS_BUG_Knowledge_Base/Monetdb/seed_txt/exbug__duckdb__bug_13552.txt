CREATE TABLE test_unpivot (id INT, column_name VARCHAR, column_value VARCHAR, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_unpivot SELECT id, CONCAT('column_', id), CONCAT('value_', id) FROM (SELECT ROW_NUMBER() OVER () AS id FROM test_unpivot LIMIT 10000);
CREATE TABLE pivoted_test AS SELECT id, MAX(CASE WHEN column_name = 'column_1' THEN column_value END) AS column_1, MAX(CASE WHEN column_name = 'column_2' THEN column_value END) AS column_2, MAX(CASE WHEN column_name = 'column_3' THEN column_value END) AS column_3, MAX(CASE WHEN column_name = 'column_4' THEN column_value END) AS column_4, MAX(CASE WHEN column_name = 'column_5' THEN column_value END) AS column_5 FROM test_unpivot GROUP BY id;
SELECT COUNT(*) FROM (SELECT id, column_name, column_value FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) AS unpivoted WHERE id IS NOT NULL);
CREATE TABLE final_result AS SELECT * FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) WHERE id < 5000;
