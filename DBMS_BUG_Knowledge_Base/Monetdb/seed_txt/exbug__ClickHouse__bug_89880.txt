CREATE TABLE T1 (K BIGINT, C0 VARCHAR(255), C1 VARCHAR(255), C2 DOUBLE PRECISION);
INSERT INTO T1 SELECT number, ARRAY(SELECT cityHash64(i * number) FROM generate_series(0, 9) AS i), CASE WHEN number % 2 = 0 THEN 'even' ELSE NULL END, number * 1.5 FROM generate_series(0, 99) AS number;

INSERT INTO T2 SELECT K, C0, CAST(K AS INT) FROM T1 WHERE C1 IS NOT NULL;
