CREATE TABLE equipment (EquipmentName VARCHAR(255), FailureReason VARCHAR(255));
INSERT INTO equipment (EquipmentName, FailureReason) VALUES ('Excavator', 'Hydraulic Failure'), ('Bulldozer', 'Engine Overheat');
CREATE VIEW v_resolve_maintenance_action AS SELECT * FROM equipment;
SELECT e.*, v.* FROM equipment e LEFT JOIN v_resolve_maintenance_action v ON e.EquipmentName = v.EquipmentName AND e.FailureReason = v.FailureReason;
CREATE TABLE maintenance_actions (ActionId INT, ActionDescription VARCHAR(255));
