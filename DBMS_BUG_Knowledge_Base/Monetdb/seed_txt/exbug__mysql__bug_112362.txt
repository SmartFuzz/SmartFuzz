CREATE TABLE c (c_id INTEGER NOT NULL PRIMARY KEY, value VARCHAR(100) NOT NULL, UNIQUE (value));
CREATE TABLE d (d_id INTEGER NOT NULL PRIMARY KEY, value VARCHAR(100) NOT NULL, INDEX (value));
INSERT INTO c (value) SELECT CONCAT('value_', seq) FROM (SELECT ROW_NUMBER() OVER () AS seq FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2 LIMIT 1000);
INSERT INTO d (value) SELECT CONCAT('value_', seq) FROM (SELECT ROW_NUMBER() OVER () AS seq FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2 LIMIT 1000000);
WITH b2 AS (SELECT DISTINCT value FROM d), a2 AS (SELECT value FROM c) SELECT value FROM b2 WHERE value NOT IN (SELECT value FROM a2);
