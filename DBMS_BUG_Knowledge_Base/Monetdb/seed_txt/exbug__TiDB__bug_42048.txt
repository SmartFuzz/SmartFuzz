CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(100), balance DECIMAL(10, 2) NOT NULL DEFAULT 0.00);
CREATE TABLE transactions (transaction_id INT PRIMARY KEY, user_id INT, amount DECIMAL(10, 2), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
