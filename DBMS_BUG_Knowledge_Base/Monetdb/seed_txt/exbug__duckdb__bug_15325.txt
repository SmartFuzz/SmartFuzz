SET MEMORY_LIMIT='300MB';
CREATE TABLE diverse_table (id INT, name VARCHAR, value DOUBLE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, is_active BOOLEAN);
INSERT INTO diverse_table SELECT i, concat('Name_', i), random() * 100, NOW(), (i % 2 = 0) FROM (SELECT ROW_NUMBER() OVER () AS i FROM sysibm.sysdummy1) AS t LIMIT 1000000;
SELECT id, name, value, created_at, is_active FROM diverse_table WHERE is_active IS NOT NULL;
SELECT COUNT(*) AS total_count, SUM(CASE WHEN value IS NULL THEN 1 ELSE 0 END) AS null_value_count FROM diverse_table;
