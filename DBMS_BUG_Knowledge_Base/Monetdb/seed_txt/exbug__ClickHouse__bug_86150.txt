DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id INT, keyword_id INT, search_date TIMESTAMP, position_on_serp INT, version INT, value VARCHAR(255));
INSERT INTO test_final_partitions SELECT (RANDOM() % 1000) AS organization_id, (RANDOM() % 100000) AS keyword_id, TIMESTAMP '2023-01-01 00:00:00' + (RANDOM() % (86400*180)) AS search_date, (RANDOM() % 50) AS position_on_serp, (ROWID() % 5) AS version, CAST(ROWID() AS VARCHAR) FROM SYSTEM GENERATE SERIES 1 TO 50000000;
SELECT COUNT(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT COUNT(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
