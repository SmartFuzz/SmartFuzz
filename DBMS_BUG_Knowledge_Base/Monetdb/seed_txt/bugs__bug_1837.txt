CREATE TABLE scene (id INT PRIMARY KEY, victim_id INT NOT NULL, description VARCHAR(255));
CREATE TABLE victim (id INT PRIMARY KEY, name VARCHAR(255), age INT CHECK (age >= 0));
CREATE TABLE inspector (id INT PRIMARY KEY, name VARCHAR(255), experience_years INT DEFAULT 0);
