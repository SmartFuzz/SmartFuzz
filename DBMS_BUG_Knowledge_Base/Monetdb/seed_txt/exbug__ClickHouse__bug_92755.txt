DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR(255), event_time TIMESTAMP, id BIGINT, msg VARCHAR(255), value FLOAT8);
INSERT INTO repro.events SELECT CAST(number % 100 AS VARCHAR(255)), NOW() - (number % 200000), number, CONCAT('m', CAST(number AS VARCHAR(255))), CAST(number % 1000 AS FLOAT8) FROM generate_series(0, 1999999);
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
