CREATE TABLE dept (
       deptno int PRIMARY KEY,
       dname varchar(15),
       loc varchar(15)
     );

SELECT e.ename, d.dname
       FROM emp e
      LEFT JOIN dept d ON e.deptno = d.deptno
     WHERE d.loc = 'DALLAS' OR e.sal > 2000.00
