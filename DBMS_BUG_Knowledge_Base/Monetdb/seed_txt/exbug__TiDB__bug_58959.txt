CREATE TABLE t1(c1 BLOB NOT NULL, c2 INT DEFAULT 0);
INSERT INTO t1 VALUES ('true'), ('false'), ('1xx'), (NULL);
SELECT CASE WHEN c1 IS NOT NULL AND c1 IN ('true', 'false') THEN 1 ELSE 0 END AS is_true FROM t1 WHERE c2 > 0;
