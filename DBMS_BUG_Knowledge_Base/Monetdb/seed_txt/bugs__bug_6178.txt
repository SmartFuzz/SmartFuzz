CREATE TABLE flights (tailnum STRING, arr_delay INTEGER, flight_date DATE);
COPY OFFSET 2 INTO flights FROM 'c:\\Users\\anthonyd\\Desktop\\flights.csv' USING DELIMITERS ',','\n','"' NULL as '';
SELECT tailnum, AVG(arr_delay) FROM flights WHERE arr_delay IS NOT NULL GROUP BY tailnum;
SELECT tailnum, AVG(arr_delay) FROM flights GROUP BY tailnum HAVING AVG(arr_delay) IS NOT NULL;
SELECT * FROM (SELECT tailnum, AVG(arr_delay) AS avg_delay FROM flights GROUP BY tailnum) AS avg_flights WHERE avg_delay > 0;
