CREATE TABLE d3.t2 (id INT, value VARCHAR(255), timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE d3.t2_dist (id INT, value VARCHAR(255), timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE VIEW d3.v3 AS SELECT id, COALESCE(value, 'default_value') AS value FROM d3.t2 WHERE id >= 10;
CREATE VIEW d3.v4 AS SELECT id, CONCAT(value, '_suffix') AS value FROM d3.t2_dist WHERE timestamp < CURRENT_TIMESTAMP - INTERVAL '1 DAY';
