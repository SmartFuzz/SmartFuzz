DROP TABLE IF EXISTS segments;
CREATE TABLE segments (meter INT NOT NULL, distance INT CHECK (distance >= 0), speed FLOAT DEFAULT 0.0);
INSERT INTO segments VALUES (1, 1, 1), (9, 9, 9), (NULL, 5, 3);
SELECT t.meter, s.distance, s.speed FROM segments AS s, LATERAL generate_series(COALESCE(s.meter, 0), COALESCE(s.meter, 0) + s.distance + 1) AS t(meter);
SELECT t.*, s.distance, s.speed FROM segments AS s, LATERAL generate_series(s.meter, s.meter + s.distance + 1) AS t(meter) WHERE s.speed IS NOT NULL;
