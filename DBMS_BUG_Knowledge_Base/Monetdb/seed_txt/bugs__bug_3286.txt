CREATE TABLE storage_summary (tblname VARCHAR(255), count BIGINT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO storage_summary (tblname, count) SELECT "table", max("count") FROM sys.storagemodel() WHERE "schema" = 'sys' GROUP BY "table";
CREATE TABLE storage_details (tblname VARCHAR(255) NOT NULL, count INT CHECK (count >= 0), description TEXT);
INSERT INTO storage_details (tblname, count, description) SELECT "table", CAST(max("count") AS INT), 'Storage details for ' || "table" FROM sys.storagemodel() WHERE "schema" = 'sys' GROUP BY "table";
SELECT tblname, count FROM storage_summary WHERE count IS NOT NULL ORDER BY count DESC;
