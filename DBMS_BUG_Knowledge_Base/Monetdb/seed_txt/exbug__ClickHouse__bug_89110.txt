CREATE TABLE IF NOT EXISTS default.initial_complex (n UInt32, description VARCHAR, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (n);
CREATE TABLE IF NOT EXISTS default.target1_complex (x UInt32, status VARCHAR, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (x);
CREATE TABLE IF NOT EXISTS default.target2_complex (x UInt32, value DOUBLE PRECISION, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (x);
CREATE MATERIALIZED VIEW IF NOT EXISTS default.initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM default.initial_complex WHERE RANDOM() < 0.5;
CREATE MATERIALIZED VIEW IF NOT EXISTS default.target1_target2_mv_complex AS SELECT x, CAST(x AS DOUBLE PRECISION) * 1.5 AS value FROM default.target1_complex WHERE status IS NOT NULL;
