DROP SCHEMA IF EXISTS dtobs CASCADE;
CREATE SCHEMA dtobs;
CREATE TABLE dtobs.span_local (links ARRAY(JSON), dummy INT) PRIMARY KEY (dummy);
CREATE TABLE dtobs.span_link_local (linked_id STRING) PRIMARY KEY (linked_id);
CREATE MATERIALIZED VIEW dtobs.mv_bad AS SELECT link['trace_id'] AS linked_id FROM (SELECT UNNEST(links) AS link FROM dtobs.span_local) WHERE link['trace_id'] IS NOT NULL;
ALTER TABLE dtobs.span_local DROP COLUMN dummy;
