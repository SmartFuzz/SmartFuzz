DROP TABLE IF EXISTS test_special_columns;
CREATE TABLE test_special_columns(key BIGINT, value VARCHAR(255), timestamp TIMESTAMP, is_active TINYINT) USING REPLACE ORDER BY key;
INSERT INTO test_special_columns VALUES (1, 'active', '2023-01-01 00:00:00', 1), (2, 'inactive', '2023-01-02 00:00:00', 0), (3, 'active', '2023-01-03 00:00:00', 1);
