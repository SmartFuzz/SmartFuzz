CREATE TABLE user_events (event_id UUID, user_id INT, event_type VARCHAR(10), event_time TIMESTAMP);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT gen_random_uuid(), FLOOR(RANDOM() * 10000), CASE FLOOR(RANDOM() * 3) WHEN 0 THEN 'click' WHEN 1 THEN 'view' ELSE 'purchase' END, NOW() - INTERVAL '1 SECOND' * (FLOOR(RANDOM() * 3600 * 24)) FROM (SELECT 1 FROM generate_series(1, 100000) AS s);
CREATE TABLE user_summary (user_id INT, event_count INT);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
