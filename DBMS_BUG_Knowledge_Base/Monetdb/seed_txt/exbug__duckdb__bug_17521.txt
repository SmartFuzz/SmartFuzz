CREATE TABLE t1 (v1 INT, v2 INT, v3 INT DEFAULT 0);
CREATE TABLE t2 (v1 INT, v2 INT, v4 VARCHAR(255), v5 FLOAT);
INSERT INTO t1 SELECT i, i, i % 2 FROM (SELECT ROW_NUMBER() OVER() - 1 AS i FROM system_range(0, 1000));
INSERT INTO t2 SELECT i, i, CONCAT('value', i), i * 1.0 FROM (SELECT ROW_NUMBER() OVER() - 1 AS i FROM system_range(0, 10000000));
SELECT t1.v2, t2.v2, COUNT(*) FROM t1 INNER JOIN t2 ON (t1.v1 = t2.v1) GROUP BY t1.v2, t2.v2 ORDER BY t1.v2 LIMIT 5;
