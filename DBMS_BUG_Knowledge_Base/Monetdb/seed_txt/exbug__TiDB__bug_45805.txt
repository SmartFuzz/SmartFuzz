CREATE TABLE signed_estimate_volume (id BIGINT NOT NULL AUTO_INCREMENT, estimate_id BIGINT NOT NULL DEFAULT 0, dealer_id VARCHAR(40) NOT NULL DEFAULT '', broker_id VARCHAR(40) NOT NULL DEFAULT '', real_t0_volume BIGINT NOT NULL DEFAULT 0, real_t1_volume BIGINT NOT NULL DEFAULT 0, manager_yid VARCHAR(32) NOT NULL DEFAULT '', estimate_volume BIGINT NOT NULL DEFAULT 0, estimate_year_month VARCHAR(8) NOT NULL DEFAULT '', is_dropped SMALLINT NOT NULL DEFAULT 0, PRIMARY KEY (id), UNIQUE (estimate_id), UNIQUE (estimate_year_month, manager_yid, broker_id, dealer_id, is_dropped));
CREATE TABLE unsigned_estimate_volume (id BIGINT NOT NULL AUTO_INCREMENT, estimate_id BIGINT NOT NULL DEFAULT 0, estimate_year_month VARCHAR(8) NOT NULL DEFAULT '', business_id BIGINT NOT NULL DEFAULT 0, broker_id VARCHAR(40) NOT NULL DEFAULT '', estimate_volume BIGINT NOT NULL DEFAULT 0, manager_yid VARCHAR(32) NOT NULL DEFAULT '', PRIMARY KEY (id), UNIQUE (estimate_id), UNIQUE (estimate_year_month, manager_yid, broker_id, business_id));

SELECT estimate_year_month, manager_yid, real_t1_volume, real_t0_volume, estimate_volume + COALESCE(b.estimate_volume, 0) AS estimate_volume, COALESCE(a.increase_sum, 0) + COALESCE(b.increase_sum, 0) AS increase_sum, COALESCE(a.decrease_sum, 0) + COALESCE(b.decrease_sum, 0) AS decrease_sum FROM (SELECT estimate_year_month, manager_yid, SUM(real_t1_volume) AS real_t1_volume, SUM(real_t0_volume) AS real_t0_volume, SUM(estimate_volume) AS estimate_volume, SUM(CASE WHEN estimate_volume > real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS increase_sum, SUM(CASE WHEN estimate_volume < real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS decrease_sum FROM signed_estimate_volume GROUP BY manager_yid) AS a LEFT JOIN (SELECT manager_yid, SUM(estimate_volume) AS estimate_volume, SUM(estimate_volume) AS increase_sum, 0 AS decrease_sum FROM unsigned_estimate_volume GROUP BY manager_yid) AS b ON a.manager_yid = b.manager_yid;
