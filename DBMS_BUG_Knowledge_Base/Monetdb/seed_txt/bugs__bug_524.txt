CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(100), created_at TIMESTAMP WITH TIME ZONE DEFAULT now());
CREATE TABLE events (event_id SERIAL PRIMARY KEY, event_time TIMESTAMP WITH TIME ZONE NOT NULL, user_id INT REFERENCES users(id));
