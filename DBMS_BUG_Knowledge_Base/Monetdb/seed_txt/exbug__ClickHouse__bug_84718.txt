DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id INT, name VARCHAR(255), value DOUBLE PRECISION, created_at TIMESTAMP);
INSERT INTO test_table SELECT number, CONCAT('name_', number::TEXT, ' ', (number % 10)::TEXT) AS name, number * 1.5 AS value, NOW() FROM generate_series(1, 100) AS number;
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
