CREATE TABLE urls(urlid bigint primary key, c bigint, d bigint, url_text varchar(255));
CREATE TABLE keys(urlid bigint, a bigint, b bigint, foreign key(urlid) references urls(urlid), created_at timestamp DEFAULT CURRENT_TIMESTAMP);
INSERT INTO urls(urlid, c, d, url_text) VALUES (1, 10, 20, 'http://example.com/1');
INSERT INTO keys(urlid, a, b) VALUES (1, 50, 150), (1, 50, 250), (2, 60, 180);
SELECT * FROM urls WHERE urlid IN ((SELECT urlid FROM keys WHERE a < 100 AND b < 200) INTERSECT (SELECT urlid FROM keys WHERE a < 100 AND b < 300));
