CREATE TABLE test_table (id INT PRIMARY KEY, name TEXT NOT NULL, value INT CHECK (value >= 0));
CREATE INDEX CaseSensitiveIndex ON test_table (name, value);
CREATE INDEX not_case_sensitive ON test_table (value, id);
SELECT COUNT(*) FROM test_table;
