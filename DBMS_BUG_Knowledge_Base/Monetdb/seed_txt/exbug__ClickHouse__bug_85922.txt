CREATE TABLE IF NOT EXISTS users (id BIGINT, name VARCHAR(255), created_at TIMESTAMP);
CREATE TABLE IF NOT EXISTS user_logs (user_id BIGINT, log_message VARCHAR(255), log_time TIMESTAMP);

CREATE TABLE IF NOT EXISTS user_summary AS SELECT user_id, COUNT(*) AS log_count FROM user_logs GROUP BY user_id;
