CREATE TABLE customer (c_id INT PRIMARY KEY, c_name VARCHAR(50), c_balance DECIMAL(10, 2) NOT NULL);
CREATE TABLE orders (o_id INT PRIMARY KEY, o_customer_id INT, o_amount DECIMAL(10, 2), FOREIGN KEY (o_customer_id) REFERENCES customer(c_id));

SELECT c.c_name, COALESCE(SUM(o.o_amount), 0) AS total_spent FROM customer c LEFT JOIN orders o ON c.c_id = o.o_customer_id GROUP BY c.c_id;
