DROP TABLE IF EXISTS xyz;
CREATE TABLE xyz (id integer PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY, name varchar(255), age integer, created_at timestamp DEFAULT CURRENT_TIMESTAMP);

CREATE OR REPLACE FUNCTION get_age() RETURNS integer AS 'BEGIN FOR user_record IN (SELECT * FROM xyz WHERE age IS NOT NULL) LOOP RETURN user_record.age;
