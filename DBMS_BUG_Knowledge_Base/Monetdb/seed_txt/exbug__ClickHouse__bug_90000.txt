CREATE TABLE T1(A BIGINT, B DOUBLE) AS SELECT number, number * 1.5 FROM generate_series(1, 100000);
CREATE TABLE T2(C VARCHAR, D INT) AS SELECT CAST(number AS VARCHAR) AS C, CASE WHEN number % 2 = 0 THEN number ELSE NULL END AS D FROM generate_series(1, 1000);
CREATE TABLE item_dict2 (id BIGINT PRIMARY KEY, a1 BIGINT, a2 INT);
INSERT INTO item_dict2 SELECT id, a1, a2 FROM T2;
SELECT DISTINCT A, (SELECT a1 FROM item_dict2 WHERE id = B) AS a1, (SELECT a2 FROM item_dict2 WHERE id = B) AS a2 FROM (SELECT A, B FROM T1);
DROP TABLE item_dict2;
