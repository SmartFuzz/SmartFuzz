DROP DATABASE IF EXISTS my_db;
CREATE DATABASE my_db;
CREATE TABLE my_db.my_table (id BIGINT, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE VIEW my_db.my_view1 AS SELECT * FROM my_db.my_table WHERE id = ?;
CREATE VIEW my_db.my_view2 AS SELECT * FROM my_db.my_view1 WHERE id = CAST(NULL AS BIGINT);
