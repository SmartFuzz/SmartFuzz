CREATE TABLE t1(x INTEGER, y INTEGER CHECK (x >= 0 AND y >= 0));
CREATE TABLE t2(a INTEGER, b INTEGER DEFAULT 10);
INSERT INTO t1(x, y) VALUES (1, 2), (3, 4), (5, NULL);
INSERT INTO t2(a, b) VALUES (0, 5), (5, 15), (NULL, 20);
SELECT * FROM t1 FULL OUTER JOIN t2 ON x BETWEEN a AND b OR y BETWEEN a AND b OR a BETWEEN x AND y;
