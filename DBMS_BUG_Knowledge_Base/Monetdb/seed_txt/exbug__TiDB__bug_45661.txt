CREATE TABLE sales ( year INT DEFAULT NULL, country VARCHAR(20) DEFAULT NULL, product VARCHAR(32) DEFAULT NULL, profit INT DEFAULT NULL );
INSERT INTO sales (year, country, product, profit) VALUES (2001, 'USA', 'Gadget', 500), (2002, 'USA', 'Gadget', 800), (2003, 'USA', 'Gadget', 1200), (2003, 'Canada', 'Gadget', 1000);
SELECT year + 1 AS y, SUM(profit) AS profit FROM sales GROUP BY year + 1 WITH ROLLUP HAVING y > 2002 ORDER BY y, profit;
