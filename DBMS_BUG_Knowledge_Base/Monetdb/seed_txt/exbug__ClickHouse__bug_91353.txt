CREATE TABLE test_table1 (id INT, value FLOAT, created_at TIMESTAMP);
CREATE TABLE test_table2 (id INT, description VARCHAR(255), is_active BOOLEAN);
CREATE MATERIALIZED VIEW test_mv1 AS SELECT id, value FROM test_table2 WHERE is_active = TRUE;
CREATE PROCEDURE refresh_mv2() BEGIN INSERT INTO test_table2 SELECT id, description FROM test_table1 WHERE created_at > NOW() - INTERVAL '5 MINUTE';
END;
CALL refresh_mv2();
