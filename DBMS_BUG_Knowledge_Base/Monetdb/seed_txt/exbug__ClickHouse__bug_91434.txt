CREATE TABLE json_test_v2 (json JSON, col_b BIGINT, col_c VARCHAR);
INSERT INTO json_test_v2 SELECT number AS a, number * 2 AS b, CAST(number * 3 AS VARCHAR) AS c FROM generate_series(1, 10000000);
SELECT col_b, col_c FROM json_test_v2 WHERE col_b IS NOT NULL;
SELECT json->>'b' AS direct_b, json->>'c' AS direct_c FROM json_test_v2 WHERE (json->>'b')::BIGINT > 5000000;
SELECT col_b, COUNT(*) FROM json_test_v2 GROUP BY col_b HAVING col_b < 100;
