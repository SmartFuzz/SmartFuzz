CREATE TABLE test_table (id INT PRIMARY KEY, flag BOOLEAN, description STRING);
INSERT INTO test_table (id, flag, description) VALUES (1, true, 'First entry'), (2, false, 'Second entry'), (3, null, 'Third entry');
CREATE FUNCTION evaluate_flag(b BOOLEAN) RETURNS STRING BEGIN IF b THEN RETURN 'true';
ELSEIF NOT b THEN RETURN 'false';
ELSE RETURN 'null';
END IF;
END;
SELECT id, evaluate_flag(flag) AS evaluation FROM test_table;
SELECT evaluate_flag(NULLIF(flag, false)) AS adjusted_evaluation FROM test_table;
