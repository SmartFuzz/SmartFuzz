CREATE OR REPLACE FUNCTION test_subfact(num INT) RETURNS TABLE (n INT, sub BIGINT) AS $$ BEGIN RETURN QUERY WITH RECURSIVE cte(n, psub, sub) AS (SELECT 0 AS n, 1 AS psub, 1 AS sub UNION ALL SELECT n + 1, sub, (n * (sub + psub)) FROM cte WHERE n < num) SELECT n, sub FROM cte;
END;
$$;
