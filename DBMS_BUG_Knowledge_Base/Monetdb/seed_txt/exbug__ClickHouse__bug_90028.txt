CREATE TABLE t1 (c0 BIT(5), c1 INT, c2 VARCHAR(255));
INSERT INTO t1 (c0, c1, c2) SELECT ARRAY[CAST(number % 10 AS FLOAT), CAST((-number) AS FLOAT)], number, 'test' FROM generate_series(0, 19);
CREATE TABLE t2 (c0 INT, c1 FLOAT, c2 TIMESTAMP);
INSERT INTO t2 (c0, c1, c2) SELECT number, NULL, CURRENT_TIMESTAMP FROM generate_series(0, 9);
SELECT t1.c0, t1.c1 FROM t1 JOIN t2 ON t1.c1 = t2.c0 ORDER BY t1.c0, t1.c1;
