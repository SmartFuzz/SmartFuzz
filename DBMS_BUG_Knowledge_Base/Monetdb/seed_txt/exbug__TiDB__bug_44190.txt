CREATE TABLE employee_info ( employee_id bigint NOT NULL, employee_name varchar(100) NOT NULL, department_id bigint NOT NULL, salary decimal(10,2) NOT NULL, UNIQUE (employee_id, department_id) );

CREATE TABLE department_info ( department_id bigint NOT NULL, department_name varchar(100) NOT NULL, PRIMARY KEY (department_id) );
