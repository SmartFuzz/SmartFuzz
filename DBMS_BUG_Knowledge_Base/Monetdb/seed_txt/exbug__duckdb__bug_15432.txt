SELECT id, SUM(COALESCE(NULLIF(values[1], ''), '0')::DOUBLE PRECISION) + SUM(COALESCE(NULLIF(values[2], ''), '0')::DOUBLE PRECISION) + SUM(COALESCE(NULLIF(values[3], ''), '0')::DOUBLE PRECISION) FROM test_data GROUP BY id;
CREATE OR REPLACE FUNCTION custom_sum(lst VARCHAR[]) RETURNS DOUBLE AS $$ SELECT SUM(COALESCE(NULLIF(lst[1], ''), '0')::DOUBLE PRECISION) + SUM(COALESCE(NULLIF(lst[2], ''), '0')::DOUBLE PRECISION) + SUM(COALESCE(NULLIF(lst[3], ''), '0')::DOUBLE PRECISION);
$$ LANGUAGE SQL;
SELECT id, custom_sum(values) FROM test_data WHERE values IS NOT NULL GROUP BY id;
