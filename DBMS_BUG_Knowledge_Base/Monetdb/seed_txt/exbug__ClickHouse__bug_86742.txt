DROP TABLE IF EXISTS job_sample;
CREATE TABLE job_sample (id INT, posted_at DATE, discovered_at TIMESTAMP, url TEXT, title TEXT, company_name TEXT, description TEXT);
COPY job_sample FROM 'https://media.theirstack.com/jobs/jan-2025-sample.parquet' USING DELIMITER ',';
ALTER TABLE job_sample ADD COLUMN description_matches_python INT DEFAULT (CASE WHEN description LIKE '%Python%' THEN 1 ELSE 0 END);
UPDATE job_sample SET description_matches_python = CASE WHEN description ILIKE '%Python%' THEN 1 ELSE 0 END;
