CREATE TABLE t1(c0 INTEGER, c1 TEXT, c2 DOUBLE PRECISION);
INSERT INTO t1(c0, c1, c2) VALUES(1, 'A', 1.5);
INSERT INTO t1(c0, c1, c2) VALUES(2, 'B', NULL);
CREATE VIEW v3 AS SELECT c1, c2 FROM t1 WHERE c0 >= 1 AND c0 <= 2;
