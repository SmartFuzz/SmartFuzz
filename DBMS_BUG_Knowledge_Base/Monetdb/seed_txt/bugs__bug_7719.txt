CREATE TABLE v1 ( id INT PRIMARY KEY, name VARCHAR(50), age INT CHECK (age >= 0) );
INSERT INTO v1 (id, name, age) VALUES (1, 'Alice', 30), (2, 'Bob', 25), (3, 'Charlie', NULL);
CREATE TABLE v2 ( group_id INT, description VARCHAR(100), FOREIGN KEY (group_id) REFERENCES v1(id) );
INSERT INTO v2 (group_id, description) VALUES (1, 'Group A'), (2, 'Group B'), (3, 'Group C');
SELECT id, name, LAST_VALUE(description) OVER (PARTITION BY group_id ORDER BY id RANGE BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING) FROM v1 LEFT JOIN v2 ON v1.id = v2.group_id;
