DROP TABLE IF EXISTS t;
CREATE TABLE t (pk1 VARCHAR(128), pk2 VARCHAR(128), pk3 VARCHAR(128), data VARCHAR(128), PRIMARY KEY (pk1, pk2, pk3));
INSERT INTO t (pk1, pk2, pk3, data) VALUES (UUID(), UUID(), UUID(), UUID()), (UUID(), UUID(), UUID(), UUID());
SELECT pk1, COUNT(*) FROM t GROUP BY pk1 ORDER BY COUNT(*) LIMIT 10;
SELECT pk1, SUM(COALESCE(data, '0')) FROM t GROUP BY pk1 HAVING SUM(COALESCE(data, '0')) IS NOT NULL;
