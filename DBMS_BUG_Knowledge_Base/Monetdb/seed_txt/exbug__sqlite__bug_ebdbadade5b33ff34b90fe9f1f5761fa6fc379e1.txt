CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0));
CREATE TABLE orders (id INT PRIMARY KEY, user_id INT, amount DOUBLE NOT NULL, FOREIGN KEY(user_id) REFERENCES users(id));
