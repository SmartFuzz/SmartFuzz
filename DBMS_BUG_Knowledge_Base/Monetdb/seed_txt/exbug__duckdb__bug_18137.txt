CREATE TABLE test_large_lists (id INTEGER, gts LIST(LIST(BYTE)));
INSERT INTO test_large_lists SELECT i, LIST(LIST(CAST(random() % 256 AS BYTE), CAST(random() % 256 AS BYTE))) FROM 1 TO 500000 AS i;
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, UNNEST(gts) AS inner_list FROM list_view;
