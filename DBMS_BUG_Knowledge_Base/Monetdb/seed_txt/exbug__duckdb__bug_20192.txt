CREATE TABLE products (id INTEGER PRIMARY KEY, name VARCHAR, description VARCHAR, price DECIMAL(10, 2), created_at TIMESTAMP);
CREATE TABLE orders (id INTEGER PRIMARY KEY, product_id INTEGER, order_date TIMESTAMP, quantity INTEGER, large_blob BYTEA, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, 'Product ' || i AS name, 'Description for product ' || i AS description, ROUND(RANDOM() * 100, 2) AS price, '2024-01-01' + INTERVAL (i) DAY AS created_at FROM (SELECT ROW_NUMBER() OVER () AS i FROM orders LIMIT 10000) AS t;
INSERT INTO orders SELECT i AS id, (i % 10000) + 1 AS product_id, '2024-02-01' + INTERVAL (i) DAY AS order_date, (i % 10) + 1 AS quantity, repeat('y', 2000) AS large_blob FROM (SELECT ROW_NUMBER() OVER () AS i FROM products LIMIT 5000000) AS t;
EXPLAIN SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01' AND '2024-04-01' AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
