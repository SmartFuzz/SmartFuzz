DROP TABLE IF EXISTS t1, t2;
CREATE TABLE t1(id int PRIMARY KEY, col1 varchar(10), col2 varchar(10), col3 varchar(10));
CREATE TABLE t2(id int PRIMARY KEY, col1 varchar(10) NOT NULL, col2 varchar(10) NOT NULL, col3 varchar(10) NOT NULL);
INSERT INTO t1 VALUES(1, NULL, NULL, NULL), (2, NULL, NULL, NULL), (3, NULL, NULL, NULL);
INSERT INTO t2 VALUES(1, 'a', 'aa', 'aaa'), (2, 'b', 'bb', 'bbb'), (3, 'c', 'cc', 'ccc');
UPDATE t1 SET col1 = (SELECT t2.col1 FROM t2 WHERE t2.id = t1.id), col2 = (SELECT t2.col2 FROM t2 WHERE t2.id = t1.id), col3 = (SELECT t2.col3 FROM t2 WHERE t2.id = t1.id) WHERE t1.id IN (SELECT id FROM t2 WHERE id > 1);
