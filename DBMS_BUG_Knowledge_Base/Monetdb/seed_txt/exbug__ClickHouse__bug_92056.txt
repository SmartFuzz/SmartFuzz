DROP TABLE IF EXISTS events;
CREATE TABLE events ( Id BIGINT, Payload VARCHAR(255), Time TIMESTAMP, Status VARCHAR(255) ) USING MEMORY;
INSERT INTO events SELECT number, CONCAT('Payload_', CAST(number AS VARCHAR)), TIMESTAMP '2024-01-01 00:00:00' + INTERVAL number MINUTE, CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM generate_series(0, 9);
DROP TABLE IF EXISTS attributes;
CREATE TABLE attributes ( EventId BIGINT, Attribute VARCHAR(255), Value DOUBLE PRECISION DEFAULT 0.0 ) USING MERGE;
SELECT COUNT(*), COUNT(CASE WHEN t1.Attribute != '' THEN 1 END), SUM(CAST(t1.Attribute AS BIGINT)) FROM events AS t0 JOIN attributes AS t1 ON t1.EventId = t0.Id;
