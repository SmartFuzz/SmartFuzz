DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INT, accepted_at TIMESTAMP(3), delivery_report_at VARCHAR(255));
INSERT INTO agg_test SELECT number AS id, TIMESTAMP '2025-09-24 00:00:00' + number * INTERVAL '1' SECOND AS accepted_at, (SELECT MAX(CASE WHEN number % 2 = 1 THEN TO_CHAR(number) END) FROM numbers(100) WHERE number <= id) AS delivery_report_at FROM numbers(100);
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
