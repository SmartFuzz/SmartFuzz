CREATE TABLE employees (id integer PRIMARY KEY, name varchar, salary decimal(10, 2), department_id integer);
CREATE TABLE departments (id integer PRIMARY KEY, name varchar NOT NULL);

SELECT d.*, e.*, RANK() OVER (ORDER BY e.salary) AS salary_rank FROM employees e JOIN departments d ON e.department_id = d.id;
