CREATE TABLE test (a INT, b VARCHAR(255), c FLOAT CHECK (c >= 0));
INSERT INTO test (a, b, c) VALUES (1, 'hello', 3.5), (2, 'world', 4.0), (1, NULL, 2.5);
SELECT a, b, char_length(b), count(*) FROM test GROUP BY a, b HAVING count(a) = char_length(b);
SELECT a, b, char_length(b), count(*) FROM test GROUP BY a, b HAVING count(a) = char_length('test');
SELECT a, b, char_length(b), count(*) FROM test WHERE c IS NOT NULL GROUP BY a, b HAVING count(a) = 1;
