CREATE TABLE IF NOT EXISTS user_activity (activity_id BIGINT, user_id INT, activity_time TIMESTAMP, activity_type VARCHAR(10), details VARCHAR(255));
INSERT INTO user_activity SELECT number AS activity_id, MOD(RANDOM(), 1000) AS user_id, NOW() + INTERVAL number * 5 SECOND AS activity_time, CASE WHEN MOD(number, 3) = 0 THEN 'login' WHEN MOD(number, 3) = 1 THEN 'logout' ELSE 'purchase' END AS activity_type, CONCAT('Activity details for ', number) AS details FROM (SELECT ROW_NUMBER() OVER () AS number FROM system.tables LIMIT 100);
CREATE TABLE IF NOT EXISTS product_sales (sale_id INT, product_id INT, sale_time TIMESTAMP, quantity SMALLINT, price FLOAT);
INSERT INTO product_sales SELECT number AS sale_id, MOD(RANDOM(), 500) AS product_id, NOW() + INTERVAL number * 2 SECOND AS sale_time, MOD(RANDOM(), 10) + 1 AS quantity, (MOD(RANDOM(), 100) + 1) * 1.5 AS price FROM (SELECT ROW_NUMBER() OVER () AS number FROM system.tables LIMIT 50);
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL '1 DAY' GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
