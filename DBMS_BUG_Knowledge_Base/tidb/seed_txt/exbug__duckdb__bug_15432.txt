CREATE TABLE test_data (id INT, values JSON);
INSERT INTO test_data VALUES (1, JSON_ARRAY('1', '2', '3')), (2, JSON_ARRAY('4', '5', NULL)), (3, JSON_ARRAY('6', '7', '8'));
SELECT id, SUM(CAST(JSON_UNQUOTE(JSON_EXTRACT(values, '$[*]')) AS DECIMAL)) FROM test_data GROUP BY id;
CREATE FUNCTION custom_sum(lst JSON) RETURNS DECIMAL DETERMINISTIC RETURN (SELECT SUM(CAST(JSON_UNQUOTE(JSON_EXTRACT(lst, '$[*]')) AS DECIMAL)));
SELECT id, custom_sum(values) FROM test_data WHERE values IS NOT NULL GROUP BY id;
