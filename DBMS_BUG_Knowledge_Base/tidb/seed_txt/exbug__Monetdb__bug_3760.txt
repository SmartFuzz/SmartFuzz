CREATE TABLE t3 (id INT PRIMARY KEY, name VARCHAR(50), value FLOAT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO t3 VALUES (1, 'test1', 10.5, NULL), (2, 'test2', 20.0, '2023-01-01 10:00:00');
SELECT name, (SELECT COUNT(*) FROM t3 AS x WHERE x.value > t3.value) AS count_value, CASE WHEN created_at IS NULL THEN 'No Date' ELSE 'Has Date' END AS date_status FROM t3;
SELECT name AS name_alias, (SELECT COUNT(*) FROM t3 AS x WHERE x.id < t3.id) AS count_id, CASE value WHEN 20.0 THEN 'Matched' END AS match_status FROM t3;
SELECT value, CASE WHEN value < 15.0 THEN 'Low' ELSE 'High' END AS value_status, (SELECT AVG(value) FROM t3) AS avg_value FROM t3;
