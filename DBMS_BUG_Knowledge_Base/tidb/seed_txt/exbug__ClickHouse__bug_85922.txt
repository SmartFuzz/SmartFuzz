CREATE TABLE IF NOT EXISTS users (id BIGINT, name VARCHAR(255), created_at DATETIME);
CREATE TABLE IF NOT EXISTS user_logs (user_id BIGINT, log_message VARCHAR(255), log_time DATETIME);
