CREATE TABLE courses (course_id INT PRIMARY KEY, course_name VARCHAR(255) NOT NULL, credits INT CHECK (credits > 0));
INSERT INTO courses VALUES (1, 'Mathematics', 4), (2, 'Physics', 3), (3, 'Chemistry', 4);
CREATE TABLE enrollments (enrollment_id INT PRIMARY KEY, sid INT, course_id INT, FOREIGN KEY (sid) REFERENCES students(sid), FOREIGN KEY (course_id) REFERENCES courses(course_id));
INSERT INTO enrollments VALUES (1, 1, 1), (2, 2, 2), (3, 3, 3);
SELECT c.course_name FROM courses c JOIN enrollments e ON c.course_id = e.course_id WHERE e.sid IS NOT NULL;
