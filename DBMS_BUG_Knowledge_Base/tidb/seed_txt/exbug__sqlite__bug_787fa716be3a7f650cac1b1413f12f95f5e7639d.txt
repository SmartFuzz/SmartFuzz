CREATE TABLE musicians (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, genre VARCHAR(255));
CREATE TABLE songs (id INT PRIMARY KEY, title VARCHAR(255) NOT NULL, musician_id INT, FOREIGN KEY (musician_id) REFERENCES musicians(id));
