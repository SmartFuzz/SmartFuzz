CREATE TABLE default.t_proj_test1 (id INT UNSIGNED, name VARCHAR(255), value DOUBLE, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO default.t_proj_test1 SELECT number, CONCAT('Name', number), number * 1.5, NOW() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 10000) AS numbers;
CREATE TABLE default.t_proj_test2 (user_id INT UNSIGNED, score INT, timestamp DATE DEFAULT CURRENT_DATE);
INSERT INTO default.t_proj_test2 SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END, CURRENT_DATE FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 5000) AS numbers;
SELECT user_id, SUM(score) AS total_score FROM default.t_proj_test2 GROUP BY user_id ORDER BY total_score DESC;
