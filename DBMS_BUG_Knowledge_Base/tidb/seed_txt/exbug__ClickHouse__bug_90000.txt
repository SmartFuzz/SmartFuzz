CREATE TABLE T1 (A BIGINT, B DOUBLE) AS SELECT number, number * 1.5 FROM numbers(100000);
CREATE TABLE T2 (C VARCHAR(255), D INT) AS SELECT CAST(number AS CHAR) AS C, CASE WHEN number % 2 = 0 THEN number ELSE NULL END AS D FROM numbers(1000);
SELECT DISTINCT A, (SELECT a1 FROM T2 WHERE id = CAST(B AS UNSIGNED)) AS a1, (SELECT a2 FROM T2 WHERE id = CAST(B AS UNSIGNED)) AS a2 FROM (SELECT A, B FROM T1);
DROP TABLE T2;
DROP TABLE T1;
