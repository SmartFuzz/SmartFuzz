CREATE TABLE test_data (id INT UNSIGNED, ts DATETIME, value VARCHAR(255), status VARCHAR(255)) CHARACTER SET utf8mb4;
INSERT INTO test_data (id, ts, value, status) SELECT number, NOW() + INTERVAL number SECOND, CAST(number % 5 AS CHAR), CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers;
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > NOW() - INTERVAL 1 DAY AND value LIKE '1%';
SELECT DISTINCT id FROM test_data WHERE ts BETWEEN '2023-01-01' AND '2023-12-31' AND status IS NOT NULL;
