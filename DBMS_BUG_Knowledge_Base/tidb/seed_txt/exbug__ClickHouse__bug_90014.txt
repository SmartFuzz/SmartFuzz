CREATE TABLE position_evaluation_v2 (
    `symbol` VARCHAR(255),
    `evaluated_at` DATETIME(3),
    `mark_price` DECIMAL(18, 8),
    `volume` INT UNSIGNED
);

SELECT mark_price, LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price
FROM position_evaluation_v2;
SELECT mark_price, LAG(mark_price, 1, NULL) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price_with_default
FROM position_evaluation_v2;
SELECT mark_price, LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price_invalid
FROM position_evaluation_v2;
