CREATE TABLE cols (c VARCHAR(255));
INSERT INTO cols (SELECT CONCAT('c', i) FROM (SELECT 1 AS i UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11 UNION ALL SELECT 12 UNION ALL SELECT 13 UNION ALL SELECT 14 UNION ALL SELECT 15 UNION ALL SELECT 16 UNION ALL SELECT 17 UNION ALL SELECT 18 UNION ALL SELECT 19 UNION ALL SELECT 20));
CREATE TABLE matrix (col1 VARCHAR(255), col2 VARCHAR(255), val TINYINT UNSIGNED NOT NULL CHECK (val < 250));
INSERT INTO matrix (SELECT c1.c, c2.c, CAST(FLOOR(RAND() * 250) AS TINYINT UNSIGNED) FROM cols c1 JOIN cols c2 ON c1.c != c2);
SELECT col1, col2, SUM(val) AS total_val FROM matrix GROUP BY col1, col2 ORDER BY col2 LIMIT 2;
