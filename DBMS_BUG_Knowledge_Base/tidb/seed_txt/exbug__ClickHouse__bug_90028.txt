CREATE TABLE t1 (c0 JSON, c1 INT, c2 VARCHAR(255));
INSERT INTO t1 (c0, c1, c2) SELECT JSON_ARRAY(CAST(number % 10 AS DECIMAL(10, 2)), CAST((-number) AS DECIMAL(10, 2)), number, 'test' FROM numbers(20);
CREATE TABLE t2 (c0 INT, c1 FLOAT, c2 DATETIME);
INSERT INTO t2 (c0, c1, c2) SELECT number, NULL, NOW() FROM numbers(10);
SELECT t1.c0, t1.c1 FROM t1 JOIN t2 ON t1.c1 = t2.c0 ORDER BY t1.c0, t1.c1;
