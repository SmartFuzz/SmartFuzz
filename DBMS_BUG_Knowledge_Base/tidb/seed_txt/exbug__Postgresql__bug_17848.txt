DROP TABLE IF EXISTS demo;
CREATE TABLE demo(val BIGINT, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE INDEX demo_idx ON demo(val);
INSERT INTO demo(val, name) SELECT val, CONCAT('Name ', val) FROM (SELECT 1 AS val UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10) AS series;
ANALYZE demo;
