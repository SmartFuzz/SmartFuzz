CREATE TABLE `sales` ( `sale_id` int NOT NULL AUTO_INCREMENT, `product_id` int NOT NULL, `sale_date` date NOT NULL, `amount` decimal(10,2) NOT NULL, PRIMARY KEY (`sale_id`), FOREIGN KEY (`product_id`) REFERENCES `products`(`id`));
