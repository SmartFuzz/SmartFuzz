CREATE TABLE products (id BIGINT PRIMARY KEY, name VARCHAR(255), price DOUBLE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO products VALUES (1, 'Product A', 10.5, '2023-01-01 10:00:00'), (2, 'Product B', 15.0, '2023-01-02 11:00:00'), (3, 'Product C', 20.0, '2023-01-03 12:00:00');
CREATE FUNCTION get_product_name(_id BIGINT) RETURNS VARCHAR(255) BEGIN DECLARE _name VARCHAR(255);
SELECT name INTO _name FROM products WHERE id = _id LIMIT 1;
RETURN _name;
END;
