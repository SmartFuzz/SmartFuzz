DROP DATABASE IF EXISTS db1;
CREATE DATABASE db1;
USE db1;
CREATE TABLE t1(c0 INT, c1 JSON, c2 VARCHAR(10) DEFAULT 'default');
INSERT INTO t1(c0, c1, c2) VALUES (1, JSON_OBJECT('key', 'value'), NULL);
CREATE VIEW v1 AS SELECT c0, JSON_EXTRACT(c1, '$.key') AS extracted_key, c2 FROM t1;
SELECT * FROM v1 NATURAL LEFT JOIN t1 WHERE (JSON_EXTRACT(c1, '$.key') IS NULL) OR (c2 = 'default');
