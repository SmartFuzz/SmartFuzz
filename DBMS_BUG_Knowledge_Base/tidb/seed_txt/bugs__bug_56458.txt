CREATE TABLE authors (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(100) NOT NULL, email VARCHAR(100) UNIQUE NOT NULL);
CREATE TABLE books (id INT PRIMARY KEY AUTO_INCREMENT, title VARCHAR(200) NOT NULL, publication_date DATE NOT NULL, author_id INT, FOREIGN KEY (author_id) REFERENCES authors(id) ON DELETE CASCADE);
