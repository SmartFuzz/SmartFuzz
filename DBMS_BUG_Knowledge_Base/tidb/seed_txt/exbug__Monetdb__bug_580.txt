CREATE TABLE product (product_id INT NOT NULL, product_name VARCHAR(100) NOT NULL, category_id INT NOT NULL, price FLOAT NOT NULL, created_at TIMESTAMP NOT NULL, PRIMARY KEY (product_id));
CREATE TABLE category (category_id INT NOT NULL, category_name VARCHAR(100) NOT NULL, PRIMARY KEY (category_id));
