CREATE TABLE t1(a INT, b VARCHAR(255), c DATE);
CREATE TABLE t2(d FLOAT, e BOOLEAN);
CREATE VIEW v1 AS SELECT a, CAST(NULL AS DATE) AS null_date FROM t1;
SELECT * FROM t1 LEFT JOIN t2 ON t1.a = CAST(t2.d AS SIGNED) WHERE t1.b IS NULL;
SELECT * FROM v1 JOIN (SELECT e FROM t2 WHERE e = TRUE) AS sub2 ON v1.null_date < CAST('2023-01-01' AS DATE);
