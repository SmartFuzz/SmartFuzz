CREATE FUNCTION json_each(input JSON) RETURNS TABLE (key VARCHAR, value JSON) AS $$ BEGIN RETURN SELECT CASE WHEN JSON_TYPE(val) = 'ARRAY' THEN CAST(JSON_UNQUOTE(JSON_EXTRACT(val, '$[*]')) AS CHAR) ELSE JSON_KEYS(val) END AS key, JSON_UNQUOTE(JSON_EXTRACT(val, key)) AS value FROM (SELECT input AS val) AS t;
END;
$$;
SELECT * FROM (SELECT '[{"key": "value1"}, {"key": "value2"}]' AS message) AS t CROSS JOIN json_each(message);
