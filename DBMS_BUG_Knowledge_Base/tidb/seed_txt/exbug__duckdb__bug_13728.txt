CREATE FUNCTION unpivot_test(vars VARCHAR(255)) RETURNS TABLE (a INT, b VARCHAR(255)) AS BEGIN RETURN (SELECT a, b FROM t2 WHERE a IN (SELECT CAST(SUBSTRING_INDEX(SUBSTRING_INDEX(vars, ',', numbers.n), ',', -1) AS UNSIGNED) FROM (SELECT 1 AS n UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4) numbers WHERE CHAR_LENGTH(vars) - CHAR_LENGTH(REPLACE(vars, ',', '')) >= numbers.n - 1));
END;

CREATE FUNCTION unpivot_with_null() RETURNS TABLE (c DOUBLE, d BOOLEAN) AS BEGIN RETURN (SELECT c, d FROM t2 WHERE c IS NOT NULL OR d IS NOT NULL);
