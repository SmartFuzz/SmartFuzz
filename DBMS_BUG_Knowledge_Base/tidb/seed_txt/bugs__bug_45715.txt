CREATE TABLE `sales` ( `year` int(11) NOT NULL, `country` varchar(20) NOT NULL, `product` varchar(32) NOT NULL, `profit` int(11) DEFAULT 0, PRIMARY KEY (`year`, `country`, `product`));

SELECT year, country, product, SUM(profit) AS total_profit FROM sales GROUP BY year, country, product WITH ROLLUP ORDER BY GROUPING(year), country;
