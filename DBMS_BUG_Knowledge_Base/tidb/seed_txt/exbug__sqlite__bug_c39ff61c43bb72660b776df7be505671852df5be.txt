CREATE TABLE users (id BIGINT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0));
CREATE TABLE comments (id BIGINT PRIMARY KEY AUTO_INCREMENT, user_id BIGINT, content VARCHAR(255) NOT NULL, FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE);
