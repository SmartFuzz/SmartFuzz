CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id INT PRIMARY KEY AUTO_INCREMENT, user_id INT, amount DECIMAL(10, 2) NOT NULL, order_date DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TRIGGER update_order_timestamp AFTER UPDATE ON orders FOR EACH ROW BEGIN UPDATE orders SET order_date = CURRENT_TIMESTAMP WHERE id = NEW.id;
END;
