DROP TABLE IF EXISTS data_local;
CREATE TABLE data_local (id INT UNSIGNED, value VARCHAR(255)) AS SELECT number, CAST(number AS CHAR) FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers;
CREATE TABLE data_distributed AS SELECT * FROM data_local;
SELECT COUNT(*) FROM (SELECT DISTINCT value FROM data_distributed) AS distinct_values;
