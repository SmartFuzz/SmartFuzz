CREATE TABLE testgroup2 (id INT, isB2b INT, priceGross DECIMAL(10, 2), priceNet DECIMAL(10, 2), discount DECIMAL(10, 2));
INSERT INTO testgroup2 VALUES (1, 1, 100.00, 80.00, 10.00), (2, 0, 150.00, 120.00, 15.00);
SELECT id, isB2b, priceGross, priceNet, discount, IF(isB2b = 1, SUM(priceGross - discount), SUM(priceNet)) AS total FROM testgroup2 GROUP BY id, isB2b, priceGross, priceNet, discount;
SELECT id, isB2b, priceGross, priceNet, discount, CASE WHEN isB2b = 1 THEN SUM(priceGross) ELSE SUM(priceNet + discount) END AS total FROM testgroup2 GROUP BY id, isB2b, priceGross, priceNet, discount;
SELECT id, isB2b, priceGross, priceNet, discount, COALESCE(SUM(IF(isB2b = 1, priceGross, NULL)), AVG(priceNet)) AS total FROM testgroup2 GROUP BY id, isB2b, priceGross, priceNet, discount;
