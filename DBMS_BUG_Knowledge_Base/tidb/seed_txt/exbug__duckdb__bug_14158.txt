CREATE TABLE json_data (MainLevel JSON);
INSERT INTO json_data VALUES ('{"MainLevel":[{"More":{}},{"More":{"Level2":[{"Field1":"A"}]}}]}');
SELECT Level2.Field1 FROM json_data, JSON_TABLE(MainLevel, '$.MainLevel[*]' COLUMNS (More JSON PATH '$.More')) AS MainLevel, JSON_TABLE(MainLevel.More, '$.Level2[*]' COLUMNS (Field1 VARCHAR(255) PATH '$.Field1')) AS Level2;
SELECT COUNT(*) FROM json_data, JSON_TABLE(MainLevel, '$.MainLevel[*]' COLUMNS (More JSON PATH '$.More')) AS MainLevel WHERE MainLevel.More IS NOT NULL;
SELECT JSON_UNQUOTE(JSON_EXTRACT(MainLevel, '$.More.Level2[0].Field1')) AS Field1 FROM json_data, JSON_TABLE(MainLevel, '$.MainLevel[*]' COLUMNS (More JSON PATH '$.More')) AS MainLevel;
