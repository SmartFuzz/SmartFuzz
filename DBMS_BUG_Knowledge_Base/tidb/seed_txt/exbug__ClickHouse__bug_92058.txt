CREATE TABLE dst (e ENUM('a', 'b'), f INT) ENGINE = Memory;
CREATE TABLE src (e ENUM('a'), g INT);
CREATE MATERIALIZED VIEW mv AS SELECT e, g FROM src;
INSERT INTO src SELECT CAST(number % 1 AS UNSIGNED) AS e, number % 10 AS g FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS generateRandom;
INSERT INTO dst SELECT e, g FROM mv WHERE g IS NOT NULL;
