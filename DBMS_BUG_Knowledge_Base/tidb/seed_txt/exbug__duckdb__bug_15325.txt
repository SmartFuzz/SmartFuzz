SET @@global.tidb_mem_quota = 300 * 1024 * 1024;
CREATE TABLE diverse_table (id INT, name VARCHAR(255), value DOUBLE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, is_active BOOLEAN);
INSERT INTO diverse_table SELECT i, CONCAT('Name_', i), RAND() * 100, NOW(), (i % 2 = 0) FROM (SELECT seq AS i FROM information_schema.SEQUENCES LIMIT 1000000) AS t;
SELECT id, name, value, created_at, is_active FROM diverse_table WHERE is_active IS NOT NULL;
SELECT COUNT(*) AS total_count, SUM(CASE WHEN value IS NULL THEN 1 ELSE 0 END) AS null_value_count FROM diverse_table;
