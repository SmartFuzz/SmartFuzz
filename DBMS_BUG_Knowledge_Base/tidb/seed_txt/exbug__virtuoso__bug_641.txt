CREATE TABLE actors (id INT PRIMARY KEY, name VARCHAR(100), birth_year INT, is_active BOOLEAN);
CREATE TABLE movies (id INT PRIMARY KEY, title VARCHAR(100), release_year INT, director_id INT, FOREIGN KEY (director_id) REFERENCES actors(id));
