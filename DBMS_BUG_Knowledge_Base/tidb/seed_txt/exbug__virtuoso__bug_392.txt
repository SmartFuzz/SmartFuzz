CREATE TABLE automobiles (id INT PRIMARY KEY, manufacturer VARCHAR(255), thumbnail VARCHAR(255), depiction VARCHAR(255), image VARCHAR(255), logo VARCHAR(255));

SELECT id, COALESCE(thumbnail, depiction, image, logo) AS image_url FROM automobiles WHERE manufacturer IS NOT NULL;
