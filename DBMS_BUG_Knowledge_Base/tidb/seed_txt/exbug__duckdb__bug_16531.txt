CREATE TABLE test_data (id INT, download INT, upload INT, app VARCHAR(255), `key` VARCHAR(44));
INSERT INTO test_data SELECT id, (FLOOR(RAND() * 3)) AS download, (FLOOR(RAND() * 3)) AS upload, CONCAT('App_', FLOOR(RAND() * 100)) AS app, LEFT(MD5(RAND()), 44) AS `key` FROM (SELECT 1 AS id UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10) AS numbers LIMIT 100000;
SELECT app, `key`, COUNT(*) AS count_download, SUM(CASE WHEN download = 0 THEN 1 ELSE 0 END) AS low_count, SUM(CASE WHEN download = 1 THEN 1 ELSE 0 END) AS average_count, SUM(CASE WHEN download = 2 THEN 1 ELSE 0 END) AS high_count FROM test_data GROUP BY app, `key`;
WITH download_summary AS (SELECT app, `key`, SUM(CASE WHEN download = 0 THEN 1 ELSE 0 END) AS low_count, SUM(CASE WHEN download = 1 THEN 1 ELSE 0 END) AS average_count, SUM(CASE WHEN download = 2 THEN 1 ELSE 0 END) AS high_count FROM test_data GROUP BY app, `key`) SELECT app, `key`, CASE WHEN low_count = GREATEST(low_count, average_count, high_count) THEN 0 WHEN average_count = GREATEST(low_count, average_count, high_count) THEN 1 ELSE 2 END AS mode_download FROM download_summary;
SELECT app, `key`, COALESCE(NULLIF(MAX(download), 0), 1) AS max_download FROM test_data GROUP BY app, `key` HAVING COUNT(*) > 10;
