CREATE TABLE employees (emp_no INT PRIMARY KEY, gender CHAR(1) NOT NULL, hire_date DATE);
CREATE TABLE salaries (emp_no INT, salary DECIMAL(10, 2), FOREIGN KEY (emp_no) REFERENCES employees(emp_no));
CREATE TABLE dept_manager (emp_no INT, dept_no VARCHAR(255), FOREIGN KEY (emp_no) REFERENCES employees(emp_no));
