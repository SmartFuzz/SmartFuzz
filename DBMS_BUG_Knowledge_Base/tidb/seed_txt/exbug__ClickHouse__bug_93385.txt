CREATE TABLE user_events (event_id CHAR(36), user_id INT UNSIGNED, event_type VARCHAR(20), event_time DATETIME);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT UUID(), FLOOR(RAND() * 10000) AS user_id, ELT(FLOOR(RAND() * 3) + 1, 'click', 'view', 'purchase') AS event_type, NOW() - INTERVAL FLOOR(RAND() * 3600 * 24) SECOND AS event_time FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10) AS numbers LIMIT 100000;
CREATE TABLE user_summary (user_id INT UNSIGNED, event_count INT UNSIGNED);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
