DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INT UNSIGNED, accepted_at DATETIME(3), delivery_report_at JSON);
INSERT INTO agg_test (id, accepted_at, delivery_report_at) SELECT number AS id, ADDTIME('2025-09-24 00:00:00', INTERVAL number SECOND) AS accepted_at, JSON_ARRAYAGG(CONCAT(number, ':', ADDTIME('2025-09-24 00:00:00', INTERVAL number SECOND))) FROM (SELECT number FROM numbers(100)) AS temp GROUP BY id, accepted_at;
SELECT id, JSON_UNQUOTE(JSON_EXTRACT(delivery_report_at, '$[0]')) AS delivered_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, JSON_UNQUOTE(JSON_EXTRACT(delivery_report_at, '$[0]')) AS delivered_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
