DROP TABLE IF EXISTS b_products;
CREATE TABLE b_products (product_id BIGINT, product_name VARCHAR(255), price DOUBLE) ENGINE=InnoDB;
INSERT INTO b_products (product_id, product_name, price) SELECT number, CONCAT('Product_', number), number * 10.5 FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers LIMIT 100;
DROP TABLE IF EXISTS b_sales;
CREATE TABLE b_sales (sale_id BIGINT, product_id BIGINT, quantity INT) ENGINE=InnoDB;
