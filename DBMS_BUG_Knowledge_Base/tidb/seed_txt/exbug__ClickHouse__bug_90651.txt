CREATE DATABASE IF NOT EXISTS test_db;
CREATE TABLE IF NOT EXISTS test_distr (num INT, date DATETIME, value FLOAT DEFAULT 0.0);
INSERT INTO test_distr SELECT number AS num, NOW() AS date, number * 1.5 AS value FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS system_numbers LIMIT 100;
CREATE TABLE IF NOT EXISTS replicated_table (num BIGINT, date DATE, metric FLOAT) ENGINE = InnoDB;
INSERT INTO replicated_table SELECT num, date, (SELECT date FROM test_distr WHERE num = replicated_table.num) AS metric FROM test_distr WHERE num IS NOT NULL;
