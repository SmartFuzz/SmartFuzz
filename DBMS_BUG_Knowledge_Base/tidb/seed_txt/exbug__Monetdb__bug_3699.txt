CREATE DATABASE financialdata;
CREATE TABLE financialdata.stocks (id INT AUTO_INCREMENT NOT NULL PRIMARY KEY, trade_time TIMESTAMP NOT NULL, trade_date TIMESTAMP, ticker VARCHAR(10) NOT NULL, currency VARCHAR(10) NOT NULL, opening_price DOUBLE NOT NULL, highest_price DOUBLE NOT NULL, lowest_price DOUBLE NOT NULL, closing_price DOUBLE NOT NULL, trade_volume DOUBLE);
INSERT INTO financialdata.stocks (trade_time, trade_date, ticker, currency, opening_price, highest_price, lowest_price, closing_price, trade_volume) VALUES (NOW(), NOW(), 'AAPL', 'USD', 150.0, 155.0, 149.0, 154.0, 10000.0);
SELECT trade_date, MIN(opening_price) AS opening_price, MAX(highest_price) AS highest_price, MIN(lowest_price) AS lowest_price, MAX(closing_price) AS closing_price, SUM(trade_volume) AS trade_volume FROM financialdata.stocks WHERE ticker='AAPL' AND trade_date IS NOT NULL AND trade_date > NOW() - INTERVAL 30 DAY GROUP BY trade_date ORDER BY trade_date DESC;
SELECT DATE(trade_date) FROM financialdata.stocks LIMIT 10;
