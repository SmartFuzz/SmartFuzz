CREATE TABLE test_table (id INT, values JSON);
INSERT INTO test_table VALUES (1, JSON_ARRAY(10, 20, 30)), (2, JSON_ARRAY(NULL, 40, 50)), (3, JSON_ARRAY(60, 70, NULL));
SELECT id, JSON_UNQUOTE(JSON_ARRAYAGG(IFNULL(JSON_UNQUOTE(JSON_EXTRACT(values, CONCAT('$[', idx, ']'))), NULL))) ORDER BY idx) AS transformed_ix FROM test_table, (SELECT 0 AS idx UNION ALL SELECT 1 UNION ALL SELECT 2) AS idxs GROUP BY id;
SELECT id, (SELECT COUNT(*) - 1 FROM JSON_TABLE(values, '$[*]' COLUMNS (value INT PATH '$')) AS jt) AS reduced_ix FROM test_table;
SELECT id, JSON_UNQUOTE(JSON_ARRAYAGG(IFNULL(JSON_UNQUOTE(JSON_EXTRACT(values, CONCAT('$[', idx, ']'))), NULL) + 1)) ORDER BY idx) AS incremented_values FROM test_table, (SELECT 0 AS idx UNION ALL SELECT 1 UNION ALL SELECT 2) AS idxs WHERE values IS NOT NULL GROUP BY id;
