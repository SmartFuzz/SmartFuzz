DROP TABLE IF EXISTS t1, t2;
CREATE TABLE t1(id int PRIMARY KEY, col1 varchar(10) DEFAULT NULL, col2 varchar(10) DEFAULT NULL, col3 varchar(10) DEFAULT NULL);
CREATE TABLE t2(id int PRIMARY KEY, col1 varchar(10) NOT NULL, col2 varchar(10) NOT NULL, col3 varchar(10) NOT NULL);

WITH tmp AS (SELECT t2.* FROM t2 WHERE t2.id > 1) UPDATE t1 SET col1 = (SELECT tmp.col1 FROM tmp WHERE tmp.id = t1.id), col2 = (SELECT tmp.col2 FROM tmp WHERE tmp.id = t1.id), col3 = (SELECT tmp.col3 FROM tmp WHERE tmp.id = t1.id) WHERE t1.id IN (SELECT id FROM tmp);
