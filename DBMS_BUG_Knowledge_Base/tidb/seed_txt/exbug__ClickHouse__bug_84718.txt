DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id INT UNSIGNED, name VARCHAR(255), value DOUBLE, created_at DATETIME) ENGINE=InnoDB;
INSERT INTO test_table (id, name, value, created_at) SELECT number, CONCAT('name_', number, ' ', number % 10) AS name, number * 1.5 AS value, NOW() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 100) AS numbers;
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
