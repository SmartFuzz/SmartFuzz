CREATE TABLE c (c_id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, value VARCHAR(100) NOT NULL COLLATE utf8mb4_bin, UNIQUE KEY ix_value_unique (value));
CREATE TABLE d (d_id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, value VARCHAR(100) NOT NULL COLLATE utf8mb4_bin, KEY ix_value (value));
INSERT INTO c (value) SELECT CONCAT('value_', seq) FROM (SELECT seq FROM (SELECT 1 AS seq UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 AS seq UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2 LIMIT 1000) AS seq_table;
INSERT INTO d (value) SELECT CONCAT('value_', seq) FROM (SELECT seq FROM (SELECT 1 AS seq UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 AS seq UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2 LIMIT 1000000) AS seq_table;
WITH b2 AS (SELECT DISTINCT value FROM d), a2 AS (SELECT value FROM c) SELECT value FROM b2 WHERE value NOT IN (SELECT value FROM a2);
