CREATE TABLE test_partitioned (id INT UNSIGNED, dt DATETIME, value FLOAT) PARTITION BY DATE_FORMAT(dt, '%Y%m') ORDER BY (dt);
INSERT INTO test_partitioned SELECT number, DATE_ADD('2015-01-01', INTERVAL (number % 3650) DAY) AS dt, RAND() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT @rownum := 0) r, information_schema.tables LIMIT 10000000) AS numbers;
CREATE TABLE test_non_partitioned (id INT UNSIGNED, dt DATETIME, value FLOAT) ORDER BY (dt);
INSERT INTO test_non_partitioned SELECT number, DATE_ADD('2015-01-01', INTERVAL (number % 3650) DAY) AS dt, RAND() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT @rownum := 0) r, information_schema.tables LIMIT 10000000) AS numbers;
SELECT dt FROM test_partitioned WHERE value IS NOT NULL ORDER BY dt LIMIT 20;
