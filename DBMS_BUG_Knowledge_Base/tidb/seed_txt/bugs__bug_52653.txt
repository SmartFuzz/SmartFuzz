CREATE TABLE `test_table` ( `id` int NOT NULL AUTO_INCREMENT, `name` varchar(100) NOT NULL, `created_at` datetime DEFAULT CURRENT_TIMESTAMP, `status` tinyint(1) DEFAULT 0, PRIMARY KEY (`id`)) ENGINE=InnoDB;
INSERT INTO `test_table` (`name`, `status`) VALUES ('Test Name', 1);
CREATE TABLE `related_table` ( `related_id` int NOT NULL AUTO_INCREMENT, `test_id` int NOT NULL, `description` text, PRIMARY KEY (`related_id`), FOREIGN KEY (`test_id`) REFERENCES `test_table`(`id`) ON DELETE CASCADE) ENGINE=InnoDB;
INSERT INTO `related_table` (`test_id`, `description`) VALUES (1, 'Related Description');
SELECT t.id, t.name, r.description FROM `test_table` t LEFT JOIN `related_table` r ON t.id = r.test_id WHERE t.status = 1 AND r.description IS NOT NULL ORDER BY t.created_at DESC LIMIT 5;
