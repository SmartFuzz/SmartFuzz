CREATE TABLE t1(a VARCHAR(10) COLLATE utf8_general_ci, b CHAR(10) COLLATE utf8_general_ci);
INSERT INTO t1 VALUES ('x', 'X '), ('y', 'Y'), ('z', 'Z'), (' ', ' ');
CREATE TABLE t2(c VARCHAR(10) COLLATE utf8_unicode_ci, d CHAR(10) COLLATE utf8_unicode_ci);
INSERT INTO t2 VALUES ('x', 'X'), ('y', 'Y '), ('z', 'Z '), (NULL, 'NULL');
SELECT * FROM t1 WHERE (a, b) IN (SELECT d, c FROM t2 WHERE c IS NOT NULL);
