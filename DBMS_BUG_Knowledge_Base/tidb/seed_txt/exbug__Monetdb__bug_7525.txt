CREATE TABLE test_vals(grp INT NOT NULL, value INT CHECK(value >= 0));
INSERT INTO test_vals VALUES (1, 100), (2, 200), (1, NULL), (2, 300);
CREATE FUNCTION jit_sum_variadic(value INT) RETURNS BIGINT DETERMINISTIC RETURN SUM(value);
SELECT grp, jit_sum_variadic(value) FROM test_vals GROUP BY grp;
SELECT grp, jit_sum_variadic(value) FROM test_vals WHERE value IS NOT NULL GROUP BY grp;
