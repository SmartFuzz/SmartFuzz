CREATE TABLE t1 (name VARCHAR(255) NOT NULL, n INT CHECK (n >= 0));
CREATE TABLE t2 (name VARCHAR(255), n INT DEFAULT 0);
CREATE TABLE t3 (name VARCHAR(255) UNIQUE, n FLOAT);
