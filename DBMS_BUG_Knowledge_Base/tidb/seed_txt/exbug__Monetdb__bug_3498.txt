CREATE TABLE test.agg_table (id INT, value DOUBLE, category VARCHAR(50));
INSERT INTO test.agg_table VALUES (1, 10.5, 'A'), (2, 20.5, 'B'), (3, NULL, 'A'), (4, 40.0, 'B'), (5, 30.0, 'C');
SELECT SUM(value) AS total_value FROM test.agg_table WHERE category = 'A' ORDER BY total_value DESC;
SELECT AVG(value) AS avg_value FROM test.agg_table GROUP BY category ORDER BY avg_value DESC LIMIT 2;
SELECT COUNT(*) AS count_value FROM test.agg_table GROUP BY category HAVING COUNT(*) > 2 ORDER BY count_value DESC LIMIT 1;
