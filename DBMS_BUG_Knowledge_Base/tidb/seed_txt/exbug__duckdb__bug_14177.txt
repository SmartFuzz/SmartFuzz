CREATE TABLE test_data (id INT PRIMARY KEY, name VARCHAR(255), age INT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
LOAD DATA INFILE 'test_duckdb_data.csv' INTO TABLE test_data FIELDS TERMINATED BY ',' LINES STARTING BY '' IGNORE 1 LINES;
CREATE VIEW valid_records AS SELECT * FROM test_data WHERE age IS NOT NULL;
CREATE VIEW rejected_records AS SELECT * FROM test_data WHERE name IS NULL;
SELECT COUNT(*) AS valid_count, (SELECT COUNT(*) FROM rejected_records) AS rejected_count FROM valid_records;
