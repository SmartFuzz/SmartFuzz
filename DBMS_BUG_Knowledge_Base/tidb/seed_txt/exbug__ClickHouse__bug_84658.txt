CREATE TABLE t1 (id INT UNSIGNED, name VARCHAR(255), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO t1 (id, name) VALUES (1, 'Alice'), (2, 'Bob'), (3, NULL);
CREATE TABLE t2 (ref_id INT UNSIGNED, value DOUBLE);
INSERT INTO t2 (ref_id, value) SELECT id, CAST(name AS DOUBLE) FROM t1 WHERE name IS NOT NULL;
SELECT t1.id, t1.name, t2.value FROM t1 LEFT JOIN t2 ON t1.id = t2.ref_id WHERE t1.created_at < NOW() - INTERVAL 1 DAY;
