CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, age INT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE comments (id INT PRIMARY KEY AUTO_INCREMENT, user_id INT, content TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
