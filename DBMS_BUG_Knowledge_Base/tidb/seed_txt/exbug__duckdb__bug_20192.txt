CREATE TABLE products (id INT PRIMARY KEY, name VARCHAR(255), description VARCHAR(255), price DECIMAL(10, 2), created_at TIMESTAMP);
CREATE TABLE orders (id INT PRIMARY KEY, product_id INT, order_date TIMESTAMP, quantity INT, large_blob BLOB, FOREIGN KEY (product_id) REFERENCES products(id));
