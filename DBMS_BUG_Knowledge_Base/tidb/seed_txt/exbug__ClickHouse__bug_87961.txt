CREATE TABLE IF NOT EXISTS user_activity (activity_id BIGINT, user_id INT, activity_time DATETIME, activity_type ENUM('login', 'logout', 'purchase'), details TEXT);
INSERT INTO user_activity SELECT number AS activity_id, RAND() % 1000 AS user_id, NOW() + INTERVAL number * 5 SECOND AS activity_time, (CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, CONCAT('Activity details for ', number) AS details FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers;
CREATE TABLE IF NOT EXISTS product_sales (sale_id INT, product_id INT, sale_time DATETIME, quantity SMALLINT, price FLOAT);
INSERT INTO product_sales SELECT number AS sale_id, RAND() % 500 AS product_id, NOW() + INTERVAL number * 2 SECOND AS sale_time, (RAND() % 10 + 1) AS quantity, (RAND() % 100 + 1) * 1.5 AS price FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers LIMIT 50;
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL 1 DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
