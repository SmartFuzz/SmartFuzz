DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR(255), event_time DATETIME, id BIGINT, msg VARCHAR(255), value FLOAT);
INSERT INTO repro.events (tenant, event_time, id, msg, value) SELECT CAST(number % 100 AS CHAR) AS tenant, NOW() - INTERVAL (number % 200000) SECOND AS event_time, number AS id, CONCAT('m', CAST(number AS CHAR)) AS msg, CAST(number % 1000 AS FLOAT) AS value FROM (SELECT number FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 2000000) AS numbers);
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
