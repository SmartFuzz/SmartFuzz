CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE events (event_id INT AUTO_INCREMENT PRIMARY KEY, event_time TIMESTAMP NOT NULL, user_id INT, FOREIGN KEY (user_id) REFERENCES users(id));
