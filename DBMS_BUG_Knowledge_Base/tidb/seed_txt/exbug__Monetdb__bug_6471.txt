INSERT INTO test_series (id, value) SELECT id, id FROM (SELECT 1 AS id UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10) AS series;
CREATE TABLE test_data (data_id INT, data_value VARCHAR(255) NOT NULL, CONSTRAINT chk_data_value CHECK (data_value <> ''));
INSERT INTO test_data (data_id, data_value) SELECT id, CONCAT('Value ', id) FROM (SELECT 1 AS id UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) AS series;
SELECT t.id, t.value, d.data_value FROM test_series t JOIN test_data d ON t.id = d.data_id WHERE t.value IS NOT NULL;
