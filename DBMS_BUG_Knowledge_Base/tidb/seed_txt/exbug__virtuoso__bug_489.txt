CREATE TABLE test_graph (id INT PRIMARY KEY, graph_uri VARCHAR(255), annotated_at TIMESTAMP);
CREATE TABLE test_address (id INT PRIMARY KEY, address_uri VARCHAR(255), target_graph INT, address_timestamp TIMESTAMP, FOREIGN KEY (target_graph) REFERENCES test_graph(id));

SELECT g.graph_uri, g.annotated_at, COUNT(a.address_timestamp) AS addresses_count, MAX(a.address_timestamp) AS most_recent_timestamp FROM test_graph g LEFT JOIN test_address a ON g.id = a.target_graph GROUP BY g.id;
