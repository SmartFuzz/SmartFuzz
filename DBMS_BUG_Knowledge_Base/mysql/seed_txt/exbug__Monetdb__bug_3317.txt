CREATE TABLE vouchers (vid INT PRIMARY KEY, date CHAR(8), cur CHAR(3), val DECIMAL(15,2), FOREIGN KEY(cur) REFERENCES prices(currency));

SELECT vid, date, cur, val, (SELECT MAX(price) FROM prices p WHERE p.valid_from = (SELECT MAX(q.valid_from) FROM prices q WHERE q.valid_from <= v.date AND q.currency = v.cur)) AS Preis FROM vouchers v;
