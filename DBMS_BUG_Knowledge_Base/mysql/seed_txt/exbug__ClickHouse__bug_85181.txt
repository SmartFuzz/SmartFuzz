CREATE DATABASE demo;
CREATE TABLE demo.dict_test (code INT, value VARCHAR(255)) ENGINE = MEMORY;
CREATE TABLE demo.test_local (code INT, name VARCHAR(255) DEFAULT 'unknown');
INSERT INTO demo.test_local (code, name) VALUES (1, 'value1'), (2, 'value2'), (3, NULL);
SELECT code, (SELECT value FROM demo.dict_test WHERE code = demo.test_local.code) AS dict_value FROM demo.test_local WHERE code IN (SELECT code FROM demo.test_local WHERE name IS NOT NULL);
