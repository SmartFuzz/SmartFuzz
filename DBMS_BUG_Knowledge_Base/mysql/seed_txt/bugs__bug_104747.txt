CREATE TABLE fig (id INT NOT NULL PRIMARY KEY, varchar_1 VARCHAR(20), num_col INT DEFAULT 0);
INSERT INTO fig (id, varchar_1, num_col) VALUES (1, 'test', NULL), (2, 'example', 5), (3, NULL, 10);
CREATE VIEW v_substr AS SELECT substr(varchar_1, IF((-(4) = 0), 1, -(4)), 4) AS `SUBSTR(varchar_1,-4,4)` FROM fig WHERE num_col IS NOT NULL;
CREATE VIEW v_join AS SELECT f.id, v.SUBSTR FROM fig f JOIN v_substr v ON f.id = 1;
