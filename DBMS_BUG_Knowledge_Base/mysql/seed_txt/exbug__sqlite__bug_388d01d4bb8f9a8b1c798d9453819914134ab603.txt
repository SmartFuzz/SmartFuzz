CREATE TABLE fts (content TEXT, FULLTEXT(content));
CREATE TABLE t1 (x TEXT PRIMARY KEY, y INT NOT NULL DEFAULT 0);
INSERT INTO t1 (x, y) VALUES ('hello world', 1);
INSERT INTO fts (content) VALUES ('hello world');
SELECT DISTINCT fts.rowid, t1.x FROM fts JOIN t1 ON fts.content = t1.x WHERE MATCH(fts.content) AGAINST('hello world') ORDER BY t1.y;
