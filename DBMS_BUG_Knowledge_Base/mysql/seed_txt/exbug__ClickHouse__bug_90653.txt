CREATE TABLE test (s VARCHAR(255), json JSON, id INT UNSIGNED) ENGINE=InnoDB;
INSERT INTO test (s, id) VALUES ('example', 1);
UPDATE test SET s = 'updated' WHERE JSON_UNQUOTE(json_extract(json, '$.s')) = 'example';
DELETE FROM test WHERE JSON_UNQUOTE(json_extract(json, '$.s')) = 'updated';
SELECT * FROM test WHERE JSON_UNQUOTE(json_extract(json, '$.s')) IS NOT NULL;
