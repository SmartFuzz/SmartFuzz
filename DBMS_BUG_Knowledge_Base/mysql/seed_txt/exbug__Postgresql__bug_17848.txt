DROP TABLE IF EXISTS demo;
CREATE TABLE demo(val BIGINT, name TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE INDEX demo_idx ON demo(val);
INSERT INTO demo(val, name) SELECT val, CONCAT('Name ', val) FROM (SELECT @rownum := @rownum + 1 AS val FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 10000000) AS series;
ANALYZE TABLE demo;
