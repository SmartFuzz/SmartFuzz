DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INT UNSIGNED, accepted_at DATETIME(3), delivery_report_at JSON);
INSERT INTO agg_test (id, accepted_at, delivery_report_at) SELECT number AS id, DATE_ADD('2025-09-24 00:00:00', INTERVAL number SECOND) AS accepted_at, JSON_OBJECT('value', MAX(CASE WHEN number % 2 = 1 THEN number END), 'timestamp', DATE_ADD('2025-09-24 00:00:00', INTERVAL number SECOND)) FROM (SELECT number FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 0 UNION SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9) AS a, (SELECT @rownum := -1) AS r) AS numbers) AS grouped GROUP BY id, accepted_at;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
