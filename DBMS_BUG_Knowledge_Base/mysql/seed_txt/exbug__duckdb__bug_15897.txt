CREATE TABLE courses (course_id INT PRIMARY KEY, course_name VARCHAR(255), credits INT CHECK (credits > 0));
CREATE TABLE enrollments (enrollment_id INT PRIMARY KEY, student_id INT, course_id INT, grade INT, FOREIGN KEY (student_id) REFERENCES students(id), FOREIGN KEY (course_id) REFERENCES courses(course_id));
