CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT);
CREATE TABLE posts (id INT AUTO_INCREMENT PRIMARY KEY, user_id INT, content TEXT, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id INT AUTO_INCREMENT PRIMARY KEY, post_id INT, user_id INT, comment TEXT, FOREIGN KEY(post_id) REFERENCES posts(id), FOREIGN KEY(user_id) REFERENCES users(id));
