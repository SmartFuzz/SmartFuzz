DROP TABLE IF EXISTS data_local;
CREATE TABLE data_local (id INT UNSIGNED, value VARCHAR(255));
INSERT INTO data_local (id, value) SELECT number, CAST(number AS CHAR) FROM (SELECT 0 AS number UNION SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9) AS numbers;
CREATE TABLE data_distributed AS SELECT * FROM data_local;
SELECT COUNT(DISTINCT value) FROM data_distributed;
