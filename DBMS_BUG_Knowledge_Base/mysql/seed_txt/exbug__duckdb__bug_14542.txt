CREATE PROCEDURE json_each(IN input JSON) BEGIN SELECT CASE JSON_TYPE(input) WHEN 'ARRAY' THEN JSON_UNQUOTE(JSON_EXTRACT(input, '$[0]')) ELSE JSON_KEYS(input) END AS key, JSON_UNQUOTE(JSON_EXTRACT(input, CONCAT('$[', key, ']'))) AS value;
END;
CALL json_each('[{"key": "value1"}, {"key": "value2"}]');
CALL json_each('[1,2,3]');
