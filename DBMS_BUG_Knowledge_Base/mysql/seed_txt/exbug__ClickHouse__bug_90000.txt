CREATE TABLE T1 (A BIGINT, B FLOAT) ENGINE = InnoDB;
INSERT INTO T1 (A, B) SELECT number, number * 1.5 FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 100000) AS numbers;
CREATE TABLE T2 (C VARCHAR(255), D INT) ENGINE = InnoDB;
INSERT INTO T2 (C, D) SELECT CAST(number AS CHAR) AS C, IF(number % 2 = 0, number, NULL) AS D FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 1000) AS numbers;
SELECT DISTINCT A, (SELECT a1 FROM T2 WHERE id = B) AS a1, (SELECT a2 FROM T2 WHERE id = B) AS a2 FROM T1;
