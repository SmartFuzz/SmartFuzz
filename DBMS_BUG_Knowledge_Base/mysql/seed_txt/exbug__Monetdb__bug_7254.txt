CREATE TABLE y (id INT PRIMARY KEY, value VARCHAR(255) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO y (id, value) SELECT i, CONCAT('Value ', i) FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 1000000) AS temp;
DELETE FROM y WHERE id IN (SELECT id FROM y WHERE value LIKE 'Value 1%');
CREATE TABLE z AS SELECT DISTINCT value FROM y WHERE created_at < NOW() - INTERVAL 1 DAY;
DELETE FROM z WHERE value IS NULL;
