CREATE TABLE `PRIMARY` (`ID` INT NOT NULL, `LEN` FLOAT NOT NULL, PRIMARY KEY (`ID`));
CREATE TABLE `FOREIGN` (`ID` INT NOT NULL, `POS` FLOAT NOT NULL, FOREIGN KEY (`ID`) REFERENCES `PRIMARY` (`ID`));

INSERT INTO `FOREIGN` (`ID`, `POS`) VALUES (1, 5.0), (2, 15.0), (3, 25.0);
