CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO users (id, name) VALUES (1, 'Alice'), (2, 'Bob'), (3, NULL);
CREATE TABLE orders (order_id INT PRIMARY KEY, user_id INT, amount DECIMAL(10, 2) CHECK (amount > 0), FOREIGN KEY (user_id) REFERENCES users(id));
INSERT INTO orders (order_id, user_id, amount) VALUES (101, 1, 150.00), (102, 2, 200.50), (103, 1, NULL);
CREATE VIEW recent_orders AS SELECT user_id, SUM(amount) AS total_amount FROM orders GROUP BY user_id;
SELECT * FROM recent_orders WHERE total_amount IS NOT NULL;
