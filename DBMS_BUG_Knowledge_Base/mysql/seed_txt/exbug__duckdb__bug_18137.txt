CREATE TABLE test_large_lists (id INT, gts JSON);
INSERT INTO test_large_lists (id, gts) SELECT i, JSON_ARRAY(JSON_ARRAY(CAST(FLOOR(RAND() * 256) AS UNSIGNED), CAST(FLOOR(RAND() * 256) AS UNSIGNED))) FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 500000) AS numbers;
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, JSON_UNQUOTE(JSON_EXTRACT(gts, '$[0]')) AS inner_list FROM list_view;
