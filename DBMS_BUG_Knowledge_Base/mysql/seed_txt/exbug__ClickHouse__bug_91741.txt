CREATE TABLE test_parallel_replicas (id INT UNSIGNED, value VARCHAR(255), timestamp DATETIME) ENGINE=InnoDB;
INSERT INTO test_parallel_replicas (id, value, timestamp) SELECT number, CAST(number AS CHAR), NOW() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT @rownum := 0) r, information_schema.tables LIMIT 100) AS numbers;
CREATE TABLE test_parallel_replicas_2 (id INT UNSIGNED, value VARCHAR(255), timestamp DATETIME);
INSERT INTO test_parallel_replicas_2 (id, value, timestamp) SELECT number, CASE WHEN number % 2 = 0 THEN CAST(number AS CHAR) ELSE NULL END, NOW() FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT @rownum := 0) r, information_schema.tables LIMIT 100) AS numbers;
SELECT COUNT(*) FROM test_parallel_replicas WHERE value IS NOT NULL AND timestamp > NOW() - INTERVAL 1 DAY;
