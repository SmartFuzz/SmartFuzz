CREATE TABLE RBIG_EXTENDED (id INT, src INT, dest INT, weight FLOAT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO RBIG_EXTENDED (id, src, dest, weight, created_at) SELECT id, src, dest, RAND() * 100, NOW() FROM RBIG;
SELECT COUNT(*) FROM RBIG_EXTENDED WHERE weight IS NULL;
SELECT R1.src, R2.dest FROM RBIG_EXTENDED R1 JOIN RBIG_EXTENDED R2 ON R1.dest = R2.src WHERE R1.weight > 50;
SELECT AVG(weight) FROM RBIG_EXTENDED WHERE created_at < NOW() - INTERVAL 1 DAY;
