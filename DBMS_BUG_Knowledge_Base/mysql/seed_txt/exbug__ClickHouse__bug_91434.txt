CREATE TABLE json_test_v2 (json JSON, col_b BIGINT, col_c VARCHAR(255));
INSERT INTO json_test_v2 (json, col_b, col_c) SELECT JSON_OBJECT('a', number, 'b', number * 2, 'c', CAST(number * 3 AS CHAR)) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 10000000) AS numbers;
SELECT col_b, col_c FROM json_test_v2 WHERE col_b IS NOT NULL;
SELECT JSON_UNQUOTE(JSON_EXTRACT(json, '$.b')) AS direct_b, JSON_UNQUOTE(JSON_EXTRACT(json, '$.c')) AS direct_c FROM json_test_v2 WHERE JSON_UNQUOTE(JSON_EXTRACT(json, '$.b')) > 5000000;
SELECT col_b, COUNT(*) FROM json_test_v2 GROUP BY col_b HAVING col_b < 100;
