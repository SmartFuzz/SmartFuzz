CREATE TABLE account (id INT PRIMARY KEY, global_account_id INT NOT NULL, latest_version_id INT);
CREATE TABLE account_version (id INT PRIMARY KEY, account_type_id INT NOT NULL, global_account_id INT NOT NULL);
CREATE TABLE account_type (id INT PRIMARY KEY, type VARCHAR(20) NOT NULL, description VARCHAR(20));
