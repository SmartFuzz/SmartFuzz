SET SESSION max_heap_table_size=300000000;
CREATE TABLE diverse_table (id INT, name VARCHAR(255), value DOUBLE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, is_active BOOLEAN);
INSERT INTO diverse_table (id, name, value, created_at, is_active) SELECT i, CONCAT('Name_', i), RAND() * 100, NOW(), (i % 2 = 0) FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) t3 LIMIT 1000000) AS numbers;
SELECT id, name, value, created_at, is_active FROM diverse_table WHERE is_active IS NOT NULL;
SELECT COUNT(*) AS total_count, SUM(CASE WHEN value IS NULL THEN 1 ELSE 0 END) AS null_value_count FROM diverse_table;
