CREATE TABLE test_random (id INT PRIMARY KEY, value DOUBLE);
INSERT INTO test_random (id, value) SELECT i, RAND() FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 100000);
SELECT COUNT(DISTINCT value) FROM test_random;
SELECT COUNT(*) FROM (SELECT DISTINCT RAND() AS a FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 100000) AS unique_random;
SELECT COUNT(*) FROM test_random WHERE value IS NOT NULL;
