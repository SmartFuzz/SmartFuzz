CREATE TABLE test_table (id INT, value INT, z INT);
INSERT INTO test_table (id, value, z) VALUES (1, 10, 5), (2, 20, NULL), (3, 30, 15);
CREATE PROCEDURE demo(IN n INT, IN z INT) BEGIN SELECT IFNULL(z, 0) AS row FROM (SELECT @rownum := @rownum + 1 AS row_num FROM (SELECT @rownum := -1) r, test_table LIMIT n) AS temp;
END;
CALL demo(3, 0);
CALL demo(2, NULL);
