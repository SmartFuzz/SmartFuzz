CREATE TABLE set_idx_rand_v2 (id INT UNSIGNED, s VARCHAR(255), d DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO set_idx_rand_v2 (id, s) SELECT number, CAST(number % 10 AS CHAR) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) AS a, (SELECT @rownum := 0) AS r LIMIT 100) AS numbers;
SELECT * FROM set_idx_rand_v2 WHERE RAND() % 1000 = 1;
SELECT * FROM set_idx_rand_v2 WHERE d IS NULL OR d > NOW() - INTERVAL 1 DAY;
SELECT COUNT(*) FROM set_idx_rand_v2 WHERE s IN (SELECT s FROM set_idx_rand_v2 WHERE id < 50) AND RAND() % 1000 = 1;
