CREATE TABLE test_table (id INT, values JSON);
INSERT INTO test_table (id, values) VALUES (1, JSON_ARRAY(10, 20, 30)), (2, JSON_ARRAY(NULL, 40, 50)), (3, JSON_ARRAY(60, 70, NULL));
SELECT id, JSON_UNQUOTE(JSON_EXTRACT(values, '$[0]')) AS first_value FROM test_table;
SELECT id, JSON_LENGTH(values) - 1 AS last_index FROM test_table;
SELECT id, JSON_ARRAYAGG(JSON_UNQUOTE(JSON_EXTRACT(values, CONCAT('$[', n.n, ']')))) AS incremented_values FROM test_table JOIN (SELECT 0 AS n UNION ALL SELECT 1 UNION ALL SELECT 2) n ON n.n < JSON_LENGTH(values) WHERE values IS NOT NULL GROUP BY id;
