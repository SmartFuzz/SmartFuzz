CREATE TABLE `sales` ( `year` int DEFAULT NULL, `country` varchar(20) DEFAULT NULL, `product` varchar(32) DEFAULT NULL, `profit` int DEFAULT NULL, `quantity` int DEFAULT NULL);
INSERT INTO `sales` (year, country, product, profit, quantity) VALUES (2001, 'USA', 'Widget', 100, 10), (2002, 'Canada', 'Gadget', 150, 20), (2003, 'USA', 'Widget', 200, 30);
SELECT year + 1 AS adjusted_year, SUM(profit) AS total_profit FROM sales GROUP BY adjusted_year WITH ROLLUP HAVING adjusted_year > 2002;
SELECT year + 2 AS adjusted_year, SUM(profit) AS profit FROM sales GROUP BY adjusted_year WITH ROLLUP HAVING adjusted_year > 2002;
SELECT country, SUM(profit) AS total_profit FROM sales GROUP BY country WITH ROLLUP HAVING country IS NOT NULL;
