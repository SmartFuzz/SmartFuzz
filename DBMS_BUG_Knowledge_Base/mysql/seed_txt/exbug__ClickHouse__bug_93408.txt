CREATE TABLE throttle_test (id BIGINT, payload VARCHAR(1000), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO throttle_test (id, payload) SELECT number, LEFT(UUID(), 8) FROM (SELECT @rownum:=@rownum+1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum:=0) t2 LIMIT 100) AS numbers;
CREATE TABLE throttle_test_backup AS SELECT * FROM throttle_test;
SELECT id, payload FROM throttle_test WHERE created_at IS NOT NULL AND MOD(id, 2) = 0;
SELECT COUNT(*) FROM throttle_test_backup WHERE payload LIKE '%test%' AND id IN (SELECT id FROM throttle_test WHERE id < 50);
