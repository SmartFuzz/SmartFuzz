CREATE TABLE t3 (id INT PRIMARY KEY, name VARCHAR(50), value FLOAT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO t3 (id, name, value, created_at) VALUES (1, 'test1', 10.5, NULL), (2, 'test2', 20.0, '2023-01-01 10:00:00');
SELECT name, (SELECT COUNT(*) FROM t3 AS x WHERE x.value > t3.value) AS value_count, IF(created_at IS NULL, 'No Date', 'Has Date') AS date_status FROM t3;
SELECT name AS name_alias, (SELECT COUNT(*) FROM t3 AS x WHERE x.id < t3.id) AS id_count, CASE value WHEN 20.0 THEN 'Matched' END AS match_status FROM t3;
SELECT value, CASE WHEN value < 15.0 THEN 'Low' ELSE 'High' END AS value_category, (SELECT AVG(value) FROM t3) AS average_value FROM t3;
