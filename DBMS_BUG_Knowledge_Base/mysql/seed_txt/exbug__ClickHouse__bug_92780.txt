CREATE TABLE test_data (id INT UNSIGNED, ts DATETIME, value VARCHAR(255), status VARCHAR(255) NULL);
INSERT INTO test_data (id, ts, value, status) SELECT number, NOW() + INTERVAL number SECOND, CAST(number % 5 AS CHAR), IF(number % 2 = 0, 'active', NULL) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 1000000) AS numbers;
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > NOW() - INTERVAL 1 DAY AND value LIKE '1%';
SELECT DISTINCT id FROM test_data WHERE ts BETWEEN '2023-01-01' AND '2023-12-31' AND status IS NOT NULL;
