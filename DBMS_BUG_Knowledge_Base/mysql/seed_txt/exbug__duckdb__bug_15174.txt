CREATE TABLE test_data (id INT PRIMARY KEY, value DOUBLE, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_data SELECT i, RAND(), CONCAT('Name ', i), NOW() FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 200000000) AS temp;
CREATE TABLE filtered_data AS SELECT * FROM test_data WHERE value IS NOT NULL;
SELECT * INTO OUTFILE '/tmp/filtered_test.csv' FIELDS TERMINATED BY ',' ENCLOSED BY '"' LINES TERMINATED BY '\n' FROM filtered_data ORDER BY value DESC;
SELECT COUNT(*) FROM filtered_data WHERE value BETWEEN 0.0 AND 1.0;
