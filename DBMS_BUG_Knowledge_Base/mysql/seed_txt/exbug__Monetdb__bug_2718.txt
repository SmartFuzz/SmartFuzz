CREATE TABLE bug (time DATETIME, val INT, description VARCHAR(255) NOT NULL);
INSERT INTO bug VALUES (CURRENT_TIMESTAMP, 1, 'First entry');
INSERT INTO bug VALUES (CURRENT_TIMESTAMP + INTERVAL 2 SECOND, 2, 'Second entry');
INSERT INTO bug VALUES (CURRENT_TIMESTAMP - INTERVAL 3 SECOND, 3, 'Third entry');
SELECT t1.time, COUNT(t2.val) FROM bug t1 JOIN bug t2 ON t2.time BETWEEN (t1.time - INTERVAL 1 SECOND) AND (t1.time + INTERVAL 1 SECOND) GROUP BY t1.time ORDER BY t1.time;
