DROP TABLE IF EXISTS events;
CREATE TABLE events ( `Id` BIGINT UNSIGNED, `Payload` VARCHAR(255), `Time` DATETIME, `Status` VARCHAR(255) NULL );
INSERT INTO events (Id, Payload, Time, Status) SELECT number, CONCAT('Payload_', number), DATE_ADD('2024-01-01 00:00:00', INTERVAL number MINUTE), CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers;
DROP TABLE IF EXISTS attributes;
CREATE TABLE attributes ( `EventId` BIGINT UNSIGNED, `Attribute` VARCHAR(255), `Value` FLOAT DEFAULT 0.0 );
