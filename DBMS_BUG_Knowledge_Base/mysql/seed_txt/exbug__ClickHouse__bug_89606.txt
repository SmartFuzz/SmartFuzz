CREATE TABLE test_single_file (ts DATETIME(3)) ENGINE = InnoDB;
INSERT INTO test_single_file (ts) SELECT STR_TO_DATE(ts, '%Y-%m-%d %H:%i:%s') FROM (SELECT '2020-01-01 14:00:00' AS ts) AS temp;
CREATE TABLE test_brace_expression (ts DATETIME(3)) ENGINE = InnoDB;
INSERT INTO test_brace_expression (ts) SELECT STR_TO_DATE(ts, '%Y-%m-%d %H:%i:%s') FROM (SELECT '2020-01-01 14:00:00' AS ts UNION ALL SELECT '2020-01-01 14:00:00' AS ts) AS temp;
CREATE TABLE test_nullable (ts DATETIME(3) NULL) ENGINE = InnoDB;
