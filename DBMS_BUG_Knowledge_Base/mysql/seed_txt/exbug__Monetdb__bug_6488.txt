CREATE TABLE dict (value INT UNIQUE);
INSERT INTO dict (value) SELECT value FROM (SELECT 1 + n.n + 100 * m.m AS value FROM (SELECT 0 AS n UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4) n, (SELECT 0 AS m UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3) m) AS numbers WHERE value <= 2050;
CREATE TABLE t (i INT);
INSERT INTO t (i) SELECT value FROM dict WHERE value % 2 = 0;
SELECT d.value FROM dict d JOIN t ON d.value = t.i;
