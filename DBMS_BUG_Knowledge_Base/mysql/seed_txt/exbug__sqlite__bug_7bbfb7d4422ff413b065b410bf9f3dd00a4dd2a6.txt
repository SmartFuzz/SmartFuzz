CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id INT AUTO_INCREMENT PRIMARY KEY, user_id INT, amount DECIMAL(10, 2) NOT NULL, order_date DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TRIGGER update_order_timestamp BEFORE UPDATE ON orders FOR EACH ROW SET NEW.order_date = CURRENT_TIMESTAMP;
