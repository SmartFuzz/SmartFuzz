CREATE TABLE Course(id BIGINT PRIMARY KEY, title VARCHAR(255) NOT NULL, credits INT CHECK (credits > 0));

CREATE TABLE Enrollment(student_id BIGINT, course_id BIGINT, enrollment_date DATE DEFAULT CURRENT_DATE, FOREIGN KEY (student_id) REFERENCES Student(id), FOREIGN KEY (course_id) REFERENCES Course(id));
