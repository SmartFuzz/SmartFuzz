CREATE TABLE test_table (id INT, value VARCHAR(255), flag BOOLEAN);
INSERT INTO test_table (id, value, flag) VALUES (1, 'test', TRUE), (2, NULL, FALSE), (3, 'example', NULL), (4, 'sample', TRUE);
SELECT GROUP_CONCAT(value SEPARATOR ', ') AS aggregated_values FROM (SELECT id, value FROM test_table WHERE flag IS NOT NULL) AS cte;
SELECT id, IF(value IS NULL, 'NULL', value) AS value_check FROM test_table;
SELECT COUNT(*) AS total, GROUP_CONCAT(value SEPARATOR '|') AS concatenated FROM test_table WHERE flag = TRUE;
