DROP TABLE IF EXISTS b_products;
CREATE TABLE b_products (product_id BIGINT, product_name VARCHAR(255), price DOUBLE) ENGINE = InnoDB;
INSERT INTO b_products (product_id, product_name, price) SELECT number, CONCAT('Product_', number), number * 10.5 FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) AS a, (SELECT @rownum := 0) AS r LIMIT 100) AS numbers;
DROP TABLE IF EXISTS b_sales;
CREATE TABLE b_sales (sale_id BIGINT, product_id BIGINT, quantity INT) ENGINE = InnoDB;
