CREATE TABLE test_table (id INT AUTO_INCREMENT PRIMARY KEY, name TEXT NOT NULL, value INT CHECK (value >= 0));
CREATE INDEX CaseSensitiveIndex ON test_table (name, value);
CREATE INDEX not_case_sensitive ON test_table (value, id);
SELECT TABLE_NAME FROM INFORMATION_SCHEMA.STATISTICS WHERE TABLE_NAME = 'test_table' AND INDEX_NAME = 'not_case_sensitive';
SELECT TABLE_NAME FROM INFORMATION_SCHEMA.STATISTICS WHERE TABLE_NAME = 'test_table' AND INDEX_NAME = 'CaseSensitiveIndex';
