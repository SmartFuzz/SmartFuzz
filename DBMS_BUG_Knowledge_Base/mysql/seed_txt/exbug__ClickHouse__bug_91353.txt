CREATE TABLE test_table1 (id INT, value FLOAT, created_at DATETIME);
CREATE TABLE test_table2 (id INT, description VARCHAR(255), is_active TINYINT);
CREATE TRIGGER test_mv1_trigger AFTER INSERT ON test_table2 FOR EACH ROW BEGIN IF NEW.is_active = 1 THEN INSERT INTO test_table1 (id, value, created_at) VALUES (NEW.id, NEW.value, NOW());
END IF;
END;
CREATE EVENT test_mv2_event ON SCHEDULE EVERY 10 MINUTE DO INSERT INTO test_table2 (id, description) SELECT id, description FROM test_table1 WHERE created_at > NOW() - INTERVAL 5 MINUTE;
