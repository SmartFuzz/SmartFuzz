CREATE DATABASE test2;
CREATE TABLE test2.raw_source_local (id INT, time DATETIME, value FLOAT DEFAULT 0.0);
INSERT INTO test2.raw_source_local (id, time, value) SELECT number AS id, ADDTIME('2024-01-01 00:00:00', INTERVAL number SECOND) AS time, RAND() * 100 AS value FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 1000) AS numbers;
CREATE TABLE test2.table_for_mv (sum_value FLOAT, date DATE, count_id INT);
CREATE VIEW test2.mv_from_raw_source_local_to_table AS SELECT SUM(value) AS sum_value, DATE(time) AS date, COUNT(*) AS count_id FROM test2.raw_source_local GROUP BY DATE(time);
