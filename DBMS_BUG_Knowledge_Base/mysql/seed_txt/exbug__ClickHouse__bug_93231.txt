DROP DATABASE IF EXISTS dtobs;
CREATE DATABASE dtobs;
CREATE TABLE span_local (links JSON, dummy TINYINT(1)) ENGINE=InnoDB;
CREATE TABLE span_link_local (linked_id VARCHAR(255)) ENGINE=InnoDB;
CREATE VIEW mv_bad AS SELECT JSON_UNQUOTE(JSON_EXTRACT(link, '$.trace_id')) AS linked_id FROM (SELECT JSON_UNQUOTE(JSON_EXTRACT(links, '$[*]')) AS link FROM span_local) AS temp WHERE linked_id IS NOT NULL;
ALTER TABLE span_local DROP COLUMN dummy;
