CREATE TABLE d3.t2 (id INT UNSIGNED, value VARCHAR(255), timestamp DATETIME DEFAULT CURRENT_TIMESTAMP) ENGINE=InnoDB;
CREATE TABLE d3.t2_dist (id INT UNSIGNED, value VARCHAR(255), timestamp DATETIME DEFAULT CURRENT_TIMESTAMP) ENGINE=InnoDB;
CREATE VIEW d3.v3 AS SELECT id, IF(value IS NULL, 'default_value', value) AS value FROM d3.t2 WHERE id >= 10;
CREATE VIEW d3.v4 AS SELECT id, CONCAT(value, '_suffix') AS value FROM d3.t2_dist WHERE timestamp < NOW() - INTERVAL 1 DAY;
