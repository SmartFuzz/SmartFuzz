CREATE TABLE test(id BIGINT, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test(id, name) SELECT @rownum := @rownum + 1, CONCAT('Record ', @rownum) FROM (SELECT @rownum := 0) r, bigtable LIMIT 20000000;
DELETE FROM test WHERE id = 1;
DELETE FROM test WHERE id = (SELECT MAX(id) FROM test);
SELECT COUNT(*) FROM test WHERE created_at IS NOT NULL;
