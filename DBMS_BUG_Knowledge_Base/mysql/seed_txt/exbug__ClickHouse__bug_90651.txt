CREATE DATABASE IF NOT EXISTS test_db;
CREATE TABLE IF NOT EXISTS test_distr (num INT, date DATETIME, value FLOAT DEFAULT 0.0);
INSERT INTO test_distr SELECT number AS num, NOW() AS date, number * 1.5 AS value FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 100) AS numbers;
CREATE TABLE IF NOT EXISTS replicated_table (num BIGINT, date DATE, metric FLOAT) ENGINE = InnoDB;
INSERT INTO replicated_table (num, date, metric) SELECT num, date, (SELECT date FROM test_distr WHERE num = test_distr.num LIMIT 1) AS metric FROM test_distr WHERE num IS NOT NULL;
