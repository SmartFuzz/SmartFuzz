DROP TABLE IF EXISTS t1, t2;
CREATE TABLE t1(id INT PRIMARY KEY, col1 VARCHAR(10) DEFAULT NULL, col2 VARCHAR(10) DEFAULT NULL, col3 VARCHAR(10) DEFAULT NULL);
CREATE TABLE t2(id INT PRIMARY KEY, col1 VARCHAR(10) NOT NULL, col2 VARCHAR(10) NOT NULL, col3 VARCHAR(10) NOT NULL);
INSERT INTO t1 VALUES(1, NULL, NULL, NULL), (2, NULL, NULL, NULL), (3, NULL, NULL, NULL);
INSERT INTO t2 VALUES(1, 'x', 'xx', 'xxx'), (2, 'y', 'yy', 'yyy'), (3, 'z', 'zz', 'zzz');
UPDATE t1 SET col1 = (SELECT t2.col1 FROM t2 WHERE t2.id = t1.id AND t2.id > 1), col2 = (SELECT t2.col2 FROM t2 WHERE t2.id = t1.id AND t2.id > 1), col3 = (SELECT t2.col3 FROM t2 WHERE t2.id = t1.id AND t2.id > 1) WHERE t1.id IN (SELECT id FROM t2 WHERE id > 1);
