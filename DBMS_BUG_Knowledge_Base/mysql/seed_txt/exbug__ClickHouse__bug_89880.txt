CREATE TABLE T1 (K BIGINT, C0 VARCHAR(255), C1 VARCHAR(255), C2 DOUBLE) ENGINE=InnoDB;
INSERT INTO T1 (K, C0, C1, C2) SELECT number, CONCAT_WS(',', number * 1), IF(number % 2 = 0, 'even', NULL), number * 1.5 FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT @rownum := 0) t2 LIMIT 100) AS numbers;
CREATE TABLE T2 (K BIGINT, C0 VARCHAR(255), C1 INT) ENGINE=InnoDB;
INSERT INTO T2 (K, C0, C1) SELECT K, C0, CAST(K AS SIGNED) FROM T1 WHERE C1 IS NOT NULL;
SELECT * FROM T1 WHERE C1 = 'even';
