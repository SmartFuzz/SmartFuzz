CREATE TABLE IF NOT EXISTS user_activity (activity_id BIGINT, user_id INT, activity_time DATETIME, activity_type ENUM('login', 'logout', 'purchase'), details TEXT);
INSERT INTO user_activity SELECT number AS activity_id, FLOOR(RAND() * 1000) AS user_id, NOW() + INTERVAL number * 5 SECOND AS activity_time, (CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, CONCAT('Activity details for ', number) AS details FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 100);
CREATE TABLE IF NOT EXISTS product_sales (sale_id INT, product_id INT, sale_time DATETIME, quantity SMALLINT, price FLOAT);
INSERT INTO product_sales SELECT number AS sale_id, FLOOR(RAND() * 500) AS product_id, NOW() + INTERVAL number * 2 SECOND AS sale_time, (FLOOR(RAND() * 10) + 1) AS quantity, (FLOOR(RAND() * 100) + 1) * 1.5 AS price FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 50);
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL 1 DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
