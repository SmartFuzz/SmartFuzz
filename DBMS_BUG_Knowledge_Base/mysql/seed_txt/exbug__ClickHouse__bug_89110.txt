CREATE TABLE IF NOT EXISTS initial_complex (n INT UNSIGNED, description VARCHAR(255), _timestamp DATETIME DEFAULT CURRENT_TIMESTAMP) ENGINE=InnoDB;
CREATE TABLE IF NOT EXISTS target1_complex (x INT UNSIGNED, status VARCHAR(255), _timestamp DATETIME DEFAULT CURRENT_TIMESTAMP) ENGINE=InnoDB;
CREATE TABLE IF NOT EXISTS target2_complex (x INT UNSIGNED, value FLOAT, _timestamp DATETIME DEFAULT CURRENT_TIMESTAMP) ENGINE=InnoDB;
CREATE VIEW initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM initial_complex WHERE RAND() < 0.5;
CREATE VIEW target1_target2_mv_complex AS SELECT x, CAST(x AS DECIMAL(10,2)) * 1.5 AS value FROM target1_complex WHERE status IS NOT NULL;
