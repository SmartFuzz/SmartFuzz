CREATE TABLE test_unpivot (id INT, column_name VARCHAR(255), column_value VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_unpivot (id, column_name, column_value) SELECT id, CONCAT('column_', id), CONCAT('value_', id) FROM (SELECT id FROM (SELECT @rownum := @rownum + 1 AS id FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 10000) AS ids);
CREATE TABLE pivoted_test AS SELECT id, MAX(CASE WHEN column_name = 'column_1' THEN column_value END) AS column_1, MAX(CASE WHEN column_name = 'column_2' THEN column_value END) AS column_2, MAX(CASE WHEN column_name = 'column_3' THEN column_value END) AS column_3, MAX(CASE WHEN column_name = 'column_4' THEN column_value END) AS column_4, MAX(CASE WHEN column_name = 'column_5' THEN column_value END) AS column_5 FROM test_unpivot GROUP BY id;
SELECT COUNT(*) FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) AS unpivoted;
CREATE TABLE final_result AS SELECT * FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) AS filtered WHERE id < 5000;
