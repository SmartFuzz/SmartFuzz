CREATE TABLE test_table (id INT PRIMARY KEY, name VARCHAR(50), value DECIMAL(10,2), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table (id, name, value) VALUES (1, 'A', 10.00), (2, 'B', NULL), (3, 'C', 20.50), (4, 'D', 30.00);
SELECT id, name, JSON_ARRAYAGG(value) OVER (ORDER BY created_at ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS json_values FROM test_table;
SELECT id, name, JSON_OBJECTAGG(name, value) OVER (PARTITION BY value IS NULL ORDER BY id) AS json_agg FROM test_table;
SELECT id, name, LEAD(value, 1) OVER (ORDER BY id) AS next_value FROM test_table WHERE value IS NOT NULL;
