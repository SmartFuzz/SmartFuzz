CREATE TABLE products (id INT PRIMARY KEY, name VARCHAR(255), description VARCHAR(255), price DECIMAL(10, 2), created_at DATETIME);
CREATE TABLE orders (id INT PRIMARY KEY, product_id INT, order_date DATETIME, quantity INT, large_blob LONGBLOB, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, CONCAT('Product ', i) AS name, CONCAT('Description for product ', i) AS description, ROUND(RAND() * 100, 2) AS price, DATE_ADD('2024-01-01', INTERVAL i DAY) AS created_at FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 10000) AS temp;
INSERT INTO orders SELECT i AS id, (i % 10000) + 1 AS product_id, DATE_ADD('2024-02-01', INTERVAL i DAY) AS order_date, (i % 10) + 1 AS quantity, REPEAT('y', 2000) AS large_blob FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers LIMIT 5000000) AS temp;
SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01' AND '2024-04-01' AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
