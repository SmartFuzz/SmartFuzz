CREATE TABLE employees (dep VARCHAR(10) NOT NULL, name VARCHAR(20) NOT NULL, salary DOUBLE NOT NULL CHECK (salary > 0), hire_date DATE DEFAULT NULL);
INSERT INTO employees (dep, name, salary, hire_date) VALUES ('mgmt', 'Smith', 81, STR_TO_DATE('10-08-2019', '%d-%m-%Y'));
INSERT INTO employees (dep, name, salary, hire_date) VALUES ('dev', 'Jones', 55, STR_TO_DATE('9-08-2019', '%d-%m-%Y'));
SELECT dep, name, salary, FLOOR((ROW_NUMBER() OVER (ORDER BY hire_date DESC) - 1) / (SELECT COUNT(*) FROM employees) * 4) + 1 AS ntile FROM employees;
SELECT dep, name, salary, FLOOR((ROW_NUMBER() OVER (ORDER BY hire_date) - 1) / (SELECT COUNT(*) FROM employees) * 4) + 1 AS ntile FROM employees WHERE salary IS NOT NULL;
