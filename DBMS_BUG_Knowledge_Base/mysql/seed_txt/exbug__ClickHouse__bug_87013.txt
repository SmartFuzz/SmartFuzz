DROP DATABASE IF EXISTS test;
CREATE DATABASE test;
CREATE TABLE IF NOT EXISTS test.mv_src_table (id BIGINT UNSIGNED, value VARCHAR(255));
INSERT INTO test.mv_src_table SELECT number, CAST(number AS CHAR) FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS numbers;
CREATE TABLE IF NOT EXISTS test.mv_refreshable_dst_table (id BIGINT UNSIGNED);
CREATE TRIGGER mv_refreshable_trigger AFTER INSERT ON test.mv_src_table FOR EACH ROW BEGIN INSERT INTO test.mv_refreshable_dst_table (id) SELECT MAX(id) FROM test.mv_src_table WHERE value IS NOT NULL;
