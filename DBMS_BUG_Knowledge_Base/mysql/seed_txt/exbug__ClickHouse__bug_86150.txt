DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id INT UNSIGNED, keyword_id INT UNSIGNED, search_date DATETIME, position_on_serp INT UNSIGNED, version INT UNSIGNED, value VARCHAR(255));
INSERT INTO test_final_partitions (organization_id, keyword_id, search_date, position_on_serp, version, value) SELECT FLOOR(RAND() * 1000) AS organization_id, FLOOR(RAND() * 100000) AS keyword_id, DATE_ADD('2023-01-01', INTERVAL FLOOR(RAND() * 180 * 86400) SECOND) AS search_date, FLOOR(RAND() * 50) AS position_on_serp, MOD(number, 5) AS version, CAST(number AS CHAR) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r LIMIT 50000000) AS numbers;
SELECT COUNT(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT COUNT(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
