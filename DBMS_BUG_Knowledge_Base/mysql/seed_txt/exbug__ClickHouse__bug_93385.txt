CREATE TABLE user_events (event_id CHAR(36), user_id INT UNSIGNED, event_type ENUM('click', 'view', 'purchase'), event_time DATETIME);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT UUID(), FLOOR(RAND() * 10000) AS user_id, ELT(FLOOR(RAND() * 3) + 1, 'click', 'view', 'purchase') AS event_type, NOW() - INTERVAL FLOOR(RAND() * 3600 * 24) SECOND AS event_time FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) AS numbers LIMIT 100000;
CREATE TABLE user_summary (user_id INT UNSIGNED, event_count INT UNSIGNED);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
