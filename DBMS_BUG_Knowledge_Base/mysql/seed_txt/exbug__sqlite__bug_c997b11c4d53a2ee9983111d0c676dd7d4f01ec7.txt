CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE posts (id INT AUTO_INCREMENT PRIMARY KEY, user_id INT, content TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id INT AUTO_INCREMENT PRIMARY KEY, post_id INT, user_id INT, comment TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(post_id) REFERENCES posts(id), FOREIGN KEY(user_id) REFERENCES users(id));
