CREATE TABLE test_data (id INT, values JSON);
INSERT INTO test_data VALUES (1, JSON_ARRAY('1', '2', '3')), (2, JSON_ARRAY('4', '5', NULL)), (3, JSON_ARRAY('6', '7', '8'));
SELECT id, JSON_LENGTH(values) AS value_count FROM test_data GROUP BY id;
SELECT id, AVG(CAST(JSON_UNQUOTE(JSON_EXTRACT(values, CONCAT('$[', n.n, ']'))) AS UNSIGNED)) AS centroid FROM test_data JOIN (SELECT 0 AS n UNION ALL SELECT 1 UNION ALL SELECT 2) n ON n.n < JSON_LENGTH(values) GROUP BY id;
SELECT id, SUM(CAST(JSON_UNQUOTE(JSON_EXTRACT(values, CONCAT('$[', n.n, ']'))) AS UNSIGNED)) AS total_sum FROM test_data JOIN (SELECT 0 AS n UNION ALL SELECT 1 UNION ALL SELECT 2) n ON n.n < JSON_LENGTH(values) GROUP BY id;
