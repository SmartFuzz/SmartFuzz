CREATE TABLE tbl1 (id INT UNSIGNED, value VARCHAR(255), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO tbl1 (id, value) SELECT number, CAST(number % 10 AS CHAR) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) AS a, (SELECT @rownum := 0) AS r LIMIT 100) AS numbers;
CREATE TABLE tbl2 (ref_id INT UNSIGNED, description VARCHAR(255), amount DOUBLE);
INSERT INTO tbl2 (ref_id, description, amount) SELECT id, CONCAT('Description ', id), id * 1.5 FROM tbl1;
SELECT tbl1.id, tbl2.amount FROM tbl1 JOIN tbl2 ON tbl1.id = tbl2.ref_id WHERE tbl1.value IS NOT NULL ORDER BY tbl1.id;
