DROP DATABASE IF EXISTS sandbox;
CREATE DATABASE sandbox;
USE sandbox;
CREATE TABLE t_ref (id bigint(20) NOT NULL AUTO_INCREMENT, PRIMARY KEY (id)) ENGINE=InnoDB;
CREATE TABLE t_main (id int(20) NOT NULL AUTO_INCREMENT, unique_id varchar(32) GENERATED ALWAYS AS ('1') VIRTUAL, t_ref_id bigint(20) NOT NULL, PRIMARY KEY (id), UNIQUE KEY (unique_id), FOREIGN KEY (t_ref_id) REFERENCES t_ref (id)) ENGINE=InnoDB;
