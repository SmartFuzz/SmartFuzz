CREATE TABLE test_vals(grp INT NOT NULL, value INT CHECK(value >= 0));
INSERT INTO test_vals (grp, value) VALUES (1, 100), (2, 200), (1, NULL), (2, 300), (3, -50);
CREATE FUNCTION jit_sum_variadic(grp INT) RETURNS BIGINT DETERMINISTIC BEGIN DECLARE total BIGINT DEFAULT 0;
SELECT SUM(value) INTO total FROM test_vals WHERE grp = grp AND value IS NOT NULL;
RETURN total;
END;
