CREATE TABLE test_table(test_column INT NOT NULL, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table(test_column, name) SELECT seq AS test_column, 'Test Name' FROM (SELECT @rownum := @rownum + 1 AS seq FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) t2, (SELECT @rownum := 0) r) AS numbers WHERE seq <= 2147483647;
SELECT COUNT(*) FROM test_table WHERE test_column IS NOT NULL;
DELETE FROM test_table WHERE test_column < 1000;
SELECT AVG(test_column) FROM test_table WHERE created_at IS NOT NULL;
