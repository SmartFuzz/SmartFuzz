DROP TABLE IF EXISTS job_sample;
CREATE TABLE job_sample (id INT UNSIGNED, posted_at DATE, discovered_at DATETIME(3), url VARCHAR(255), title VARCHAR(255), company_name VARCHAR(255), description TEXT);
LOAD DATA INFILE 'https://media.theirstack.com/jobs/jan-2025-sample.parquet' INTO TABLE job_sample FIELDS TERMINATED BY ',';
ALTER TABLE job_sample ADD COLUMN description_matches_python TINYINT(1) DEFAULT 0;
UPDATE job_sample SET description_matches_python = CASE WHEN description REGEXP 'Python' THEN 1 ELSE 0 END;
