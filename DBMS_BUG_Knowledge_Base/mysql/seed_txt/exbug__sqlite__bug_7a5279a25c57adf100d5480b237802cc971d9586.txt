CREATE TABLE users(id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255), age INT, email VARCHAR(255) UNIQUE);

SELECT u.name FROM users u JOIN users v ON u.id = v.id WHERE u.age IN (SELECT MAX(age) FROM users WHERE age IS NOT NULL);
SELECT u.email FROM users u WHERE u.id IN (SELECT id FROM users WHERE age < 25 UNION SELECT id FROM users WHERE email IS NULL);
