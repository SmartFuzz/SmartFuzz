DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR(255), event_time DATETIME, id BIGINT, msg VARCHAR(255), value FLOAT NULL);
INSERT INTO repro.events (tenant, event_time, id, msg, value) SELECT CAST(number % 100 AS CHAR) AS tenant, NOW() - INTERVAL (number % 200000) SECOND AS event_time, number AS id, CONCAT('m', CAST(number AS CHAR)) AS msg, CAST(number % 1000 AS FLOAT) AS value FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT @rownum := 0) r, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9) t1, (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9) t2 LIMIT 2000000) AS numbers;
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
