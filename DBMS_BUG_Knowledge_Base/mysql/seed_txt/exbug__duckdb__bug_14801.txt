CREATE FUNCTION s(str VARCHAR(255), re VARCHAR(255), replacement VARCHAR(255)) RETURNS VARCHAR(255) DETERMINISTIC BEGIN DECLARE result VARCHAR(255);
SELECT CONCAT(SUBSTRING_INDEX(SUBSTRING_INDEX(str, 'r', 1), 'r', -1), replacement, SUBSTRING_INDEX(str, 'r', -1)) INTO result FROM (SELECT str, re) AS cte WHERE str REGEXP re LIMIT 1;
RETURN result;
END;
