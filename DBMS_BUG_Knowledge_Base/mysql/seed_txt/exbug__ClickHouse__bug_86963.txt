DROP TABLE IF EXISTS test_data;
CREATE TABLE test_data (id BIGINT UNSIGNED, value VARCHAR(255), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_data (id, value) SELECT number, CAST(number AS CHAR) FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 100) AS numbers;
SELECT id, value FROM test_data WHERE MOD(id, 2) = 0;
