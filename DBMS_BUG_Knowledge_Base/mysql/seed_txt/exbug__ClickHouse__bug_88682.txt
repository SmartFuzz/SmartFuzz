CREATE TABLE c (a BIGINT, b BIGINT NULL);
INSERT INTO c SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) AS numbers, (SELECT @rownum := 0) AS init);
CREATE TABLE d (m BIGINT, n BIGINT);
INSERT INTO d SELECT number, number + 1 FROM (SELECT @rownum := @rownum + 1 AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5) AS numbers, (SELECT @rownum := 0) AS init);
SELECT c.a, d.n FROM c LEFT JOIN d ON c.a > d.m;
