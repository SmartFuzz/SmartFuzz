CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0));
CREATE TABLE comments (id INT AUTO_INCREMENT PRIMARY KEY, user_id INT, content TEXT NOT NULL, FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE);
