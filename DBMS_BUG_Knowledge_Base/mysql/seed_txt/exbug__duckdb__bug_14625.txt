CREATE TABLE test_strings (id INT PRIMARY KEY, name VARCHAR(3000) NOT NULL);
INSERT INTO test_strings SELECT i, REPEAT('abc', 1000) FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 100) AS seq;
CREATE TABLE test_numbers (num INT, name VARCHAR(3000));
INSERT INTO test_numbers SELECT i, REPEAT('xyz', 200) FROM (SELECT @rownum := @rownum + 1 AS i FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) t1, (SELECT @rownum := 0) t2 LIMIT 50) AS seq;
SELECT t1.id, t2.name FROM test_strings t1 JOIN test_numbers t2 ON t1.id = t2.num WHERE t1.name IS NOT NULL;
