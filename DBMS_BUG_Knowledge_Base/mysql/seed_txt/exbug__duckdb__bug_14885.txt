CREATE TABLE test_dates (id INT PRIMARY KEY, dt DATE NOT NULL, value INT);

CREATE FUNCTION yearqtr(x DATE) RETURNS INT DETERMINISTIC RETURN (YEAR(x) * 10 + FLOOR((MONTH(x) - 1) / 3) + 1);
SELECT dt, yearqtr(dt) AS yq, yearqtr(dt) DIV 10 AS yq2, yearqtr(dt) DIV 3 AS yq3, yearqtr(dt) MOD 3 AS yq4 FROM test_dates WHERE dt = '2050-12-31';
