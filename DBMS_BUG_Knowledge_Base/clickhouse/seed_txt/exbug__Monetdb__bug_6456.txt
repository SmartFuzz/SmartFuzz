CREATE TABLE data_local (origin String, dest String, value Nullable(Int32));

CREATE TABLE data_slave1 AS SELECT origin, dest, ifNull(value, 0) AS value FROM data_local;
CREATE TABLE data_slave2 AS SELECT origin, dest FROM data_local WHERE value IS NULL;
SELECT a.origin, a.dest, b.value FROM data_slave1 a LEFT JOIN data_slave2 b ON a.origin = b.origin;
