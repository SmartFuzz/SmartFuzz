CREATE TABLE ap (a1 Int32, a2 Int32, prob Float32) ENGINE = MergeTree() ORDER BY (a1, a2);
CREATE TABLE bp (a1 Int32, a2 Int32, prob Float32) ENGINE = MergeTree() ORDER BY (a1, a2);
INSERT INTO ap VALUES (1, 1, 0.5), (2, 1, 0.3), (3, 2, 0.2);
INSERT INTO bp VALUES (1, 1, 0.4), (2, 1, 0.6), (3, 2, 0.1);
SELECT ap.a1, bp.a1 AS a3, SUM(ap.prob * bp.prob) AS total_prob FROM ap JOIN bp ON ap.a2 = bp.a2 GROUP BY ap.a1, a3;
