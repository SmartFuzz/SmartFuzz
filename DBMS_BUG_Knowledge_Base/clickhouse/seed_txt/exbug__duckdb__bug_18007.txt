CREATE TABLE cols (c String);
INSERT INTO cols SELECT concat('c', toString(number)) FROM numbers(21);
CREATE TABLE matrix (col1 String, col2 String, val UInt8) ENGINE = MergeTree() ORDER BY col1;
INSERT INTO matrix SELECT c1.c, c2.c, CAST(rand() % 250 AS UInt8) FROM cols c1 JOIN cols c2 ON c1.c != c2;
SELECT col2, SUM(val) AS total FROM matrix GROUP BY col2 ORDER BY col2 LIMIT 2;
