CREATE TABLE categories(id UInt32, name String);
CREATE TABLE products(id UInt32, name String, category_id UInt32);
INSERT INTO categories VALUES(1, 'Electronics'), (2, 'Books'), (3, 'Clothing');

SELECT c.name, COUNT(p.id) FROM categories c LEFT JOIN products p ON c.id = p.category_id GROUP BY c.name ORDER BY c.name;
