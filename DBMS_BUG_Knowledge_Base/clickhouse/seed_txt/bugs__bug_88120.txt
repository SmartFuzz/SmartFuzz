CREATE TABLE IF NOT EXISTS logs (id UInt32, date Date, host_id UInt32, message String, status LowCardinality(String), value Float64, count UInt64, created_at DateTime, updated_at DateTime, user_id UInt32, error_code Int32, description String, tags Array(String), metadata Map(String, String), priority Enum('low', 'medium', 'high'), session_id UUID, ip_address IPv4, region String, duration UInt32, response_time Float32, is_active UInt8) ENGINE = MergeTree() PARTITION BY toYYYYMM(date) ORDER BY (toDate(date), host_id) TTL date + INTERVAL 90 DAY SETTINGS index_granularity = 8192, min_bytes_for_wide_part = 0;
