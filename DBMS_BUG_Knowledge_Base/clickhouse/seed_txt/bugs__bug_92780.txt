CREATE TABLE test_data (id UInt32, ts DateTime, value String, status Nullable(String)) ENGINE = MergeTree() ORDER BY (id, ts) SETTINGS index_granularity = 8192;
INSERT INTO test_data SELECT number, now() + INTERVAL number SECOND, toString(number % 5), if(number % 2 = 0, 'active', NULL) FROM numbers(1000000);
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > now() - INTERVAL 1 DAY AND value LIKE '1%';
