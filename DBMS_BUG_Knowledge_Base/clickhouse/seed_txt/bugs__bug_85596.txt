CREATE TABLE tbl1 (id UInt32, value Nullable(String), created_at DateTime DEFAULT now()) ENGINE = MergeTree() ORDER BY id;
INSERT INTO tbl1 SELECT number, toString(number % 10) FROM numbers(100);
CREATE TABLE tbl2 (ref_id UInt32, description String, amount Float64) ENGINE = MergeTree() ORDER BY ref_id;
INSERT INTO tbl2 SELECT id, concat('Description ', toString(id)), id * 1.5 FROM tbl1;
