CREATE FUNCTION normalize_si(sva String) RETURNS String AS (substring(sva, 9, 2) || substring(sva, 7, 2) || substring(sva, 5, 2) || substring(sva, 1, 4));
CREATE FUNCTION si_lt(a String, b String) RETURNS UInt8 AS (IF(normalize_si(a) < normalize_si(b), 1, 0));
CREATE FUNCTION si_same(a String, b String) RETURNS Int8 AS (IF(normalize_si(a) < normalize_si(b), -1, IF(normalize_si(a) > normalize_si(b), 1, 0)));
