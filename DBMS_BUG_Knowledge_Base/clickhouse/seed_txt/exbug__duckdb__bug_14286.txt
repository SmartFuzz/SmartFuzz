CREATE TABLE test_data (id Int32, name String, category String);

SELECT category, arrayDistinct(arrayFilter(x -> x IS NOT NULL, arrayMap(x -> name, groupArray(name))) ) AS a_list FROM test_data GROUP BY category;
SELECT category, arrayFilter(x -> x IS NOT NULL, arrayMap(x -> name, groupArray(name))) AS b_list FROM test_data GROUP BY category;
SELECT category, arrayMap(x -> (id, name, category), groupArray((id, name, category))) AS c_list FROM test_data GROUP BY category;
