DROP TABLE IF EXISTS data_local;
CREATE TABLE data_local (id UInt32, value String) ENGINE=Memory() AS SELECT number, toString(number) FROM numbers(10);
CREATE TABLE data_distributed AS data_local ENGINE=Distributed(test_cluster_two_shards, currentDatabase(), data_local, id);
SELECT count() FROM (SELECT DISTINCT value FROM data_distributed) SETTINGS optimize_skip_unused_shards=0;
SELECT count() FROM (SELECT DISTINCT value FROM data_distributed) SETTINGS optimize_skip_unused_shards=1;
