SELECT id, name, arrayJoin(groupArray(value)) OVER (ORDER BY created_at ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS json_values FROM test_table;
SELECT id, name, groupArrayIf(value, value IS NULL) OVER (PARTITION BY value IS NULL ORDER BY id) AS json_agg FROM test_table;
