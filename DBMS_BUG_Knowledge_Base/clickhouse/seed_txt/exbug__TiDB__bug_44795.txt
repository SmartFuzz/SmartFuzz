CREATE TABLE customer (c_custkey Int32, c_name String, c_address String, c_comment String);
CREATE TABLE orders (o_orderkey Int32, o_custkey Int32, o_comment String, FOREIGN KEY (o_custkey) REFERENCES customer(c_custkey));
INSERT INTO customer (c_custkey, c_name, c_address, c_comment) VALUES (1, 'Alice', '123 Main St', 'No special requests'), (2, 'Bob', '456 Elm St', 'Special requests'), (3, 'Charlie', '789 Oak St', 'Regular customer');
INSERT INTO orders (o_orderkey, o_custkey, o_comment) VALUES (1, 1, 'Order 1'), (2, 1, 'Order 2'), (3, 2, 'Order 3'), (4, 3, 'Special requests'), (5, 1, 'Order 4');
SELECT c_count, count(*) as custdist FROM ( SELECT c_custkey, count(o_orderkey) as c_count FROM customer LEFT JOIN orders ON c_custkey = o_custkey AND NOT like(o_comment, '%special%requests%') GROUP BY c_custkey ) c_orders GROUP BY c_count ORDER BY custdist DESC, c_count DESC;
