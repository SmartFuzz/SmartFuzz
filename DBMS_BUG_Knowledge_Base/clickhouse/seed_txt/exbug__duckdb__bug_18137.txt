CREATE TABLE test_large_lists (id UInt32, gts Array(Array(UInt8)));
INSERT INTO test_large_lists SELECT i, array(array(rand() % 256, rand() % 256)) FROM numbers(500000) AS i;
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, arrayJoin(gts) AS inner_list FROM list_view;
