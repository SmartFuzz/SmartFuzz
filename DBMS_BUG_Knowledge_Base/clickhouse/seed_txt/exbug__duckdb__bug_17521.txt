CREATE TABLE t1 (v1 Int32, v2 Int32, v3 Int32 DEFAULT 0);
CREATE TABLE t2 (v1 Int32, v2 Int32, v4 String, v5 Float64);
INSERT INTO t1 SELECT i, i, i % 2 FROM numbers(1000);
INSERT INTO t2 SELECT i, i, concat('value', toString(i)), i * 1.0 FROM numbers(10000000);
SELECT t1.v2, t2.v2, COUNT(*) FROM t1 INNER JOIN t2 ON (t1.v1 = t2.v1) GROUP BY t1.v2, t2.v2 ORDER BY t1.v2 LIMIT 5;
