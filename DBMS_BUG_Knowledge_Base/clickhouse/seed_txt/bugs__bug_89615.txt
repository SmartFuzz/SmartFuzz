CREATE TABLE set_idx_rand_v2
(
  id UInt32,
  s String,
  d DateTime DEFAULT now(),
  INDEX idx_set s TYPE set(0) GRANULARITY 1
)
ENGINE=MergeTree
ORDER BY id;
INSERT INTO set_idx_rand_v2 SELECT number, toString(number % 10) FROM numbers(100);
SELECT * FROM set_idx_rand_v2 WHERE rand() % 1000 == 1 SETTINGS use_skip_indexes=1;
SELECT * FROM set_idx_rand_v2 WHERE d IS NULL OR d > now() - INTERVAL 1 DAY;
SELECT COUNT(*) FROM set_idx_rand_v2 WHERE s IN (SELECT s FROM set_idx_rand_v2 WHERE id < 50) AND rand() % 1000 == 1;
