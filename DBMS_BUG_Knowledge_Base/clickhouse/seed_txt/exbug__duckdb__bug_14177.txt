CREATE TABLE test_data (id Int32, name String, age Int32, created_at DateTime DEFAULT now());
INSERT INTO test_data SELECT * FROM file('test_duckdb_data.csv', format='CSVWithNames', skip_header=1);
CREATE VIEW valid_records AS SELECT * FROM test_data WHERE age IS NOT NULL;
CREATE VIEW rejected_records AS SELECT * FROM file('test_duckdb_data.csv', format='CSVWithNames', skip_header=1) WHERE name IS NULL;
SELECT COUNT(*) AS valid_count, (SELECT COUNT(*) FROM rejected_records) AS rejected_count FROM valid_records;
