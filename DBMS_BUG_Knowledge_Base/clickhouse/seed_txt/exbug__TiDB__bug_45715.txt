CREATE TABLE sales (year Int32 NOT NULL, country String NOT NULL, product String NOT NULL, profit Int32 DEFAULT 0) ENGINE = MergeTree() PRIMARY KEY (year, country, product);
INSERT INTO sales (year, country, product, profit) VALUES (2020, 'USA', 'Gadget', 1000), (2020, 'USA', 'Widget', 1500), (2020, 'Canada', 'Gadget', 2000), (2021, 'USA', 'Gadget', 1200);
SELECT year, country, product, SUM(profit) AS total_profit FROM sales GROUP BY year, country, product WITH ROLLUP ORDER BY year, country;
CREATE TABLE products (product_id Int32 NOT NULL, product_name String NOT NULL, category String, PRIMARY KEY (product_id)) ENGINE = MergeTree();
