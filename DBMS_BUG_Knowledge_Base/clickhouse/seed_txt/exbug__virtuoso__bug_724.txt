CREATE TABLE images (id UInt32, img_url String, description String) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE audio (id UInt32, audio_url String, duration UInt32) ENGINE = MergeTree() ORDER BY id;
INSERT INTO images (id, img_url, description) VALUES (1, 'http://example.com/image1.jpg', 'Image 1 description'), (2, 'http://example.com/image2.jpg', 'Image 2 description'), (3, 'http://example.com/image3.jpg', 'Image 3 description');
INSERT INTO audio (id, audio_url, duration) VALUES (1, 'http://example.com/audio1.mp3', 120), (2, 'http://example.com/audio2.mp3', 150), (3, 'http://example.com/audio3.mp3', 90);
SELECT img.img_url, aud.audio_url FROM images img JOIN audio aud ON img.id = aud.id WHERE aud.duration IS NOT NULL AND img.id <= 3;
