CREATE TABLE urls(urlid Int64, c Int64, d Int64, url_text String) ENGINE = MergeTree() ORDER BY urlid;
CREATE TABLE keys(urlid Int64, a Int64, b Int64, created_at DateTime DEFAULT now()) ENGINE = MergeTree() ORDER BY urlid;
INSERT INTO urls(urlid, c, d, url_text) VALUES (1, 10, 20, 'http://example.com/1');
