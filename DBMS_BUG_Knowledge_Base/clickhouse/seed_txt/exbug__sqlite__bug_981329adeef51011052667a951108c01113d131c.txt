CREATE TABLE test_table (id UInt64, value String, depth Int32 DEFAULT 0);
INSERT INTO test_table (value, depth) VALUES ('Initial', 0);
WITH c AS (SELECT number AS x FROM numbers(400000)) SELECT countIf(x >= 1) FROM c;
SELECT * FROM test_table WHERE depth IS NULL;
UPDATE test_table SET value = 'Updated' WHERE id = (SELECT id FROM test_table ORDER BY depth DESC LIMIT 1);
