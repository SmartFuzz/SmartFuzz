CREATE TABLE test_unpivot (id Int32, column_name String, column_value String, created_at DateTime DEFAULT now());
INSERT INTO test_unpivot SELECT id, concat('column_', id), concat('value_', id) FROM generateSeries(1, 10000) AS id;
CREATE TABLE pivoted_test AS SELECT id, arrayJoin([column_1, column_2, column_3, column_4, column_5]) AS column_value, concat('column_', id) AS column_name FROM test_unpivot;
SELECT COUNT(*) FROM (SELECT id, column_name, column_value FROM pivoted_test WHERE column_value IS NOT NULL) AS unpivoted;
CREATE TABLE final_result AS SELECT * FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) WHERE id < 5000;
