CREATE TABLE json_data (MainLevel Nested(More Nested(Level2 Nested(Field1 String))));

SELECT Level2.Field1 FROM json_data ARRAY JOIN MainLevel AS MainLevel ARRAY JOIN MainLevel.More.Level2 AS Level2;
