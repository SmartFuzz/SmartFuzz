DROP TABLE IF EXISTS test_left;
DROP TABLE IF EXISTS test_right;
CREATE TABLE test_left (a Int64, b String, c LowCardinality(String), d Nullable(Int32)) ENGINE = MergeTree() ORDER BY a;
CREATE TABLE test_right (a Int64, b String, c LowCardinality(String), d Nullable(Int32)) ENGINE = MergeTree() ORDER BY a;
