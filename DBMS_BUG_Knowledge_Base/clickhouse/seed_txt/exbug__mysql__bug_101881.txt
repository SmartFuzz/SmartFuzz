DROP TABLE IF EXISTS TestSequence;
CREATE TABLE TestSequence (num String PRIMARY KEY, value Int32 NOT NULL DEFAULT 0);
INSERT INTO TestSequence (num, value) VALUES ('0:ZERO', 0), ('1:ONE', 1), ('2:TWO', 2), ('3:THREE', 3), ('4:FOUR', 4), ('5:FIVE', 5), ('6:SIX', 6), ('7:SEVEN', 7), ('8:EIGHT', 8), ('9:NINE', 9);
SELECT arrayStringConcat(arrayMap(x -> x.num, arraySort((SELECT TS FROM TestSequence TS WHERE TS.value IS NOT NULL)), 'ORDER BY value ASC')) AS ss;
