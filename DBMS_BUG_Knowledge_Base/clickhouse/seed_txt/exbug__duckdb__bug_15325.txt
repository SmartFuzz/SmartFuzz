SET max_memory_usage='300000000';
CREATE TABLE diverse_table (id UInt32, name String, value Float64, created_at DateTime DEFAULT now(), is_active UInt8);
INSERT INTO diverse_table SELECT i, concat('Name_', toString(i)), rand() * 100, now(), (i % 2 = 0) FROM numbers(1000000);
SELECT id, name, value, created_at, is_active FROM diverse_table WHERE is_active IS NOT NULL;
SELECT COUNT(*) AS total_count, SUM(CASE WHEN value IS NULL THEN 1 ELSE 0 END) AS null_value_count FROM diverse_table;
