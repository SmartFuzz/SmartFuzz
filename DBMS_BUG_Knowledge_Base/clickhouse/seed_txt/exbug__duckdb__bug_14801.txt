CREATE FUNCTION s(str String, re String, replacement String) RETURNS String AS (SELECT concat(substring(str, 1, position(str, 'r') - 1), replacement, substring(str, position(str, 'r') + 1)) FROM (SELECT regexp_extract(str, re) AS r FROM (SELECT str, re FROM test_data WHERE id = 1) AS cte));
