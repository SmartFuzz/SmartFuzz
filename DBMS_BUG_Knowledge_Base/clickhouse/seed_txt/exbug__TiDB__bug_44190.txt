CREATE TABLE employee_info ( employee_id UInt64 NOT NULL, employee_name String NOT NULL, department_id UInt64 NOT NULL, salary Decimal(10, 2) NOT NULL, UNIQUE INDEX employee_unique(employee_id, department_id) ) ENGINE = MergeTree() ORDER BY employee_id;

CREATE TABLE department_info ( department_id UInt64 NOT NULL, department_name String NOT NULL, PRIMARY KEY (department_id) ) ENGINE = MergeTree() ORDER BY department_id;
