CREATE TABLE users (id Int32, name String, age Int32) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE orders (id Int32, user_id Int32, amount Decimal(10, 2)) ENGINE = MergeTree() ORDER BY id;
