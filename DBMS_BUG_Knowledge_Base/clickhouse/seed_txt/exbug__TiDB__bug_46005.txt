DROP TABLE IF EXISTS t;
CREATE TABLE t(a Int32, b String, c Decimal(10, 2)) ENGINE = MergeTree() ORDER BY a;
INSERT INTO t(a, b, c) VALUES (1, 'one', 1.00), (2, 'two', 2.50), (NULL, 'three', NULL);
SELECT * FROM t WHERE a IS NULL OR b = 'one' ORDER BY c LIMIT 10;
UPDATE t SET c = c + 1 WHERE b LIKE 't%';
