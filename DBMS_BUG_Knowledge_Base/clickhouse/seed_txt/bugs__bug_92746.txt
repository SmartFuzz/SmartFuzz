CREATE DATABASE test_db ON CLUSTER '{cluster}';
CREATE TABLE test_db.sample_table ON CLUSTER '{cluster}' (id UInt64, name String, created_at DateTime, score Float64) ENGINE = MergeTree() ORDER BY id;
CREATE VIEW test_db.sample_view ON CLUSTER '{cluster}' AS SELECT id, name, created_at FROM test_db.sample_table WHERE score > 50.0;
