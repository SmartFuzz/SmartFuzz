CREATE TABLE storage_summary (tblname String, count Int64, created_at DateTime DEFAULT now());
INSERT INTO storage_summary (tblname, count) SELECT "table", max("count") FROM system.parts WHERE database = 'sys' GROUP BY "table";
CREATE TABLE storage_details (tblname String NOT NULL, count Int32 CHECK (count >= 0), description String);
INSERT INTO storage_details (tblname, count, description) SELECT "table", toInt32(max("count")), concat('Storage details for ', "table") FROM system.parts WHERE database = 'sys' GROUP BY "table";
SELECT tblname, count FROM storage_summary WHERE count IS NOT NULL ORDER BY count DESC;
