DROP TABLE IF EXISTS mock_data_in;
CREATE TABLE mock_data_in (i Int32, j Int32, k Int32, l UInt32) ENGINE = MergeTree() ORDER BY (i, j);
INSERT INTO mock_data_in SELECT i, j, k, (i * 10000 + j * 100 + k) % 100 AS l FROM numbers(100) AS s CROSS JOIN numbers(10) AS t CROSS JOIN numbers(100) AS u;
CREATE VIEW base_view AS SELECT * FROM mock_data_in;
SELECT * FROM base_view WHERE i = 10 AND j = 10 AND k IS NOT NULL;
