CREATE TABLE equipment (EquipmentName String, FailureReason String) ENGINE = Memory;
INSERT INTO equipment (EquipmentName, FailureReason) VALUES ('Excavator', 'Hydraulic Failure'), ('Bulldozer', 'Engine Overheat');
CREATE VIEW v_resolve_maintenance_action AS WITH {EquipmentName: String, FailureReason: String} AS p_params SELECT * FROM equipment WHERE EquipmentName = p_params.EquipmentName AND FailureReason = p_params.FailureReason;
SELECT * FROM equipment e LEFT JOIN v_resolve_maintenance_action ON e.EquipmentName = v_resolve_maintenance_action.EquipmentName AND e.FailureReason = v_resolve_maintenance_action.FailureReason;
CREATE TABLE maintenance_actions (ActionId Int32, ActionDescription String) ENGINE = Memory;
