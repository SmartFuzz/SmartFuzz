DROP TABLE IF EXISTS source_table;
CREATE TABLE source_table (date Date, value Nullable(Int32)) ENGINE = MergeTree() ORDER BY date;
CREATE MATERIALIZED VIEW mv_table (date Date, number UInt32) ENGINE = MergeTree() PARTITION BY date ORDER BY (date, number) AS SELECT date, value FROM source_table WHERE value IS NOT NULL;
