DROP TABLE IF EXISTS events;
CREATE TABLE events ( `Id` UInt64, `Payload` String, `Time` DateTime, `Status` Nullable(String) ) ENGINE = Memory;
INSERT INTO events SELECT number, concat('Payload_', toString(number)), toDateTime('2024-01-01 00:00:00') + INTERVAL number MINUTES, if(number % 2 = 0, 'active', NULL) FROM numbers(10);
DROP TABLE IF EXISTS attributes;
CREATE TABLE attributes ( `EventId` UInt64, `Attribute` String, `Value` Float64 DEFAULT 0.0 ) ENGINE = MergeTree ORDER BY EventId;
