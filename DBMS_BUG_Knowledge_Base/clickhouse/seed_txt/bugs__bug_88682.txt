CREATE TABLE c (a UInt64, b Nullable(UInt64)) ENGINE = MergeTree ORDER BY a;
INSERT INTO c SELECT number, if(number % 2 = 0, number, NULL) FROM numbers(10);
CREATE TABLE d (m UInt64, n UInt64) ENGINE = MergeTree ORDER BY m;
INSERT INTO d SELECT number, number + 1 FROM numbers(5);
SELECT c.a, d.n FROM c LEFT OUTER JOIN d ON c.a > d.m;
