CREATE TABLE t2 (id Int32, c1 String, d1 Int32 CHECK (d1 >= 0), d2 Int32 DEFAULT 0);
INSERT INTO t2 VALUES (1, 'X', 100, 200), (2, 'Y', 150, 300), (3, 'Z', 0, 0), (4, 'X', 250, 400), (5, 'Y', 300, 500);
SELECT c1, SUM(d1) AS total_d1, SUM(d2) AS total_d2 FROM t2 GROUP BY c1 HAVING SUM(d1) < IF(SUM(d2) > 100, 600, 500);
SELECT c1, COUNT(*) AS count, AVG(d1) AS avg_d1 FROM t2 GROUP BY c1 HAVING AVG(d1) > IF(COUNT(*) > 2, 100, 50);
SELECT c1, SUM(d1) AS total_d1, SUM(d2) AS total_d2, IF(SUM(d1) IS NULL, 0, SUM(d1)) AS safe_sum FROM t2 GROUP BY c1 HAVING safe_sum < 300;
