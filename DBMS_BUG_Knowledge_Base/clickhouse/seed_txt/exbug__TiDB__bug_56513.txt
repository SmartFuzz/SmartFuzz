CREATE TABLE t1(a Int32, b String, c Date DEFAULT today());
CREATE TABLE t2(a Int32, b Int32, c Float64 DEFAULT 0.0) ENGINE = MergeTree() ORDER BY a;
CREATE TABLE t3(a Int32, b Int32, c String) ENGINE = MergeTree() ORDER BY a;
INSERT INTO t1(a, b) VALUES (1, 'test1'), (2, 'test2'), (3, 'test3');
