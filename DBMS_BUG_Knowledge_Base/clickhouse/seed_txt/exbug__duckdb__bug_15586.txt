CREATE TABLE t1(a Int32, b String, c Date);
CREATE TABLE t2(d Float64, e UInt8);
CREATE VIEW v1 AS SELECT a, NULL AS null_date FROM t1;
SELECT * FROM t1 LEFT JOIN t2 ON t1.a = CAST(t2.d AS Int32) WHERE b IS NULL;
SELECT * FROM v1 JOIN (SELECT e FROM t2 WHERE e = 1) AS sub2 ON v1.null_date < '2023-01-01';
