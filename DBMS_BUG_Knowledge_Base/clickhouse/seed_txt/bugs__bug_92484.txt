DROP TABLE IF EXISTS demo_ts;
CREATE TABLE demo_ts (id UInt32, labels String) ENGINE = Memory;
INSERT INTO demo_ts VALUES (1, '{"http.status_code":"200"}'), (2, '{"http.status_code":"503"}'), (3, '{"http.status_code":"404"}'), (4, '{"http.status_code":"208"}');
SELECT count(*) FROM demo_ts WHERE JSONHas(labels, 'http.status_code') AND JSONExtractString(labels, 'http.status_code') > '100';
SELECT id, JSONExtractString(labels, 'http.status_code') AS sc FROM demo_ts WHERE JSONExtractString(labels, 'http.status_code') IS NOT NULL ORDER BY sc;
