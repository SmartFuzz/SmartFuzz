SET allow_experimental_json_type = 1;
CREATE TABLE span_local (links Array(JSON), dummy UInt8) ENGINE = MergeTree ORDER BY tuple();
CREATE TABLE span_link_local (linked_id String) ENGINE = MergeTree ORDER BY tuple();
CREATE MATERIALIZED VIEW mv_bad TO span_link_local AS SELECT link.`trace_id` AS linked_id FROM (SELECT arrayJoin(links) AS link FROM span_local) WHERE linked_id IS NOT NULL;
ALTER TABLE span_local DROP COLUMN dummy;
