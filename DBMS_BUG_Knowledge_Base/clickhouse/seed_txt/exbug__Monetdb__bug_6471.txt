CREATE TABLE test_series (id Int32 PRIMARY KEY, value Int32 NOT NULL);
INSERT INTO test_series (id, value) SELECT number, number FROM system.generateSeries(1, 10, 1);
CREATE TABLE test_data (data_id Int32, data_value String NOT NULL) ENGINE = MergeTree() ORDER BY data_id;
INSERT INTO test_data (data_id, data_value) SELECT number, concat('Value ', number) FROM system.generateSeries(1, 5, 1);
SELECT t.id, t.value, d.data_value FROM test_series t JOIN test_data d ON t.id = d.data_id WHERE t.value IS NOT NULL;
