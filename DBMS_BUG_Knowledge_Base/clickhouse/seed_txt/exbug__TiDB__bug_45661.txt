CREATE TABLE sales (year Int32, country String, product String, profit Int32);

SELECT year + 1 AS y, SUM(profit) AS profit FROM sales GROUP BY year + 1 WITH ROLLUP HAVING y > 2002 ORDER BY y, profit;
