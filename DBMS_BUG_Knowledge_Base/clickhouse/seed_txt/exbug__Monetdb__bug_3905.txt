CREATE TABLE t2 (f1 Int32, f2 Int32 DEFAULT 0, f3 Float32) ENGINE = MergeTree() ORDER BY f1;
INSERT INTO t1 (f1, f2) VALUES (1, 'A'), (2, 'B'), (3, 'C');
INSERT INTO t2 (f1, f2, f3) VALUES (1, 10, 1.5), (2, NULL, 2.5), (3, 20, NULL);
