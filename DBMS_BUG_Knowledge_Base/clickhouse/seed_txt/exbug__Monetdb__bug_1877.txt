CREATE TABLE table1 (col1 Int32, col2 Int32 NOT NULL, col3 String) ENGINE = MergeTree() PRIMARY KEY col1;
CREATE TABLE table2 (col1 Int32, col2 Int32 DEFAULT 0, col3 Float32) ENGINE = MergeTree() PRIMARY KEY col1;
INSERT INTO table1 (col1, col2, col3) VALUES (1, 10, 'test1'), (2, 20, 'test2');
INSERT INTO table2 (col1, col2, col3) VALUES (1, 5, 1.5), (2, NULL, 2.5);
UPDATE table1 SET col2 = (SELECT SUM(col2) FROM table2 WHERE table2.col1 = table1.col1) WHERE col1 = 1;
