CREATE TABLE T1(A UInt64, B Float64) ENGINE = MergeTree() ORDER BY A AS SELECT number, number * 1.5 FROM numbers(100000);
CREATE TABLE T2(C String, D Nullable(Int32)) ENGINE = MergeTree() ORDER BY C AS SELECT toString(number) AS C, if(number % 2 = 0, number, NULL) AS D FROM numbers(1000);
CREATE DICTIONARY item_dict2 (id UInt64, a1 UInt64, a2 Int32) PRIMARY KEY id SOURCE(CLICKHOUSE(TABLE T2 DB 'default')) LIFETIME(MIN 0 MAX 0) LAYOUT(HASHED());
SELECT DISTINCT A, dictGetUInt64('item_dict2', 'a1', toUInt64(B)) AS a1, dictGetInt32('item_dict2', 'a2', toUInt64(B)) AS a2 FROM (SELECT A, B FROM T1) FORMAT Null;
DROP DICTIONARY item_dict2;
