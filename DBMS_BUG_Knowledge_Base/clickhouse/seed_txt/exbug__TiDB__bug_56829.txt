CREATE TABLE cache_test (cache_key String NOT NULL, cache_value String NOT NULL, created_at DateTime NOT NULL DEFAULT now(), updated_at DateTime NOT NULL DEFAULT now(), expires Int32 DEFAULT -1, expired_at DateTime DEFAULT (if(expires > 0, updated_at + expires, updated_at + INTERVAL 99 YEAR)), PRIMARY KEY (cache_key)) ENGINE = MergeTree() ORDER BY cache_key;
