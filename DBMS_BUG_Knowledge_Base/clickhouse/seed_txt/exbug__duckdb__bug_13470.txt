CREATE TABLE testgroup2 (id Int32, isB2b Int32, priceGross Decimal(10, 2), priceNet Decimal(10, 2), discount Decimal(10, 2));
INSERT INTO testgroup2 VALUES (1, 1, 100.00, 80.00, 10.00), (2, 0, 150.00, 120.00, 15.00);
SELECT id, isB2b, SUM(CASE WHEN isB2b = 1 THEN priceGross - discount ELSE 0 END) AS totalGross, SUM(CASE WHEN isB2b = 0 THEN priceNet ELSE 0 END) AS totalNet FROM testgroup2 GROUP BY id, isB2b;
SELECT id, isB2b, SUM(priceGross) AS totalGross, SUM(priceNet + discount) AS totalNet FROM testgroup2 GROUP BY id, isB2b;
SELECT id, isB2b, COALESCE(SUM(CASE WHEN isB2b = 1 THEN priceGross ELSE NULL END), AVG(priceNet)) AS result FROM testgroup2 GROUP BY id, isB2b;
