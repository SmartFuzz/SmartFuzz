CREATE TABLE departments (id Int32, name String) ENGINE = MergeTree() ORDER BY id;
INSERT INTO departments VALUES (1, 'Management'), (2, 'Development'), (3, 'Sales');
CREATE TABLE employees (dep_id Int32, name String, salary Float64, hire_date Date) ENGINE = MergeTree() ORDER BY dep_id;
INSERT INTO employees VALUES (1, 'Smith', 81, '2019-08-10'), (2, 'Jones', 55, '2019-08-09'), (3, 'Williams', 55, '2019-07-14');
SELECT dep_id, name, salary, SUM(salary) OVER (ORDER BY toYear(hire_date)) AS salary_sum FROM employees;
