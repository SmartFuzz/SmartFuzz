CREATE TABLE users (id Int32, name String, age Int32, created_at DateTime DEFAULT now());
CREATE TABLE orders (order_id Int32, user_id Int32, amount Decimal(10, 2), order_date Date, PRIMARY KEY (order_id), FOREIGN KEY (user_id) REFERENCES users(id));
