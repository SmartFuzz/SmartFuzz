DROP TABLE IF EXISTS t;
CREATE TABLE t(a String, b String, c Nullable(Int32));
INSERT INTO t (a, b, c) VALUES ('a', 'A', 1), ('b', 'B', 2), ('c', 'C', NULL);
SELECT * FROM t WHERE (CASE WHEN a = 'A' THEN 1 ELSE 0 END) + (CASE WHEN b = 'A' THEN 1 ELSE 0 END) > 1;
SELECT * FROM t WHERE c IS NULL OR (CASE WHEN a = 'B' THEN 1 ELSE 0 END) + (CASE WHEN b = 'B' THEN 1 ELSE 0 END) > 1;
