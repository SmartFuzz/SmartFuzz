CREATE TABLE users (id UInt32, name String, age Nullable(UInt32), created_at DateTime DEFAULT now());
CREATE TABLE posts (id UInt32, user_id UInt32, content String, created_at DateTime DEFAULT now(), INDEX idx_user_id (user_id) TYPE minmax);
CREATE TABLE comments (id UInt32, post_id UInt32, user_id UInt32, comment String, created_at DateTime DEFAULT now(), INDEX idx_post_id (post_id) TYPE minmax, INDEX idx_user_id (user_id) TYPE minmax);

SELECT u.name, COUNT(c.id) AS comment_count FROM users u LEFT JOIN posts p ON u.id = p.user_id LEFT JOIN comments c ON p.id = c.post_id GROUP BY u.id ORDER BY comment_count DESC;
