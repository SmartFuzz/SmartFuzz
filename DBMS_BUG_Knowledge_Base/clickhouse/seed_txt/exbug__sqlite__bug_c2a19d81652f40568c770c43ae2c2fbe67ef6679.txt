CREATE TABLE fruits (id UInt32, name String, color String);
INSERT INTO fruits (name, color) VALUES ('apple', 'red'), ('banana', 'yellow'), ('grape', 'purple');
CREATE TABLE colors (id UInt32, color_name String) ENGINE = MergeTree() ORDER BY id;
INSERT INTO colors (color_name) VALUES ('red'), ('yellow'), ('green');
SELECT * FROM (SELECT name, color FROM fruits) a LEFT JOIN (SELECT color_name FROM colors) b ON a.color = b.color_name;
