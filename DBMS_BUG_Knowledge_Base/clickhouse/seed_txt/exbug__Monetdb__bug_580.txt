CREATE TABLE sys.product (product_id Int32 NOT NULL, product_name String NOT NULL, category_id Int32 NOT NULL, price Float32 NOT NULL, created_at DateTime NOT NULL, PRIMARY KEY (product_id));
CREATE TABLE sys.category (category_id Int32 NOT NULL, category_name String NOT NULL, PRIMARY KEY (category_id));

SELECT c.category_name, AVG(p.price) as avg_price, (SELECT MAX(p2.price) FROM sys.product p2 WHERE p2.category_id = c.category_id) as max_price FROM sys.category c LEFT JOIN sys.product p ON c.category_id = p.category_id GROUP BY c.category_name;
