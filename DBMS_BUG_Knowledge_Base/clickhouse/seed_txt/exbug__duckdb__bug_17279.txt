CREATE TABLE orders (OrderID UInt32, UserID UInt32, Amount Decimal(10, 2), OrderDate DateTime);

WITH latest_orders AS (SELECT UserID, Amount FROM orders WHERE Amount IS NOT NULL ORDER BY OrderDate DESC) SELECT u.Username, lo.Amount FROM users u JOIN (SELECT DISTINCT UserID, Amount FROM latest_orders) lo ON u.UserID = lo.UserID ORDER BY u.Username;
