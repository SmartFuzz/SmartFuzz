CREATE TABLE position_evaluation_v2
(
    `symbol` String,
    `evaluated_at` DateTime64(3),
    `mark_price` Decimal(18, 8),
    `volume` UInt32
)
ENGINE = MergeTree
ORDER BY (symbol, evaluated_at);
INSERT INTO position_evaluation_v2 SELECT * FROM generateRandom('symbol String, evaluated_at DateTime64(3), mark_price Decimal(18, 8), volume UInt32', 1) LIMIT 100;
SELECT lag(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price
FROM position_evaluation_v2;
SELECT lag(mark_price, 1, NULL) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price_with_default
FROM position_evaluation_v2;
SELECT lag(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC ROWS BETWEEN 1 PRECEDING AND CURRENT ROW) AS last_mark_price_invalid
FROM position_evaluation_v2;
