CREATE TABLE IF NOT EXISTS sasp.system_asset_info ON CLUSTER cluster_1st (asset String, asset_type Enum8('ip' = 1, 'domain' = 2), ip_country String, ip_province String, ip_city String, ip_district String, ip_network_type String, ip_isp_cate String, asset_source String, evidence_domain LowCardinality(String), unit_name String, update_time DateTime DEFAULT now()) ENGINE = ReplicatedReplacingMergeTree('/clickhouse/tables/{layer}-{shard}/sasp/system_asset_info', '{replica}', update_time) PARTITION BY toYYYYMM(update_time) PRIMARY KEY asset ORDER BY asset SETTINGS index_granularity = 8192;
