CREATE TABLE test_vals(grp Int32, value Int32) ENGINE = MergeTree() ORDER BY grp;
INSERT INTO test_vals VALUES (1, 100), (2, 200), (1, NULL), (2, 300), (3, -50);
CREATE FUNCTION jit_sum_variadic(value Int32) RETURNS Int64 AS (SELECT sum(value) FROM test_vals GROUP BY grp);
SELECT grp, jit_sum_variadic(value) FROM test_vals GROUP BY grp;
SELECT grp, jit_sum_variadic(value) FROM test_vals WHERE value IS NOT NULL GROUP BY grp;
