DROP DATABASE IF EXISTS my_db ON CLUSTER analytics;
CREATE DATABASE my_db ON CLUSTER analytics;
CREATE TABLE my_db.my_table (id Int64, name String, created_at DateTime DEFAULT now()) ENGINE = ReplicatedMergeTree() ORDER BY id;
CREATE VIEW my_db.my_view1 AS SELECT * FROM my_db.my_table WHERE id = {param_id:Int64};
CREATE VIEW my_db.my_view2 AS SELECT * FROM my_db.my_view1(param_id = CAST(NULL AS Int64));
