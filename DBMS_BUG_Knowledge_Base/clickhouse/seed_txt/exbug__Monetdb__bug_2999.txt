CREATE TABLE test_table(id Int32, value String, order_num Int32 NOT NULL) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_table(id, value, order_num) VALUES (1, 'first', 10), (2, 'second', 20), (3, 'third', 30);
CREATE TABLE order_table(order_id Int32, order_value Float64) ENGINE = MergeTree() ORDER BY order_id;
INSERT INTO order_table(order_id, order_value) VALUES (1, 100.50), (2, 200.75), (3, 300.25);
SELECT t.id, t.value, o.order_value FROM test_table t JOIN order_table o ON t.id = o.order_id ORDER BY o.order_value DESC;
