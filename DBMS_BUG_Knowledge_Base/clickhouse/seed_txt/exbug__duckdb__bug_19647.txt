CREATE TABLE test_table (id Int32, value String, flag Nullable(Bool));

WITH cte AS (SELECT id, value FROM test_table WHERE flag IS NOT NULL) SELECT arrayStringConcat(groupArray(value), ', ') AS aggregated_values FROM cte;
SELECT id, IF(value IS NULL, 'NULL', value) AS value_check FROM test_table;
