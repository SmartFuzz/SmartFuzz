CREATE TABLE test_table2 (id Int32, description String, is_active UInt8) ENGINE = MergeTree() ORDER BY id;
CREATE MATERIALIZED VIEW test_mv1 REFRESH EVERY 1 MINUTE APPEND TO test_table1 AS SELECT id, CAST(value AS Float64) FROM test_table2 WHERE is_active = 1;
CREATE MATERIALIZED VIEW test_mv2 REFRESH AFTER 10 MINUTE DEPENDS ON test_mv1 APPEND TO test_table2 AS SELECT id, description FROM test_table1 WHERE created_at > now() - INTERVAL 5 MINUTE;
