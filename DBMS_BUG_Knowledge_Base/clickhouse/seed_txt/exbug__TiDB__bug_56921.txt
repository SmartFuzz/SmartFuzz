DROP TABLE IF EXISTS t1;
CREATE TABLE t1(c0 Int32, c1 String, c2 Decimal(10,2));
INSERT INTO t1(c0, c1, c2) VALUES (1, 'A', 10.50), (2, 'B', NULL), (3, 'C', 20.00);
SELECT COUNT(*) FROM t1 WHERE c1 >= toString(c0) OR c1 LIKE 'A%';
