DROP TABLE IF EXISTS tree;
CREATE TABLE tree(id UInt32, parent_id Nullable(UInt32), name String, depth UInt32 DEFAULT 0);
INSERT INTO tree(parent_id, name, depth) VALUES (NULL, 'Root', 0), (1, 'Child1', 1), (1, 'Child2', 1), (2, 'Grandchild1', 2), (2, 'Grandchild2', 2);
WITH RECURSIVE mtree AS (SELECT id, name, depth FROM tree WHERE id = 1 UNION ALL SELECT t.id, t.name, t.depth FROM tree AS t INNER JOIN mtree AS m ON t.parent_id = m.id) SELECT arrayJoin(mtree) AS node, node.depth FROM mtree;
WITH RECURSIVE mtree AS (SELECT id, name FROM tree WHERE id = 1 UNION ALL SELECT t.id, t.name FROM tree AS t INNER JOIN mtree AS m ON t.parent_id = m.id) SELECT node.id, node.name FROM mtree;
