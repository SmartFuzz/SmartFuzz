CREATE TABLE c (c_id UInt32 NOT NULL, value String, UNIQUE INDEX ix_value_unique (value)) ENGINE = MergeTree() ORDER BY c_id;
CREATE TABLE d (d_id UInt32 NOT NULL, value String, INDEX ix_value (value) TYPE minmax) ENGINE = MergeTree() ORDER BY d_id;
INSERT INTO c (value) SELECT CONCAT('value_', seq) FROM numbers(1000);
INSERT INTO d (value) SELECT CONCAT('value_', seq) FROM numbers(1000000);
SELECT DISTINCT value FROM d WHERE value NOT IN (SELECT value FROM c);
