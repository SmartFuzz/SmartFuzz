CREATE TABLE books (id UInt32, title String, int_value Int32, double_value Float64, float_value Float32, decimal_value Decimal(10, 2), datetime_value DateTime);

SELECT id, title, float_value, decimal_value, datetime_value FROM books WHERE datetime_value > '2014-01-01 00:00:00';
