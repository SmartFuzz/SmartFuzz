CREATE TABLE test (a Int32, b String, c Float32) ENGINE = MergeTree() ORDER BY a;
INSERT INTO test (a, b, c) VALUES (1, 'hello', 3.5), (2, 'world', 4.0), (1, NULL, 2.5);
SELECT a, b, length(b) AS b_length, count(*) FROM test GROUP BY a, b HAVING count(a) = length(b);
SELECT a, b, length(b) AS b_length, count(*) FROM test GROUP BY a, b HAVING count(a) = length('test');
SELECT a, b, length(b) AS b_length, count(*) FROM test WHERE c IS NOT NULL GROUP BY a, b HAVING count(a) = 1;
