DROP DATABASE IF EXISTS test_db ON CLUSTER analytics;
CREATE DATABASE test_db ON CLUSTER analytics;
CREATE TABLE test_db.test_table ON CLUSTER analytics (id Int64, name String, created_at DateTime) ENGINE = MergeTree() ORDER BY id;
CREATE VIEW test_db.test_view1 ON CLUSTER analytics AS SELECT id, name FROM test_db.test_table WHERE created_at > now() - INTERVAL 1 DAY;
SELECT * FROM test_db.test_view1 WHERE id = CAST(NULL AS Int64);
