DROP TABLE IF EXISTS b_products;
CREATE TABLE b_products (product_id Int64, product_name String, price Float64) ENGINE = MergeTree ORDER BY product_id;
INSERT INTO b_products SELECT number, concat('Product_', number), number * 10.5 FROM system.numbers LIMIT 100;
DROP TABLE IF EXISTS b_sales;
CREATE TABLE b_sales (sale_id Int64, product_id Int64, quantity Int32) ENGINE = MergeTree ORDER BY sale_id;
