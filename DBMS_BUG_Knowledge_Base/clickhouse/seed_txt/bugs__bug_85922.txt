CREATE TABLE IF NOT EXISTS {db:Identifier}.users (id UInt64, name String, created_at DateTime) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE IF NOT EXISTS {db:Identifier}.user_logs (user_id UInt64, log_message String, log_time DateTime) ENGINE = MergeTree() ORDER BY log_time;
