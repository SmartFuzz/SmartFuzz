CREATE TABLE fig (id Int32 NOT NULL, varchar_1 String, num_col Int32 DEFAULT 0) ENGINE = MergeTree() ORDER BY id;
INSERT INTO fig (id, varchar_1, num_col) VALUES (1, 'test', NULL), (2, 'example', 5), (3, NULL, 10);
CREATE VIEW v_substr AS SELECT substr(varchar_1, if(-(4) = 0, 1, -(4)), 4) AS SUBSTR_varchar_1 FROM fig WHERE num_col IS NOT NULL;
CREATE VIEW v_join AS SELECT f.id, v.SUBSTR_varchar_1 FROM fig f JOIN v_substr v ON f.id = 1;
SELECT * FROM v_join;
