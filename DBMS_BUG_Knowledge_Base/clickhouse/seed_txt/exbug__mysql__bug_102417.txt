DROP TABLE IF EXISTS ts;
CREATE TABLE ts (id Int32, purchased Date, amount Decimal(10, 2) DEFAULT 0.00) ENGINE = MergeTree() PARTITION BY toYYYYMM(purchased) ORDER BY id;
