CREATE TABLE organisations (id String, name String, acronym_id String, site_id String) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE acronyms (id String, short_form String) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE sites (id String, nuts_code String, geographical_location String) ENGINE = MergeTree() ORDER BY id;
