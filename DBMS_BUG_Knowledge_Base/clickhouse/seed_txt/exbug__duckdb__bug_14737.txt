CREATE TABLE test_table (id UInt32, value Int32) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_table (id, value) VALUES (1, 10), (2, 20), (3, 10), (4, 30);
SELECT arrayDistinct(arraySort(groupArray(value))) AS distinct_values FROM test_table;
SELECT arrayDistinct(arraySort(groupArray(value % 3))) AS mod_values FROM test_table;
SELECT arrayDistinct(arraySort(groupArray(CASE WHEN value > 15 THEN 'high' ELSE 'low' END))) AS categories FROM test_table;
