CREATE TABLE test_strings (id UInt32, name String) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_strings SELECT number, repeat('abc', 1000) FROM numbers(100);
CREATE TABLE test_numbers (num UInt32, name String) ENGINE = MergeTree() ORDER BY num;
INSERT INTO test_numbers SELECT number, repeat('xyz', 200) FROM numbers(50);
SELECT t1.id, t2.name FROM test_strings t1 JOIN test_numbers t2 ON t1.id = t2.num WHERE t1.name IS NOT NULL;
