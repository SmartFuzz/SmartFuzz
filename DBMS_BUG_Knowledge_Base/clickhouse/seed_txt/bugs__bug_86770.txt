CREATE TABLE releases (
    version REAL PRIMARY KEY COMMENT 'Version of the registry.'
) ENGINE = MergeTree;
CREATE TABLE releases_v2 (
    version FLOAT COMMENT 'Version of the registry.'
) ENGINE = MergeTree PRIMARY KEY(version);
CREATE TABLE releases_v3 (
    id UInt32,
    version REAL,
    description String COMMENT 'Description of the release.'
) ENGINE = MergeTree PRIMARY KEY(id);
