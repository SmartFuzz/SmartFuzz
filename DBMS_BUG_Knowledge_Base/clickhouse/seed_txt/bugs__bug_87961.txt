CREATE TABLE IF NOT EXISTS user_activity ON CLUSTER 'default' (activity_id UInt64, user_id UInt32, activity_time DateTime, activity_type Enum('login' = 1, 'logout' = 2, 'purchase' = 3), details String) ENGINE = ReplicatedMergeTree('/clickhouse/tables/{shard}/{table}', '{replica}') PARTITION BY toYYYYMM(activity_time) ORDER BY (user_id, activity_time);
INSERT INTO user_activity SELECT number AS activity_id, rand() % 1000 AS user_id, now() + number * 5 AS activity_time, (CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, concat('Activity details for ', toString(number)) AS details FROM numbers(100);
CREATE TABLE IF NOT EXISTS product_sales ON CLUSTER 'default' (sale_id UInt32, product_id UInt32, sale_time DateTime, quantity UInt16, price Float32) ENGINE = ReplicatedMergeTree('/clickhouse/tables/{shard}/{table}', '{replica}') PARTITION BY toYYYYMM(sale_time) ORDER BY (product_id, sale_time);
INSERT INTO product_sales SELECT number AS sale_id, rand() % 500 AS product_id, now() + number * 2 AS sale_time, (rand() % 10 + 1) AS quantity, (rand() % 100 + 1) * 1.5 AS price FROM numbers(50);
SELECT ua.user_id, count(*) AS activity_count, sum(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= now() - INTERVAL 1 DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
