CREATE TABLE t1(x Int32, y Int32) ENGINE = MergeTree() ORDER BY x;
CREATE TABLE t2(a Int32, b Int32 DEFAULT 10) ENGINE = MergeTree() ORDER BY a;
INSERT INTO t1(x, y) VALUES (1, 2), (3, 4), (5, NULL);
INSERT INTO t2(a, b) VALUES (0, 5), (5, 15), (NULL, 20);
SELECT * FROM t1 CROSS JOIN t2 WHERE (x BETWEEN a AND b OR y BETWEEN a AND b OR a BETWEEN x AND y);
