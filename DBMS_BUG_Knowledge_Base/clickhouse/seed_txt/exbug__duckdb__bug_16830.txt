CREATE TABLE test_regex (id UInt32, pattern String, test_string String);
INSERT INTO test_regex (id, pattern, test_string) VALUES (1, 'bc', 'abcd'), (2, 'a.c', 'abcd'), (3, 'a.*d', 'abcd'), (4, '(b|x)', 'abcd'), (5, '^a', 'abcd'), (6, '^(b|c)', 'abcd');
SELECT id, match(test_string, pattern) AS match_result FROM test_regex;
SELECT id, match(test_string, 'a.*d') AS match_result FROM test_regex WHERE pattern = 'a.*d';
SELECT id, match(test_string, '^(b|c)') AS match_result FROM test_regex WHERE test_string IS NOT NULL;
