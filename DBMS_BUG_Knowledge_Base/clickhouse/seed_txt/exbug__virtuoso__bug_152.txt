CREATE TABLE test_graph (id Int32, name String, cost Nullable(Int32)) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_graph (id, name, cost) VALUES (1, 'Concept A', NULL), (2, 'Concept A1', 1), (3, 'Concept A2', 20), (4, 'Concept A3', 7), (5, 'Concept A5', 1);
SELECT id, name, COALESCE(cost, 0) AS cost FROM test_graph WHERE cost IS NOT NULL;
SELECT SUM(cost) AS total_cost FROM test_graph WHERE cost IS NOT NULL GROUP BY name;
SELECT id, name, IF(cost IS NULL, 'No Cost', 'Has Cost') AS cost_status FROM test_graph;
