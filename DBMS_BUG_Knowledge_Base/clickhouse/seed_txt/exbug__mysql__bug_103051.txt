CREATE TABLE customer (c_id Int32, c_name String, c_balance Decimal(10, 2)) ENGINE = MergeTree() ORDER BY c_id;
CREATE TABLE orders (o_id Int32, o_customer_id Int32, o_amount Decimal(10, 2)) ENGINE = MergeTree() ORDER BY o_id;

SELECT c.c_name, COALESCE(SUM(o.o_amount), 0) AS total_spent FROM customer c LEFT JOIN orders o ON c.c_id = o.o_customer_id GROUP BY c.c_id;
