CREATE TABLE automobiles (id UInt32, manufacturer String, thumbnail String, depiction String, image String, logo String);

SELECT id, COALESCE(thumbnail, depiction, image, logo) AS image_url FROM automobiles WHERE manufacturer IS NOT NULL;
SELECT id, countIf(thumbnail IS NOT NULL OR depiction IS NOT NULL OR image IS NOT NULL OR logo IS NOT NULL) AS optional_count FROM automobiles GROUP BY id HAVING optional_count > 0;
