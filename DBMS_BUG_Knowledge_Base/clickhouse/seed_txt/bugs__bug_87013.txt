DROP DATABASE IF EXISTS test SYNC;
CREATE DATABASE test ENGINE=Atomic;
CREATE TABLE IF NOT EXISTS test.mv_src_table (id UInt64, value String) ENGINE=MergeTree() ORDER BY id;
INSERT INTO test.mv_src_table SELECT number, toString(number) FROM numbers(10);
CREATE MATERIALIZED VIEW IF NOT EXISTS test.mv_refreshable REFRESH EVERY 1 MINUTE TO test.mv_refreshable_dst_table EMPTY AS SELECT max(id) AS id FROM test.mv_src_table WHERE value IS NOT NULL;
