CREATE TABLE scene (id Int32, victim_id Int32 NOT NULL, description String) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE victim (id Int32, name String, age Int32) ENGINE = MergeTree() ORDER BY id;
CREATE TABLE inspector (id Int32, name String, experience_years Int32 DEFAULT 0) ENGINE = MergeTree() ORDER BY id;
