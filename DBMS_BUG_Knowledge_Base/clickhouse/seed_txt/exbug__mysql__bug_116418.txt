DROP TABLE IF EXISTS t1;
CREATE TABLE t1 (id UInt32, pad1 String NOT NULL, pad2 String NOT NULL, pad3 String NOT NULL, pad4 String NOT NULL) ENGINE = MergeTree() ORDER BY id;
