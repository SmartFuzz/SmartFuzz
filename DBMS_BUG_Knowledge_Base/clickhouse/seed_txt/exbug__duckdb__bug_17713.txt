CREATE TABLE test_floats (id UInt32, value Float64);
INSERT INTO test_floats (id, value) VALUES (1, 0.99), (2, 1.01), (3, NULL);
SELECT IF(value IS NULL, 'NULL', toString(value)) AS float_check FROM test_floats;
SELECT IF(value < 1.0, 'Less than one', 'One or more') AS float_category FROM test_floats;
SELECT AVG(value) AS average_value FROM test_floats WHERE value IS NOT NULL;
