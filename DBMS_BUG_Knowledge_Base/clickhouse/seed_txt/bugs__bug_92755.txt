DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant String, event_time DateTime, id UInt64, msg String, value Nullable(Float64)) ENGINE = MergeTree PARTITION BY toYYYYMMDD(event_time) ORDER BY (tenant, event_time, id);
INSERT INTO repro.events SELECT toString(number % 100) AS tenant, now() - (number % 200000) AS event_time, number AS id, concat('m', toString(number)) AS msg, CAST(number % 1000 AS Nullable(Float64)) AS value FROM numbers(2000000);
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
