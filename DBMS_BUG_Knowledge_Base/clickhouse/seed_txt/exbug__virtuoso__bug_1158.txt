CREATE TABLE observation_collection (id UInt32, phenomenon_time DateTime, result_time DateTime, has_member String);
INSERT INTO observation_collection (phenomenon_time, result_time, has_member) VALUES (now(), now(), 'member1'), (now() - INTERVAL 1 DAY, now() - INTERVAL 1 HOUR, 'member2'), (now() - INTERVAL 2 DAY, now() - INTERVAL 2 HOUR, 'member3');
SELECT id, MAX(phenomenon_time) AS max_phenomenon_time, MAX(result_time) AS max_result_time FROM observation_collection GROUP BY id ORDER BY max_phenomenon_time LIMIT 100 OFFSET 0;
SELECT id, MAX(phenomenon_time) AS max_phenomenon_time FROM (SELECT id, phenomenon_time FROM observation_collection WHERE has_member IS NOT NULL) AS subquery GROUP BY id ORDER BY max_phenomenon_time LIMIT 100 OFFSET 20000;
SELECT DISTINCT id FROM observation_collection WHERE phenomenon_time IS NOT NULL ORDER BY id LIMIT 100 OFFSET 14000;
