CREATE TABLE fts (rowid UInt64, content String) ENGINE = MergeTree() ORDER BY rowid;
CREATE TABLE t1 (x String, y Int32) ENGINE = MergeTree() ORDER BY x;
INSERT INTO t1 (x, y) VALUES ('hello world', 1);
INSERT INTO fts (rowid, content) VALUES (1, 'hello world');
SELECT DISTINCT fts.rowid, t1.x FROM fts JOIN t1 ON fts.content = t1.x WHERE fts.content = 'hello world' ORDER BY t1.y;
