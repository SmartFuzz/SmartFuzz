DROP TABLE IF EXISTS t1, t2;
CREATE TABLE t1(id Int32, col1 String DEFAULT NULL, col2 String DEFAULT NULL, col3 String DEFAULT NULL);
CREATE TABLE t2(id Int32, col1 String NOT NULL, col2 String NOT NULL, col3 String NOT NULL);
INSERT INTO t1 VALUES(1, NULL, NULL, NULL), (2, NULL, NULL, NULL), (3, NULL, NULL, NULL);
INSERT INTO t2 VALUES(1, 'a', 'aa', 'aaa'), (2, 'b', 'bb', 'bbb'), (3, 'c', 'cc', 'ccc');
UPDATE t1 SET col1 = (SELECT col1 FROM t2 WHERE t2.id = t1.id), col2 = (SELECT col2 FROM t2 WHERE t2.id = t1.id), col3 = (SELECT col3 FROM t2 WHERE t2.id = t1.id) WHERE t1.id IN (SELECT id FROM t2 WHERE id > 1);
