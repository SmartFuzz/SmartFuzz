CREATE TABLE t1 (c0 QBit(Float64, 5), c1 Int32, c2 String) ENGINE = MergeTree() ORDER BY c1;
INSERT INTO t1 (c0, c1, c2) SELECT [CAST(number % 10 AS Float64), CAST((-number) AS Float64)], number, 'test' FROM numbers(20);
CREATE TABLE t2 (c0 Int32, c1 Nullable(Float32), c2 DateTime) ENGINE = MergeTree() ORDER BY c0;
INSERT INTO t2 (c0, c1, c2) SELECT number, CAST(NULL AS Float32), now() FROM numbers(10);
SELECT c0, c1 FROM t1 JOIN t2 ON t1.c1 = t2.c0 ORDER BY c0, c1;
