CREATE TABLE bigaggr (a1 Int32, a2 Int32, a3 Int32 DEFAULT 0);
INSERT INTO bigaggr (a1, a2, a3) VALUES (1, 2, NULL), (3, 4, 5), (NULL, NULL, 6);
CREATE TABLE bigaggr_agg AS SELECT a1, SUM(a2) AS total_a2 FROM bigaggr GROUP BY a1;
SELECT COUNT(*) FROM bigaggr_agg WHERE total_a2 IS NOT NULL;
CREATE TABLE bigaggr_join AS SELECT b.a1, b.a2 FROM bigaggr b JOIN bigaggr_agg a ON b.a1 = a.a1;
