CREATE TABLE EMPLOYEES (EMPNUM Int32, NAME String, SALARY Decimal(10, 2));
CREATE TABLE DEPARTMENTS (DEPTID Int32, DEPTNAME String);
CREATE VIEW EMP_DEPT AS SELECT E.EMPNUM, E.NAME, D.DEPTNAME FROM EMPLOYEES E LEFT JOIN DEPARTMENTS D ON E.EMPNUM = D.DEPTID;
