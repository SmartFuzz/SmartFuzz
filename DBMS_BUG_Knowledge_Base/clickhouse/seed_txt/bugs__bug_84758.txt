CREATE TABLE genai_eva_trace.otel_traces_main_for_sampling_extended
(
    `trace_id` String,
    `feature` String,
    `account_id` String,
    `extension_id` String,
    `start_timestamp` DateTime64(9),
    `status` Nullable(String)
)
ENGINE = MergeTree
PARTITION BY toDate(start_timestamp)
ORDER BY (feature,
 account_id,
 extension_id,
 start_timestamp,
 cityHash64(trace_id))
SAMPLE BY cityHash64(trace_id)
TTL toDate(start_timestamp) + toIntervalDay(7)
SETTINGS index_granularity = 8192;
