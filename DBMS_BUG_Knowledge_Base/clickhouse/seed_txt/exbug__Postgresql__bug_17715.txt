CREATE FUNCTION cube(p Int32) RETURNS Int32 AS (p * p * p);
CREATE FUNCTION myfunc2(p Int32) RETURNS Int32 AS (cube(p) + 2);
CREATE TABLE numbers (value Int32 NOT NULL) ENGINE = MergeTree() ORDER BY value;
INSERT INTO numbers VALUES (5), (10), (15);
CREATE MATERIALIZED VIEW numview ENGINE = MergeTree() AS SELECT myfunc2(value) FROM numbers;
