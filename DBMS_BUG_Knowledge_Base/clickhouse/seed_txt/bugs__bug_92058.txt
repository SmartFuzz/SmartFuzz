CREATE TABLE dst (e Enum('a' = 0, 'b' = 1), f Int32) ENGINE = Memory;
CREATE TABLE src (e Enum('a' = 0), g Nullable(Int32)) ENGINE = Memory;
CREATE MATERIALIZED VIEW mv TO dst (e Enum('a' = 0, 'b' = 1), f Int32) AS SELECT e, g FROM src;
INSERT INTO src SELECT toEnum8(number % 1) AS e, number % 10 AS g FROM generateRandom() LIMIT 10;
INSERT INTO dst SELECT e, g FROM mv WHERE g IS NOT NULL;
