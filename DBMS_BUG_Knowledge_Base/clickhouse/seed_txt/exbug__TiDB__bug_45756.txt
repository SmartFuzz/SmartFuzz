CREATE TABLE sales (year Int32 NOT NULL, country String NOT NULL, product String NOT NULL, profit Int32 DEFAULT 0) ENGINE = MergeTree() ORDER BY (year, country, product);
INSERT INTO sales (year, country, product, profit) VALUES (2020, 'USA', 'Gadget', 1500), (2020, 'USA', 'Widget', 2000), (2021, 'Canada', 'Gadget', 2500);
SELECT year, country, product, (year IS NOT NULL) + (country IS NOT NULL) + (product IS NOT NULL) AS grouping FROM sales GROUP BY year, country, product WITH ROLLUP HAVING grouping = 0;
CREATE TABLE products (product_id Int32 NOT NULL, product_name String NOT NULL, PRIMARY KEY (product_id)) ENGINE = MergeTree() ORDER BY product_id;
