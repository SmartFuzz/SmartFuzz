DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id UInt32, name String, value Float64, created_at DateTime) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_table SELECT number, concat('name_', toString(number), ' ', toString(number % 10) ) AS name, number * 1.5 AS value, now() FROM numbers(100);
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10 SETTINGS max_bytes_before_external_sort = 100, max_bytes_ratio_before_external_sort = 0.0;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
