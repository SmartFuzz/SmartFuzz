CREATE TABLE employees (emp_no Int32, gender String, hire_date Date) ENGINE = MergeTree() ORDER BY emp_no;
CREATE TABLE salaries (emp_no Int32, salary Decimal(10, 2)) ENGINE = MergeTree() ORDER BY emp_no;
CREATE TABLE dept_manager (emp_no Int32, dept_no String) ENGINE = MergeTree() ORDER BY emp_no;
