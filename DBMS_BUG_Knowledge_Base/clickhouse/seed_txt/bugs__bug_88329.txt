CREATE TABLE test2.raw_source_local (id Int32, time DateTime, value Float64 DEFAULT 0.0) ENGINE=ReplicatedMergeTree() PRIMARY KEY id ORDER BY id;
INSERT INTO test2.raw_source_local (id, time, value) SELECT number AS id, toDateTime('2024-01-01 00:00:00') + INTERVAL number SECOND AS time, rand() * 100 AS value FROM numbers(1, 1000);
CREATE TABLE test2.table_for_mv (sum_value Float64, date Date, count_id UInt32) ENGINE=ReplicatedMergeTree() PRIMARY KEY date ORDER BY date;
CREATE MATERIALIZED VIEW test2.mv_from_raw_source_local_to_table REFRESH EVERY 1 MINUTE TO test2.table_for_mv AS SELECT sum(value) AS sum_value, toDate(time) AS date, count() AS count_id FROM test2.raw_source_local GROUP BY toDate(time);
