CREATE TABLE users (id UInt32, name String, age Nullable(UInt32), created_at DateTime DEFAULT now());
CREATE TABLE orders (order_id UInt32, user_id UInt32, amount Decimal(10, 2), order_date Date, FOREIGN KEY (user_id) REFERENCES users(id));
