CREATE TABLE users (id Int32 PRIMARY KEY, name String, balance Decimal(10, 2) DEFAULT 0.00);
CREATE TABLE transactions (transaction_id Int32 DEFAULT 0, user_id Int32, amount Decimal(10, 2), created_at DateTime DEFAULT now(), PRIMARY KEY (transaction_id));
