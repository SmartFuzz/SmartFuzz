CREATE TABLE nyc_taxi.trips_large (
    trip_id             UInt32,
    pickup_datetime     DateTime,
    dropoff_datetime    DateTime,
    pickup_longitude    Nullable(Float64),
    pickup_latitude     Nullable(Float64),
    dropoff_longitude   Nullable(Float64),
    dropoff_latitude    Nullable(Float64),
    passenger_count     UInt8,
    trip_distance       Float32,
    fare_amount         Float32,
    extra               Float32,
    tip_amount          Float32,
    tolls_amount        Float32,
    total_amount        Float32,
    payment_type        Enum('CSH' = 1, 'CRE' = 2, 'NOC' = 3, 'DIS' = 4, 'UNK' = 5),
    pickup_ntaname      LowCardinality(String),
    dropoff_ntaname     LowCardinality(String)
)
ENGINE = MergeTree
PRIMARY KEY (pickup_datetime, dropoff_datetime);
INSERT INTO nyc_taxi.trips_large
SELECT
    trip_id,
    pickup_datetime,
    dropoff_datetime,
    pickup_longitude,
    pickup_latitude,
    dropoff_longitude,
    dropoff_latitude,
    passenger_count,
    trip_distance,
    fare_amount,
    extra,
    tip_amount,
    tolls_amount,
    total_amount,
    payment_type,
    pickup_ntaname,
    dropoff_ntaname
FROM gcs(
    'https://storage.googleapis.com/clickhouse-public-datasets/nyc-taxi/trips_{0..2}.gz',
    'TabSeparatedWithNames'
);
CREATE TABLE nyc_taxi.agg_h3_10_large (
    pickup_latitude Float64,
    pickup_longitude Float64,
    h3_cell UInt64,
    cnt AggregateFunction(count, String),
    pickup_ntaname_uniq AggregateFunction(uniq, String),
    pickup_ntaname_top_app AggregateFunction(approx_top_k(1000), String)
)
ENGINE = AggregatingMergeTree
ORDER BY (pickup_latitude, pickup_longitude);
INSERT INTO nyc_taxi.agg_h3_10_large
SELECT
    pickup_latitude,
    pickup_longitude,
    h3_cell,
    count() AS cnt,
    uniq(pickup_ntaname) AS pickup_ntaname_uniq,
    approx_top_k(pickup_ntaname, 1000) AS pickup_ntaname_top_app
FROM nyc_taxi.trips_large
GROUP BY pickup_latitude, pickup_longitude, h3_cell;
SELECT
    pickup_latitude,
    pickup_longitude,
    cnt
FROM nyc_taxi.agg_h3_10_large
WHERE cnt > 1000;
