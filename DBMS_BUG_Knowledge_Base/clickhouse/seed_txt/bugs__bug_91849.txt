DROP TABLE IF EXISTS test_special_columns;
CREATE TABLE test_special_columns(key Int64, value String, timestamp DateTime, is_active UInt8) ENGINE = ReplacingMergeTree(timestamp) ORDER BY key;
INSERT INTO test_special_columns VALUES (1, 'active', '2023-01-01 00:00:00', 1), (2, 'inactive', '2023-01-02 00:00:00', 0), (3, 'active', '2023-01-03 00:00:00', 1);
SELECT key, value FROM test_special_columns FINAL PREWHERE is_active = 1 AND timestamp > '2023-01-01 00:00:00' ORDER BY key;
SELECT count() FROM test_special_columns FINAL PREWHERE is_active = 0;
