CREATE TABLE test(id INTEGER, name TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test(id, name) SELECT ROW_NUMBER() OVER(), 'Record ' || ROW_NUMBER() OVER() FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) LIMIT 20000000;
DELETE FROM test WHERE id = 1;
DELETE FROM test WHERE id = (SELECT MAX(id) FROM test);
SELECT COUNT(*) FROM test WHERE created_at IS NOT NULL;
