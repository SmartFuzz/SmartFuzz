DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id INTEGER, name TEXT, value REAL, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table SELECT number, 'name_' || number || ' ' || (number % 10) AS name, number * 1.5 AS value, datetime('now') FROM (SELECT rowid AS number FROM sqlite_sequence LIMIT 100);
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
