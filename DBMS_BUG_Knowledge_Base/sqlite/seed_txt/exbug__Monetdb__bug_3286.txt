CREATE TABLE storage_summary (tblname TEXT, count INTEGER, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO storage_summary (tblname, count) SELECT "table", MAX("count") FROM sys.storagemodel() WHERE "schema" = 'sys' GROUP BY "table";
CREATE TABLE storage_details (tblname TEXT NOT NULL, count INTEGER CHECK (count >= 0), description TEXT);
INSERT INTO storage_details (tblname, count, description) SELECT "table", CAST(MAX("count") AS INTEGER), 'Storage details for ' || "table" FROM sys.storagemodel() WHERE "schema" = 'sys' GROUP BY "table";
SELECT tblname, count FROM storage_summary WHERE count IS NOT NULL ORDER BY count DESC;
