CREATE TABLE product (product_id INTEGER NOT NULL PRIMARY KEY, product_name TEXT NOT NULL, category_id INTEGER NOT NULL, price REAL NOT NULL, created_at DATETIME NOT NULL);
CREATE TABLE category (category_id INTEGER NOT NULL PRIMARY KEY, category_name TEXT NOT NULL);

SELECT c.category_name, AVG(p.price) as avg_price, (SELECT MAX(p2.price) FROM product p2 WHERE p2.category_id = c.category_id) as max_price FROM category c LEFT JOIN product p ON c.category_id = p.category_id GROUP BY c.category_name;
