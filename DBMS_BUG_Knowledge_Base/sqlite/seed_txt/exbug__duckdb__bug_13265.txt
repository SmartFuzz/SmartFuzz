CREATE TABLE users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (order_id INTEGER PRIMARY KEY, user_id INTEGER, amount REAL, order_date DATE, FOREIGN KEY (user_id) REFERENCES users(id));
