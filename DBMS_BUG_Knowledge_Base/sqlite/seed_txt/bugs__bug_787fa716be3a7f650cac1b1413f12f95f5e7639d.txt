CREATE TABLE musicians (id INTEGER PRIMARY KEY, name TEXT NOT NULL, genre TEXT);
CREATE TABLE songs (id INTEGER PRIMARY KEY, title TEXT NOT NULL, musician_id INTEGER REFERENCES musicians(id));
