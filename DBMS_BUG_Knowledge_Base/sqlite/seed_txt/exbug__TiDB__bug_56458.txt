CREATE TABLE authors (id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, email TEXT UNIQUE NOT NULL);
CREATE TABLE books (id INTEGER PRIMARY KEY AUTOINCREMENT, title TEXT NOT NULL, publication_date TEXT NOT NULL, author_id INTEGER, FOREIGN KEY (author_id) REFERENCES authors(id) ON DELETE CASCADE);
