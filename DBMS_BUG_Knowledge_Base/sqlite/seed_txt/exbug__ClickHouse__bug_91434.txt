CREATE TABLE json_test_v2 (json TEXT, col_b INTEGER, col_c TEXT);
INSERT INTO json_test_v2 (json, col_b, col_c) SELECT json_object('a', a, 'b', b, 'c', c) AS json, number * 2 AS col_b, CAST(number * 3 AS TEXT) AS col_c FROM (SELECT rowid AS number FROM sqlite_sequence LIMIT 10000000);
SELECT col_b, col_c FROM json_test_v2 WHERE col_b IS NOT NULL;
SELECT json_extract(json, '$.b') AS direct_b, json_extract(json, '$.c') AS direct_c FROM json_test_v2 WHERE json_extract(json, '$.b') > 5000000;
SELECT col_b, COUNT(*) FROM json_test_v2 GROUP BY col_b HAVING col_b < 100;
