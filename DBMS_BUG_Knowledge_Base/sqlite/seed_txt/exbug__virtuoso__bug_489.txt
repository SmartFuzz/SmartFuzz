CREATE TABLE test_graph (id INTEGER PRIMARY KEY, graph_uri TEXT, annotated_at TEXT);
CREATE TABLE test_address (id INTEGER PRIMARY KEY, address_uri TEXT, target_graph INTEGER, address_timestamp TEXT, FOREIGN KEY (target_graph) REFERENCES test_graph(id));

SELECT g.graph_uri, g.annotated_at, COUNT(a.address_timestamp) AS addresses_count, MAX(a.address_timestamp) AS most_recent_timestamp FROM test_graph g LEFT JOIN test_address a ON g.id = a.target_graph GROUP BY g.id;
