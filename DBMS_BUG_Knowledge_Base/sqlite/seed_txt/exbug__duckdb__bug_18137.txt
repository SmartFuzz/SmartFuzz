CREATE TABLE test_large_lists (id INTEGER, gts TEXT);
INSERT INTO test_large_lists SELECT i, '[' || GROUP_CONCAT('[' || CAST(random() % 256 AS INTEGER) || ',' || CAST(random() % 256 AS INTEGER) || ']' ) || ']' FROM (SELECT ROW_NUMBER() OVER() AS i FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS x LIMIT 500000);
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, json_each.value AS inner_list FROM list_view, json_each(gts);
