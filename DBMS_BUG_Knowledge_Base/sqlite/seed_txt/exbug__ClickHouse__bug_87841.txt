CREATE TABLE table_src (order_id INTEGER, random_date DATE, amount REAL);
CREATE TABLE table_dst (order_id INTEGER, random_date DATE, amount REAL);
INSERT INTO table_src SELECT rowid AS order_id, DATE('now', '-' || (ABS(RANDOM()) % 3650) || ' days') AS random_date, ABS(RANDOM() % 10000) / 100.0 AS amount FROM (SELECT rowid FROM sqlite_master LIMIT 1000000);
INSERT INTO table_dst SELECT order_id, random_date, amount FROM table_src WHERE amount > 50.0;
INSERT INTO table_dst SELECT order_id, random_date, amount FROM table_src ORDER BY strftime('%Y-%m', random_date);
