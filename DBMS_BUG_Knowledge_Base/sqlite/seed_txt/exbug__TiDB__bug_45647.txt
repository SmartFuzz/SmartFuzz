CREATE TABLE products (id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, price REAL NOT NULL, quantity INTEGER DEFAULT 0);

CREATE TABLE sales (sale_id INTEGER PRIMARY KEY AUTOINCREMENT, product_id INTEGER NOT NULL, sale_date TEXT NOT NULL, amount REAL NOT NULL, FOREIGN KEY (product_id) REFERENCES products(id));

SELECT strftime('%Y', sale_date) + 1 AS sale_year, SUM(amount) AS total_sales FROM sales GROUP BY sale_year HAVING total_sales > 2000 ORDER BY sale_year;
