DROP TABLE IF EXISTS test_lwu;
CREATE TABLE test_lwu (a INTEGER, b TEXT, c REAL DEFAULT 0.0);

DELETE FROM test_lwu WHERE (ABS(RANDOM() % 10)) = 0;
