CREATE TABLE IF NOT EXISTS user_data (user_id INTEGER PRIMARY KEY, username TEXT NOT NULL, age INTEGER CHECK (age >= 0));
CREATE UNIQUE INDEX IF NOT EXISTS idx_username ON user_data(upper(username));
