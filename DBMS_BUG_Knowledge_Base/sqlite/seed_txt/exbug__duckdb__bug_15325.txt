CREATE TABLE diverse_table (id INTEGER, name TEXT, value REAL, created_at DATETIME DEFAULT CURRENT_TIMESTAMP, is_active BOOLEAN);
INSERT INTO diverse_table SELECT i, 'Name_' || i, RANDOM() % 100, CURRENT_TIMESTAMP, (i % 2 = 0) FROM (SELECT ROW_NUMBER() OVER() AS i FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS a, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS b, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS c LIMIT 1000);
SELECT id, name, value, created_at, is_active FROM diverse_table WHERE is_active IS NOT NULL;
SELECT COUNT(*) AS total_count, SUM(CASE WHEN value IS NULL THEN 1 ELSE 0 END) AS null_value_count FROM diverse_table;
