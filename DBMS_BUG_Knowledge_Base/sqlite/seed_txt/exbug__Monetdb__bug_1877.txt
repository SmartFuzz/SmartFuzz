CREATE TABLE table1 (col1 INTEGER PRIMARY KEY, col2 INTEGER NOT NULL, col3 TEXT);
CREATE TABLE table2 (col1 INTEGER PRIMARY KEY, col2 INTEGER DEFAULT 0, col3 REAL);
INSERT INTO table1 (col1, col2, col3) VALUES (1, 10, 'test1'), (2, 20, 'test2');
INSERT INTO table2 (col1, col2, col3) VALUES (1, 5, 1.5), (2, NULL, 2.5);
UPDATE table1 SET col2 = (SELECT COALESCE(SUM(col2), 0) FROM table2 WHERE table2.col1 = table1.col1) WHERE col1 = 1;
