CREATE TABLE kvk (kvk INTEGER PRIMARY KEY, bedrijfsnaam TEXT NOT NULL, adres TEXT, postcode TEXT, plaats TEXT, type TEXT CHECK (type IN ('Type1', 'Type2')), kvks INTEGER, sub INTEGER);
CREATE TABLE concernrelaties (kvk INTEGER, ouder TEXT, postcode TEXT, plaats TEXT, ouderkvk INTEGER, FOREIGN KEY (kvk) REFERENCES kvk (kvk));
