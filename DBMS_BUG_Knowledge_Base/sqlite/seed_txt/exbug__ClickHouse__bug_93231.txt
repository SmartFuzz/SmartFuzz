DROP TABLE IF EXISTS span_local;
CREATE TABLE span_local (links TEXT, dummy INTEGER);
DROP TABLE IF EXISTS span_link_local;
CREATE TABLE span_link_local (linked_id TEXT);
CREATE VIEW mv_bad AS SELECT json_extract(link, '$.trace_id') AS linked_id FROM (SELECT json_each.value AS link FROM span_local, json_each(links)) WHERE json_extract(link, '$.trace_id') IS NOT NULL;
ALTER TABLE span_local DROP COLUMN dummy;
