CREATE TABLE t2 (
id INTEGER PRIMARY KEY AUTOINCREMENT,
custom_prefix TEXT NOT NULL,
created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
CHECK (custom_prefix GLOB '[a-z0-9-]*')
);
INSERT INTO t2 (custom_prefix) VALUES ('valid-prefix'), ('invalid_prefix!');
SELECT * FROM t2 WHERE custom_prefix IS NOT NULL;
UPDATE t2 SET custom_prefix = NULL WHERE id = 1;
CREATE TEMP TABLE t2_backup AS SELECT * FROM t2;
