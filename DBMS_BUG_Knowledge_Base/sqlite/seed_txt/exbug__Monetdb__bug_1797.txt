CREATE TABLE EMPLOYEES (EMPNUM INTEGER PRIMARY KEY, NAME TEXT, SALARY REAL);
CREATE TABLE DEPARTMENTS (DEPTID INTEGER PRIMARY KEY, DEPTNAME TEXT);
CREATE VIEW EMP_DEPT AS SELECT E.EMPNUM, E.NAME, D.DEPTNAME FROM EMPLOYEES E LEFT JOIN DEPARTMENTS D ON E.EMPNUM = D.DEPTID;
