CREATE TABLE IF NOT EXISTS user_activity (activity_id INTEGER, user_id INTEGER, activity_time DATETIME, activity_type TEXT, details TEXT);
INSERT INTO user_activity SELECT number AS activity_id, ABS(RANDOM() % 1000) AS user_id, DATETIME('now', '+' || number * 5 || ' seconds') AS activity_time, (CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, 'Activity details for ' || number AS details FROM (SELECT ROW_NUMBER() OVER() AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) LIMIT 100);
CREATE TABLE IF NOT EXISTS product_sales (sale_id INTEGER, product_id INTEGER, sale_time DATETIME, quantity INTEGER, price REAL);
INSERT INTO product_sales SELECT number AS sale_id, ABS(RANDOM() % 500) AS product_id, DATETIME('now', '+' || number * 2 || ' seconds') AS sale_time, (ABS(RANDOM() % 10) + 1) AS quantity, (ABS(RANDOM() % 100) + 1) * 1.5 AS price FROM (SELECT ROW_NUMBER() OVER() AS number FROM (SELECT 1 UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5 UNION SELECT 6 UNION SELECT 7 UNION SELECT 8 UNION SELECT 9 UNION SELECT 10) LIMIT 50);
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= DATETIME('now', '-1 day') GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
