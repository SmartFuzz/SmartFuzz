CREATE TABLE orders (OrderID INTEGER PRIMARY KEY, UserID INTEGER, Amount REAL, OrderDate DATETIME, FOREIGN KEY (UserID) REFERENCES users(UserID));

WITH latest_orders AS (SELECT UserID, Amount FROM orders WHERE Amount IS NOT NULL ORDER BY UserID, OrderDate DESC LIMIT 1) SELECT u.Username, lo.Amount FROM users u JOIN latest_orders lo ON u.UserID = lo.UserID ORDER BY u.Username;
