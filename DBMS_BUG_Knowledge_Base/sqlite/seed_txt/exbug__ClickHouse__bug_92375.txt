DROP TABLE IF EXISTS data_local;
CREATE TABLE data_local (id INTEGER, value TEXT);
INSERT INTO data_local (id, value) SELECT number, CAST(number AS TEXT) FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9);
CREATE TABLE data_distributed AS SELECT * FROM data_local;
SELECT COUNT(DISTINCT value) FROM data_distributed;
