DROP TABLE IF EXISTS t1;
DROP TABLE IF EXISTS t2;
CREATE TABLE t1 (c0 INTEGER, c1 INTEGER, c2 TEXT);
CREATE TABLE t2 (c0 INTEGER, c1 INTEGER, c2 REAL);
CREATE TABLE t3 (c0 INTEGER, c1 INTEGER, c2 DATE NOT NULL);
SELECT * FROM t1 JOIN t2 ON t1.c0 = t2.c0 WHERE t2.c1 IS NULL OR t2.c1 > 0 AND t1.c1 > 0;
