CREATE TABLE throttle_test (id INTEGER PRIMARY KEY, payload TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO throttle_test (id, payload) SELECT number, random() % 1000 || 'test' FROM (SELECT ROW_NUMBER() OVER () AS number FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS t1, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS t2 LIMIT 100);
CREATE TABLE throttle_test_backup AS SELECT * FROM throttle_test;
SELECT id, payload FROM throttle_test WHERE created_at IS NOT NULL AND id % 2 = 0;
SELECT COUNT(*) FROM throttle_test_backup WHERE payload LIKE '%test%' AND id IN (SELECT id FROM throttle_test WHERE id < 50);
