CREATE VIEW yearqtr AS SELECT dt, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) AS yq FROM test_dates;
SELECT dt, yq, yq / 10 AS yq2, yq / 3 AS yq3, yq % 3 AS yq4 FROM (SELECT dt, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) AS yq FROM test_dates) WHERE dt = '2050-12-31';
SELECT dt, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) AS yq, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) / 10 AS yq2, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) / 3 AS yq3, ((strftime('%Y', dt) * 10) + ((strftime('%m', dt) - 1) / 3) + 1) % 3 AS yq4 FROM test_dates WHERE dt IS NOT NULL;
