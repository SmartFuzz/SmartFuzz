CREATE TABLE test_table1 (id INTEGER, value REAL, created_at DATETIME);
CREATE TABLE test_table2 (id INTEGER, description TEXT, is_active INTEGER);
CREATE TRIGGER test_mv1_trigger AFTER INSERT ON test_table2 BEGIN INSERT INTO test_table1 (id, value) SELECT id, value FROM test_table2 WHERE is_active = 1;
END;
CREATE TRIGGER test_mv2_trigger AFTER INSERT ON test_table1 BEGIN INSERT INTO test_table2 (id, description) SELECT id, description FROM test_table1 WHERE created_at > datetime('now', '-5 minutes');
END;
