CREATE TABLE user_events (event_id TEXT, user_id INTEGER, event_type TEXT, event_time DATETIME);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT LOWER(hex(randomblob(16))) AS event_id, ABS(random() % 10000) AS user_id, CASE ABS(random() % 3) WHEN 0 THEN 'click' WHEN 1 THEN 'view' ELSE 'purchase' END AS event_type, datetime('now', '-' || (ABS(random() % 3600 * 24)) || ' seconds') AS event_time;
CREATE TABLE user_summary (user_id INTEGER, event_count INTEGER);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
