DROP TABLE IF EXISTS repro_events;
CREATE TABLE repro_events (tenant TEXT, event_time DATETIME, id INTEGER, msg TEXT, value REAL);
INSERT INTO repro_events (tenant, event_time, id, msg, value) SELECT CAST(number % 100 AS TEXT), DATETIME('now', '-' || (number % 200000) || ' seconds'), number, 'm' || CAST(number AS TEXT), CAST(number % 1000 AS REAL) FROM (SELECT ROW_NUMBER() OVER() - 1 AS number FROM (SELECT 0 UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS a, (SELECT 0 UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS b LIMIT 2000000);
SELECT * FROM repro_events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
