CREATE TABLE test_data (id INTEGER, ts DATETIME, value TEXT, status TEXT);
INSERT INTO test_data SELECT number, datetime('now', '+' || number || ' seconds'), CAST(number % 5 AS TEXT), CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM (SELECT rowid AS number FROM sqlite_sequence LIMIT 1000000);
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > datetime('now', '-1 day') AND value LIKE '1%';
SELECT DISTINCT id FROM test_data WHERE ts BETWEEN '2023-01-01' AND '2023-12-31' AND status IS NOT NULL;
