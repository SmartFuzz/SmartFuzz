CREATE TABLE set_idx_rand_v2 (id INTEGER, s TEXT, d DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO set_idx_rand_v2 SELECT number, CAST(number % 10 AS TEXT) FROM (SELECT ROW_NUMBER() OVER () AS number FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10) LIMIT 100);
SELECT * FROM set_idx_rand_v2 WHERE RANDOM() % 1000 = 1;
SELECT * FROM set_idx_rand_v2 WHERE d IS NULL OR d > DATETIME('now', '-1 day');
SELECT COUNT(*) FROM set_idx_rand_v2 WHERE s IN (SELECT s FROM set_idx_rand_v2 WHERE id < 50) AND RANDOM() % 1000 = 1;
