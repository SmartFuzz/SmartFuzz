CREATE TABLE urls(urlid INTEGER PRIMARY KEY, c INTEGER, d INTEGER, url_text TEXT);
CREATE TABLE keys(urlid INTEGER, a INTEGER, b INTEGER, FOREIGN KEY(urlid) REFERENCES urls(urlid), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
