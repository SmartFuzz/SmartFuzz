CREATE TABLE users (id INTEGER PRIMARY KEY, name TEXT NOT NULL, age INTEGER CHECK(age >= 0), created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id INTEGER PRIMARY KEY, user_id INTEGER, amount REAL NOT NULL, order_date DATETIME DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TRIGGER update_order_timestamp AFTER UPDATE ON orders FOR EACH ROW BEGIN UPDATE orders SET order_date = CURRENT_TIMESTAMP WHERE id = NEW.id;
END;
