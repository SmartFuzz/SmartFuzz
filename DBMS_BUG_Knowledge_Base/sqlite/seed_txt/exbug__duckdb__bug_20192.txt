CREATE TABLE products (id INTEGER PRIMARY KEY, name TEXT, description TEXT, price REAL, created_at TEXT);
CREATE TABLE orders (id INTEGER PRIMARY KEY, product_id INTEGER, order_date TEXT, quantity INTEGER, large_blob BLOB, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, 'Product ' || i AS name, 'Description for product ' || i AS description, ROUND(RANDOM() * 100, 2) AS price, DATE('2024-01-01', '+' || i || ' days') AS created_at FROM (SELECT ROW_NUMBER() OVER() AS i FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) t1, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) t2 LIMIT 10000);
INSERT INTO orders SELECT i AS id, (i % 10000) + 1 AS product_id, DATE('2024-02-01', '+' || i || ' days') AS order_date, (i % 10) + 1 AS quantity, repeat('y', 2000) AS large_blob FROM (SELECT ROW_NUMBER() OVER() AS i FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) t1, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) t2 LIMIT 5000000);
SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01' AND '2024-04-01' AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
