CREATE TABLE customer (c_id INTEGER PRIMARY KEY, c_name TEXT, c_balance REAL NOT NULL);
CREATE TABLE orders (o_id INTEGER PRIMARY KEY, o_customer_id INTEGER, o_amount REAL, FOREIGN KEY (o_customer_id) REFERENCES customer(c_id));

SELECT c.c_name, COALESCE(SUM(o.o_amount), 0) AS total_spent FROM customer c LEFT JOIN orders o ON c.c_id = o.o_customer_id GROUP BY c.c_id;
