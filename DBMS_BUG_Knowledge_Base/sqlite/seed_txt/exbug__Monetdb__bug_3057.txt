CREATE TABLE images (imageid INTEGER PRIMARY KEY, ds_id INTEGER, bmaj REAL);
CREATE TABLE xtrsrc (xtrsrcid INTEGER PRIMARY KEY, image_id INTEGER, zone INTEGER, ra REAL, decl REAL, ra_err REAL, decl_err REAL, x REAL, y REAL, z REAL, FOREIGN KEY (image_id) REFERENCES images (imageid));
