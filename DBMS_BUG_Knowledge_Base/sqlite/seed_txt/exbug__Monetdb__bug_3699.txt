CREATE TABLE stocks (id INTEGER PRIMARY KEY AUTOINCREMENT, trade_time TEXT NOT NULL, trade_date TEXT, ticker TEXT NOT NULL, currency TEXT NOT NULL, opening_price REAL NOT NULL, highest_price REAL NOT NULL, lowest_price REAL NOT NULL, closing_price REAL NOT NULL, trade_volume REAL);
INSERT INTO stocks (trade_time, trade_date, ticker, currency, opening_price, highest_price, lowest_price, closing_price, trade_volume) VALUES (DATETIME('now'), DATETIME('now'), 'AAPL', 'USD', 150.0, 155.0, 149.0, 154.0, 10000.0);
SELECT trade_date, opening_price, MAX(highest_price) AS highest_price, MIN(lowest_price) AS lowest_price, closing_price, SUM(trade_volume) FROM stocks WHERE ticker='AAPL' AND trade_date IS NOT NULL AND DATE(trade_date) > DATE('now', '-30 days') GROUP BY trade_date ORDER BY trade_date DESC;
SELECT DISTINCT DATE(trade_date) FROM stocks LIMIT 10;
