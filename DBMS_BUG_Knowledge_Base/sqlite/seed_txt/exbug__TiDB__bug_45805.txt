CREATE TABLE signed_estimate_volume (id INTEGER PRIMARY KEY AUTOINCREMENT, estimate_id INTEGER NOT NULL DEFAULT 0, dealer_id TEXT NOT NULL DEFAULT '', broker_id TEXT NOT NULL DEFAULT '', real_t0_volume INTEGER NOT NULL DEFAULT 0, real_t1_volume INTEGER NOT NULL DEFAULT 0, manager_yid TEXT NOT NULL DEFAULT '', estimate_volume INTEGER NOT NULL DEFAULT 0, estimate_year_month TEXT NOT NULL DEFAULT '', is_dropped INTEGER NOT NULL DEFAULT 0);
CREATE TABLE unsigned_estimate_volume (id INTEGER PRIMARY KEY AUTOINCREMENT, estimate_id INTEGER NOT NULL DEFAULT 0, estimate_year_month TEXT NOT NULL DEFAULT '', business_id INTEGER NOT NULL DEFAULT 0, broker_id TEXT NOT NULL DEFAULT '', estimate_volume INTEGER NOT NULL DEFAULT 0, manager_yid TEXT NOT NULL DEFAULT '');

SELECT estimate_year_month, manager_yid, real_t1_volume, real_t0_volume, (COALESCE(a_estimate_volume, 0) + COALESCE(b_estimate_volume, 0)) AS estimate_volume, (COALESCE(increase_sum, 0) + COALESCE(b.increase_sum, 0)) AS increase_sum, (COALESCE(decrease_sum, 0) + COALESCE(b.decrease_sum, 0)) AS decrease_sum FROM (SELECT estimate_year_month, manager_yid, SUM(real_t1_volume) AS real_t1_volume, SUM(real_t0_volume) AS real_t0_volume, SUM(estimate_volume) AS a_estimate_volume, SUM(CASE WHEN estimate_volume > real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS increase_sum, SUM(CASE WHEN estimate_volume < real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS decrease_sum FROM signed_estimate_volume GROUP BY manager_yid) AS a LEFT JOIN (SELECT manager_yid, SUM(estimate_volume) AS b_estimate_volume, SUM(estimate_volume) AS increase_sum, 0 AS decrease_sum FROM unsigned_estimate_volume GROUP BY manager_yid) AS b ON a.manager_yid = b.manager_yid;
