DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INTEGER, accepted_at DATETIME, delivery_report_at TEXT);
INSERT INTO agg_test (id, accepted_at, delivery_report_at) SELECT number AS id, datetime('2025-09-24 00:00:00', '+' || number || ' seconds') AS accepted_at, (SELECT max(CASE WHEN number % 2 = 1 THEN number END) FROM (SELECT number + i AS number FROM (SELECT 0 AS i UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS t)) AS delivery_report_at FROM (SELECT number FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9) AS t) LIMIT 100;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
