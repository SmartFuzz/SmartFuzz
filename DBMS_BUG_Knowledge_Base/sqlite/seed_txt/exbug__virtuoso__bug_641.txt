CREATE TABLE actors (id INTEGER PRIMARY KEY, name TEXT, birth_year INTEGER, is_active INTEGER);
CREATE TABLE movies (id INTEGER PRIMARY KEY, title TEXT, release_year INTEGER, director_id INTEGER, FOREIGN KEY (director_id) REFERENCES actors(id));
