CREATE TABLE equipment (EquipmentName TEXT, FailureReason TEXT);
INSERT INTO equipment (EquipmentName, FailureReason) VALUES ('Excavator', 'Hydraulic Failure'), ('Bulldozer', 'Engine Overheat');
CREATE TABLE maintenance_actions (ActionId INTEGER, ActionDescription TEXT);
SELECT e.*, m.ActionDescription FROM equipment e LEFT JOIN maintenance_actions m ON e.EquipmentName = m.ActionDescription WHERE e.FailureReason = 'Hydraulic Failure';
