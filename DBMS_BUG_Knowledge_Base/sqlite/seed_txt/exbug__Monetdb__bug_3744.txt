CREATE TABLE test_casts (id INTEGER, dec_value DECIMAL(10, 7), dbl_value REAL);
INSERT INTO test_casts (id, dec_value, dbl_value) VALUES (1, 13.8, 13.8);
SELECT id, CAST(dec_value AS INTEGER) AS dec_to_int, CAST(dbl_value AS INTEGER) AS dbl_to_int FROM test_casts;
SELECT id, CAST(CAST(dec_value AS DECIMAL(10, 7)) AS INTEGER) AS dec_casted, CAST(CAST(dbl_value AS REAL) AS INTEGER) AS dbl_casted FROM test_casts;
SELECT id, CASE WHEN dec_value IS NULL THEN 'NULL' ELSE CAST(dec_value AS INTEGER) END AS dec_null_handling FROM test_casts;
