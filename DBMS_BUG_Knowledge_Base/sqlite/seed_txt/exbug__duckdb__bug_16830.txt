CREATE TABLE test_regex (id INTEGER PRIMARY KEY, pattern TEXT, test_string TEXT);
INSERT INTO test_regex (id, pattern, test_string) VALUES (1, 'bc', 'abcd'), (2, 'a.c', 'abcd'), (3, 'a.*d', 'abcd'), (4, '(b|x)', 'abcd'), (5, '^a', 'abcd'), (6, '^(b|c)', 'abcd');
SELECT id, test_string LIKE pattern AS match_result FROM test_regex;
SELECT id, test_string LIKE 'a.%d' AS match_result FROM test_regex WHERE pattern = 'a.*d';
SELECT id, test_string LIKE '^(b|c)' AS match_result FROM test_regex WHERE test_string IS NOT NULL;
