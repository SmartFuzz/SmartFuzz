CREATE TABLE test_subfact(n INTEGER, sub INTEGER);
WITH RECURSIVE cte(n, psub, sub) AS (SELECT 0 AS n, 1 AS psub, 1 AS sub UNION ALL SELECT n + 1, sub, (n * (sub + psub)) FROM cte WHERE n < 20) INSERT INTO test_subfact SELECT n, sub FROM cte;
SELECT * FROM test_subfact;
