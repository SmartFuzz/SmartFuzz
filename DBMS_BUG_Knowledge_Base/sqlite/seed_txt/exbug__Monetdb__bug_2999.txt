CREATE TABLE test_table(id INTEGER PRIMARY KEY, value TEXT, order_num INTEGER NOT NULL);

CREATE TABLE order_table(order_id INTEGER, order_value REAL, FOREIGN KEY(order_id) REFERENCES test_table(id));
