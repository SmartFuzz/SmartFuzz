CREATE TABLE scene (id INTEGER PRIMARY KEY, victim_id INTEGER NOT NULL, description TEXT);
CREATE TABLE victim (id INTEGER PRIMARY KEY, name TEXT, age INTEGER CHECK (age >= 0));
CREATE TABLE inspector (id INTEGER PRIMARY KEY, name TEXT, experience_years INTEGER DEFAULT 0);
