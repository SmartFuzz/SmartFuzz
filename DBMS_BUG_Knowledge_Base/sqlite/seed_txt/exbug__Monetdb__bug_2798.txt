CREATE TABLE test_table (id INTEGER PRIMARY KEY, value REAL, category TEXT);
INSERT INTO test_table (id, value, category) VALUES (1, 100.00, 'A'), (2, 200.50, 'B'), (3, NULL, 'A');
SELECT category AS cat, COUNT(id) AS count_id, AVG(value) AS avg_value FROM test_table GROUP BY cat;
SELECT category AS cat, SUM(value) AS total_value, (SUM(value) / (SELECT COUNT(*) FROM test_table)) AS avg_per_category FROM test_table GROUP BY cat;
SELECT category AS cat, SUM(value) AS total_value, (2 * (SUM(value) / NULLIF((SELECT COUNT(*) FROM test_table), 0))) AS adjusted_value FROM test_table GROUP BY cat;
