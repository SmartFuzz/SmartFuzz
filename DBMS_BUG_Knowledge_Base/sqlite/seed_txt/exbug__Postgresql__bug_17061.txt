DROP TABLE IF EXISTS tree;
CREATE TABLE tree(id INTEGER PRIMARY KEY AUTOINCREMENT, parent_id INTEGER, name TEXT NOT NULL, depth INTEGER DEFAULT 0);

WITH RECURSIVE mtree(id, name, depth) AS (SELECT id, name, depth FROM tree WHERE id = 1 UNION ALL SELECT t.id, t.name, t.depth FROM tree AS t INNER JOIN mtree AS m ON t.parent_id = m.id) SELECT json_object('depth', depth) FROM mtree;
