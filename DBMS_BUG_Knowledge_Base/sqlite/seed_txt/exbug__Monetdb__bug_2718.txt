CREATE TABLE bug (time DATETIME, val INTEGER, description TEXT NOT NULL);

SELECT t1.time, COUNT(t2.val) FROM bug t1, bug t2 WHERE t2.time BETWEEN (t1.time - INTERVAL '1' SECOND) AND (t1.time + INTERVAL '1' SECOND) GROUP BY t1.time ORDER BY t1.time;
