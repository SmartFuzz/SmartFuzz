CREATE TABLE test (a INTEGER, b TEXT, c REAL CHECK (c >= 0));
INSERT INTO test (a, b, c) VALUES (1, 'hello', 3.5), (2, 'world', 4.0), (1, NULL, 2.5);
SELECT a, b, LENGTH(b), COUNT(*) FROM test GROUP BY a, b HAVING COUNT(a) = LENGTH(b);
SELECT a, b, LENGTH(b), COUNT(*) FROM test GROUP BY a, b HAVING COUNT(a) = LENGTH('test');
SELECT a, b, LENGTH(b), COUNT(*) FROM test WHERE c IS NOT NULL GROUP BY a, b HAVING COUNT(a) = 1;
