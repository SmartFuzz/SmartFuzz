CREATE TABLE IF NOT EXISTS mv_src_table (id INTEGER, value TEXT);
INSERT INTO mv_src_table SELECT number, CAST(number AS TEXT) FROM (SELECT 0 AS number UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9);
CREATE TABLE IF NOT EXISTS mv_refreshable_dst_table (id INTEGER);
CREATE TRIGGER mv_refreshable AFTER INSERT ON mv_src_table BEGIN INSERT INTO mv_refreshable_dst_table (id) SELECT max(id) FROM mv_src_table WHERE value IS NOT NULL;
END;
