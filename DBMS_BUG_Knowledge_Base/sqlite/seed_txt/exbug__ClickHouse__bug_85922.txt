CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT, created_at DATETIME);
CREATE TABLE IF NOT EXISTS user_logs (user_id INTEGER, log_message TEXT, log_time DATETIME);
INSERT INTO users (id, name, created_at) VALUES (1, 'Alice', CURRENT_TIMESTAMP);
INSERT INTO user_logs (user_id, log_message, log_time) SELECT id, 'User logged in', CURRENT_TIMESTAMP FROM users WHERE id = 1;
CREATE TABLE IF NOT EXISTS user_summary AS SELECT user_id, COUNT(*) AS log_count FROM user_logs GROUP BY user_id;
