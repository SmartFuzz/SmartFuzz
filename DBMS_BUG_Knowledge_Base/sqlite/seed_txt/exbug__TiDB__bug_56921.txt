DROP TABLE IF EXISTS t1;
CREATE TABLE t1(c0 INT, c1 TEXT, c2 REAL);
INSERT INTO t1(c0, c1, c2) VALUES (1, 'A', 10.50), (2, 'B', NULL), (3, 'C', 20.00);
SELECT COUNT(*) FROM t1 WHERE c1 >= CAST(c0 AS TEXT) OR c1 LIKE 'A%';
SELECT SUM(count) FROM (SELECT (c1 >= CAST(c0 AS TEXT) OR c1 LIKE 'B%') AS count FROM t1);
