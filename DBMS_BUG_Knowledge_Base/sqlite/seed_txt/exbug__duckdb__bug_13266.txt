CREATE TABLE mixed_timestamps (id INTEGER, time_column TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO mixed_timestamps (id, time_column) VALUES (1, '2021-10-11T18:59:14Z'), (2, '2021-10-11T12:55Z'), (3, '2021-10-11T12:55:00');
CREATE TABLE parsed_timestamps AS SELECT id, time_column, CASE WHEN time_column GLOB '*T*' THEN datetime(time_column) ELSE NULL END AS parsed_time FROM mixed_timestamps;
SELECT id, time_column, parsed_time FROM parsed_timestamps WHERE parsed_time IS NULL;
SELECT COUNT(*) FROM parsed_timestamps WHERE parsed_time IS NOT NULL;
