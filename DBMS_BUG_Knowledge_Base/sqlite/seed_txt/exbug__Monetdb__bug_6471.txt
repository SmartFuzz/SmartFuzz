CREATE TABLE test_series (id INTEGER PRIMARY KEY, value INTEGER NOT NULL);
INSERT INTO test_series (id, value) SELECT id, id FROM (SELECT 1 AS id UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10);
CREATE TABLE test_data (data_id INTEGER, data_value TEXT NOT NULL CHECK (data_value <> ''));
INSERT INTO test_data (data_id, data_value) SELECT id, 'Value ' || id FROM (SELECT 1 AS id UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5);
SELECT t.id, t.value, d.data_value FROM test_series t JOIN test_data d ON t.id = d.data_id WHERE t.value IS NOT NULL;
