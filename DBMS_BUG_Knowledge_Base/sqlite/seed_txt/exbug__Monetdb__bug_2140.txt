CREATE TABLE users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER CHECK (age >= 0));
CREATE TABLE orders (id INTEGER PRIMARY KEY, user_id INTEGER, amount REAL, FOREIGN KEY (user_id) REFERENCES users(id));
