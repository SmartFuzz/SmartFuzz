CREATE TABLE A (id INTEGER PRIMARY KEY, entity_id INTEGER NOT NULL, organization_id INTEGER NOT NULL);
CREATE TABLE B (id INTEGER PRIMARY KEY, entity_id INTEGER NOT NULL, organization_id INTEGER NOT NULL, status TEXT NOT NULL CHECK(status IN ('ACTIVE', 'INACTIVE')));
CREATE TABLE C (id INTEGER PRIMARY KEY, status TEXT NOT NULL);
INSERT INTO A (id, entity_id, organization_id) VALUES (1, 100, 200), (2, 101, 201);
INSERT INTO B (id, entity_id, organization_id, status) VALUES (1, 100, 200, 'ACTIVE'), (2, 101, 201, 'INACTIVE');
SELECT A.id AS aid, A.entity_id AS entity_id, subquery.id AS bid FROM A LEFT OUTER JOIN (SELECT B.id AS bid, B.organization_id AS organization_id, B.entity_id AS entity_id FROM B INNER JOIN C ON C.id = B.entity_id WHERE B.status = 'ACTIVE' AND C.status = 'ACTIVE') AS subquery ON A.entity_id = subquery.entity_id AND A.organization_id = subquery.organization_id;
