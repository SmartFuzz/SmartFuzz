DROP TABLE IF EXISTS test_data;
CREATE TABLE test_data (id INTEGER PRIMARY KEY, value TEXT, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_data (id, value) SELECT number, CAST(number AS TEXT) FROM (SELECT ROW_NUMBER() OVER () AS number LIMIT 100);
SELECT id, value FROM test_data WHERE id % 2 = 0;
