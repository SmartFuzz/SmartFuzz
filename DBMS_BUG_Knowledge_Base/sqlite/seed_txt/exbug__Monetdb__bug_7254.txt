CREATE TABLE y (id INTEGER PRIMARY KEY, value TEXT NOT NULL, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
INSERT INTO y (id, value) SELECT i, 'Value ' || i FROM (SELECT ROW_NUMBER() OVER() AS i FROM (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS a, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS b, (SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1 UNION ALL SELECT 1) AS c LIMIT 1000000);
DELETE FROM y WHERE id IN (SELECT id FROM y WHERE value LIKE 'Value 1%');
CREATE TABLE z AS SELECT DISTINCT value FROM y WHERE created_at < DATETIME('now', '-1 day');
DELETE FROM z WHERE value IS NULL;
