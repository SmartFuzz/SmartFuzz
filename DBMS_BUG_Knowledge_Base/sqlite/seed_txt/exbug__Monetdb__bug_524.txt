CREATE TABLE users (id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, created_at DATETIME DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE events (event_id INTEGER PRIMARY KEY AUTOINCREMENT, event_time DATETIME NOT NULL, user_id INTEGER, FOREIGN KEY (user_id) REFERENCES users(id));
