CREATE TABLE fruits (id INTEGER PRIMARY KEY, name TEXT NOT NULL, color TEXT);
INSERT INTO fruits (name, color) VALUES ('apple', 'red'), ('banana', 'yellow'), ('grape', 'purple');
CREATE TABLE colors (id INTEGER PRIMARY KEY, color_name TEXT UNIQUE);
INSERT INTO colors (color_name) VALUES ('red'), ('yellow'), ('green');
SELECT * FROM (SELECT name FROM fruits) a LEFT JOIN (SELECT color_name FROM colors) b ON a.color = b.color_name;
