DROP TABLE IF EXISTS TestSequence;
CREATE TABLE TestSequence (num TEXT PRIMARY KEY, value INTEGER NOT NULL DEFAULT 0);
INSERT INTO TestSequence(num, value) VALUES ('0:ZERO', 0), ('1:ONE', 1), ('2:TWO', 2), ('3:THREE', 3), ('4:FOUR', 4), ('5:FIVE', 5), ('6:SIX', 6), ('7:SEVEN', 7), ('8:EIGHT', 8), ('9:NINE', 9);
SELECT GROUP_CONCAT(TS.num ORDER BY TS.value ASC) AS ss FROM TestSequence TS WHERE TS.value IS NOT NULL;
