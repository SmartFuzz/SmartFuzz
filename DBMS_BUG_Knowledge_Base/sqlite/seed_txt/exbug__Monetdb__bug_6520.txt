CREATE TABLE "PRIMARY" ("ID" INTEGER NOT NULL, "LEN" REAL NOT NULL, PRIMARY KEY ("ID"));
CREATE TABLE "FOREIGN" ("ID" INTEGER NOT NULL, "POS" REAL NOT NULL, FOREIGN KEY ("ID") REFERENCES "PRIMARY" ("ID"));
INSERT INTO "PRIMARY" ("ID", "LEN") VALUES (1, 10.5), (2, 20.0), (3, NULL);
INSERT INTO "FOREIGN" ("ID", "POS") VALUES (1, 5.0), (2, 15.0), (3, 25.0);
UPDATE "PRIMARY" SET "LEN" = (SELECT AVG("LEN") FROM "PRIMARY" WHERE "ID" IN (SELECT "ID" FROM "FOREIGN" WHERE "POS" < (SELECT MAX("LEN") FROM "PRIMARY"))) ) WHERE "ID" IN (SELECT "ID" FROM "FOREIGN" WHERE "POS" < (SELECT MAX("LEN") FROM "PRIMARY"));
SELECT * FROM "PRIMARY";
