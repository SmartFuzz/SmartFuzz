CREATE TABLE test_json_1 AS SELECT json('{"is_active": false, "is_true": true}') AS data;
CREATE TABLE test_json_2 AS SELECT json('{"is_active": null, "is_true": false}') AS data;
SELECT * FROM test_json_1 WHERE NOT (json_extract(data, '$.is_active'));
SELECT * FROM test_json_2 WHERE NOT (json_extract(data, '$.is_true'));
SELECT * FROM test_json_1 WHERE NOT (json_extract(data, '$.is_active')) AND (json_extract(data, '$.is_true')) IS NOT NULL;
