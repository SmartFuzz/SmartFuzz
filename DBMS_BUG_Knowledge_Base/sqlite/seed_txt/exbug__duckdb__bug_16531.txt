CREATE TABLE test_data (id INTEGER, download INTEGER, upload INTEGER, app TEXT, key TEXT);
INSERT INTO test_data SELECT id, (ABS(RANDOM() % 3)) AS download, (ABS(RANDOM() % 3)) AS upload, 'App_' || (ABS(RANDOM() % 100)) AS app, SUBSTR(HEX(RANDOMBLOB(22)), 1, 44) AS key FROM (SELECT ROW_NUMBER() OVER() AS id FROM (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) AS a, (SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5) AS b LIMIT 100000);
SELECT app, key, COUNT(*) AS count_download, SUM(CASE WHEN download = 0 THEN 1 ELSE 0 END) AS low_count, SUM(CASE WHEN download = 1 THEN 1 ELSE 0 END) AS average_count, SUM(CASE WHEN download = 2 THEN 1 ELSE 0 END) AS high_count FROM test_data GROUP BY app, key;
WITH download_summary AS (SELECT app, key, SUM(CASE WHEN download = 0 THEN 1 ELSE 0 END) AS low_count, SUM(CASE WHEN download = 1 THEN 1 ELSE 0 END) AS average_count, SUM(CASE WHEN download = 2 THEN 1 ELSE 0 END) AS high_count FROM test_data GROUP BY app, key) SELECT app, key, CASE WHEN low_count = MAX(low_count, average_count, high_count) THEN 0 WHEN average_count = MAX(low_count, average_count, high_count) THEN 1 ELSE 2 END AS mode_download FROM download_summary;
SELECT app, key, COALESCE(NULLIF(MAX(download), 0), 1) AS max_download FROM test_data GROUP BY app, key HAVING COUNT(*) > 10;
