CREATE TABLE test_table (id INTEGER PRIMARY KEY, value REAL, status TEXT);
INSERT INTO test_table (id, value, status) VALUES (1, 50.5, 'active'), (2, NULL, 'inactive'), (3, -10.2, 'active');
SELECT id, CASE WHEN value IS NULL THEN 'no value' WHEN value < 0 THEN 'negative' ELSE 'positive' END AS value_status FROM test_table;
SELECT COUNT(*), CASE WHEN COUNT(*) > 2 THEN 'multiple' ELSE 'single' END AS count_status FROM test_table;
SELECT id, CASE WHEN value = (SELECT AVG(value) FROM test_table WHERE status = 'active') THEN 'average' ELSE 'not average' END AS avg_status FROM test_table;
