CREATE SCHEMA test_schema;
CREATE TABLE test_schema.test_table (id INT AUTO_INCREMENT PRIMARY KEY, path VARCHAR(255) NOT NULL, description TEXT);
CREATE FUNCTION test_schema.trigger_function() RETURNS trigger AS $$ BEGIN RETURN NEW;
END;
$$;
CREATE TRIGGER path_change_trigger AFTER UPDATE ON test_schema.test_table FOR EACH ROW WHEN (NEW.path <> OLD.path) EXECUTE PROCEDURE test_schema.trigger_function();
