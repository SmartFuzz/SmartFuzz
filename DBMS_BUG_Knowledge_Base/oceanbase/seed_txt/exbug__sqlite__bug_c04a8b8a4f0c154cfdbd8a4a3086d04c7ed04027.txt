CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0), email VARCHAR(255) UNIQUE);
CREATE VIEW user_info AS SELECT id, name, age FROM users;
