DROP TABLE IF EXISTS test.t2;
CREATE TABLE test.t2 (id INT PRIMARY KEY, value DECIMAL(10, 2), status VARCHAR(10), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

SELECT id, value, LAST_VALUE(value) OVER (PARTITION BY status ORDER BY created_at ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS last_val FROM test.t2;
