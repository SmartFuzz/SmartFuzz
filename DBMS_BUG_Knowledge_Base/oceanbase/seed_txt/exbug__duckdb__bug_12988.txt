CREATE TABLE employees (id integer PRIMARY KEY, name varchar(255), salary decimal(10, 2), department_id integer);
CREATE TABLE departments (id integer PRIMARY KEY, name varchar(255) NOT NULL);
INSERT INTO employees VALUES (1, 'John Doe', 50000.00, 1), (2, 'Jane Smith', 60000.00, 2), (3, 'Alice Johnson', 55000.00, 1);

SELECT d.*, e.*, RANK() OVER (ORDER BY e.salary) AS salary_rank FROM departments d JOIN employees e ON d.id = e.department_id;
