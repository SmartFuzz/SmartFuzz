CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id INT PRIMARY KEY AUTO_INCREMENT, user_id INT, amount FLOAT NOT NULL, order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TRIGGER update_order_timestamp BEFORE UPDATE ON orders FOR EACH ROW SET NEW.order_date = CURRENT_TIMESTAMP;
