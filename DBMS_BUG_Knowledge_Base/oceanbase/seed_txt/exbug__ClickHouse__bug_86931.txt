CREATE DATABASE reproduction_test;
CREATE TABLE reproduction_test.tree_nodes (date DATE, c1 VARCHAR(255), c2 INT UNSIGNED) ENGINE = OBTree PARTITION BY EXTRACT(YEAR_MONTH FROM date) ORDER BY (date, c1, c2);
INSERT INTO reproduction_test.tree_nodes VALUES ('2023-01-01', 'node1', 1), ('2023-01-01', 'node2', 2), ('2023-01-01', 'node3', 3);
SELECT date, COUNT(DISTINCT c2) AS unique_count FROM reproduction_test.tree_nodes GROUP BY date WITH ROLLUP;
