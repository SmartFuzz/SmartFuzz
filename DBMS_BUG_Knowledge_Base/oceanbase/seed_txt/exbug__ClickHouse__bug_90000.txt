CREATE TABLE T1(A UInt64, B Float64) ENGINE = ObRowStore AS SELECT number, number * 1.5 FROM dual WHERE number < 100000;
CREATE TABLE T2(C String, D Nullable(Int32)) ENGINE = ObRowStore AS SELECT toString(number) AS C, CASE WHEN number % 2 = 0 THEN number ELSE NULL END AS D FROM dual WHERE number < 1000;
CREATE TABLE item_dict2 (id UInt64, a1 UInt64, a2 Int32) PRIMARY KEY (id) AS SELECT id, a1, a2 FROM T2;
SELECT DISTINCT A, (SELECT a1 FROM item_dict2 WHERE id = toUInt64(B)) AS a1, (SELECT a2 FROM item_dict2 WHERE id = toUInt64(B)) AS a2 FROM (SELECT A, B FROM T1);
DROP TABLE item_dict2;
