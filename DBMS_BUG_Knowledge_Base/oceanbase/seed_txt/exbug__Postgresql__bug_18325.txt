DROP TABLE IF EXISTS xyz;
CREATE TABLE xyz (id integer PRIMARY KEY AUTO_INCREMENT, name varchar(255), age integer, created_at timestamp DEFAULT CURRENT_TIMESTAMP);

BEGIN FOR user_record IN (SELECT * FROM xyz WHERE age IS NOT NULL) LOOP RETURN user_record.age;
