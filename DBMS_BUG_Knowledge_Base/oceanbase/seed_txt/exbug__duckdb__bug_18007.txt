CREATE TABLE cols (c VARCHAR(255));
INSERT INTO cols SELECT 'c' || i FROM (SELECT LEVEL AS i FROM dual CONNECT BY LEVEL <= 21);
CREATE TABLE matrix (col1 VARCHAR(255), col2 VARCHAR(255), val TINYINT NOT NULL CHECK (val < 250));
INSERT INTO matrix SELECT c1.c, c2.c, CAST(FLOOR(DBMS_RANDOM.VALUE() * 250) AS TINYINT) FROM cols c1, cols c2 WHERE c1.c != c2;
SELECT col1, col2, SUM(val) AS total_val FROM matrix GROUP BY col1, col2 ORDER BY col2 FETCH FIRST 2 ROWS ONLY;
