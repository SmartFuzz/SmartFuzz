CREATE TABLE IF NOT EXISTS default.initial_complex (n UInt32, description VARCHAR(255), _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) ENGINE = REPLACE INTO default.initial_complex (_timestamp) ORDER BY (n);
CREATE TABLE IF NOT EXISTS default.target1_complex (x UInt32, status VARCHAR(255), _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) ENGINE = REPLACE INTO default.target1_complex (_timestamp) ORDER BY (x);
CREATE TABLE IF NOT EXISTS default.target2_complex (x UInt32, value FLOAT, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) ENGINE = REPLACE INTO default.target2_complex (_timestamp) ORDER BY (x);
CREATE MATERIALIZED VIEW IF NOT EXISTS default.initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM default.initial_complex WHERE MOD(FLOOR(RAND() * 100), 2) = 0;
CREATE MATERIALIZED VIEW IF NOT EXISTS default.target1_target2_mv_complex AS SELECT x, CAST(x AS FLOAT) * 1.5 AS value FROM default.target1_complex WHERE status IS NOT NULL;
