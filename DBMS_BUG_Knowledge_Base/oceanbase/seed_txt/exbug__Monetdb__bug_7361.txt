CREATE TABLE test_table(test_column INT NOT NULL, name VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table(test_column, name) SELECT seq, 'Test Name' FROM (SELECT ROW_NUMBER() OVER () AS seq FROM dual CONNECT BY LEVEL <= 2147483647) WHERE seq <= 2147483647;
SELECT COUNT(*) FROM test_table WHERE test_column IS NOT NULL;
DELETE FROM test_table WHERE test_column < 1000;
SELECT AVG(test_column) FROM test_table WHERE created_at IS NOT NULL;
