CREATE TABLE test_table (id INT PRIMARY KEY, value VARCHAR(50), amount DECIMAL(10, 2), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table (id, value, amount) VALUES (1, 'test', 5.50), (2, 'example', NULL), (3, 'sample', 10.00);
SELECT id, value, CASE WHEN amount IS NULL THEN 'No Amount' WHEN amount < 5 THEN 'Low' WHEN amount BETWEEN 5 AND 10 THEN 'Medium' ELSE 'High' END AS amount_category FROM test_table;
SELECT id, CASE id WHEN 1 THEN 'One' WHEN 2 THEN 'Two' ELSE 'Other' END AS id_description FROM test_table;
SELECT id, value, CASE WHEN created_at < NOW() - INTERVAL '1 DAY' THEN 'Old' ELSE 'Recent' END AS age_category FROM test_table;
