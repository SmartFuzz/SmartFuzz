CREATE TABLE departments (id INT PRIMARY KEY, name VARCHAR(50) NOT NULL);
INSERT INTO departments VALUES (1, 'Management'), (2, 'Development'), (3, 'Sales');
CREATE TABLE employees (dep_id INT, name VARCHAR(20), salary DOUBLE, hire_date DATE, FOREIGN KEY (dep_id) REFERENCES departments(id));
INSERT INTO employees VALUES (1, 'Smith', 81, TO_DATE('2019-08-10', 'YYYY-MM-DD')), (2, 'Jones', 55, TO_DATE('2019-08-09', 'YYYY-MM-DD')), (3, 'Williams', 55, TO_DATE('2019-07-14', 'YYYY-MM-DD'));
SELECT dep_id, name, salary, SUM(salary) OVER (ORDER BY EXTRACT(YEAR FROM hire_date)) AS salary_sum FROM employees;
