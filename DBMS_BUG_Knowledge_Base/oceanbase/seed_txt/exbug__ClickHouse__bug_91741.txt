CREATE TABLE test_parallel_replicas (id UInt32, value String, timestamp DateTime) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_parallel_replicas SELECT number, toString(number) AS value, now() FROM dual WHERE number < 100;
CREATE TABLE test_parallel_replicas_2 (id UInt32, value Nullable(String), timestamp DateTime) ENGINE = MergeTree() ORDER BY id;
INSERT INTO test_parallel_replicas_2 SELECT number, CASE WHEN number % 2 = 0 THEN toString(number) ELSE NULL END AS value, now() FROM dual WHERE number < 100;
SELECT COUNT(*) FROM test_parallel_replicas WHERE value IS NOT NULL AND timestamp > NOW() - INTERVAL '1' DAY;
