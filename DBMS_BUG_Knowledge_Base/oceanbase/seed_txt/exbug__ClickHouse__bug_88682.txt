CREATE TABLE c (a UInt64, b Nullable(UInt64));
INSERT INTO c SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END FROM dual CONNECT BY LEVEL <= 10;
CREATE TABLE d (m UInt64, n UInt64);
INSERT INTO d SELECT number, number + 1 FROM dual CONNECT BY LEVEL <= 5;
SELECT c.a, d.n FROM c LEFT OUTER JOIN d ON c.a > d.m;
