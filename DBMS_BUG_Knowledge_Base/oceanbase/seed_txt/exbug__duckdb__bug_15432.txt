CREATE TABLE test_data (id INT, values VARCHAR(255)[]);
INSERT INTO test_data VALUES (1, ARRAY['1', '2', '3']), (2, ARRAY['4', '5', NULL]), (3, ARRAY['6', '7', '8']);
SELECT id, SUM(CASE WHEN value IS NOT NULL THEN TO_NUMBER(value) ELSE 0 END) AS total FROM test_data, UNNEST(values) AS value GROUP BY id;
CREATE OR REPLACE FUNCTION custom_sum(lst VARCHAR(255)[]) RETURNS FLOAT AS (SELECT SUM(CASE WHEN value IS NOT NULL THEN TO_NUMBER(value) ELSE 0 END) FROM UNNEST(lst) AS value);
SELECT id, custom_sum(values) FROM test_data WHERE values IS NOT NULL GROUP BY id;
