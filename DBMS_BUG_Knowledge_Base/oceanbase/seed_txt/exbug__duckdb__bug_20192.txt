CREATE TABLE products (id INTEGER PRIMARY KEY, name VARCHAR(255), description VARCHAR(255), price DECIMAL(10, 2), created_at TIMESTAMP);
CREATE TABLE orders (id INTEGER PRIMARY KEY, product_id INTEGER, order_date TIMESTAMP, quantity INTEGER, large_blob BLOB, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, CONCAT('Product ', i) AS name, CONCAT('Description for product ', i) AS description, ROUND(RANDOM() * 100, 2) AS price, TIMESTAMPADD(DAY, i, '2024-01-01') AS created_at FROM (SELECT LEVEL AS i FROM dual CONNECT BY LEVEL <= 10000);
INSERT INTO orders SELECT i AS id, MOD(i, 10000) + 1 AS product_id, TIMESTAMPADD(DAY, i, '2024-02-01') AS order_date, MOD(i, 10) + 1 AS quantity, REPEAT('y', 2000) AS large_blob FROM (SELECT LEVEL AS i FROM dual CONNECT BY LEVEL <= 5000000);
EXPLAIN PLAN FOR SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01' AND '2024-04-01' AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
