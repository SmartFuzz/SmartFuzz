CREATE TABLE tbl1 (id INT, value VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO tbl1 SELECT number, CAST(number % 10 AS CHAR) FROM dual CONNECT BY LEVEL <= 100;
CREATE TABLE tbl2 (ref_id INT, description VARCHAR(255), amount DOUBLE) ENGINE = MergeTree() ORDER BY ref_id;
INSERT INTO tbl2 SELECT id, CONCAT('Description ', CAST(id AS CHAR)), id * 1.5 FROM tbl1;
SELECT tbl1.id, tbl2.amount FROM tbl1 JOIN tbl2 ON tbl1.id = tbl2.ref_id WHERE tbl1.value IS NOT NULL ORDER BY tbl1.id;
