DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id UInt32, accepted_at TIMESTAMP(3), delivery_report_at BLOB) ENGINE = OBServer;
INSERT INTO agg_test SELECT number AS id, TIMESTAMP '2025-09-24 00:00:00.000' + INTERVAL number SECOND AS accepted_at, argMaxState(toString(number), TIMESTAMP '2025-09-24 00:00:00.000' + INTERVAL number SECOND) AS delivery_report_at FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM dual CONNECT BY LEVEL <= 100);
SELECT id, finalizeAggregation(delivery_report_at) AS delivered_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, finalizeAggregation(delivery_report_at) AS delivered_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
