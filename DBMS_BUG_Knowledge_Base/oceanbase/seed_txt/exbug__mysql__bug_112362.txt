CREATE TABLE c (c_id INTEGER NOT NULL PRIMARY KEY, value VARCHAR(100) NOT NULL, UNIQUE KEY ix_value_unique (value));
CREATE TABLE d (d_id INTEGER NOT NULL PRIMARY KEY, value VARCHAR(100) NOT NULL, KEY ix_value (value));
INSERT INTO c (value) SELECT CONCAT('value_', seq) FROM (SELECT ROW_NUMBER() OVER () AS seq FROM dual CONNECT BY LEVEL <= 1000);
INSERT INTO d (value) SELECT CONCAT('value_', seq) FROM (SELECT ROW_NUMBER() OVER () AS seq FROM dual CONNECT BY LEVEL <= 1000000);
SELECT DISTINCT value FROM d WHERE value NOT IN (SELECT value FROM c);
