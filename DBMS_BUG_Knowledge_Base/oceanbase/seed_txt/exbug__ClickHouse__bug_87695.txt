CREATE TABLE t1 (id INT UNSIGNED, data VARCHAR(255)) ENGINE = OBStore;
INSERT INTO t1 VALUES (1, '{"key":"value1"}'), (2, '{"key":"value2"}'), (3, NULL);
CREATE TABLE t2 (id INT UNSIGNED, json_data VARCHAR(255)) ENGINE = OBStore;
INSERT INTO t2 VALUES (1, '{"c1":1}'), (2, '{"c1":2}'), (3, NULL);
SELECT COUNT(*) FROM t1 WHERE data = '{"key":"value1"}';
SELECT COUNT(*) FROM t2 WHERE json_data = '{"c1":1}';
