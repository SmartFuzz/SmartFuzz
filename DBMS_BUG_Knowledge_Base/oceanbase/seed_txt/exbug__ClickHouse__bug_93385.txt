CREATE TABLE user_events (event_id VARCHAR(36), user_id INT UNSIGNED, event_type VARCHAR(20), event_time TIMESTAMP) PARTITION BY HASH(user_id) PARTITIONS 4;
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT UUID() AS event_id, FLOOR(RAND() * 10000) AS user_id, CASE FLOOR(RAND() * 3) WHEN 0 THEN 'click' WHEN 1 THEN 'view' ELSE 'purchase' END AS event_type, NOW() - INTERVAL FLOOR(RAND() * 3600 * 24) SECOND AS event_time FROM DUAL LIMIT 100000;
CREATE TABLE user_summary (user_id INT UNSIGNED, event_count INT UNSIGNED) PARTITION BY HASH(user_id) PARTITIONS 4;
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
