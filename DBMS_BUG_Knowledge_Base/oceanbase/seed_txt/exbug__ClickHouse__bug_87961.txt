CREATE TABLE IF NOT EXISTS user_activity (activity_id UInt64, user_id UInt32, activity_time DateTime, activity_type VARCHAR(20), details VARCHAR(255)) PARTITION BY EXTRACT(YEAR_MONTH FROM activity_time) ORDER BY (user_id, activity_time);
INSERT INTO user_activity SELECT number AS activity_id, MOD(rand(), 1000) AS user_id, NOW() + number * INTERVAL '5' SECOND AS activity_time, (CASE WHEN MOD(number, 3) = 0 THEN 'login' WHEN MOD(number, 3) = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, CONCAT('Activity details for ', CAST(number AS CHAR)) AS details FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 100);
CREATE TABLE IF NOT EXISTS product_sales (sale_id UInt32, product_id UInt32, sale_time DateTime, quantity UInt16, price FLOAT) PARTITION BY EXTRACT(YEAR_MONTH FROM sale_time) ORDER BY (product_id, sale_time);
INSERT INTO product_sales SELECT number AS sale_id, MOD(rand(), 500) AS product_id, NOW() + number * INTERVAL '2' SECOND AS sale_time, (MOD(rand(), 10) + 1) AS quantity, (MOD(rand(), 100) + 1) * 1.5 AS price FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 50);
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL '1' DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
