CREATE TABLE T1 (K BIGINT, C0 VARCHAR(255), C1 VARCHAR(255), C2 DOUBLE) ENGINE=OBSERVER;
INSERT INTO T1 SELECT number, ARRAY_AGG(CAST(number * i AS VARCHAR(255))) AS C0, CASE WHEN number % 2 = 0 THEN 'even' ELSE NULL END AS C1, number * 1.5 AS C2 FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 100);
CREATE TABLE T2 (K BIGINT, C0 VARCHAR(255), C1 INT) ENGINE=OBSERVER;
INSERT INTO T2 SELECT K, C0, CAST(K AS INT) FROM T1 WHERE C1 IS NOT NULL;
SELECT * FROM T1 WHERE C1 = 'even';
