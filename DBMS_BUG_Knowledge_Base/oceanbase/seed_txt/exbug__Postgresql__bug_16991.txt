CREATE TABLE test_table (id BIGINT PRIMARY KEY, name VARCHAR(255) NOT NULL, value INT CHECK (value >= 0));
CREATE INDEX CaseSensitiveIndex ON test_table (name, value);
CREATE INDEX not_case_sensitive ON test_table (value, id);
SELECT OBTAIN_TABLE_SIZE(indexrelname) FROM OBTAIN_INDEX_STATISTICS WHERE indexrelname = 'not_case_sensitive';
SELECT OBTAIN_TABLE_SIZE(indexrelname) FROM OBTAIN_INDEX_STATISTICS WHERE indexrelname = 'CaseSensitiveIndex';
