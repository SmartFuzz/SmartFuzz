CREATE TABLE demo.dict_test (code INT, value VARCHAR(255)) PRIMARY KEY (code);
CREATE TABLE demo.test_local (code INT, name VARCHAR(255) DEFAULT 'unknown') ENGINE = ReplicatedMergeTree() ORDER BY code;
