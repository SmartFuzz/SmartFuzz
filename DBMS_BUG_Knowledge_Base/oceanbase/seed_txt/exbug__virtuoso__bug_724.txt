CREATE TABLE images (id INT PRIMARY KEY, img_url VARCHAR(255) NOT NULL, description TEXT);
CREATE TABLE audio (id INT PRIMARY KEY, audio_url VARCHAR(255) NOT NULL, duration INT CHECK (duration >= 0));
