CREATE TABLE test_partitioned (id INT, dt TIMESTAMP, value FLOAT) PARTITION BY EXTRACT(YEAR_MONTH FROM dt) ORDER BY (dt);
INSERT INTO test_partitioned SELECT number, DATE_ADD('2015-01-01', INTERVAL (number % 3650) DAY) AS dt, RAND() FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 10000000);
CREATE TABLE test_non_partitioned (id INT, dt TIMESTAMP, value FLOAT) ORDER BY (dt);
INSERT INTO test_non_partitioned SELECT number, DATE_ADD('2015-01-01', INTERVAL (number % 3650) DAY) AS dt, RAND() FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 10000000);
SELECT dt FROM test_partitioned WHERE value IS NOT NULL ORDER BY dt LIMIT 20;
