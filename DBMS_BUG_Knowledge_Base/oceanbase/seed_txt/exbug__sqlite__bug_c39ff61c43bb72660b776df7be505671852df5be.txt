CREATE TABLE users (id BIGINT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0));
CREATE TABLE comments (id BIGINT PRIMARY KEY, user_id BIGINT, content VARCHAR(255) NOT NULL, CONSTRAINT fk_user FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE);
