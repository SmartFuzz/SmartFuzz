CREATE TABLE t(p BOOL, q INT, r VARCHAR(255) DEFAULT 'default');
INSERT INTO t(p, q) VALUES (NULL, 1);
SELECT ob_advisory_lock(2);
WITH lock_t AS (SELECT ROWID FROM t WHERE p IS NULL FOR UPDATE) UPDATE t SET q = q + 1 WHERE ROWID IN (SELECT l.ROWID FROM lock_t l) RETURNING p, q;
WITH lock_t AS (SELECT ROWID FROM t FOR UPDATE) UPDATE t SET r = 'updated' WHERE ROWID IN (SELECT l.ROWID FROM lock_t l) RETURNING r;
