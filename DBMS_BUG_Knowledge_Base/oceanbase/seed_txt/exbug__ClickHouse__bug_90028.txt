CREATE TABLE t1 (c0 VARBINARY(5), c1 INT, c2 VARCHAR(255));
INSERT INTO t1 (c0, c1, c2) SELECT CAST(CONCAT(CAST(number % 10 AS CHAR), ',', CAST((-number) AS CHAR)) AS VARBINARY), number, 'test' FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM dual CONNECT BY LEVEL <= 20);
CREATE TABLE t2 (c0 INT, c1 FLOAT, c2 TIMESTAMP);
INSERT INTO t2 (c0, c1, c2) SELECT number, NULL, NOW() FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM dual CONNECT BY LEVEL <= 10);
SELECT t1.c0, t1.c1 FROM t1 JOIN t2 ON t1.c1 = t2.c0 ORDER BY t1.c0, t1.c1;
