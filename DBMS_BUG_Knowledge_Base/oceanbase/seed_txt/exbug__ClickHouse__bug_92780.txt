CREATE TABLE test_data (id INT UNSIGNED, ts TIMESTAMP, value VARCHAR(255), status VARCHAR(255)) PARTITION BY RANGE (YEAR(ts)) SUBPARTITION BY HASH (id) ENABLE ROW LEVEL LOCK;
INSERT INTO test_data SELECT number, NOW() + INTERVAL number SECOND, CAST(number % 5 AS CHAR), CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM (SELECT ROWNUM AS number FROM dual CONNECT BY LEVEL <= 1000000);
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > NOW() - INTERVAL '1' DAY AND value LIKE '1%';
SELECT DISTINCT id FROM test_data WHERE ts BETWEEN '2023-01-01 00:00:00' AND '2023-12-31 23:59:59' AND status IS NOT NULL;
