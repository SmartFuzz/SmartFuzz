CREATE TABLE test_large_lists (id INTEGER, gts ARRAY(ARRAY(UINT8)));
INSERT INTO test_large_lists SELECT i, ARRAY(ARRAY(CAST(MOD(RAND(), 256) AS UINT8), CAST(MOD(RAND(), 256) AS UINT8))) FROM (SELECT ROW_NUMBER() OVER () AS i FROM dual CONNECT BY LEVEL <= 500000);
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, gts[1] AS inner_list FROM list_view;
