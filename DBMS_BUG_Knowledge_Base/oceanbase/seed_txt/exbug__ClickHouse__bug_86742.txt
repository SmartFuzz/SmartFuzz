DROP TABLE IF EXISTS job_sample;
CREATE TABLE job_sample (id UNSIGNED INT, posted_at DATE, discovered_at TIMESTAMP(3), url VARCHAR(255), title VARCHAR(255), company_name VARCHAR(255), description VARCHAR(255));
INSERT INTO job_sample SELECT * FROM OB_DATA_SOURCE('https://media.theirstack.com/jobs/jan-2025-sample.parquet', 'Parquet');
ALTER TABLE job_sample ADD COLUMN description_matches_python TINYINT GENERATED ALWAYS AS (CASE WHEN description REGEXP 'Python' THEN 1 ELSE 0 END);
UPDATE job_sample SET description_matches_python = CASE WHEN description REGEXP '(?i)Python' THEN 1 ELSE 0 END;
