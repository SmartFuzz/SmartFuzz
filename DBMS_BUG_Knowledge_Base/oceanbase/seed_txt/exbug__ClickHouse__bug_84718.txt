DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id UInt32, name String, value Float64, created_at DateTime) ENGINE = ObRowStore ORDER BY id;
INSERT INTO test_table SELECT number, CONCAT('name_', TO_CHAR(number), ' ', TO_CHAR(number % 10)) AS name, number * 1.5 AS value, NOW() FROM (SELECT LEVEL AS number FROM dual CONNECT BY LEVEL <= 100);
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
