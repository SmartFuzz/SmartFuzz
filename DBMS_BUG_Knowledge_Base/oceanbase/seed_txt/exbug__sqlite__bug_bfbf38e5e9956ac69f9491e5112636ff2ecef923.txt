CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0));
CREATE TABLE posts (id INT PRIMARY KEY, user_id INT, content VARCHAR(255), FOREIGN KEY(user_id) REFERENCES users(id));
