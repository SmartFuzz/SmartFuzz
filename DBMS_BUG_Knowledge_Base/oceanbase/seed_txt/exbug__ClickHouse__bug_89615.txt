INSERT INTO set_idx_rand_v2 SELECT number, CAST(number % 10 AS CHAR) FROM dual CONNECT BY LEVEL <= 100;

SELECT * FROM set_idx_rand_v2 WHERE d IS NULL OR d > NOW() - INTERVAL '1' DAY;
