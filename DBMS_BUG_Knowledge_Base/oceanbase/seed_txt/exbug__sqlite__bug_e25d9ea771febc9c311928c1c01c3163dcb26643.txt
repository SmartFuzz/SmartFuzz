CREATE TABLE Users (id BIGINT PRIMARY KEY, name VARCHAR(255) NOT NULL, email VARCHAR(255) UNIQUE);
CREATE TABLE Comments (id BIGINT PRIMARY KEY, user_id BIGINT, content VARCHAR(255), FOREIGN KEY (user_id) REFERENCES Users(id) ON DELETE CASCADE);
