SELECT json_data AS json_data, SUBSTRING(json_data, POSITION('"id":' IN json_data) + 6, POSITION(',' IN json_data) - POSITION('"id":' IN json_data) - 7) AS id, SUBSTRING(json_data, POSITION('"parent":' IN json_data) + 11, POSITION('}' IN json_data) - POSITION('"parent":' IN json_data) - 11) AS parent FROM test_data WHERE json_data IS NOT NULL ORDER BY id;
