CREATE TABLE test_regex (id INTEGER PRIMARY KEY, pattern VARCHAR(255), test_string VARCHAR(255));
INSERT INTO test_regex (id, pattern, test_string) VALUES (1, 'bc', 'abcd'), (2, 'a.c', 'abcd'), (3, 'a.*d', 'abcd'), (4, '(b|x)', 'abcd'), (5, '^a', 'abcd'), (6, '^(b|c)', 'abcd');
SELECT id, CASE WHEN REGEXP_LIKE(test_string, pattern) THEN 'true' ELSE 'false' END AS match_result FROM test_regex;
SELECT id, CASE WHEN REGEXP_LIKE(test_string, 'a.*d') THEN 'true' ELSE 'false' END AS match_result FROM test_regex WHERE pattern = 'a.*d';
SELECT id, CASE WHEN REGEXP_LIKE(test_string, '^(b|c)') THEN 'true' ELSE 'false' END AS match_result FROM test_regex WHERE test_string IS NOT NULL;
