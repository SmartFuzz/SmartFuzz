CREATE TABLE users (UserID integer PRIMARY KEY, Username varchar(255) NOT NULL, CreatedAt timestamp);

CREATE TABLE orders (OrderID integer PRIMARY KEY, UserID integer, Amount decimal(10, 2), OrderDate timestamp, FOREIGN KEY (UserID) REFERENCES users(UserID));
INSERT INTO orders(OrderID, UserID, Amount, OrderDate) VALUES (1, 1, 100.50, '2023-01-01 10:30:00'), (2, 1, 150.75, '2023-01-02 10:30:00'), (3, 2, 200.00, '2023-01-03 10:30:00');
WITH latest_orders AS (SELECT UserID, Amount FROM orders WHERE Amount IS NOT NULL ORDER BY UserID, OrderDate DESC) SELECT u.Username, lo.Amount FROM users u JOIN (SELECT DISTINCT UserID, Amount FROM latest_orders) lo ON u.UserID = lo.UserID ORDER BY u.Username;
