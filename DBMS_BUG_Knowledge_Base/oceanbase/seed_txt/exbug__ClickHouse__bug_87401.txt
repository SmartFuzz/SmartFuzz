DROP TABLE IF EXISTS b_products;
CREATE TABLE b_products (product_id BIGINT, product_name VARCHAR(255), price DOUBLE) ENGINE=OBTree;
INSERT INTO b_products SELECT number, CONCAT('Product_', number), number * 10.5 FROM dual CONNECT BY LEVEL <= 100;
DROP TABLE IF EXISTS b_sales;
CREATE TABLE b_sales (sale_id BIGINT, product_id BIGINT, quantity INT) ENGINE=OBTree;
