CREATE TABLE ranges(x_min decimal(8,5), y_min decimal(7,5), width integer, nb integer);
CREATE TABLE trips(tripid bigint not null, x decimal(8,5) not null, y decimal(8,5) not null, time bigint not null);
INSERT INTO ranges(x_min, y_min, width, nb) VALUES (1.23456, 2.34567, 10, 0);
INSERT INTO trips(tripid, x, y, time) VALUES (1, 1.5, 2.5, 1622548800);
UPDATE ranges SET nb = (SELECT COUNT(*) FROM trips T WHERE T.x BETWEEN x_min AND x_min + width AND T.y BETWEEN y_min AND y_min + width);
