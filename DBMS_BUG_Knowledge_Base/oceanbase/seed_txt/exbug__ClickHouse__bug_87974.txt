DROP TABLE IF EXISTS test_left;
DROP TABLE IF EXISTS test_right;
CREATE TABLE test_left (a BIGINT, b VARCHAR(255), c VARCHAR(255), d INT) PARTITION BY HASH(a) PARTITIONS 4;
CREATE TABLE test_right (a BIGINT, b VARCHAR(255), c VARCHAR(255), d INT) PARTITION BY HASH(a) PARTITIONS 4;
INSERT INTO test_left SELECT MOD(number, 10000), CAST(MOD(number, 10000) AS CHAR), CAST(MOD(number, 10000) AS CHAR), MOD(number, 5) FROM dual CONNECT BY LEVEL <= 100000;
INSERT INTO test_right SELECT MOD(number, 10), CAST(MOD(number, 10) AS CHAR), CAST(MOD(number, 10) AS CHAR), NULL FROM dual CONNECT BY LEVEL <= 10000;
