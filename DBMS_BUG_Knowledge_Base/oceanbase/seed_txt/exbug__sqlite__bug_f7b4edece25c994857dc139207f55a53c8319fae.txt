CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT CHECK(age >= 0), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id INT PRIMARY KEY, user_id INT, amount FLOAT NOT NULL, order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
