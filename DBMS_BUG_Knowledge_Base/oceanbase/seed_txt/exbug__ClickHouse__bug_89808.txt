CREATE DATABASE nyc_taxi;
CREATE TABLE nyc_taxi.trips_large (
    trip_id             UInt32,
    pickup_datetime     DateTime,
    dropoff_datetime    DateTime,
    pickup_longitude    Float64,
    pickup_latitude     Float64,
    dropoff_longitude   Float64,
    dropoff_latitude    Float64,
    passenger_count     UInt8,
    trip_distance       Float32,
    fare_amount         Float32,
    extra               Float32,
    tip_amount          Float32,
    tolls_amount        Float32,
    total_amount        Float32,
    payment_type        VARCHAR,
    pickup_ntaname      VARCHAR,
    dropoff_ntaname     VARCHAR
);
INSERT INTO nyc_taxi.trips_large
SELECT
    trip_id,
    pickup_datetime,
    dropoff_datetime,
    pickup_longitude,
    pickup_latitude,
    dropoff_longitude,
    dropoff_latitude,
    passenger_count,
    trip_distance,
    fare_amount,
    extra,
    tip_amount,
    tolls_amount,
    total_amount,
    payment_type,
    pickup_ntaname,
    dropoff_ntaname
FROM gcs(
    'https://storage.googleapis.com/clickhouse-public-datasets/nyc-taxi/trips_{0..2}.gz',
    'TabSeparatedWithNames'
);
CREATE TABLE nyc_taxi.agg_h3_10_large (
    pickup_latitude Float64,
    pickup_longitude Float64,
    h3_cell UInt64,
    cnt BIGINT,
    pickup_ntaname_uniq VARCHAR,
    pickup_ntaname_top_app VARCHAR
);
INSERT INTO nyc_taxi.agg_h3_10_large
SELECT
    pickup_latitude,
    pickup_longitude,
    h3_cell,
    COUNT(*) AS cnt,
    NULL AS pickup_ntaname_uniq,
    NULL AS pickup_ntaname_top_app
FROM nyc_taxi.trips_large
GROUP BY pickup_latitude, pickup_longitude, h3_cell;
SELECT
    pickup_latitude,
    pickup_longitude,
    cnt
FROM nyc_taxi.agg_h3_10_large
WHERE cnt > 1000;
