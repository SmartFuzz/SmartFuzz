DROP TABLE IF EXISTS products;
CREATE TABLE products (product_id INTEGER PRIMARY KEY, name VARCHAR(255) NOT NULL, price DECIMAL(10,2) CHECK(price >= 0), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO products VALUES (1, 'Widget', 19.99, '2023-01-01 10:00:00'), (2, 'Gadget', 29.99, '2023-02-01 11:00:00'), (3, 'Thingamajig', 39.99, '2023-03-01 12:00:00');
SELECT p.product_id, p.name, p.price FROM products p WHERE p.price > (SELECT AVG(price) FROM products) AND p.created_at < CURRENT_TIMESTAMP;
WITH recent_products AS (SELECT * FROM products WHERE created_at >= '2023-01-01 00:00:00') SELECT rp.product_id FROM recent_products rp WHERE EXISTS (SELECT 1 FROM recent_products WHERE price < rp.price);
