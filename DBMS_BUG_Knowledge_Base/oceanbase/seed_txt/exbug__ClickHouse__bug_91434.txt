CREATE TABLE default.json_test_v2 (json JSON, col_b Int64, col_c String);
INSERT INTO default.json_test_v2 SELECT JSON_OBJECT('a', number, 'b', number * 2, 'c', toString(number * 3)) FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 10000000);
SELECT col_b, col_c FROM default.json_test_v2 WHERE col_b IS NOT NULL;
SELECT json->>'b' AS direct_b, json->>'c' AS direct_c FROM default.json_test_v2 WHERE json->>'b' > '5000000';
SELECT col_b, COUNT(*) FROM default.json_test_v2 GROUP BY col_b HAVING col_b < 100;
