CREATE TABLE test_data (id INT PRIMARY KEY, value INT, name VARCHAR(50), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_data (id, value, name) VALUES (1, 10, 'A'), (2, NULL, 'B'), (3, 30, 'C');
CREATE OR REPLACE FUNCTION python_avg(value INT) RETURNS FLOAT AS $$ BEGIN RETURN (SELECT AVG(value) FROM test_data WHERE value IS NOT NULL);
END;
$$ LANGUAGE SQL;
SELECT python_avg(value) FROM test_data GROUP BY name;
