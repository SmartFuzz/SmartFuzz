CREATE TABLE employees (dep VARCHAR(10) NOT NULL, name VARCHAR(20) NOT NULL, salary DOUBLE CHECK (salary > 0), hire_date DATE DEFAULT NULL);
INSERT INTO employees (dep, name, salary, hire_date) VALUES ('mgmt', 'Smith', 81, TO_DATE('2019-10-08', 'YYYY-MM-DD'));

SELECT dep, name, salary, ROW_NUMBER() OVER (ORDER BY hire_date DESC) AS ntile FROM employees;
