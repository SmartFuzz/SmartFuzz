CREATE OR REPLACE FUNCTION json_each(input JSON) RETURNS TABLE (key VARCHAR, value JSON) AS $$ BEGIN RETURN QUERY SELECT CASE WHEN JSON_TYPE(val) = 'ARRAY' THEN idx::VARCHAR ELSE JSON_KEYS(val) END AS key, JSON_EXTRACT(val, key) AS value FROM (SELECT input AS val, generate_series(0, JSON_ARRAY_LENGTH(val) - 1) AS idx) AS subquery;
END;
$$ LANGUAGE plpgsql;
SELECT * FROM (SELECT '[{"key": "value1"}, {"key": "value2"}]' AS message) AS msg CROSS JOIN json_each(message);
