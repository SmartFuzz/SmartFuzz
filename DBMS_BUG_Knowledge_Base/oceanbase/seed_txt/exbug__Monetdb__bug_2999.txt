CREATE TABLE test_table(id INT PRIMARY KEY, value VARCHAR(50), order_num INT NOT NULL);

CREATE TABLE order_table(order_id INT, order_value FLOAT, CONSTRAINT fk_order FOREIGN KEY(order_id) REFERENCES test_table(id));
