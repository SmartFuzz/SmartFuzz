DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR(255), event_time TIMESTAMP, id BIGINT, msg VARCHAR(255), value FLOAT) PARTITION BY EXTRACT(YEAR FROM event_time), EXTRACT(MONTH FROM event_time), EXTRACT(DAY FROM event_time);
INSERT INTO repro.events SELECT CAST(number % 100 AS VARCHAR(255)) AS tenant, NOW() - INTERVAL (number % 200000) SECOND AS event_time, number AS id, CONCAT('m', CAST(number AS VARCHAR(255))) AS msg, CAST(number % 1000 AS FLOAT) AS value FROM dual CONNECT BY LEVEL <= 2000000;
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
