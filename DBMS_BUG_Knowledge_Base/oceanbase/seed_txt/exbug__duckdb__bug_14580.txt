CREATE OR REPLACE FUNCTION test_subfact(num INT) RETURNS TABLE (n INT, sub BIGINT) AS (WITH RECURSIVE cte(n, psub, sub) AS (SELECT 0 AS n, 1 AS psub, 1 AS sub UNION ALL SELECT n + 1 AS n, sub AS psub, (n * (sub + psub)) AS sub FROM cte WHERE n < num) SELECT n, sub FROM cte);
