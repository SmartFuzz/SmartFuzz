CREATE TABLE t_199999 (s1 VARCHAR(255), s2 VARCHAR(255), s3 INT CHECK (s3 >= 0));
CREATE TABLE t_200000 (s1 VARCHAR(255), s2 VARCHAR(255), s3 INT CHECK (s3 >= 0));
INSERT INTO t_199999 (s1, s2, s3) SELECT CONCAT('value', i), CONCAT('value', i), i FROM (SELECT ROW_NUMBER() OVER () AS i FROM dual CONNECT BY LEVEL <= 199999);
INSERT INTO t_200000 (s1, s2, s3) SELECT CONCAT('value', i), CONCAT('value', i), i FROM (SELECT ROW_NUMBER() OVER () AS i FROM dual CONNECT BY LEVEL <= 200000);
SELECT count(1) FROM t_200000 WHERE similarity(s1, s2) > 0.5 AND s3 IS NOT NULL;
