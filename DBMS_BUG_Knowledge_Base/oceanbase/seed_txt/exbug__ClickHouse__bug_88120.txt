CREATE TABLE IF NOT EXISTS logs (id INT UNSIGNED, date DATE, host_id INT UNSIGNED, message VARCHAR(255), status VARCHAR(255), value DOUBLE, count BIGINT UNSIGNED, created_at TIMESTAMP, updated_at TIMESTAMP, user_id INT UNSIGNED, error_code INT, description VARCHAR(255), tags JSON, metadata JSON, priority ENUM('low', 'medium', 'high'), session_id CHAR(36), ip_address VARCHAR(15), region VARCHAR(255), duration INT UNSIGNED, response_time FLOAT, is_active TINYINT) PARTITION BY EXTRACT(YEAR_MONTH FROM date) ORDER BY (date, host_id);
