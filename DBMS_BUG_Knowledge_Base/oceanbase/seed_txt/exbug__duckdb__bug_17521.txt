CREATE TABLE t2 (v1 INT, v2 INT, v4 VARCHAR(255), v5 FLOAT);
INSERT INTO t1 SELECT i, i, MOD(i, 2) FROM (SELECT LEVEL - 1 AS i FROM dual CONNECT BY LEVEL <= 1001);
INSERT INTO t2 SELECT i, i, CONCAT('value', i), i * 1.0 FROM (SELECT LEVEL - 1 AS i FROM dual CONNECT BY LEVEL <= 10000001);
