DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id UInt32, keyword_id UInt32, search_date DateTime, position_on_serp UInt32, version UInt32, value String) PARTITION BY EXTRACT(YEAR_MONTH FROM search_date) ORDER BY (organization_id, keyword_id, search_date, position_on_serp);
INSERT INTO test_final_partitions SELECT MOD(RAND(), 1000) AS organization_id, MOD(RAND(), 100000) AS keyword_id, TIMESTAMPADD(SECOND, MOD(RAND(), 86400 * 180), '2023-01-01 00:00:00') AS search_date, MOD(RAND(), 50) AS position_on_serp, MOD(number, 5) AS version, CAST(number AS CHAR) FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 50000000);
SELECT COUNT(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT COUNT(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
