DROP DATABASE IF EXISTS test_db;
CREATE DATABASE test_db;
CREATE TABLE test_db.sample_table (id BIGINT, name VARCHAR(255), created_at TIMESTAMP, score FLOAT) PRIMARY KEY (id);
CREATE VIEW test_db.sample_view AS SELECT id, name, created_at FROM test_db.sample_table WHERE score > 50.0;
INSERT INTO test_db.sample_table SELECT number, CONCAT('Name', CAST(number AS CHAR)), NOW(), number % 100 FROM (SELECT ROW_NUMBER() OVER () AS number FROM dual CONNECT BY LEVEL <= 100);
