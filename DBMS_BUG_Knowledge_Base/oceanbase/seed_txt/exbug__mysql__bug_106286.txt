CREATE TABLE test (id INT PRIMARY KEY, ts TIMESTAMP NULL);
INSERT INTO test (id, ts) VALUES (1, '2023-01-01 00:00:00');
CREATE TRIGGER tr_bi_check_uniqueness_with_nulls BEFORE INSERT ON test FOR EACH ROW BEGIN DECLARE new_ts TIMESTAMP;
SET new_ts = NEW.ts;
IF EXISTS ( SELECT NULL FROM test WHERE test.id = NEW.id AND test.ts = new_ts ) THEN SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Duplicated values not allowed.';
END IF;
