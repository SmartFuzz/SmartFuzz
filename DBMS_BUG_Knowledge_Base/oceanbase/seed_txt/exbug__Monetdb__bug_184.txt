CREATE TABLE artists (id INT PRIMARY KEY, name VARCHAR(255), age INT CHECK (age >= 0));
CREATE TABLE songs (id INT PRIMARY KEY, title VARCHAR(255), artist_id INT, FOREIGN KEY (artist_id) REFERENCES artists(id));
