CREATE TABLE mixed_timestamps (id INTEGER, time_column VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO mixed_timestamps (id, time_column) VALUES (1, '2021-10-11T18:59:14Z'), (2, '2021-10-11T12:55Z'), (3, '2021-10-11T12:55:00');
CREATE TABLE parsed_timestamps AS SELECT id, time_column, CASE WHEN time_column IS NOT NULL AND REGEXP_LIKE(time_column, '^[0-9]{4}-[0-9]{2}-[0-9]{2}T[0-9]{2}:[0-9]{2}Z?$') THEN CAST(time_column AS TIMESTAMP) END AS parsed_time FROM mixed_timestamps;
SELECT id, time_column, parsed_time FROM parsed_timestamps WHERE parsed_time IS NULL;
SELECT COUNT(*) FROM parsed_timestamps WHERE parsed_time IS NOT NULL;
