DROP TABLE IF EXISTS test_lwu;
CREATE TABLE test_lwu (a BIGINT, b VARCHAR(255), c FLOAT DEFAULT 0.0);
INSERT INTO test_lwu SELECT number, CASE WHEN number % 5 = 0 THEN NULL ELSE CAST(number % 5 AS VARCHAR(255)) END, number / 2.0 FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM dual CONNECT BY LEVEL <= 100);
