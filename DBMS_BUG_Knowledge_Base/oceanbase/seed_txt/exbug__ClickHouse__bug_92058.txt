CREATE TABLE dst (e ENUM('a', 'b'), f INT) ENGINE = MEMORY;
CREATE TABLE src (e ENUM('a'), g INT NULL) ENGINE = MEMORY;
CREATE MATERIALIZED VIEW mv TO dst (e ENUM('a', 'b'), f INT) AS SELECT e, g FROM src;
INSERT INTO src SELECT CASE WHEN number % 1 = 0 THEN 'a' ELSE NULL END AS e, number % 10 AS g FROM dual CONNECT BY LEVEL <= 10;
INSERT INTO dst SELECT e, g FROM mv WHERE g IS NOT NULL;
