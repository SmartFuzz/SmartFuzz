DROP TABLE IF EXISTS demo_ts;
CREATE TABLE demo_ts (id INT UNSIGNED, labels VARCHAR(255));
INSERT INTO demo_ts VALUES (1, '{"http.status_code":"200"}'), (2, '{"http.status_code":"503"}'), (3, '{"http.status_code":"404"}'), (4, '{"http.status_code":"208"}');
SELECT COUNT(*) FROM demo_ts WHERE JSON_CONTAINS(labels, 'http.status_code') AND CAST(JSON_UNQUOTE(JSON_EXTRACT(labels, '$.http.status_code')) AS UNSIGNED) > 100;
SELECT id, JSON_UNQUOTE(JSON_EXTRACT(labels, '$.http.status_code')) AS sc FROM demo_ts WHERE JSON_UNQUOTE(JSON_EXTRACT(labels, '$.http.status_code')) IS NOT NULL ORDER BY sc;
