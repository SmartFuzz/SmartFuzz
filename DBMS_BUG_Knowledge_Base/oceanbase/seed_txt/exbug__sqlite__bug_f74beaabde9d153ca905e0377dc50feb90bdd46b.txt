CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(255) NOT NULL, age INT);
CREATE TABLE posts (id INT PRIMARY KEY, user_id INT, content VARCHAR(255), CONSTRAINT fk_user FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id INT PRIMARY KEY, post_id INT, user_id INT, comment VARCHAR(255), CONSTRAINT fk_post FOREIGN KEY(post_id) REFERENCES posts(id), CONSTRAINT fk_user_comment FOREIGN KEY(user_id) REFERENCES users(id));
