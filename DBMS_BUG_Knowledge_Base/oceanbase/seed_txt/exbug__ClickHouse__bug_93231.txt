DROP DATABASE IF EXISTS dtobs;
CREATE DATABASE dtobs;
CREATE TABLE span_local (links Array(VARCHAR(255)), dummy TINYINT) ENGINE = MERGE TREE ORDER BY (dummy);
CREATE TABLE span_link_local (linked_id VARCHAR(255)) ENGINE = MERGE TREE ORDER BY (linked_id);
CREATE MATERIALIZED VIEW mv_bad AS SELECT link.trace_id AS linked_id FROM (SELECT UNNEST(links) AS link FROM span_local) WHERE link.trace_id IS NOT NULL;
ALTER TABLE span_local DROP COLUMN dummy;
