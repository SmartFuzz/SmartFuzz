CREATE TABLE test_data (id INT, name VARCHAR(255), category VARCHAR(255));
INSERT INTO test_data VALUES (1, '大熊猫', '熊'), (2, '大熊猫', '猫'), (3, '小熊猫', '猫');
SELECT category, GROUP_CONCAT(DISTINCT name) AS a_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
SELECT category, GROUP_CONCAT(name) AS b_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
SELECT category, JSON_ARRAYAGG(JSON_OBJECT('id', id, 'name', name, 'category', category)) AS c_list FROM test_data GROUP BY category;
