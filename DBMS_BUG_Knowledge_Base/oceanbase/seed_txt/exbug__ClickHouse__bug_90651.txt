CREATE DATABASE IF NOT EXISTS test_db;
CREATE TABLE IF NOT EXISTS test_distr (num INT, date TIMESTAMP, value FLOAT DEFAULT 0.0) ENGINE = OBDistributed('{cluster}', 'default', 'test');
INSERT INTO test_distr SELECT number AS num, NOW() AS date, number * 1.5 AS value FROM dual WHERE number < 100;
CREATE TABLE IF NOT EXISTS replicated_table (num BIGINT, date DATE, metric FLOAT) ENGINE = OBReplicatedMergeTree('/oceanbase/tables/{shard}/{database}.{table}', '{replica}') ORDER BY num;
INSERT INTO replicated_table SELECT num, date, (SELECT date FROM test_db.test_dict WHERE num = num) AS metric FROM test_distr WHERE num IS NOT NULL;
