DROP DATABASE IF EXISTS test;
CREATE DATABASE test;
CREATE TABLE IF NOT EXISTS test.mv_src_table (id UInt64, value VARCHAR(255));
INSERT INTO test.mv_src_table SELECT number, CAST(number AS VARCHAR(255)) FROM dual CONNECT BY LEVEL <= 10;
CREATE MATERIALIZED VIEW IF NOT EXISTS test.mv_refreshable AS SELECT max(id) AS id FROM test.mv_src_table WHERE value IS NOT NULL;
