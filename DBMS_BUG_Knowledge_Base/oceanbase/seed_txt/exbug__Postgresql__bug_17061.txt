DROP TABLE IF EXISTS tree;
CREATE TABLE tree(id INT PRIMARY KEY, parent_id INT, name TEXT NOT NULL, depth INT DEFAULT 0);
INSERT INTO tree(parent_id, name, depth) VALUES (NULL, 'Root', 0), (1, 'Child1', 1), (1, 'Child2', 1), (2, 'Grandchild1', 2), (2, 'Grandchild2', 2);
WITH RECURSIVE mtree(id, name, depth) AS (SELECT id, name, depth FROM tree WHERE id = 1 UNION ALL SELECT t.id, t.name, t.depth FROM tree AS t INNER JOIN mtree AS m ON t.parent_id = m.id) SELECT row_to_json(mtree) -> 'depth' AS depth FROM mtree;
