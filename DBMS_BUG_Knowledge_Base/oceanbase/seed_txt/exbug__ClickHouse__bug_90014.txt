CREATE TABLE position_evaluation_v2 (
    symbol VARCHAR(255),
    evaluated_at TIMESTAMP(3),
    mark_price DECIMAL(18, 8),
    volume INT UNSIGNED
);
INSERT INTO position_evaluation_v2 SELECT symbol, evaluated_at, mark_price, volume FROM (SELECT 'AAPL' AS symbol, NOW() AS evaluated_at, 150.12345678 AS mark_price, 100 AS volume) AS temp LIMIT 100;
SELECT mark_price, LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at) AS last_mark_price FROM position_evaluation_v2;
SELECT mark_price, LAG(mark_price, 1, NULL) OVER (PARTITION BY symbol ORDER BY evaluated_at) AS last_mark_price_with_default FROM position_evaluation_v2;
SELECT mark_price, LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at) AS last_mark_price_invalid FROM position_evaluation_v2;
