CREATE DATABASE test_fuzz;
CREATE TABLE test_fuzz.t_local (id Int32, c1 VARCHAR(255), c2 VARCHAR(255), d1 VARCHAR(255), d2 VARCHAR(255));
INSERT INTO test_fuzz.t_local (id, c1, c2, d1, d2) SELECT id, c1, c2, CONCAT(IFNULL(c1, 'NULL'), '-d1'), CONCAT(c2, '-d2') FROM (VALUES (1, 'id1-1', 'id2-1'), (2, NULL, 'id2-2'), (3, 'id1-1', NULL), (4, 'id1-2', 'id2-2')) AS t(id, c1, c2);
CREATE TABLE test_fuzz.t AS SELECT * FROM test_fuzz.t_local;
SELECT id, d1, d2 FROM test_fuzz.t ORDER BY d2;
