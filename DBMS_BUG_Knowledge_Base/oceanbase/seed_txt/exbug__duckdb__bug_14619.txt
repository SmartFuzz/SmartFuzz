CREATE TABLE test_table (id INT, values ARRAY<INT>);
INSERT INTO test_table VALUES (1, ARRAY[10, 20, 30]), (2, ARRAY[NULL, 40, 50]), (3, ARRAY[60, 70, NULL]);
SELECT id, ARRAY(SELECT ROW_NUMBER() OVER () - 1 FROM UNNEST(values)) AS transformed_ix FROM test_table;
SELECT id, MAX(ROW_NUMBER() OVER () - 1) AS reduced_ix FROM test_table WHERE values IS NOT NULL;
SELECT id, ARRAY(SELECT x + 1 FROM UNNEST(values) AS x) AS incremented_values FROM test_table WHERE values IS NOT NULL;
