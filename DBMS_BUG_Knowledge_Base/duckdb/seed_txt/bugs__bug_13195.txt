CREATE TABLE courses (course_id INTEGER PRIMARY KEY, course_name VARCHAR NOT NULL, credits INTEGER CHECK (credits > 0));
INSERT INTO courses VALUES (1, 'Mathematics', 4), (2, 'Physics', 3), (3, 'Chemistry', 4);
CREATE TABLE enrollments (enrollment_id INTEGER PRIMARY KEY, sid INTEGER, course_id INTEGER, FOREIGN KEY (sid) REFERENCES students(sid), FOREIGN KEY (course_id) REFERENCES courses(course_id));

EXPLAIN SELECT c.course_name FROM courses c JOIN enrollments e ON c.course_id = e.course_id WHERE e.sid IS NOT NULL;
