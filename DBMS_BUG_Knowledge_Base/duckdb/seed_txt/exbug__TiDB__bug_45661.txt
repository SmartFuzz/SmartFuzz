CREATE TABLE sales (year INTEGER, country VARCHAR(20), product VARCHAR(32), profit INTEGER);

SELECT year + 1 AS y, SUM(profit) AS profit, CASE WHEN year + 1 IS NOT NULL THEN 0 ELSE 1 END AS grouping_year FROM sales GROUP BY year + 1 WITH ROLLUP HAVING y > 2002 ORDER BY y, profit;
