CREATE TABLE users (id INTEGER PRIMARY KEY, name TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP);

CREATE TABLE orders (order_id INTEGER PRIMARY KEY, user_id INTEGER REFERENCES users(id), amount DECIMAL(10, 2) CHECK (amount > 0));
