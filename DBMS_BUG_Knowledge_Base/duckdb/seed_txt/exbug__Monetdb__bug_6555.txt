SELECT DISTINCT CASE WHEN (SELECT COUNT(*) FROM test_table) > 1 THEN COALESCE(SUM(value), 0) ELSE NULL END AS result FROM test_table WHERE value IS NOT NULL;
SELECT COUNT(CASE WHEN value IS NULL THEN 1 END) AS null_count, COUNT(CASE WHEN value IS NOT NULL THEN 1 END) AS not_null_count FROM test_table;
