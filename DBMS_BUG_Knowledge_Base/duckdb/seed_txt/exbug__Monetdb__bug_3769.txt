CREATE TABLE union_a (id BIGINT, mytimestamp TIMESTAMP, value VARCHAR);
CREATE TABLE union_b (id BIGINT, mytimestamp TIMESTAMP, value INT CHECK (value >= 0));
INSERT INTO union_a (id, mytimestamp, value) VALUES (1, '2023-10-01 10:00:00', 'test1');
INSERT INTO union_b (id, mytimestamp, value) VALUES (2, '2023-10-01 11:00:00', 100);
SELECT id, mytimestamp, CAST(value AS VARCHAR) AS alias_value FROM union_a UNION ALL SELECT id, mytimestamp, CAST(value AS VARCHAR) AS alias_value FROM union_b;
