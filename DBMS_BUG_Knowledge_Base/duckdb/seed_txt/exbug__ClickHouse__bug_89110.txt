CREATE TABLE IF NOT EXISTS initial_complex (n UInt32, description VARCHAR, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (n);
CREATE TABLE IF NOT EXISTS target1_complex (x UInt32, status VARCHAR, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (x);
CREATE TABLE IF NOT EXISTS target2_complex (x UInt32, value FLOAT, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP) PRIMARY KEY (x);
CREATE MATERIALIZED VIEW initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM initial_complex WHERE RANDOM() < 0.5;
CREATE MATERIALIZED VIEW target1_target2_mv_complex AS SELECT x, CAST(x AS FLOAT) * 1.5 AS value FROM target1_complex WHERE status IS NOT NULL;
