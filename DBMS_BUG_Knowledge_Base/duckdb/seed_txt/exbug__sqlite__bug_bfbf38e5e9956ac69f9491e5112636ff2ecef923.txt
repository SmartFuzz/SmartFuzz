CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR NOT NULL, age INTEGER CHECK(age >= 0));
CREATE TABLE posts (id INTEGER PRIMARY KEY, user_id INTEGER, content VARCHAR, FOREIGN KEY(user_id) REFERENCES users(id));

SELECT u.name, (SELECT COUNT(*) FROM posts p WHERE p.user_id = u.id) AS post_count FROM users u LEFT JOIN posts p ON u.id = p.user_id WHERE u.age IS NOT NULL;
