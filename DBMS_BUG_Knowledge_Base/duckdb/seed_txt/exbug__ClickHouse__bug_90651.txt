CREATE DATABASE IF NOT EXISTS test_db;
CREATE TABLE IF NOT EXISTS test_distr (num INT, date TIMESTAMP, value DOUBLE DEFAULT 0.0);
INSERT INTO test_distr SELECT number AS num, NOW() AS date, number * 1.5 AS value FROM generate_series(0, 99);
CREATE TABLE IF NOT EXISTS replicated_table (num BIGINT, date DATE, metric DOUBLE);
INSERT INTO replicated_table SELECT num, date, (SELECT date FROM test_distr WHERE num = replicated_table.num) AS metric FROM test_distr WHERE num IS NOT NULL;
