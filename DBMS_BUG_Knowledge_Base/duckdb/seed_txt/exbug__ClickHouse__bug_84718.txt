DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id INTEGER, name VARCHAR, value DOUBLE, created_at TIMESTAMP);
INSERT INTO test_table SELECT number, concat('name_', toString(number), ' ', toString(number % 10)), number * 1.5, now() FROM generate_series(0, 99) AS number;
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
