CREATE TABLE set_idx_rand_v2 (id INTEGER, s VARCHAR, d TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO set_idx_rand_v2 SELECT number, CAST(number % 10 AS VARCHAR) FROM range(100);
SELECT * FROM set_idx_rand_v2 WHERE MOD(RANDOM(), 1000) = 1;
SELECT * FROM set_idx_rand_v2 WHERE d IS NULL OR d > CURRENT_TIMESTAMP - INTERVAL '1 DAY';
SELECT COUNT(*) FROM set_idx_rand_v2 WHERE s IN (SELECT s FROM set_idx_rand_v2 WHERE id < 50) AND MOD(RANDOM(), 1000) = 1;
