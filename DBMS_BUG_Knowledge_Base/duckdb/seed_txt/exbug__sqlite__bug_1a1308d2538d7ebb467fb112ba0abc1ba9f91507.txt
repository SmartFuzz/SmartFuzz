CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR NOT NULL, age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE comments (id INTEGER PRIMARY KEY, user_id INTEGER, content VARCHAR, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));

SELECT u.name, (SELECT COUNT(*) FROM comments c WHERE c.user_id = u.id) AS comment_count FROM users u WHERE u.age > (SELECT AVG(age) FROM users) ORDER BY comment_count DESC;
