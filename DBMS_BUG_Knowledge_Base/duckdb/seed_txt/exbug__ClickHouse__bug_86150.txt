DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id INTEGER, keyword_id INTEGER, search_date TIMESTAMP, position_on_serp INTEGER, version INTEGER, value VARCHAR);
INSERT INTO test_final_partitions SELECT (RANDOM() % 1000) AS organization_id, (RANDOM() % 100000) AS keyword_id, TIMESTAMP '2023-01-01 00:00:00' + (RANDOM() % (86400*180)) AS search_date, (RANDOM() % 50) AS position_on_serp, (ROW_NUMBER() OVER()) % 5 AS version, CAST(ROW_NUMBER() OVER() AS VARCHAR) FROM range(50000000);
SELECT COUNT(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT COUNT(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
