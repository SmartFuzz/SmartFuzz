CREATE TABLE scene (id INTEGER PRIMARY KEY, victim_id INTEGER NOT NULL, description VARCHAR);
CREATE TABLE victim (id INTEGER PRIMARY KEY, name VARCHAR, age INTEGER CHECK (age >= 0));
CREATE TABLE inspector (id INTEGER PRIMARY KEY, name VARCHAR, experience_years INTEGER DEFAULT 0);

SELECT i.name, v1.name, v2.name FROM scene s JOIN scene b ON s.id <> b.id JOIN victim v1 ON v1.id = s.victim_id JOIN victim v2 ON v2.id = b.victim_id JOIN inspector i ON TRUE;
