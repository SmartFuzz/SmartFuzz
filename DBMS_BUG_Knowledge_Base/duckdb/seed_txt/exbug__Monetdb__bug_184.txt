CREATE TABLE artists (id INT PRIMARY KEY, name VARCHAR, age INT CHECK (age >= 0));
CREATE TABLE songs (id INT PRIMARY KEY, title VARCHAR, artist_id INT REFERENCES artists(id));

SELECT a.name, s.title, CASE WHEN a.name > s.title THEN 1 ELSE 0 END AS comparison FROM artists a JOIN songs s ON a.id = s.artist_id WHERE CASE WHEN a.name > s.title THEN 1 ELSE 0 END > 0 LIMIT 5;
