DROP TABLE IF EXISTS t1;
CREATE TABLE t1(c1 BLOB NOT NULL, c2 INT DEFAULT 0);
INSERT INTO t1 VALUES ('true'), ('false'), ('1xx'), (NULL);
SELECT CASE WHEN CAST(t1.c1 AS VARCHAR) = 'true' THEN TRUE ELSE FALSE END AS is_true FROM t1 WHERE c2 > 0;
SELECT CASE WHEN CAST(t1.c1 AS VARCHAR) IS NOT NULL THEN TRUE ELSE FALSE END AS is_not_null FROM t1 WHERE c2 IS NULL;
