CREATE TABLE sales ( year INTEGER NOT NULL, country VARCHAR(20) NOT NULL, product VARCHAR(32) NOT NULL, profit INTEGER DEFAULT 0, PRIMARY KEY (year, country, product) );
INSERT INTO sales (year, country, product, profit) VALUES (2020, 'USA', 'Gadget', 1500), (2020, 'USA', 'Widget', 2000), (2021, 'Canada', 'Gadget', 2500);
SELECT year, country, product, CASE WHEN GROUPING(year) = 1 THEN NULL ELSE year END AS year_grouped FROM sales GROUP BY year, country, product WITH ROLLUP HAVING GROUPING(year) = 0;
CREATE TABLE products ( product_id INTEGER NOT NULL, product_name VARCHAR(50) NOT NULL, PRIMARY KEY (product_id) );
