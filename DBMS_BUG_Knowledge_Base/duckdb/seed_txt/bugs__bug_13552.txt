CREATE TABLE test_unpivot (id INT, column_name VARCHAR, column_value VARCHAR, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_unpivot SELECT id, concat('column_', id), concat('value_', id) FROM (SELECT unnest(generate_series(1, 10000)) AS id);
CREATE TABLE pivoted_test AS SELECT * FROM test_unpivot PIVOT (FIRST(column_value) FOR column_name IN (column_1, column_2, column_3, column_4, column_5));
SELECT COUNT(*) FROM (UNPIVOT pivoted_test ON COLUMNS(* EXCLUDE id) INTO name 'column_name' VALUE 'column_value');
CREATE TABLE final_result AS SELECT * FROM (SELECT id, column_name, column_value FROM test_unpivot WHERE column_value IS NOT NULL) WHERE id < 5000;
