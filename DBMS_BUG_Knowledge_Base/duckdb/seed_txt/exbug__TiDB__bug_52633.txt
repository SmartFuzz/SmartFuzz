DROP TABLE IF EXISTS p1;
CREATE TABLE p1 ( id BIGINT NOT NULL, code VARCHAR(64) NOT NULL DEFAULT '', third_user_name VARCHAR(64) NOT NULL DEFAULT '', PRIMARY KEY (id, code));
DROP TABLE IF EXISTS t1;
CREATE TABLE t1 ( id BIGINT NOT NULL, code VARCHAR(64) NOT NULL DEFAULT '', third_user_name VARCHAR(64) NOT NULL DEFAULT '', PRIMARY KEY (id, code));
CREATE INDEX idx_third_user_name ON t1 (third_user_name);
