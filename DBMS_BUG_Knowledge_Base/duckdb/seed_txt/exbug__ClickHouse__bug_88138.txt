CREATE TABLE t1 (id INT, value DOUBLE, name VARCHAR);
INSERT INTO t1 (id, value, name) VALUES (1, 1.5, 'A'), (2, NULL, 'B'), (3, 3.5, 'C');
CREATE TABLE t2 (ref_id INT, score FLOAT);
INSERT INTO t2 (ref_id, score) SELECT id, CAST(value AS FLOAT) FROM t1 WHERE value IS NOT NULL;
SELECT ref_id, AVG(score) AS avg_score FROM t2 GROUP BY ref_id HAVING avg_score > 2.0;
