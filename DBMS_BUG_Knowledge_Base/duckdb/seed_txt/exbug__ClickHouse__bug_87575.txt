DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INTEGER, accepted_at TIMESTAMP, delivery_report_at VARCHAR);
INSERT INTO agg_test SELECT number AS id, TIMESTAMP '2025-09-24 00:00:00' + INTERVAL number SECOND AS accepted_at, MAX(CASE WHEN number % 2 = 1 THEN TO_CHAR(number) END) AS delivery_report_at FROM (SELECT number FROM range(100)) GROUP BY id, accepted_at;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, delivery_report_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
