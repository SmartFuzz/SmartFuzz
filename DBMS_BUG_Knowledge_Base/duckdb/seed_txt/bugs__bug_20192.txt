CREATE TABLE products (id INTEGER PRIMARY KEY, name VARCHAR, description VARCHAR, price DECIMAL(10, 2), created_at TIMESTAMP);
CREATE TABLE orders (id INTEGER PRIMARY KEY, product_id INTEGER, order_date TIMESTAMP, quantity INTEGER, large_blob BLOB, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, 'Product ' || i AS name, 'Description for product ' || i AS description, ROUND(RANDOM() * 100, 2) AS price, '2024-01-01'::TIMESTAMP + INTERVAL (i) DAY AS created_at FROM range(1, 10000) t(i);
INSERT INTO orders SELECT i AS id, (i % 10000) + 1 AS product_id, '2024-02-01'::TIMESTAMP + INTERVAL (i) DAY AS order_date, (i % 10) + 1 AS quantity, repeat('y', 2000)::BLOB AS large_blob FROM range(1, 5000000) t(i);
EXPLAIN ANALYZE SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01'::TIMESTAMP AND '2024-04-01'::TIMESTAMP AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
