CREATE TABLE t1(a VARCHAR, b INTEGER, c DOUBLE, d BLOB);
CREATE TABLE x1(x INTEGER, y INTEGER, z INTEGER, w VARCHAR NOT NULL);
CREATE OR REPLACE FUNCTION regexp_match(pattern VARCHAR, value VARCHAR) RETURNS BOOLEAN AS $$ SELECT value LIKE '%' || pattern || '%';
$$ LANGUAGE SQL;
CREATE TRIGGER tr1 AFTER INSERT ON t1 FOR EACH ROW INSERT INTO x1 VALUES(CASE WHEN regexp_match('abc', new.a) THEN 1 ELSE 0 END, CASE WHEN regexp_match('abc', new.b::VARCHAR) THEN 1 ELSE 0 END, CASE WHEN regexp_match('abc', new.c::VARCHAR) THEN 1 ELSE 0 END, new.d);
CREATE TRIGGER tr2 AFTER INSERT ON t1 FOR EACH ROW INSERT INTO x1 VALUES(CASE WHEN regexp_match('def', new.a) THEN 1 ELSE 0 END, CASE WHEN regexp_match('def', new.b::VARCHAR) THEN 1 ELSE 0 END, CASE WHEN regexp_match('def', new.c::VARCHAR) THEN 1 ELSE 0 END, NULL);
