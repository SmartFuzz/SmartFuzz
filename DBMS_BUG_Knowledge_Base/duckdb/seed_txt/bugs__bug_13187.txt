CREATE MACRO avg_sales(x DATE) AS (SELECT avg(amount) OVER (ORDER BY sale_date ASC RANGE BETWEEN INTERVAL 1 MONTHS PRECEDING AND INTERVAL 1 MONTHS FOLLOWING) FROM sales WHERE sale_date = x);
SELECT avg_sales('2023-01-01') AS average_sales;
