CREATE TABLE test_data (id INTEGER PRIMARY KEY, name VARCHAR, age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
COPY test_data FROM 'test_duckdb_data.csv' (DELIMITER ',', HEADER TRUE);
CREATE VIEW valid_records AS SELECT * FROM test_data WHERE age IS NOT NULL;
CREATE VIEW rejected_records AS SELECT * FROM read_csv_auto('test_duckdb_data.csv', header=True, ignore_errors=True, delim=',', all_varchar=True) WHERE name IS NULL;
SELECT COUNT(*) AS valid_count, (SELECT COUNT(*) FROM rejected_records) AS rejected_count FROM valid_records;
