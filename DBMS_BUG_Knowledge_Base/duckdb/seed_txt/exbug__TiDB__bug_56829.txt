CREATE TABLE cache_test (cache_key VARCHAR(512) NOT NULL, cache_value JSON NOT NULL, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, expires INTEGER DEFAULT -1, expired_at TIMESTAMP GENERATED ALWAYS AS (CASE WHEN expires > 0 THEN updated_at + INTERVAL expires SECOND ELSE updated_at + INTERVAL 99 YEAR END) VIRTUAL, PRIMARY KEY (cache_key));
