DROP TABLE IF EXISTS t_ref;
CREATE TABLE t_ref (id BIGINT NOT NULL, PRIMARY KEY (id));
DROP TABLE IF EXISTS t_main;
CREATE TABLE t_main (id INT NOT NULL, unique_id VARCHAR(32) GENERATED ALWAYS AS ('1'), t_ref_id BIGINT NOT NULL, PRIMARY KEY (id), UNIQUE (unique_id));
ALTER TABLE t_main ADD FOREIGN KEY (t_ref_id) REFERENCES t_ref (id);
