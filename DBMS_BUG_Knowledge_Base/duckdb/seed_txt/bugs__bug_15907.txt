CREATE TABLE test_table (id INTEGER, value FLOAT, name VARCHAR(50), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table (id, value, name) VALUES (1, 10.5, 'Alice'), (2, 20.0, 'Bob'), (3, NULL, 'Charlie');
SELECT id, lead(value ORDER BY id) OVER (ORDER BY id ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING EXCLUDE TIES) AS next_value FROM test_table;
SELECT id, lead(value ORDER BY id) OVER w AS next_value FROM test_table WINDOW w AS (ORDER BY id ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING EXCLUDE TIES);
SELECT id, value, name FROM test_table WHERE value IS NOT NULL ORDER BY created_at DESC;
