CREATE TABLE test_vals(grp INTEGER NOT NULL, value INTEGER CHECK(value >= 0));
INSERT INTO test_vals VALUES (1, 100), (2, 200), (1, NULL), (2, 300);
CREATE OR REPLACE FUNCTION jit_sum_variadic(value INTEGER[]) RETURNS BIGINT AS $$ SELECT COALESCE(SUM(v), 0) FROM UNNEST(value) AS v $$ LANGUAGE SQL;
SELECT grp, jit_sum_variadic(ARRAY_AGG(value)) FROM test_vals GROUP BY grp;
SELECT grp, jit_sum_variadic(ARRAY_AGG(value)) FROM test_vals WHERE value IS NOT NULL GROUP BY grp;
