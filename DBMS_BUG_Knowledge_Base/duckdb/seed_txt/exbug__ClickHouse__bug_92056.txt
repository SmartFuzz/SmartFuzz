DROP TABLE IF EXISTS events;
CREATE TABLE events ( Id BIGINT, Payload VARCHAR, Time TIMESTAMP, Status VARCHAR );
INSERT INTO events SELECT number, concat('Payload_', toString(number)), timestamp '2024-01-01 00:00:00' + INTERVAL number MINUTE, CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM generate_series(0, 9) AS number;
DROP TABLE IF EXISTS attributes;
CREATE TABLE attributes ( EventId BIGINT, Attribute VARCHAR, Value DOUBLE DEFAULT 0.0 );
