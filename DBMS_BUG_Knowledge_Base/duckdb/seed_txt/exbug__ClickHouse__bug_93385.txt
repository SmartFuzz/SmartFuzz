CREATE TABLE user_events (event_id UUID, user_id INTEGER, event_type VARCHAR, event_time TIMESTAMP);
INSERT INTO user_events SELECT generate_uuid() AS event_id, random() % 10000 AS user_id, CASE (random() % 3) WHEN 0 THEN 'click' WHEN 1 THEN 'view' ELSE 'purchase' END AS event_type, now() - INTERVAL (random() % 3600 * 24) SECOND AS event_time FROM range(100000);
CREATE TABLE user_summary (user_id INTEGER, event_count INTEGER);
INSERT INTO user_summary SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
