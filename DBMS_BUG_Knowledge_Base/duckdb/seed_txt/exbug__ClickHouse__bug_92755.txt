DROP SCHEMA IF EXISTS repro;
CREATE SCHEMA repro;
CREATE TABLE repro.events (tenant VARCHAR, event_time TIMESTAMP, id BIGINT, msg VARCHAR, value DOUBLE);
INSERT INTO repro.events SELECT CAST(number % 100 AS VARCHAR) AS tenant, CURRENT_TIMESTAMP - INTERVAL (number % 200000) SECOND AS event_time, number AS id, CONCAT('m', CAST(number AS VARCHAR)) AS msg, CAST(number % 1000 AS DOUBLE) AS value FROM range(2000000);
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
