CREATE TABLE t1 (id INT, name VARCHAR, score DOUBLE) AS SELECT * FROM (VALUES (1, 'Alice', 95.5), (2, 'Bob', 88.0), (3, 'Charlie', 76.5));
CREATE TABLE t2 (user_id INT, age TINYINT, active BOOLEAN) AS SELECT * FROM (VALUES (1, 25, TRUE), (2, 30, NULL), (3, 22, FALSE));
CREATE TABLE t3 AS SELECT user_id, COUNT(*) AS login_count FROM t2 GROUP BY user_id;
SELECT * FROM t1 JOIN t3 ON t1.id = t3.user_id WHERE t1.score > 80.0 ORDER BY t1.name;
