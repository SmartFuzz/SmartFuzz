CREATE OR REPLACE MACRO yearqtr(x) AS (SELECT (((date_part('year', CAST(x AS DATE)) * 10) + ((date_part('month', CAST(x AS DATE)) - 1) // 3)) + 1));
SELECT dt, yearqtr(dt) AS yq, yearqtr(dt) // 10 AS yq2, yearqtr(dt) // 3 AS yq3, yearqtr(dt) % 3 AS yq4 FROM test_dates WHERE dt = '2050-12-31';
