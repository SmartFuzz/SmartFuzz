CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR NOT NULL, age INTEGER, email VARCHAR UNIQUE);
CREATE TABLE comments (id INTEGER PRIMARY KEY, user_id INTEGER, content VARCHAR, FOREIGN KEY(user_id) REFERENCES users(id));
