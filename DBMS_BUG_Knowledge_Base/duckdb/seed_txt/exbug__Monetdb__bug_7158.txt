DROP AGGREGATE IF EXISTS python_aggregate;
CREATE OR REPLACE FUNCTION python_aggregate(val INTEGER) RETURNS INTEGER AS $$ import numpy as np;
def aggregate(val): unique = np.unique(val);
if unique.size == 0: return 0;
x = np.zeros(shape=(unique.size));
for i in range(0, unique.size): x[i] = np.sum(val[val == unique[i]]);
