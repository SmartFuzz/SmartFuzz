CREATE OR REPLACE MACRO json_each(input) AS TABLE (SELECT CASE json_type (val::json) WHEN 'ARRAY' THEN unnest(RANGE (json_array_length(val::json)::bigint)) ::varchar ELSE unnest(json_keys (val::json)) END AS key, json_extract (val::json, key) AS value FROM (SELECT input as val));
SELECT * FROM (SELECT '[{"key": "value1"}, {"key": "value2"}]'::json[] as message) CROSS JOIN json_each(message);
SELECT * FROM (SELECT '[1,2,3]'::json[] as message) CROSS JOIN json_each(message);
