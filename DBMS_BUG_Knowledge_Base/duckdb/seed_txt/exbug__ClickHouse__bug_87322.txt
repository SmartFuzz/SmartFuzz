CREATE TABLE t1 (id INTEGER, name VARCHAR, value DOUBLE) AS SELECT * FROM (VALUES (1, 'test1', 10.5), (2, 'test2', NULL), (3, 'test3', 20.0));
CREATE TABLE t2 (id INTEGER, ref_id INTEGER, description VARCHAR) AS SELECT * FROM (VALUES (1, 1, 'desc1'), (2, 2, 'desc2'), (3, 3, 'desc3'));
SELECT t1.name, t2.description FROM t1 LEFT JOIN t2 ON t1.id = t2.ref_id WHERE t1.value IS NULL;
SELECT COUNT(*) FROM t1;
SELECT COUNT(*) FROM t2;
