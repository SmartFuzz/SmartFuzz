CREATE TABLE table_src (order_id BIGINT, random_date DATE, amount DOUBLE) AS SELECT number AS order_id, toDate(rand() % 3650) AS random_date, rand() * 100 AS amount FROM range(1000000);
CREATE TABLE table_dst (order_id BIGINT, random_date DATE, amount DOUBLE) PARTITION BY month(random_date);
INSERT INTO table_dst SELECT order_id, random_date, amount FROM table_src WHERE amount > 50.0;
