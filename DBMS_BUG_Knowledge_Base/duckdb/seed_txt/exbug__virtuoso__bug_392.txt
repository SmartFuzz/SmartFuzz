CREATE TABLE automobiles (id INT PRIMARY KEY, manufacturer VARCHAR, thumbnail VARCHAR, depiction VARCHAR, image VARCHAR, logo VARCHAR);

SELECT id, COALESCE(thumbnail, depiction, image, logo) AS image_url FROM automobiles WHERE manufacturer IS NOT NULL;
SELECT id, COUNT(*) AS optional_count FROM automobiles GROUP BY id HAVING optional_count > 0;
