CREATE TABLE t1(a INTEGER, b BIGINT, c FLOAT, d DOUBLE, e DECIMAL, f VARCHAR);
INSERT INTO t1 VALUES(1, 1000, 6.8, 3.45, 8.3, 'test1'), (1, 3998, -3.4, 5.12, 9.3, 'test2'), (1, 288, 9.2, 6.08, 1, 'test3');
CREATE TABLE t2(g INTEGER, h FLOAT, i DECIMAL(10, 2), j DOUBLE);
INSERT INTO t2 SELECT a, c, e, d FROM t1 WHERE b > 2000;
SELECT VARIANCE(h) AS variance_h, VARIANCE(i) AS variance_i, VARIANCE(j) AS variance_j FROM t2 GROUP BY g;
