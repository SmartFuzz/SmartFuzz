CREATE TABLE tbl1 (id INTEGER, value VARCHAR, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO tbl1 SELECT number, CAST(number % 10 AS VARCHAR) FROM range(100);
CREATE TABLE tbl2 (ref_id INTEGER, description VARCHAR, amount DOUBLE);
INSERT INTO tbl2 SELECT id, 'Description ' || CAST(id AS VARCHAR), id * 1.5 FROM tbl1;
SELECT tbl1.id, tbl2.amount FROM tbl1 JOIN tbl2 ON tbl1.id = tbl2.ref_id WHERE tbl1.value IS NOT NULL ORDER BY tbl1.id;
