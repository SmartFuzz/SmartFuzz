CREATE TABLE parent_view (id INTEGER PRIMARY KEY, name VARCHAR NOT NULL);
CREATE TABLE child_view (id INTEGER PRIMARY KEY, parent_id INTEGER, value INTEGER, FOREIGN KEY(parent_id) REFERENCES parent_view(id));
