DROP TABLE IF EXISTS mock_data_in;
CREATE TABLE mock_data_in (i INT, j INT, k INT, l UINT, PRIMARY KEY (i, j));
INSERT INTO mock_data_in SELECT i, j, k, hash(i * 10000 + j * 100 + k) % 100 AS l FROM generate_series(1, 100) s(i) CROSS JOIN generate_series(1, 10) t(j) CROSS JOIN generate_series(1, 100) u(k);
CREATE VIEW base_view AS SELECT * FROM mock_data_in;
SELECT * FROM base_view WHERE i = 10 AND j = 10 AND k IS NOT NULL;
