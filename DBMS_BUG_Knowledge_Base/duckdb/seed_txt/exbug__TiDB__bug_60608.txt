CREATE TABLE mysql_test (id INT PRIMARY KEY, name VARCHAR(100) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, status VARCHAR(10) CHECK (status IN ('active', 'inactive')) NOT NULL);
