CREATE TABLE dst (e VARCHAR, f INT) AS SELECT 'a' AS e, 0 AS f LIMIT 0;
CREATE TABLE src (e VARCHAR, g INT) AS SELECT 'a' AS e, NULL AS g LIMIT 0;
CREATE VIEW mv AS SELECT e, g FROM src;
INSERT INTO src SELECT CASE WHEN number % 1 = 0 THEN 'a' ELSE NULL END AS e, number % 10 AS g FROM generate_series(1, 10);
INSERT INTO dst SELECT e, g FROM mv WHERE g IS NOT NULL;
