CREATE TABLE releases (version REAL, PRIMARY KEY (version));
COMMENT ON COLUMN releases.version IS 'Version of the registry.';
CREATE TABLE releases_v2 (version FLOAT);
COMMENT ON COLUMN releases_v2.version IS 'Version of the registry.';
CREATE TABLE releases_v3 (id INTEGER, version REAL, description VARCHAR);
COMMENT ON COLUMN releases_v3.description IS 'Description of the release.';
