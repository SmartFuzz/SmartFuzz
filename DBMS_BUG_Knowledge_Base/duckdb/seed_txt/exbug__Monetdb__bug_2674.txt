CREATE TABLE t1 (ra FLOAT NOT NULL, dec FLOAT NOT NULL);
SELECT ra, dec, (ACOS(SIN(213.7849600 * PI() / 180) * SIN(ra * PI() / 180) + COS(213.7849600 * PI() / 180) * COS(ra * PI() / 180) * COS((dec - -0.4932472) * PI() / 180)) * 180 / PI()) * 60 AS distance FROM t1;
CREATE OR REPLACE FUNCTION f1() RETURNS FLOAT AS $$ BEGIN RETURN NULL;
END;
$$ LANGUAGE plpgsql;
