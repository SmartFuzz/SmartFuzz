CREATE TABLE cols (c VARCHAR);
INSERT INTO cols (SELECT 'c'||i FROM range(1, 21));
CREATE TABLE matrix (col1 VARCHAR, col2 VARCHAR, val UTINYINT NOT NULL CHECK (val < 250));
INSERT INTO matrix (SELECT c1.c, c2.c, CAST(random()*250 AS UTINYINT) FROM cols c1 JOIN cols c2 ON c1.c != c2);
PIVOT matrix ON col1 USING SUM(val)::UTINYINT ORDER BY col2 LIMIT 2;
