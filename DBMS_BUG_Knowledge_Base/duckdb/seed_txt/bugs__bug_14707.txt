CREATE TABLE test_table (foo INTEGER, year INTEGER, month INTEGER, day INTEGER);
INSERT INTO test_table VALUES (1, 2024, 11, 5);
COPY (SELECT foo, year, month, day FROM test_table) TO '.' (FORMAT PARQUET, OVERWRITE_OR_IGNORE, PARTITION_BY (year, month, day), WRITE_PARTITION_COLUMNS false, FILENAME_PATTERN 'test');
COPY (SELECT foo, year, month, day FROM test_table WHERE year IS NOT NULL) TO '.' (FORMAT CSV, OVERWRITE_OR_IGNORE, PARTITION_BY (year, month, day), WRITE_PARTITION_COLUMNS true, FILENAME_PATTERN 'test_csv');
SELECT * FROM parquet_metadata('./test/year=2024/month=11/day=5/test.parquet');
