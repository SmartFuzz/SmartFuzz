CREATE SCHEMA test_schema;
CREATE TABLE test_schema.test_table (id INTEGER PRIMARY KEY AUTOINCREMENT, path VARCHAR NOT NULL, description TEXT);
CREATE FUNCTION test_schema.trigger_function() RETURNS trigger AS $$ BEGIN RETURN NEW;
END;
$$;
CREATE TRIGGER path_change_trigger AFTER UPDATE ON test_schema.test_table FOR EACH ROW WHEN (NEW.path <> OLD.path) EXECUTE FUNCTION test_schema.trigger_function();
