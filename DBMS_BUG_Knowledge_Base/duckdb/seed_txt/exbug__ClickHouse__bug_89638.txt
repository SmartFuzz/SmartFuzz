CREATE TABLE test_partitioned (id INTEGER, dt TIMESTAMP, value DOUBLE) AS SELECT number, TIMESTAMP '2015-01-01' + INTERVAL (number % 3650) DAY AS dt, RANDOM() FROM range(10000000);
CREATE TABLE test_non_partitioned (id INTEGER, dt TIMESTAMP, value DOUBLE) AS SELECT number, TIMESTAMP '2015-01-01' + INTERVAL (number % 3650) DAY AS dt, RANDOM() FROM range(10000000);
SELECT dt FROM test_partitioned WHERE value IS NOT NULL ORDER BY dt LIMIT 20;
SELECT dt FROM test_non_partitioned WHERE value IS NOT NULL ORDER BY dt LIMIT 20;
