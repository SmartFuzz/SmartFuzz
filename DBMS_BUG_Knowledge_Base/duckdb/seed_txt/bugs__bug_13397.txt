CREATE TABLE test_table (id INTEGER, value INTEGER, description VARCHAR);
INSERT INTO test_table (id, value, description) VALUES (1, 0, 'zero'), (2, 1, 'one'), (3, 2, 'two');
SELECT list_reduce((SELECT ARRAY_AGG(value) FROM test_table), (sum,x) -> sum + CASE WHEN x = 0 THEN 1 ELSE 0 END) AS count_zeros;
SELECT list_reduce((SELECT ARRAY_AGG(value) FROM test_table WHERE value IS NOT NULL), (sum,x) -> (CASE WHEN sum IS NULL THEN 0 ELSE sum END) + (CASE WHEN x = 1 THEN 1 ELSE 0 END)) AS count_ones;
SELECT list_reduce([NULL, 0, 1, 2], (sum,x) -> sum + COALESCE(CASE WHEN x = 0 THEN 1 ELSE 0 END, 0)) AS count_zeros;
