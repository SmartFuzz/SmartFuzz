DROP TABLE IF EXISTS xyz;
CREATE TABLE xyz (id INTEGER PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY, name VARCHAR(255), age INTEGER);
INSERT INTO xyz (name, age) VALUES ('Alice', 30), ('Bob', NULL);
CREATE OR REPLACE FUNCTION get_age() RETURNS INTEGER AS $$ DECLARE user_record xyz;
BEGIN FOR user_record IN SELECT * FROM xyz WHERE age IS NOT NULL LOOP RETURN user_record.age;
END LOOP;
