CREATE TABLE T1 (K INT, C0 VARCHAR, C1 VARCHAR, C2 DOUBLE);
INSERT INTO T1 SELECT number, ARRAY_AGG(CASE WHEN number % 2 = 0 THEN city_hash(number * number) END) AS C0, CASE WHEN number % 2 = 0 THEN 'even' ELSE NULL END AS C1, number * 1.5 AS C2 FROM range(100);
CREATE TABLE T2 (K INT, C0 VARCHAR, C1 INT);
INSERT INTO T2 SELECT K, C0, CAST(K AS INT) FROM T1 WHERE C1 IS NOT NULL;
SELECT * FROM T1 WHERE C1 = 'even';
