DROP TABLE IF EXISTS test_db.sample_table;
CREATE TABLE test_db.sample_table (id BIGINT, name VARCHAR, created_at TIMESTAMP, score DOUBLE) AS SELECT * FROM (VALUES (0, '', NOW(), 0.0)) AS t(id, name, created_at, score);
CREATE VIEW test_db.sample_view AS SELECT id, name, created_at FROM test_db.sample_table WHERE score > 50.0;
INSERT INTO test_db.sample_table SELECT number, concat('Name', toString(number)), NOW(), number % 100 FROM range(100);
SELECT id, name, created_at FROM test_db.sample_view ORDER BY created_at DESC LIMIT 10;
