CREATE TABLE SQL_TEST ( columna INT NOT NULL, columnb INT DEFAULT 0, columnc VARCHAR(20) CHECK (LENGTH(columnc) <= 20));
INSERT INTO SQL_TEST (columna, columnb, columnc) VALUES (1, 10, 'test1'), (2, 20, 'test2'), (3, 30, NULL), (4, 40, 'test1');
SELECT SUM(DISTINCT columna) AS sum_distinct_columna, COUNT(DISTINCT columnc) AS unique_count FROM SQL_TEST WHERE columnb > 15;
SELECT COUNT(DISTINCT columnc) AS distinct_columnc FROM SQL_TEST WHERE columna IS NOT NULL;
SELECT SUM(DISTINCT columna) AS sum_distinct_columna FROM SQL_TEST WHERE columnb < 25;
