CREATE TABLE a_table(id INTEGER PRIMARY KEY, value TEXT NOT NULL);
CREATE TABLE b_table(id INTEGER PRIMARY KEY, ref_id INTEGER, FOREIGN KEY(ref_id) REFERENCES a_table(id));
CREATE OR REPLACE FUNCTION after_insert_a() RETURNS VOID AS $$ BEGIN INSERT INTO b_table(ref_id) SELECT id FROM a_table WHERE value IS NULL;
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER after_insert_a_trigger AFTER INSERT ON a_table EXECUTE FUNCTION after_insert_a();
