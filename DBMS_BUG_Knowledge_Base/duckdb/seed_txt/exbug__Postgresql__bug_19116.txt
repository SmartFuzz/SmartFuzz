CREATE TABLE t(p BOOLEAN, q INTEGER, r TEXT DEFAULT 'default');
INSERT INTO t(p, q) VALUES (NULL, 1);
SELECT 1 AS advisory_lock;
WITH lock_t AS (SELECT ROW_NUMBER() OVER () AS rn FROM t WHERE p IS NULL) UPDATE t SET q = q + 1 WHERE ROW_NUMBER() OVER () IN (SELECT rn FROM lock_t) RETURNING p, q;
WITH lock_t AS (SELECT ROW_NUMBER() OVER () AS rn FROM t) UPDATE t SET r = 'updated' WHERE ROW_NUMBER() OVER () IN (SELECT rn FROM lock_t) RETURNING r;
