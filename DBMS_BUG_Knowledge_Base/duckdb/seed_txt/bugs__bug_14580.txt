SELECT id, value, value * 2::INT32 AS doubled_value FROM test_numbers;
CREATE OR REPLACE FUNCTION test_subfact(num INT) AS TABLE (n INT, sub UHUGEINT) AS (WITH RECURSIVE cte(n, psub, sub) AS (SELECT 0 AS n, 1::UHUGEINT AS psub, 1::UHUGEINT AS sub UNION ALL SELECT n + 1 AS n, sub AS psub, (n * (sub + psub)) AS sub FROM cte WHERE n < num) SELECT n, sub FROM cte);
SELECT * FROM test_subfact(20);
