CREATE TABLE fuzz_test (id INT NOT NULL, name VARCHAR(50) NOT NULL, value DECIMAL(10,2) NOT NULL, PRIMARY KEY (id));
CREATE OR REPLACE FUNCTION fuzz_test_BEFORE_INSERT() RETURNS TRIGGER AS $$ BEGIN NEW.value := COALESCE(NEW.value, 0);
RETURN NEW;
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER fuzz_test_BEFORE_INSERT BEFORE INSERT ON fuzz_test FOR EACH ROW EXECUTE FUNCTION fuzz_test_BEFORE_INSERT();
