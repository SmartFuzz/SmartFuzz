CREATE TABLE comments (id BIGINT NOT NULL PRIMARY KEY, post_id BIGINT, content TEXT NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (post_id) REFERENCES posts(id));
