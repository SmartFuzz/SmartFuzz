DROP TABLE IF EXISTS test_left;
DROP TABLE IF EXISTS test_right;
CREATE TABLE test_left (a BIGINT, b VARCHAR, c VARCHAR, d INTEGER);
CREATE TABLE test_right (a BIGINT, b VARCHAR, c VARCHAR, d INTEGER);
INSERT INTO test_left SELECT number % 10000, CAST(number % 10000 AS VARCHAR), CAST(number % 10000 AS VARCHAR), number % 5 FROM generate_series(0, 99999) AS number;
INSERT INTO test_right SELECT number % 10, CAST(number % 10 AS VARCHAR), CAST(number % 10 AS VARCHAR), NULL FROM generate_series(0, 9999) AS number;
