CREATE TABLE IF NOT EXISTS user_activity (activity_id BIGINT, user_id INTEGER, activity_time TIMESTAMP, activity_type VARCHAR, details VARCHAR);
INSERT INTO user_activity SELECT number AS activity_id, rand() % 1000 AS user_id, now() + number * INTERVAL '5' SECOND AS activity_time, (CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END) AS activity_type, concat('Activity details for ', number) AS details FROM range(100);
CREATE TABLE IF NOT EXISTS product_sales (sale_id INTEGER, product_id INTEGER, sale_time TIMESTAMP, quantity SMALLINT, price FLOAT);
INSERT INTO product_sales SELECT number AS sale_id, rand() % 500 AS product_id, now() + number * INTERVAL '2' SECOND AS sale_time, (rand() % 10 + 1) AS quantity, (rand() % 100 + 1) * 1.5 AS price FROM range(50);
SELECT ua.user_id, count(*) AS activity_count, sum(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= now() - INTERVAL '1' DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
