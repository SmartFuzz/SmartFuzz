CREATE TABLE courses (course_id SERIAL PRIMARY KEY, course_name VARCHAR NOT NULL, credits INTEGER CHECK (credits > 0));
INSERT INTO courses (course_id, course_name, credits) VALUES (1, 'Mathematics', 4), (2, 'Physics', 3), (3, 'Chemistry', 4);
CREATE TABLE enrollments (enrollment_id SERIAL PRIMARY KEY, sid INTEGER, course_id INTEGER, FOREIGN KEY (sid) REFERENCES students(sid), FOREIGN KEY (course_id) REFERENCES courses(course_id));
INSERT INTO enrollments (enrollment_id, sid, course_id) VALUES (1, 1, 1), (2, 2, 2), (3, 3, 3);
EXPLAIN ANALYZE SELECT c.course_name FROM courses c JOIN enrollments e ON c.course_id = e.course_id WHERE e.sid IS NOT NULL;
