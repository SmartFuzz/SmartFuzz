DROP SCHEMA IF EXISTS dtobs CASCADE;
CREATE SCHEMA dtobs;
CREATE TABLE dtobs.span_local (links JSONB[], dummy SMALLINT);
CREATE TABLE dtobs.span_link_local (linked_id TEXT);
CREATE MATERIALIZED VIEW dtobs.mv_bad AS SELECT (link->>'trace_id') AS linked_id FROM (SELECT unnest(links) AS link FROM dtobs.span_local) AS sub WHERE (link->>'trace_id') IS NOT NULL;
ALTER TABLE dtobs.span_local DROP COLUMN dummy;
