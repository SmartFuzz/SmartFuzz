CREATE EXTENSION IF NOT EXISTS fts5;
CREATE VIRTUAL TABLE vt1 USING fts5(content TEXT, id SERIAL PRIMARY KEY);
CREATE VIRTUAL TABLE vt2 USING fts5(comment TEXT, user_id INTEGER);
INSERT INTO vt1 (content) VALUES ('Sample content');
