CREATE TABLE json_data (MainLevel JSONB);
INSERT INTO json_data VALUES ('{"MainLevel":[{"More":{}},{"More":{"Level2":[{"Field1":"A"}]}}]}');
SELECT Level2.Field1 FROM json_data, jsonb_array_elements(MainLevel) AS MainLevel, jsonb_array_elements(MainLevel->'More'->'Level2') AS Level2;
SELECT COUNT(*) FROM json_data, jsonb_array_elements(MainLevel) AS MainLevel WHERE MainLevel->'More' IS NOT NULL;
SELECT MainLevel->'More'->'Level2'->0->>'Field1' AS Field1 FROM json_data, jsonb_array_elements(MainLevel) AS MainLevel;
