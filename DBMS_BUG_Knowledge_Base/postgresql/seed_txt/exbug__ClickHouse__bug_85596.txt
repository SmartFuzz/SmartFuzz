CREATE TABLE tbl1 (id SERIAL PRIMARY KEY, value TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO tbl1 (id, value) SELECT number, (number % 10)::TEXT FROM generate_series(1, 100) AS number;
CREATE TABLE tbl2 (ref_id INT, description TEXT, amount DOUBLE PRECISION);
INSERT INTO tbl2 (ref_id, description, amount) SELECT id, 'Description ' || id::TEXT, id * 1.5 FROM tbl1;
SELECT tbl1.id, tbl2.amount FROM tbl1 JOIN tbl2 ON tbl1.id = tbl2.ref_id WHERE tbl1.value IS NOT NULL ORDER BY tbl1.id;
