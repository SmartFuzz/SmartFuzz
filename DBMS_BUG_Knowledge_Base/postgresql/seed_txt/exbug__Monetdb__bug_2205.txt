CREATE TABLE kvk (kvk bigint PRIMARY KEY, bedrijfsnaam varchar(255) NOT NULL, adres varchar(64), postcode varchar(6), plaats varchar(32), type varchar(16) CHECK (type IN ('Type1', 'Type2')), kvks int, sub int);
CREATE TABLE concernrelaties (kvk bigint REFERENCES kvk (kvk), ouder varchar(30), postcode varchar(6), plaats varchar(32), ouderkvk bigint);
