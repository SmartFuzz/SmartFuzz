CREATE TABLE tst (k INTEGER NOT NULL, name CHAR(20) NOT NULL, age INTEGER DEFAULT NULL);
INSERT INTO tst (k, name, age) VALUES (1, 'Alice', 30), (2, 'Bob', NULL), (3, 'Charlie', 25);
CREATE VIEW v1 AS SELECT MAX(k) AS max_k FROM tst GROUP BY name;
CREATE VIEW v2 AS SELECT name, COUNT(*) AS count FROM tst GROUP BY name;
SELECT * FROM v1;
