CREATE TABLE test_large_lists (id INTEGER, gts INTEGER[][]);
INSERT INTO test_large_lists SELECT i, ARRAY[ARRAY[CAST(FLOOR(RANDOM() * 256) AS INTEGER), CAST(FLOOR(RANDOM() * 256) AS INTEGER)]] FROM generate_series(1, 500000) AS i;
CREATE VIEW list_view AS SELECT id, gts FROM test_large_lists;
SELECT COUNT(*) FROM list_view WHERE gts IS NOT NULL;
SELECT id, UNNEST(gts) AS inner_list FROM list_view;
