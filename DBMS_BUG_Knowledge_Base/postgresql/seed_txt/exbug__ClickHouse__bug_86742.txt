DROP TABLE IF EXISTS job_sample;
CREATE TABLE job_sample (id SERIAL PRIMARY KEY, posted_at DATE, discovered_at TIMESTAMP WITH TIME ZONE, url TEXT, title TEXT, company_name TEXT, description TEXT);
COPY job_sample (posted_at, discovered_at, url, title, company_name, description) FROM 'https://media.theirstack.com/jobs/jan-2025-sample.parquet' WITH (FORMAT 'parquet');
ALTER TABLE job_sample ADD COLUMN description_matches_python BOOLEAN;
UPDATE job_sample SET description_matches_python = (description ILIKE '%Python%');
SELECT company_name, COUNT(*) AS initial_matches FROM job_sample WHERE description_matches_python = TRUE GROUP BY company_name ORDER BY COUNT(*) DESC LIMIT 10;
