CREATE TABLE test2.raw_source_local (id INT, time TIMESTAMP, value FLOAT8 DEFAULT 0.0);
INSERT INTO test2.raw_source_local (id, time, value) SELECT generate_series(1, 1000) AS id, '2024-01-01 00:00:00'::timestamp + (generate_series(1, 1000) * interval '1 second') AS time, random() * 100 AS value;
CREATE TABLE test2.table_for_mv (sum_value FLOAT8, date DATE, count_id INT);
CREATE MATERIALIZED VIEW test2.mv_from_raw_source_local_to_table AS SELECT SUM(value) AS sum_value, DATE(time) AS date, COUNT(*) AS count_id FROM test2.raw_source_local GROUP BY DATE(time);
