CREATE TABLE mixed_timestamps (id SERIAL PRIMARY KEY, time_column VARCHAR, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

CREATE TABLE parsed_timestamps AS SELECT id, time_column, NULLIF(time_column, '')::TIMESTAMP AS parsed_time FROM mixed_timestamps;
SELECT id, time_column, parsed_time FROM parsed_timestamps WHERE parsed_time IS NULL;
SELECT COUNT(*) FROM parsed_timestamps WHERE parsed_time IS NOT NULL;
