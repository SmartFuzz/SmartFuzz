CREATE TABLE test_table (id SERIAL PRIMARY KEY, value INTEGER NOT NULL);
INSERT INTO test_table (id, value) VALUES (1, 10), (2, 20), (3, 10), (4, 30);
SELECT ARRAY(SELECT DISTINCT value FROM test_table ORDER BY value) AS distinct_values;
SELECT ARRAY(SELECT DISTINCT value % 3 FROM test_table ORDER BY id) AS mod_values;
SELECT ARRAY(SELECT DISTINCT CASE WHEN value > 15 THEN 'high' ELSE 'low' END FROM test_table ORDER BY value) AS categories;
