CREATE TABLE TestTable (id SERIAL PRIMARY KEY, name TEXT NOT NULL, value REAL DEFAULT 0.0);
CREATE OR REPLACE FUNCTION TestTriggerFunction() RETURNS TRIGGER AS $$ BEGIN UPDATE TestTable SET value = value + 1 WHERE id = NEW.id;
RETURN NEW;
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER TestTrigger AFTER INSERT ON TestTable FOR EACH ROW EXECUTE FUNCTION TestTriggerFunction();
