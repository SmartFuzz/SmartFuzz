DROP TABLE IF EXISTS products;
CREATE TABLE products (product_id SERIAL PRIMARY KEY, name TEXT NOT NULL, price DECIMAL(10,2) CHECK(price >= 0), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

SELECT p.product_id, p.name, p.price FROM products p WHERE p.price > (SELECT AVG(price) FROM products) AND p.created_at < CURRENT_TIMESTAMP;
WITH recent_products AS (SELECT * FROM products WHERE created_at >= TIMESTAMP '2023-01-01 00:00:00') SELECT rp.product_id FROM recent_products rp WHERE EXISTS (SELECT 1 FROM recent_products WHERE price < rp.price);
