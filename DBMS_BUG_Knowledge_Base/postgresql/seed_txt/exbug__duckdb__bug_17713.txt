CREATE TABLE test_floats (id SERIAL PRIMARY KEY, value FLOAT);
INSERT INTO test_floats (id, value) VALUES (1, 0.99), (2, 1.01), (3, NULL);
SELECT CASE WHEN value IS NULL THEN 'NULL' ELSE value::TEXT END AS float_check FROM test_floats;
SELECT CASE WHEN value < 1.0 THEN 'Less than one' ELSE 'One or more' END AS float_category FROM test_floats;
SELECT (SELECT AVG(value) FROM test_floats WHERE value IS NOT NULL) AS average_value;
