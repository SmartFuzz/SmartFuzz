CREATE TABLE users (id SERIAL PRIMARY KEY, name TEXT NOT NULL, age INTEGER CHECK(age >= 0), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id SERIAL PRIMARY KEY, user_id INTEGER, amount REAL NOT NULL, order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE OR REPLACE FUNCTION update_order_timestamp() RETURNS TRIGGER AS $$ BEGIN NEW.order_date = CURRENT_TIMESTAMP;
RETURN NEW;
END;
$$ LANGUAGE plpgsql;
