CREATE TABLE position_evaluation_v2 (symbol VARCHAR, evaluated_at TIMESTAMP(3), mark_price NUMERIC(18, 8), volume INTEGER);
INSERT INTO position_evaluation_v2 SELECT symbol, evaluated_at, mark_price, volume FROM generate_series(1, 100) AS id(symbol, evaluated_at, mark_price, volume);
SELECT LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price FROM position_evaluation_v2;
SELECT LAG(mark_price, 1, NULL) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price_with_default FROM position_evaluation_v2;
SELECT LAG(mark_price) OVER (PARTITION BY symbol ORDER BY evaluated_at ASC) AS last_mark_price_invalid FROM position_evaluation_v2;
