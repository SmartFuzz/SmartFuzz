CREATE TABLE T1(A BIGINT, B DOUBLE PRECISION);
INSERT INTO T1 SELECT number, number * 1.5 FROM generate_series(1, 100000) AS number;
CREATE TABLE T2(C TEXT, D INT);
INSERT INTO T2 SELECT to_char(number) AS C, CASE WHEN number % 2 = 0 THEN number ELSE NULL END AS D FROM generate_series(1, 1000) AS number;
SELECT DISTINCT A, (SELECT a1 FROM T2 WHERE id = B) AS a1, (SELECT a2 FROM T2 WHERE id = B) AS a2 FROM T1;
