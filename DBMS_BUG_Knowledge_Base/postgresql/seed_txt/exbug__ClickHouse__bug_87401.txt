DROP TABLE IF EXISTS b_products;
CREATE TABLE b_products (product_id BIGINT, product_name TEXT, price DOUBLE PRECISION);
INSERT INTO b_products (product_id, product_name, price) SELECT number, 'Product_' || number, number * 10.5 FROM generate_series(1, 100) AS number;
DROP TABLE IF EXISTS b_sales;
CREATE TABLE b_sales (sale_id BIGINT, product_id BIGINT, quantity INTEGER);
