CREATE TABLE sales (id SERIAL PRIMARY KEY, amount NUMERIC(10, 2), sale_date DATE) PARTITION BY RANGE (EXTRACT(YEAR FROM sale_date)) (PARTITION p0 FOR VALUES FROM (MINVALUE) TO (2020), PARTITION p1 FOR VALUES FROM (2020) TO (2021), PARTITION p2 FOR VALUES FROM (2021) TO (2022), PARTITION p3 FOR VALUES FROM (2022) TO (2023));
