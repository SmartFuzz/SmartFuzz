CREATE TABLE prices (currency char(3), valid_from char(8), price decimal(15,2), PRIMARY KEY(currency, valid_from));

CREATE TABLE vouchers (vid int PRIMARY KEY, date char(8), cur char(3), val decimal(15,2), FOREIGN KEY(cur) REFERENCES prices(currency));

SELECT vid, date, cur, val, (SELECT max(price) FROM prices p WHERE p.valid_from = (SELECT max(q.valid_from) FROM prices q WHERE q.valid_from <= vouchers.date AND q.currency = vouchers.cur)) AS Preis FROM vouchers;
