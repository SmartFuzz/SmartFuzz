CREATE TABLE t1 (v1 INT, v2 INT, v3 INT DEFAULT 0);
CREATE TABLE t2 (v1 INT, v2 INT, v4 VARCHAR, v5 FLOAT);
INSERT INTO t1 SELECT i, i, i % 2 FROM generate_series(0, 1000) AS t(i);
INSERT INTO t2 SELECT i, i, 'value' || i, i::FLOAT FROM generate_series(0, 10000000) AS t(i);
SELECT t1.v2, t2.v2, COUNT(*) FROM t1 INNER JOIN t2 ON (t1.v1 = t2.v1) GROUP BY t1.v2, t2.v2 ORDER BY t1.v2 LIMIT 5;
