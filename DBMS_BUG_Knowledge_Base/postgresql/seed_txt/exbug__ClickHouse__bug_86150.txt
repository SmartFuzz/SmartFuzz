DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id INTEGER, keyword_id INTEGER, search_date TIMESTAMP, position_on_serp INTEGER, version INTEGER, value TEXT);
INSERT INTO test_final_partitions SELECT (random() * 1000)::INTEGER AS organization_id, (random() * 100000)::INTEGER AS keyword_id, '2023-01-01'::TIMESTAMP + (random() * (86400 * 180)) * INTERVAL '1 second' AS search_date, (random() * 50)::INTEGER AS position_on_serp, (row_number() OVER ()) % 5 AS version, (row_number() OVER ())::TEXT FROM generate_series(1, 50000000);
SELECT count(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT count(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
