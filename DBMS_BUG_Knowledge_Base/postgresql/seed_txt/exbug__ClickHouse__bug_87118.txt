CREATE TABLE t_proj_test1 (id SERIAL PRIMARY KEY, name TEXT, value FLOAT8, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO t_proj_test1 (id, name, value, created_at) SELECT number, concat('Name', number::TEXT), number * 1.5, CURRENT_TIMESTAMP FROM generate_series(1, 10000) AS number;
CREATE TABLE t_proj_test2 (user_id SERIAL PRIMARY KEY, score INT, timestamp DATE DEFAULT CURRENT_DATE);
INSERT INTO t_proj_test2 (user_id, score, timestamp) SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END, CURRENT_DATE FROM generate_series(1, 5000) AS number;
SELECT user_id, SUM(score) AS total_score FROM t_proj_test2 GROUP BY user_id ORDER BY total_score DESC;
