CREATE TABLE test_table (id INTEGER, a FLOAT, b VARCHAR);
INSERT INTO test_table VALUES (1, 1.2, 'hello world');
CREATE TABLE another_table AS SELECT id, a FROM test_table WHERE a IS NOT NULL;
CREATE TABLE json_table AS SELECT a::FLOAT, b FROM json_populate_recordset(NULL::test_table, '[{"a": 1.2, "b": "hello world"}]');
SELECT * FROM json_table LEFT JOIN test_table ON json_table.a = test_table.a;
