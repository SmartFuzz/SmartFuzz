CREATE TABLE user_events (event_id UUID, user_id INTEGER, event_type VARCHAR, event_time TIMESTAMP);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT gen_random_uuid() AS event_id, FLOOR(RANDOM() * 10000) AS user_id, (ARRAY['click', 'view', 'purchase'])[FLOOR(RANDOM() * 3) + 1] AS event_type, NOW() - INTERVAL '1 second' * FLOOR(RANDOM() * 3600 * 24) AS event_time FROM generate_series(1, 100000);
CREATE TABLE user_summary (user_id INTEGER, event_count INTEGER);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) AS event_count FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
