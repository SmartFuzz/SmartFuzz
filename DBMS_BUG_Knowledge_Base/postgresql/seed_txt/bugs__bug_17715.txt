CREATE FUNCTION cube (p integer) RETURNS integer AS $$ BEGIN RETURN p * p * p;
END;
$$ LANGUAGE plpgsql;
CREATE FUNCTION myfunc2 (p integer) RETURNS integer AS $$ BEGIN RETURN cube(p) + 2;
END;
$$ LANGUAGE plpgsql;
CREATE TABLE numbers (value integer NOT NULL CHECK (value > 0));
INSERT INTO numbers VALUES (5), (10), (15);
CREATE MATERIALIZED VIEW numview AS (SELECT myfunc2(value) FROM numbers);
