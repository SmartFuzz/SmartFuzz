DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INTEGER, accepted_at TIMESTAMP(3), delivery_report_at JSONB);
INSERT INTO agg_test (id, accepted_at, delivery_report_at) SELECT number AS id, '2025-09-24 00:00:00'::timestamp(3) + number AS accepted_at, jsonb_build_object('value', max(CASE WHEN number % 2 = 1 THEN to_char('2025-09-24 00:00:00'::timestamp(3) + number, 'YYYY-MM-DD HH24:MI:SS.MS') END)) FROM generate_series(0, 99) AS number GROUP BY id, accepted_at;
SELECT id, delivery_report_at->>'value' AS delivered_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, delivery_report_at->>'value' AS delivered_at FROM agg_test ORDER BY accepted_at DESC LIMIT 10 OFFSET 5;
