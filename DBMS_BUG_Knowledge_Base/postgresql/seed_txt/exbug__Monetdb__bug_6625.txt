CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(50), age INTEGER);
CREATE TABLE orders (order_id SERIAL PRIMARY KEY, user_id INTEGER REFERENCES users(id), amount DECIMAL(10, 2), status VARCHAR(20));
