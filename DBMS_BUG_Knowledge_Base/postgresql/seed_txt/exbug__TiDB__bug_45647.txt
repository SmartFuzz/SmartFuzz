CREATE TABLE products ( id SERIAL PRIMARY KEY, name VARCHAR(50) NOT NULL, price DECIMAL(10,2) NOT NULL, quantity INT DEFAULT 0);

CREATE TABLE sales ( sale_id SERIAL PRIMARY KEY, product_id INT NOT NULL, sale_date DATE NOT NULL, amount DECIMAL(10,2) NOT NULL, FOREIGN KEY (product_id) REFERENCES products(id));
