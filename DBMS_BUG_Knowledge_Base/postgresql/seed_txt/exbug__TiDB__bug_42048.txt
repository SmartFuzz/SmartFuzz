CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(100), balance DECIMAL(10, 2) NOT NULL DEFAULT 0.00);
CREATE TABLE transactions (transaction_id SERIAL PRIMARY KEY, user_id INT, amount DECIMAL(10, 2), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (user_id) REFERENCES users(id));
