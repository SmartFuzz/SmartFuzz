CREATE SCHEMA repro;
CREATE TABLE repro.file_load_history (filename VARCHAR NOT NULL, load_timestamp TIMESTAMP NOT NULL);
CREATE TABLE repro.load_on_first_run (id INT, who VARCHAR, event_date DATE);
INSERT INTO repro.load_on_first_run (id, who, event_date) SELECT 0, 'system', CURRENT_TIMESTAMP::DATE WHERE (SELECT COUNT(1) FROM repro.file_load_history) <= 1;
EXPLAIN SELECT * FROM repro.load_on_first_run;
