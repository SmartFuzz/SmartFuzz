CREATE TABLE test_schema.test_table (id SERIAL PRIMARY KEY, path ltree NOT NULL, description TEXT);
CREATE FUNCTION test_schema.trigger_function() RETURNS trigger LANGUAGE plpgsql AS $$ BEGIN RETURN NEW;
END;
$$;
CREATE TRIGGER path_change_trigger AFTER UPDATE ON test_schema.test_table FOR EACH ROW WHEN ((NEW.path IS DISTINCT FROM OLD.path)) EXECUTE FUNCTION test_schema.trigger_function();
