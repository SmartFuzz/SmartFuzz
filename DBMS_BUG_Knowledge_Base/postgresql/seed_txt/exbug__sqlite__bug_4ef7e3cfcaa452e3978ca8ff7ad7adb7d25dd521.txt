CREATE TABLE a_table(id SERIAL PRIMARY KEY, value TEXT NOT NULL);
CREATE TABLE b_table(id SERIAL PRIMARY KEY, ref_id INTEGER, FOREIGN KEY(ref_id) REFERENCES a_table(id));
CREATE OR REPLACE FUNCTION after_insert_a() RETURNS TRIGGER AS $$ BEGIN INSERT INTO b_table(ref_id) SELECT id FROM a_table WHERE value IS NULL;
RETURN NEW;
END;
$$ LANGUAGE plpgsql;
