CREATE TABLE IF NOT EXISTS initial_complex (n INTEGER, description TEXT, _timestamp TIMESTAMP DEFAULT now());
CREATE TABLE IF NOT EXISTS target1_complex (x INTEGER, status TEXT, _timestamp TIMESTAMP DEFAULT now());
CREATE TABLE IF NOT EXISTS target2_complex (x INTEGER, value FLOAT8, _timestamp TIMESTAMP DEFAULT now());
CREATE MATERIALIZED VIEW initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM initial_complex WHERE random() < 0.5;
CREATE MATERIALIZED VIEW target1_target2_mv_complex AS SELECT x, CAST(x AS FLOAT8) * 1.5 AS value FROM target1_complex WHERE status IS NOT NULL;
