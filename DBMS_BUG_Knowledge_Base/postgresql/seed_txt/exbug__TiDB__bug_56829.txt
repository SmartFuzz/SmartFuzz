CREATE TABLE cache_test (cache_key varchar(512) NOT NULL, cache_value json NOT NULL, created_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP, updated_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP, expires integer DEFAULT -1, expired_at timestamp GENERATED ALWAYS AS (CASE WHEN expires > 0 THEN updated_at + (expires * interval '1 second') ELSE updated_at + interval '99 years' END) STORED, PRIMARY KEY (cache_key), INDEX idx_created_at (created_at), INDEX idx_expired_at (expired_at));
