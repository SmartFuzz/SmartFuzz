DROP TABLE IF EXISTS test_table;
CREATE TABLE test_table (id SERIAL PRIMARY KEY, name TEXT, value FLOAT8, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO test_table (id, name, value, created_at) SELECT number, concat('name_', number::TEXT, ' ', (number % 10)::TEXT), number * 1.5, NOW() FROM generate_series(1, 100) AS number;
SELECT * FROM test_table WHERE value > 50 ORDER BY created_at LIMIT 10;
SELECT id, name FROM test_table WHERE id IN (SELECT id FROM test_table WHERE value < 30) ORDER BY value DESC LIMIT 5;
