CREATE TABLE orders (order_id SERIAL PRIMARY KEY, product_id INT, quantity INT, order_date DATE);

SELECT product_name, price, NTH_VALUE(price, 1) OVER (ORDER BY price ASC) AS cheapest_product FROM products;
SELECT o.order_id, p.product_name, COALESCE(LAG(o.quantity) OVER (PARTITION BY o.product_id ORDER BY o.order_date), 0) AS latest_quantity FROM orders o JOIN products p ON o.product_id = p.product_id;
SELECT product_name, price, NTH_VALUE(product_name, 1) OVER (ORDER BY price DESC) AS most_expensive_product FROM products;
