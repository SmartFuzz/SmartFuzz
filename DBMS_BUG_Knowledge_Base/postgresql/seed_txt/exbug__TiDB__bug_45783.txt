CREATE TABLE categories (id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name varchar(255) NOT NULL, description text);
CREATE TABLE comments (id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, post_id bigint, content text NOT NULL, created_at timestamp DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (post_id) REFERENCES posts(id));
