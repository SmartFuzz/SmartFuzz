DROP DATABASE IF EXISTS sandbox;
CREATE DATABASE sandbox;
DROP TABLE IF EXISTS t_ref;
CREATE TABLE t_ref (id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY);
DROP TABLE IF EXISTS t_main;
CREATE TABLE t_main (id serial PRIMARY KEY, unique_id varchar(32) GENERATED ALWAYS AS ('1')::varchar, t_ref_id bigint NOT NULL, CONSTRAINT unique_key UNIQUE (unique_id), FOREIGN KEY (t_ref_id) REFERENCES t_ref (id));
