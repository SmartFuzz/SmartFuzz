CREATE TABLE sales ( year int DEFAULT NULL, country varchar(20) DEFAULT NULL, product varchar(32) DEFAULT NULL, profit int DEFAULT NULL );

SELECT year + 1 AS y, SUM(profit) AS profit, CASE WHEN (year + 1) IS NOT NULL THEN 0 ELSE 1 END AS grouping FROM sales GROUP BY year + 1 WITH ROLLUP HAVING y > 2002 ORDER BY y, profit;
