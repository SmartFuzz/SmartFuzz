CREATE TABLE customer (c_id SERIAL PRIMARY KEY, c_name VARCHAR(50), c_balance NUMERIC(10, 2) NOT NULL);
CREATE TABLE orders (o_id SERIAL PRIMARY KEY, o_customer_id INT, o_amount NUMERIC(10, 2), FOREIGN KEY (o_customer_id) REFERENCES customer(c_id));

SELECT c.c_name, COALESCE(SUM(o.o_amount), 0) AS total_spent FROM customer c LEFT JOIN orders o ON c.c_id = o.o_customer_id GROUP BY c.c_id;
