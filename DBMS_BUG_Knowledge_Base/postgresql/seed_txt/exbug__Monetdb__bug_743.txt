CREATE TABLE urls(urlid bigint PRIMARY KEY, c bigint, d bigint, url_text varchar(255));
CREATE TABLE keys(urlid bigint, a bigint, b bigint, created_at timestamp DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(urlid) REFERENCES urls(urlid));
INSERT INTO urls(urlid, c, d, url_text) VALUES (1, 10, 20, 'http://example.com/1');
