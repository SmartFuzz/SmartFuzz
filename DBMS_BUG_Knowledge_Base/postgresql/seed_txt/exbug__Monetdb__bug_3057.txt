CREATE TABLE images (imageid SERIAL PRIMARY KEY, ds_id INT, bmaj DOUBLE PRECISION);
CREATE TABLE xtrsrc (xtrsrcid SERIAL PRIMARY KEY, image_id INT REFERENCES images(imageid), zone INT, ra DOUBLE PRECISION, decl DOUBLE PRECISION, ra_err DOUBLE PRECISION, decl_err DOUBLE PRECISION, x DOUBLE PRECISION, y DOUBLE PRECISION, z DOUBLE PRECISION);
