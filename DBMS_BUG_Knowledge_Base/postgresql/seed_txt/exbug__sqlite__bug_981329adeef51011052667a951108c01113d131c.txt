CREATE TABLE test_table (id SERIAL PRIMARY KEY, value TEXT, depth INTEGER NOT NULL DEFAULT 0);
INSERT INTO test_table (value, depth) VALUES ('Initial', 0);
WITH RECURSIVE c(x) AS (SELECT 1 UNION ALL SELECT x + 1 FROM c WHERE x < 400000) SELECT jsonb_valid('[' || string_agg(x::text, ',') || ']') FROM c;
SELECT * FROM test_table WHERE depth IS NULL;
UPDATE test_table SET value = 'Updated' WHERE id = (SELECT id FROM test_table ORDER BY depth DESC LIMIT 1);
