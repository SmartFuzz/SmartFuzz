CREATE TABLE t(p BOOL, q INT, r TEXT DEFAULT 'default');
INSERT INTO t(p, q) VALUES (NULL, 1);
SELECT pg_advisory_lock(2);
WITH lock_t AS (SELECT ctid FROM t WHERE p IS NULL FOR UPDATE) UPDATE t SET q = q + 1 FROM lock_t l WHERE t.ctid = l.ctid RETURNING p, q;
WITH lock_t AS (SELECT ctid FROM t FOR UPDATE) UPDATE t SET r = 'updated' WHERE t.ctid = (SELECT l.ctid FROM lock_t l) RETURNING r;
