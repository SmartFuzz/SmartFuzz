DROP TABLE IF EXISTS t;
CREATE TABLE t(a VARCHAR(10) COLLATE "C", b VARCHAR(10) COLLATE "C", c INT);
INSERT INTO t (a, b, c) VALUES ('a', 'A', 1), ('b', 'B', 2), ('c', 'C', NULL);
SELECT * FROM t WHERE (CASE WHEN LOWER('A') = LOWER(a) THEN 1 ELSE 0 END) + (CASE WHEN LOWER('A') = LOWER(b) THEN 1 ELSE 0 END) > 1;
SELECT * FROM t WHERE c IS NULL OR (CASE WHEN LOWER('B') = LOWER(a) THEN 1 ELSE 0 END) + (CASE WHEN LOWER('B') = LOWER(b) THEN 1 ELSE 0 END) > 1;
