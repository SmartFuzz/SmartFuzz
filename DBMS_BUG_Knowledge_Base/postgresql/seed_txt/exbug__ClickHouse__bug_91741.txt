CREATE TABLE test_parallel_replicas_2 (id SERIAL PRIMARY KEY, value TEXT, timestamp TIMESTAMP);
INSERT INTO test_parallel_replicas_2 (id, value, timestamp) SELECT number, CASE WHEN number % 2 = 0 THEN to_char(number, 'FM9999999999') ELSE NULL END AS value, now() FROM generate_series(0, 99) AS number;
