DROP TABLE IF EXISTS events;
CREATE TABLE events ( Id BIGINT, Payload TEXT, Time TIMESTAMP, Status TEXT );
INSERT INTO events (Id, Payload, Time, Status) SELECT number, concat('Payload_', number::text), '2024-01-01 00:00:00'::timestamp + (number * interval '1 minute'), CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM generate_series(0, 9) AS number;
DROP TABLE IF EXISTS attributes;
CREATE TABLE attributes ( EventId BIGINT, Attribute TEXT, Value FLOAT8 DEFAULT 0.0 );
