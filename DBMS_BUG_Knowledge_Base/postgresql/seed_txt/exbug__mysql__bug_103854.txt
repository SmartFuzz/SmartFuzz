CREATE TABLE B (id SERIAL PRIMARY KEY, entity_id INT NOT NULL, organization_id INT NOT NULL, status VARCHAR(10) NOT NULL CHECK (status IN ('ACTIVE', 'INACTIVE')));
CREATE TABLE C (id SERIAL PRIMARY KEY, status VARCHAR(10) NOT NULL);
