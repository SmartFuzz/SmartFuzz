CREATE TABLE cols (c VARCHAR);
INSERT INTO cols (SELECT 'c' || i FROM generate_series(1, 21) AS i);
CREATE TABLE matrix (col1 VARCHAR, col2 VARCHAR, val SMALLINT NOT NULL CHECK (val < 250));
INSERT INTO matrix (SELECT c1.c, c2.c, CAST(FLOOR(random() * 250) AS SMALLINT) FROM cols c1 JOIN cols c2 ON c1.c != c2);
SELECT col1, col2, SUM(val)::SMALLINT FROM matrix GROUP BY col1, col2 ORDER BY col2 LIMIT 2;
