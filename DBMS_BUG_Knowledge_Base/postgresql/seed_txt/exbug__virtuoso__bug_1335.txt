CREATE TABLE OperationalPoint (id SERIAL PRIMARY KEY, country VARCHAR(100), value INT);
INSERT INTO OperationalPoint (country, value) VALUES ('CountryA', 10), ('CountryB', 20), ('CountryA', 15);
CREATE TABLE CountryStats (country VARCHAR(100), total INT);
INSERT INTO CountryStats (country, total) SELECT country, COUNT(*) FROM OperationalPoint GROUP BY country HAVING COUNT(*) > 1;
SELECT country, (SELECT COUNT(*) FROM OperationalPoint WHERE country = CountryStats.country) AS country_count FROM CountryStats WHERE country_count > 1;
