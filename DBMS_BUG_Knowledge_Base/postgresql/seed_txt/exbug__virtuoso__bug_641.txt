CREATE TABLE actors (id SERIAL PRIMARY KEY, name VARCHAR(100), birth_year INT, is_active BOOLEAN);
CREATE TABLE movies (id SERIAL PRIMARY KEY, title VARCHAR(100), release_year INT, director_id INT, FOREIGN KEY (director_id) REFERENCES actors(id));
