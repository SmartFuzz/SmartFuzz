DROP TABLE IF EXISTS xyz;
CREATE TABLE xyz (id serial PRIMARY KEY, name varchar(255), age integer, created_at timestamp DEFAULT CURRENT_TIMESTAMP);
INSERT INTO xyz (name, age) VALUES ('Alice', 30), ('Bob', NULL);
CREATE OR REPLACE FUNCTION get_age() RETURNS integer AS $body$ DECLARE user_record xyz;
BEGIN FOR user_record IN SELECT * FROM xyz WHERE age IS NOT NULL LOOP RETURN user_record.age;
END LOOP;
RETURN NULL;
END $body$ LANGUAGE plpgsql;
ALTER TABLE xyz DROP COLUMN age;
