DROP TABLE IF EXISTS testUUID;
CREATE TABLE testUUID (ID SERIAL PRIMARY KEY, new_ID VARCHAR(100) NOT NULL, new_ID2 VARCHAR(100), new_ID3 BYTEA UNIQUE);
