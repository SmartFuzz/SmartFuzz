CREATE TABLE test_casts (id INT, dec_value DECIMAL(10, 7), dbl_value DOUBLE PRECISION);
INSERT INTO test_casts (id, dec_value, dbl_value) VALUES (1, 13.8, 13.8);
SELECT id, CAST(dec_value AS INT) AS dec_to_int, CAST(dbl_value AS INT) AS dbl_to_int FROM test_casts;
SELECT id, CAST(CAST(dec_value AS DECIMAL(10, 7)) AS INT) AS dec_casted, CAST(CAST(dbl_value AS DOUBLE PRECISION) AS INT) AS dbl_casted FROM test_casts;
SELECT id, CASE WHEN dec_value IS NULL THEN 'NULL' ELSE CAST(dec_value AS INT) END AS dec_null_handling FROM test_casts;
