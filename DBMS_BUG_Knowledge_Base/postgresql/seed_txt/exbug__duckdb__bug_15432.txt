CREATE TABLE test_data (id INT, values VARCHAR[]);
INSERT INTO test_data VALUES (1, ARRAY['1', '2', '3']), (2, ARRAY['4', '5', NULL]), (3, ARRAY['6', '7', '8']);
SELECT id, ARRAY_LENGTH(values) AS value_count FROM test_data GROUP BY id;
CREATE OR REPLACE FUNCTION custom_sum(lst VARCHAR[]) RETURNS NUMERIC AS $$ SELECT COALESCE(SUM(CAST(unnest(lst) AS NUMERIC)), 0);
$$ LANGUAGE SQL;
SELECT id, custom_sum(values) FROM test_data WHERE values IS NOT NULL GROUP BY id;
