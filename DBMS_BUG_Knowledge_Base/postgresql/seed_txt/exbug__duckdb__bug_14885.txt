CREATE OR REPLACE FUNCTION yearqtr(x DATE) RETURNS INTEGER AS $$ SELECT (((EXTRACT(YEAR FROM x) * 10) + ((EXTRACT(MONTH FROM x) - 1) / 3) + 1));
$$ LANGUAGE SQL;
SELECT dt, yearqtr(dt) AS yq, yearqtr(dt) / 10 AS yq2, yearqtr(dt) / 3 AS yq3, yearqtr(dt) % 3 AS yq4 FROM test_dates WHERE dt = '2050-12-31';
