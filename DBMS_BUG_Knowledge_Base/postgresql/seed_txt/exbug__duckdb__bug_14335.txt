CREATE OR REPLACE FUNCTION dynamic_table(col1_value INTEGER, col2_value VARCHAR) RETURNS TABLE(column1 INTEGER, column2 VARCHAR) AS $$ SELECT col1_value AS column1, col2_value AS column2;
$$ LANGUAGE SQL;
CREATE OR REPLACE FUNCTION dynamic_table_overloaded(col1_value INTEGER, col2_value VARCHAR, col3_value FLOAT) RETURNS TABLE(column1 INTEGER, column2 VARCHAR, column3 FLOAT) AS $$ SELECT col1_value AS column1, col2_value AS column2, col3_value AS column3;
$$ LANGUAGE SQL;
