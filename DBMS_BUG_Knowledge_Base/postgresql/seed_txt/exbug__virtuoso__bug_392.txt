CREATE TABLE automobiles (id SERIAL PRIMARY KEY, manufacturer VARCHAR(255), thumbnail VARCHAR(255), depiction VARCHAR(255), image VARCHAR(255), logo VARCHAR(255));

SELECT id, COUNT(*) AS optional_count FROM automobiles WHERE thumbnail IS NOT NULL OR depiction IS NOT NULL OR image IS NOT NULL OR logo IS NOT NULL GROUP BY id HAVING COUNT(*) > 0;
