CREATE TABLE test_data (id SERIAL PRIMARY KEY, name VARCHAR, age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
COPY test_data(name, age) FROM 'test_duckdb_data.csv' WITH (FORMAT csv, HEADER true);
CREATE VIEW valid_records AS SELECT * FROM test_data WHERE age IS NOT NULL;
CREATE VIEW rejected_records AS SELECT * FROM (SELECT * FROM read_csv('test_duckdb_data.csv', HEADER true)) AS temp WHERE name IS NULL;
SELECT COUNT(*) AS valid_count, (SELECT COUNT(*) FROM rejected_records) AS rejected_count FROM valid_records;
