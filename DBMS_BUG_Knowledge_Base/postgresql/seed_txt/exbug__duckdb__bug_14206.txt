CREATE TABLE nested_types(a JSONB);
INSERT INTO nested_types VALUES (NULL::JSONB), ('{"b": NULL, "c": 1}'::JSONB), ('{"b": NULL, "c": NULL}'::JSONB);
SELECT (a IS NULL) AND (a->>'b' IS NULL) FROM nested_types;
SELECT NOT ((a IS NULL) AND (a->>'c' IS NULL)) FROM nested_types;
SELECT COUNT(*) FROM nested_types WHERE (a IS NULL) OR (a->>'b' IS NULL) AND (a->>'c' IS NULL);
