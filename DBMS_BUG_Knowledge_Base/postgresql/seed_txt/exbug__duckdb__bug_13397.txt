CREATE TABLE test_table (id INTEGER, value INTEGER, description VARCHAR);
INSERT INTO test_table (id, value, description) VALUES (1, 0, 'zero'), (2, 1, 'one'), (3, 2, 'two');
SELECT COUNT(*) FILTER (WHERE value = 0) AS count_zeros FROM test_table;
SELECT COUNT(*) FILTER (WHERE value = 1) AS count_ones FROM test_table;
SELECT COUNT(*) FILTER (WHERE value IS NULL OR value = 0) AS count_zeros_with_nulls FROM (SELECT NULL AS value UNION ALL SELECT value FROM test_table) AS combined;
