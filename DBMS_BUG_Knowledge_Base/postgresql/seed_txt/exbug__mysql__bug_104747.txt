CREATE TABLE fig (id SERIAL PRIMARY KEY, varchar_1 VARCHAR(20), num_col INT DEFAULT 0);
INSERT INTO fig (id, varchar_1, num_col) VALUES (1, 'test', NULL), (2, 'example', 5), (3, NULL, 10);
CREATE VIEW v_substr AS SELECT SUBSTR(varchar_1, CASE WHEN (-(4) = 0) THEN 1 ELSE -(4) END, 4) AS SUBSTR_varchar_1 FROM fig WHERE num_col IS NOT NULL;
CREATE VIEW v_join AS SELECT f.id, v.SUBSTR_varchar_1 FROM fig f JOIN v_substr v ON f.id = 1;
SELECT * FROM v_join;
