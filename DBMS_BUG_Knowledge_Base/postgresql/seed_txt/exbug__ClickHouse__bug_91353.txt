CREATE TABLE test_table1 (id INT, value FLOAT8, created_at TIMESTAMP);
CREATE TABLE test_table2 (id INT, description TEXT, is_active SMALLINT);
CREATE MATERIALIZED VIEW test_mv1 AS SELECT id, value::FLOAT8 FROM test_table2 WHERE is_active = 1;
CREATE OR REPLACE FUNCTION refresh_test_mv2() RETURNS VOID AS $$ BEGIN INSERT INTO test_table2 (id, description) SELECT id, description FROM test_table1 WHERE created_at > NOW() - INTERVAL '5 minutes';
END;
$$ LANGUAGE plpgsql;
