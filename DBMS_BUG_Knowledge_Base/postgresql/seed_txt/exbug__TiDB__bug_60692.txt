CREATE TABLE t1(a INT, b VARCHAR(10), c FLOAT, d DATE, e INT);
CREATE TABLE t2(a INT, b INT, c INT, d INT, e INT DEFAULT 0);
INSERT INTO t1(a, b, c, d, e) VALUES (1, 'test', 1.5, '2023-01-01', 10);
INSERT INTO t2(a, b, c, d, e) VALUES (2, 1, 3, 4, 5);
EXPLAIN SELECT tmp4.b FROM t1 JOIN t2 ON t1.a = t2.b JOIN (SELECT a, b, c, ROW_NUMBER() OVER (PARTITION BY b ORDER BY e ASC) AS rn FROM t2) tmp4 ON tmp4.b = t1.b;
