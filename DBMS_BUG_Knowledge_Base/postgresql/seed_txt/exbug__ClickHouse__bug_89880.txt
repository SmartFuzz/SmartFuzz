CREATE TABLE T1 (K BIGINT, C0 TEXT, C1 TEXT, C2 DOUBLE PRECISION);
INSERT INTO T1 SELECT number, ARRAY(SELECT (i * number) FROM generate_series(0, 9) AS i), CASE WHEN number % 2 = 0 THEN 'even' ELSE NULL END, number * 1.5 FROM generate_series(1, 100) AS number;
CREATE TABLE T2 (K BIGINT, C0 TEXT, C1 INT);
INSERT INTO T2 SELECT K, C0, CAST(K AS INT) FROM T1 WHERE C1 IS NOT NULL;
SELECT * FROM T1 WHERE C1 = 'even';
