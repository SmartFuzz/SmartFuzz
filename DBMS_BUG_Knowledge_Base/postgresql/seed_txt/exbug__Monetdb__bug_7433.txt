CREATE TABLE v1 ( id SERIAL PRIMARY KEY, name VARCHAR(100), age INT CHECK (age >= 0) );
INSERT INTO v1 (id, name, age) VALUES (1, 'Alice', 30), (2, 'Bob', 25), (3, 'Charlie', NULL);
CREATE TABLE v2 ( id INT, score FLOAT, FOREIGN KEY (id) REFERENCES v1(id) );
INSERT INTO v2 (id, score) VALUES (1, 85.5), (2, 90.0), (3, NULL);
UPDATE v1 SET age = (SELECT COUNT(*) FROM v2 WHERE score > 80) WHERE id = 1;
