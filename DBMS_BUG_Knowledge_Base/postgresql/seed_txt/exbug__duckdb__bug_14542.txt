CREATE OR REPLACE FUNCTION json_each(input jsonb[]) RETURNS TABLE(key varchar, value jsonb) AS $$ BEGIN RETURN QUERY SELECT CASE WHEN jsonb_typeof(val) = 'array' THEN unnest(array_length(val::jsonb, 1))::varchar ELSE unnest(jsonb_object_keys(val)) END AS key, val->key AS value FROM unnest(input) AS val;
END;
$$ LANGUAGE plpgsql;
SELECT * FROM (SELECT '[{"key": "value1"}, {"key": "value2"}]'::jsonb[] as message) AS msg CROSS JOIN json_each(message);
