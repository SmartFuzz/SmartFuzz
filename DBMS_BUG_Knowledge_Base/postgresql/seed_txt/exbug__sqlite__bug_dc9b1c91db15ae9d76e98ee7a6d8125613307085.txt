CREATE TABLE t1(a TEXT, b INTEGER, c REAL, d BYTEA);
CREATE TABLE x1(x INTEGER, y INTEGER, z INTEGER, w TEXT NOT NULL);
CREATE OR REPLACE FUNCTION regexp_match(text, text) RETURNS BOOLEAN AS $$ SELECT $1 ~ $2;
$$ LANGUAGE SQL;
CREATE TRIGGER tr1 AFTER INSERT ON t1 FOR EACH ROW EXECUTE PROCEDURE INSERT INTO x1 VALUES(regexp_match(NEW.a, 'abc'), regexp_match(NEW.b::TEXT, 'abc'), regexp_match(NEW.c::TEXT, 'abc'), NEW.d);
CREATE TRIGGER tr2 AFTER INSERT ON t1 FOR EACH ROW EXECUTE PROCEDURE INSERT INTO x1 VALUES(regexp_match(NEW.a, 'def'), regexp_match(NEW.b::TEXT, 'def'), regexp_match(NEW.c::TEXT, 'def'), NULL);
