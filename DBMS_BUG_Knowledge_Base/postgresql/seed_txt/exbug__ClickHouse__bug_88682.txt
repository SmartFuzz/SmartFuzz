CREATE TABLE c (a BIGINT, b BIGINT);
INSERT INTO c SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END FROM generate_series(0, 9);
CREATE TABLE d (m BIGINT, n BIGINT);
INSERT INTO d SELECT number, number + 1 FROM generate_series(0, 4);
SELECT c.a, d.n FROM c LEFT OUTER JOIN d ON c.a > d.m;
