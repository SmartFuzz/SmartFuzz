CREATE EXTENSION IF NOT EXISTS fts5;
CREATE TABLE t1 (id SERIAL PRIMARY KEY, c1 TEXT NOT NULL, c2 BYTEA);
INSERT INTO t1(c1, c2) VALUES (E'\x00', E'\x01');
CREATE VIRTUAL TABLE t2 USING fts5(c3);
INSERT INTO t2(c3) VALUES ((SELECT c1 FROM t1 WHERE id = 1)::TEXT);
SELECT * FROM t2 WHERE SUBSTRING(c3 FROM 1 FOR 1) > '0';
