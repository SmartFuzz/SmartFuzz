CREATE TABLE test_table (id INT, values INT[]);
INSERT INTO test_table VALUES (1, ARRAY[10, 20, 30]), (2, ARRAY[NULL, 40, 50]), (3, ARRAY[60, 70, NULL]);
SELECT id, ARRAY(SELECT ix FROM generate_series(1, array_length(values, 1)) AS ix) AS transformed_ix FROM test_table;
SELECT id, (SELECT ix FROM generate_series(0, array_length(values, 1) - 1) AS ix ORDER BY ix DESC LIMIT 1) AS reduced_ix FROM test_table;
SELECT id, ARRAY(SELECT x + 1 FROM unnest(values) AS x) AS incremented_values FROM test_table WHERE values IS NOT NULL;
