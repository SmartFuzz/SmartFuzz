CREATE TABLE test_table (id SERIAL PRIMARY KEY, flag BOOLEAN, description TEXT);
INSERT INTO test_table (flag, description) VALUES (true, 'First entry'), (false, 'Second entry'), (NULL, 'Third entry');
CREATE OR REPLACE FUNCTION evaluate_flag(b BOOLEAN) RETURNS TEXT AS $$ BEGIN IF b THEN RETURN 'true';
ELSIF NOT b THEN RETURN 'false';
ELSE RETURN 'null';
END IF;
