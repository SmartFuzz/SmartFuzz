CREATE TABLE t1 (c0 FLOAT8[], c1 INT, c2 TEXT);
INSERT INTO t1 (c0, c1, c2) SELECT ARRAY[CAST(number % 10 AS FLOAT8), CAST((-number) AS FLOAT8)], number, 'test' FROM generate_series(0, 19) AS number;
CREATE TABLE t2 (c0 INT, c1 FLOAT8, c2 TIMESTAMP);
INSERT INTO t2 (c0, c1, c2) SELECT number, NULL::FLOAT8, NOW() FROM generate_series(0, 9) AS number;
SELECT t1.c0, t1.c1 FROM t1 JOIN t2 ON t1.c1 = t2.c0 ORDER BY t1.c0, t1.c1;
