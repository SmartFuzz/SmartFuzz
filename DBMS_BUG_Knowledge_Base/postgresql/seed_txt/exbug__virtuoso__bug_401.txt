CREATE TABLE test_dates (id SERIAL PRIMARY KEY, event_date DATE NOT NULL);
INSERT INTO test_dates (event_date) VALUES (CURRENT_DATE), (CURRENT_DATE - INTERVAL '1 day'), (CURRENT_DATE - INTERVAL '2 days');
SELECT id, event_date, CURRENT_DATE - INTERVAL '1 day' AS yesterday FROM test_dates WHERE event_date < CURRENT_DATE;
SELECT id, event_date FROM test_dates WHERE event_date = CURRENT_DATE - INTERVAL '1 day' OR event_date IS NULL;
SELECT DISTINCT CURRENT_DATE AS today, CURRENT_DATE - INTERVAL '1 day' AS yesterday;
