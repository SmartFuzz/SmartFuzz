CREATE TABLE test_graph (id SERIAL PRIMARY KEY, graph_uri VARCHAR(255), annotated_at TIMESTAMPTZ);
CREATE TABLE test_address (id SERIAL PRIMARY KEY, address_uri VARCHAR(255), target_graph INT, address_timestamp TIMESTAMPTZ, FOREIGN KEY (target_graph) REFERENCES test_graph(id));

SELECT g.graph_uri, g.annotated_at, COUNT(a.address_timestamp) AS addresses_count, MAX(a.address_timestamp) AS most_recent_timestamp FROM test_graph g LEFT JOIN test_address a ON g.id = a.target_graph GROUP BY g.id;
