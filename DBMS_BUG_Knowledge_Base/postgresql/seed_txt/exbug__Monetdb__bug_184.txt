CREATE TABLE artists (id SERIAL PRIMARY KEY, name VARCHAR(255), age INT CHECK (age >= 0));
CREATE TABLE songs (id SERIAL PRIMARY KEY, title VARCHAR(255), artist_id INT REFERENCES artists(id));
