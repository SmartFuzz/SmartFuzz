CREATE TABLE test_table (id serial PRIMARY KEY, name text NOT NULL, value integer CHECK (value >= 0));
CREATE INDEX "CaseSensitiveIndex" ON test_table (name, value);
CREATE INDEX not_case_sensitive ON test_table (value, id);
SELECT pg_relation_size(indexrelname::regclass) FROM pg_stat_all_indexes WHERE indexrelname = 'not_case_sensitive';
SELECT pg_relation_size(indexrelname::regclass) FROM pg_stat_all_indexes WHERE indexrelname = '"CaseSensitiveIndex"';
