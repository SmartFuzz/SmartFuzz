CREATE TABLE scene (id SERIAL PRIMARY KEY, victim_id INT NOT NULL, description VARCHAR(255));
CREATE TABLE victim (id SERIAL PRIMARY KEY, name VARCHAR(255), age INT CHECK (age >= 0));
CREATE TABLE inspector (id SERIAL PRIMARY KEY, name VARCHAR(255), experience_years INT DEFAULT 0);
