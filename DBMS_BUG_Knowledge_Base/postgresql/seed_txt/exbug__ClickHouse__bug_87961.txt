CREATE TABLE IF NOT EXISTS user_activity (activity_id BIGINT, user_id INTEGER, activity_time TIMESTAMP, activity_type VARCHAR(10), details TEXT);
INSERT INTO user_activity SELECT number AS activity_id, FLOOR(RANDOM() * 1000) AS user_id, NOW() + (number * INTERVAL '5 seconds') AS activity_time, CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END AS activity_type, 'Activity details for ' || number AS details FROM generate_series(1, 100) AS number;
CREATE TABLE IF NOT EXISTS product_sales (sale_id INTEGER, product_id INTEGER, sale_time TIMESTAMP, quantity SMALLINT, price REAL);
INSERT INTO product_sales SELECT number AS sale_id, FLOOR(RANDOM() * 500) AS product_id, NOW() + (number * INTERVAL '2 seconds') AS sale_time, (FLOOR(RANDOM() * 10) + 1) AS quantity, (FLOOR(RANDOM() * 100) + 1) * 1.5 AS price FROM generate_series(1, 50) AS number;
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL '1 day' GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
