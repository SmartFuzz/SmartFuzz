CREATE OR REPLACE FUNCTION fetch_data(url VARCHAR, format VARCHAR) RETURNS TABLE(data JSONB) AS $$ BEGIN RETURN QUERY SELECT CASE WHEN format = 'JSON' THEN (SELECT * FROM jsonb_populate_recordset(NULL::record, pg_read_file(url))) ELSE (SELECT * FROM csv_read(url)) END;
END;
$$ LANGUAGE plpgsql;
