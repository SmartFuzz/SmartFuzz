CREATE TABLE courses (course_id SERIAL PRIMARY KEY, course_name VARCHAR, credits INTEGER CHECK (credits > 0));
CREATE TABLE enrollments (enrollment_id SERIAL PRIMARY KEY, student_id INTEGER REFERENCES students(id), course_id INTEGER REFERENCES courses(course_id), grade INTEGER);
