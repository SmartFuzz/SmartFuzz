CREATE TABLE t1(c0 SERIAL PRIMARY KEY, c1 TEXT, c2 INTEGER GENERATED ALWAYS AS (c0 + 1) STORED, FOREIGN KEY(c2) REFERENCES t1(c0));
CREATE TABLE t2(c0 INTEGER, c1 REAL, c2 TEXT, c3 INTEGER, UNIQUE(c0, c1));
INSERT INTO t1 (c0, c1) VALUES (1, 'test1'), (2, 'test2');
INSERT INTO t2 (c0, c1, c2, c3) VALUES (1, 1.5, 'data1', 1), (2, 2.5, 'data2', 2);
UPDATE t1 SET c1 = (SELECT c1 FROM t2 WHERE c0 = t1.c0) WHERE EXISTS (SELECT 1 FROM t2 WHERE c0 = t1.c0);
