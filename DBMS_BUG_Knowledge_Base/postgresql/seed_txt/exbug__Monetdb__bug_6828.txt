CREATE TABLE departments (id SERIAL PRIMARY KEY, name VARCHAR(50) NOT NULL);
INSERT INTO departments (id, name) VALUES (1, 'Management'), (2, 'Development'), (3, 'Sales');
CREATE TABLE employees (dep_id INT, name VARCHAR(20), salary DOUBLE PRECISION, hire_date DATE, FOREIGN KEY (dep_id) REFERENCES departments(id));
INSERT INTO employees (dep_id, name, salary, hire_date) VALUES (1, 'Smith', 81, TO_DATE('10-08-2019', 'DD-MM-YYYY')), (2, 'Jones', 55, TO_DATE('09-08-2019', 'DD-MM-YYYY')), (3, 'Williams', 55, TO_DATE('14-07-2019', 'DD-MM-YYYY'));
SELECT dep_id, name, salary, SUM(salary) OVER (ORDER BY EXTRACT(YEAR FROM hire_date)) AS salary_sum FROM employees;
