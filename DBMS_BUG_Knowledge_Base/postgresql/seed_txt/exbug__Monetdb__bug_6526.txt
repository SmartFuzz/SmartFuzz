CREATE TABLE t2 (id SERIAL PRIMARY KEY, c1 VARCHAR(50) NOT NULL, d1 INT CHECK (d1 >= 0), d2 INT DEFAULT 0);
INSERT INTO t2 (c1, d1, d2) VALUES ('X', 100, 200), ('Y', 150, 300), ('Z', 0, 0), ('X', 250, 400), ('Y', 300, 500);
SELECT c1, SUM(d1) AS total_d1, SUM(d2) AS total_d2 FROM t2 GROUP BY c1 HAVING SUM(d1) < CASE WHEN SUM(d2) > 100 THEN 600 ELSE 500 END;
SELECT c1, COUNT(*) AS count, AVG(d1) AS avg_d1 FROM t2 GROUP BY c1 HAVING AVG(d1) > CASE WHEN COUNT(*) > 2 THEN 100 ELSE 50 END;
SELECT c1, SUM(d1) AS total_d1, SUM(d2) AS total_d2, COALESCE(SUM(d1), 0) AS safe_sum FROM t2 GROUP BY c1 HAVING safe_sum < 300;
