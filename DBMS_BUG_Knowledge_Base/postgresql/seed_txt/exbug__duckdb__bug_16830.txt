CREATE TABLE test_regex (id SERIAL PRIMARY KEY, pattern VARCHAR, test_string VARCHAR);
INSERT INTO test_regex (pattern, test_string) VALUES ('bc', 'abcd'), ('a.c', 'abcd'), ('a.*d', 'abcd'), ('(b|x)', 'abcd'), ('^a', 'abcd'), ('^(b|c)', 'abcd');
SELECT id, test_string ~ pattern AS match_result FROM test_regex;
SELECT id, test_string ~ 'a.*d' AS match_result FROM test_regex WHERE pattern = 'a.*d';
SELECT id, test_string ~ '^(b|c)' AS match_result FROM test_regex WHERE test_string IS NOT NULL;
