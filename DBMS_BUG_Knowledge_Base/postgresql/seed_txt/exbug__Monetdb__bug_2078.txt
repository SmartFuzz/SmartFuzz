CREATE TABLE test3 (id SERIAL PRIMARY KEY, value DOUBLE PRECISION NOT NULL, description TEXT);
INSERT INTO test3 (value, description) VALUES (0.0, 'Zero value'), (1.0, 'Positive value'), (-1.0, 'Negative value');
SELECT CASE WHEN value > 0 THEN 1.0/value ELSE NULL END AS inverse FROM test3;
SELECT CASE WHEN (SELECT COUNT(*) FROM test3 WHERE value = 0) > 0 THEN 'Division by zero' ELSE 'Safe' END AS status;
SELECT t1.id, t1.value, t2.value, CASE WHEN (t2.value - t1.value) = 0 THEN NULL ELSE (t2.value + t1.value)/(t2.value - t1.value) END AS ratio FROM test3 t1 JOIN test3 t2 ON t1.id <> t2.id;
