CREATE TABLE users (id SERIAL PRIMARY KEY, name TEXT NOT NULL, age INTEGER);
CREATE TABLE posts (id SERIAL PRIMARY KEY, user_id INTEGER, content TEXT, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id SERIAL PRIMARY KEY, post_id INTEGER, user_id INTEGER, comment TEXT, FOREIGN KEY(post_id) REFERENCES posts(id), FOREIGN KEY(user_id) REFERENCES users(id));
