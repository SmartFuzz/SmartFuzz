CREATE TABLE test_graph (id SERIAL PRIMARY KEY, name VARCHAR(255), cost INT);

SELECT id, name, COALESCE(cost, 0) AS cost FROM test_graph WHERE cost IS NOT NULL;
SELECT SUM(cost) AS total_cost FROM test_graph WHERE cost IS NOT NULL GROUP BY name;
SELECT id, name, CASE WHEN cost IS NULL THEN 'No Cost' ELSE 'Has Cost' END AS cost_status FROM test_graph;
