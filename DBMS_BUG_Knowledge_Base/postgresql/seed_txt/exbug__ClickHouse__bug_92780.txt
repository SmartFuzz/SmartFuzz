CREATE TABLE test_data (id SERIAL PRIMARY KEY, ts TIMESTAMP, value TEXT, status TEXT);
INSERT INTO test_data (id, ts, value, status) SELECT number, NOW() + INTERVAL '1 second' * number, (number % 5)::TEXT, CASE WHEN number % 2 = 0 THEN 'active' ELSE NULL END FROM generate_series(1, 1000000) AS number;
CREATE TABLE filtered_data AS SELECT id, ts FROM test_data WHERE status IS NOT NULL;
SELECT COUNT(*) FROM filtered_data WHERE ts > NOW() - INTERVAL '1 DAY' AND value LIKE '1%';
SELECT DISTINCT id FROM test_data WHERE ts BETWEEN '2023-01-01' AND '2023-12-31' AND status IS NOT NULL;
