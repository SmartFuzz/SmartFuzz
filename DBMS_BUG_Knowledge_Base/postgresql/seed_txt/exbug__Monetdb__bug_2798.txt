CREATE TABLE test_table (id SERIAL PRIMARY KEY, value NUMERIC(10, 2), category VARCHAR(50));
INSERT INTO test_table (id, value, category) VALUES (1, 100.00, 'A'), (2, 200.50, 'B'), (3, NULL, 'A');
SELECT category AS cat, COUNT(id) AS count_id, AVG(value) AS avg_value FROM test_table GROUP BY category;
SELECT category AS cat, SUM(value) AS total_value, (SUM(value) / NULLIF(COUNT(*), 0)) AS avg_per_category FROM test_table GROUP BY category;
SELECT category AS cat, SUM(value) AS total_value, (2 * (SUM(value) / NULLIF(COUNT(*), 0))) AS adjusted_value FROM test_table GROUP BY category;
