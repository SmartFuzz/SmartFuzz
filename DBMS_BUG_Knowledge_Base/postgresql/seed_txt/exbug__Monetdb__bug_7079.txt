CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(50), age INT CHECK (age >= 0));
CREATE TABLE orders (order_id SERIAL PRIMARY KEY, user_id INT, amount DECIMAL(10, 2), FOREIGN KEY (user_id) REFERENCES users(id));
