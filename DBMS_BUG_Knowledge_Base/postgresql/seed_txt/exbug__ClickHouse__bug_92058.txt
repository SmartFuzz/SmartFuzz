CREATE TABLE dst (e VARCHAR CHECK (e IN ('a', 'b')), f INT);
CREATE TABLE src (e VARCHAR CHECK (e = 'a'), g INT);
CREATE MATERIALIZED VIEW mv AS SELECT e, g FROM src;
INSERT INTO src (e, g) SELECT CASE WHEN number % 1 = 0 THEN 'a' END, number % 10 FROM generate_series(1, 10) AS number;
INSERT INTO dst (e, f) SELECT e, g FROM mv WHERE g IS NOT NULL;
