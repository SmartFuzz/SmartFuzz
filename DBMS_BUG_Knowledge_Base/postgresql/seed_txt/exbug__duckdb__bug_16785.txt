CREATE TABLE IF NOT EXISTS users (id SERIAL PRIMARY KEY, username VARCHAR NOT NULL UNIQUE);
CREATE TABLE IF NOT EXISTS orders (order_id SERIAL PRIMARY KEY, user_id INTEGER, amount DOUBLE PRECISION NOT NULL, FOREIGN KEY (user_id) REFERENCES users(id));
