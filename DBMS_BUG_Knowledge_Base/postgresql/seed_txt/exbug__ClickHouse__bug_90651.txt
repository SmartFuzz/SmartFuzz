CREATE DATABASE test_db;
CREATE TABLE test_distr (num INT, date TIMESTAMP, value FLOAT DEFAULT 0.0);
INSERT INTO test_distr SELECT generate_series(1, 100) AS num, now() AS date, generate_series(1, 100) * 1.5 AS value;
CREATE TABLE replicated_table (num BIGINT, date DATE, metric FLOAT);
INSERT INTO replicated_table SELECT num, date, (SELECT date FROM test_distr WHERE num = replicated_table.num) AS metric FROM test_distr WHERE num IS NOT NULL;
