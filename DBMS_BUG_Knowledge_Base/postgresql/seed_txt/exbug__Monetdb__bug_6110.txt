CREATE TABLE t_fuzzing (id SERIAL PRIMARY KEY, flag BOOLEAN, description VARCHAR(10));
INSERT INTO t_fuzzing (flag, description) VALUES (true, 'True');
INSERT INTO t_fuzzing (flag, description) VALUES (false, 'False');
SELECT id, flag, flag::VARCHAR(10) as flag_str FROM t_fuzzing ORDER BY id;
SELECT id, flag, description::BOOLEAN as desc_to_bool FROM t_fuzzing WHERE flag IS NOT NULL ORDER BY id;
