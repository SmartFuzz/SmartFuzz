CREATE TABLE users (id SERIAL PRIMARY KEY, name VARCHAR(100), age INT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (order_id SERIAL PRIMARY KEY, user_id INT, amount NUMERIC(10, 2), order_date DATE, FOREIGN KEY (user_id) REFERENCES users(id));
