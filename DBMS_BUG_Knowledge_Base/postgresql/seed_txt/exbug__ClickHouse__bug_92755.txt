DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR, event_time TIMESTAMP, id BIGINT, msg VARCHAR, value FLOAT8);
INSERT INTO repro.events SELECT (number % 100)::VARCHAR AS tenant, NOW() - (number % 200000) * INTERVAL '1 second' AS event_time, number AS id, 'm' || (number::VARCHAR) AS msg, (number % 1000)::FLOAT8 AS value FROM generate_series(0, 1999999) AS number;
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
