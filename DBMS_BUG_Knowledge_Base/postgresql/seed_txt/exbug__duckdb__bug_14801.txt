CREATE OR REPLACE FUNCTION s(str VARCHAR, re VARCHAR, replacement VARCHAR) RETURNS VARCHAR AS $$ DECLARE r RECORD;
BEGIN SELECT regexp_matches(str, re) INTO r;
RETURN r[1] || replacement || r[2];
END;
