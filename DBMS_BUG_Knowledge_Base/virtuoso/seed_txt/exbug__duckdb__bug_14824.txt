CREATE TABLE RBIG_EXTENDED (id INTEGER, src INTEGER, dest INTEGER, weight FLOAT, created_at TIMESTAMP DEFAULT NOW());
INSERT INTO RBIG_EXTENDED SELECT *, RAND() * 100 FROM RBIG;
SELECT COUNT(*) FROM RBIG_EXTENDED WHERE weight IS NULL;
SELECT R1.src, R2.dest FROM RBIG_EXTENDED AS R1 JOIN RBIG_EXTENDED AS R2 ON R1.dest = R2.src WHERE R1.weight > 50;
SELECT AVG(weight) FROM RBIG_EXTENDED WHERE created_at < NOW() - INTERVAL '1 day';
