DROP DATABASE IF EXISTS test_db;
CREATE DATABASE test_db;
CREATE TABLE test_db.test_table (id xsd:integer, name xsd:string, created_at xsd:dateTime);
CREATE VIEW test_db.test_view1 AS SELECT id, name FROM test_db.test_table WHERE created_at > NOW() - INTERVAL '1' DAY;
SELECT * FROM test_db.test_view1 WHERE id = NULL;
