CREATE TABLE employees (dep VARCHAR(10) NOT NULL, name VARCHAR(20) NOT NULL, salary DOUBLE PRECISION CHECK (salary > 0), hire_date DATE DEFAULT NULL);
INSERT INTO employees (dep, name, salary, hire_date) VALUES ('mgmt', 'Smith', 81, '2019-10-08');
INSERT INTO employees (dep, name, salary, hire_date) VALUES ('dev', 'Jones', 55, '2019-09-08');
SELECT dep, name, salary, FLOOR((ROW_NUMBER() OVER (ORDER BY hire_date DESC) - 1) / (COUNT(*) OVER () / 4)) + 1 AS ntile FROM employees;
SELECT dep, name, salary, FLOOR((ROW_NUMBER() OVER (ORDER BY hire_date) - 1) / (COUNT(*) OVER () / 4)) + 1 AS ntile FROM employees WHERE salary IS NOT NULL;
