DROP DATABASE IF EXISTS repro;
CREATE DATABASE repro;
CREATE TABLE repro.events (tenant VARCHAR, event_time TIMESTAMP, id BIGINT, msg VARCHAR, value FLOAT);
INSERT INTO repro.events SELECT CAST(number % 100 AS VARCHAR) AS tenant, NOW() - (number % 200000) AS event_time, number AS id, CONCAT('m', CAST(number AS VARCHAR)) AS msg, CAST(number % 1000 AS FLOAT) AS value FROM (SELECT ROW_NUMBER() OVER () AS number FROM rdf::get_numbers(2000000));
SELECT * FROM repro.events WHERE value IS NULL ORDER BY event_time DESC LIMIT 5;
