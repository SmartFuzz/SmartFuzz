CREATE TABLE cache_test (cache_key varchar(512) NOT NULL, cache_value varchar(2048) NOT NULL, created_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP, updated_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP, expires integer DEFAULT -1, expired_at timestamp AS (CASE WHEN expires > 0 THEN updated_at + expires * INTERVAL '1' SECOND ELSE updated_at + INTERVAL '99' YEAR END) NOT NULL, PRIMARY KEY (cache_key), INDEX idx_created_at (created_at), INDEX idx_expired_at (expired_at));
