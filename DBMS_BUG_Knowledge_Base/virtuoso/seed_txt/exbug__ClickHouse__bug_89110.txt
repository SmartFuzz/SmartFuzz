CREATE TABLE IF NOT EXISTS initial_complex (n          INTEGER, description VARCHAR, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE IF NOT EXISTS target1_complex (x          INTEGER, status     VARCHAR NULL, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE IF NOT EXISTS target2_complex (x          INTEGER, value      FLOAT, _timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE MATERIALIZED VIEW IF NOT EXISTS initial_target1_mv_complex AS SELECT n AS x, CASE WHEN n % 2 = 0 THEN 'even' ELSE 'odd' END AS status FROM initial_complex WHERE RAND() < 0.5;
CREATE MATERIALIZED VIEW IF NOT EXISTS target1_target2_mv_complex AS SELECT x, CAST(x AS FLOAT) * 1.5 AS value FROM target1_complex WHERE status IS NOT NULL;
INSERT INTO initial_complex(n, description) SELECT number AS n, 'sample' AS description FROM DB.DUAL WHERE number < 11;
