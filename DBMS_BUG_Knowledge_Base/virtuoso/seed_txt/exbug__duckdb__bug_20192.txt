CREATE TABLE products (id INTEGER PRIMARY KEY, name VARCHAR, description VARCHAR, price DECIMAL(10, 2), created_at TIMESTAMP);
CREATE TABLE orders (id INTEGER PRIMARY KEY, product_id INTEGER, order_date TIMESTAMP, quantity INTEGER, large_blob BLOB, FOREIGN KEY (product_id) REFERENCES products(id));
INSERT INTO products SELECT i AS id, CONCAT('Product ', i) AS name, CONCAT('Description for product ', i) AS description, ROUND(RAND() * 100, 2) AS price, TIMESTAMPADD(DAY, i, '2024-01-01 00:00:00') AS created_at FROM (SELECT ROW_NUMBER() OVER () AS i FROM rdf_data);
INSERT INTO orders SELECT i AS id, (i % 10000) + 1 AS product_id, TIMESTAMPADD(DAY, i, '2024-02-01 00:00:00') AS order_date, (i % 10) + 1 AS quantity, CAST(REPEAT('y', 2000) AS BLOB) AS large_blob FROM (SELECT ROW_NUMBER() OVER () AS i FROM rdf_data);
SELECT o.product_id, o.large_blob FROM orders o WHERE o.order_date BETWEEN '2024-02-01 00:00:00' AND '2024-04-01 00:00:00' AND o.product_id IN (SELECT p.id FROM products p WHERE p.name LIKE 'Product 5%');
