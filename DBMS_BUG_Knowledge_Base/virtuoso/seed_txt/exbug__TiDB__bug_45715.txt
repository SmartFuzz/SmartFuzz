CREATE TABLE sales ( year INTEGER NOT NULL, country VARCHAR(20) NOT NULL, product VARCHAR(32) NOT NULL, profit INTEGER DEFAULT 0, PRIMARY KEY (year, country, product));
INSERT INTO sales (year, country, product, profit) VALUES (2020, 'USA', 'Gadget', 1000), (2020, 'USA', 'Widget', 1500), (2020, 'Canada', 'Gadget', 2000), (2021, 'USA', 'Gadget', 1200);
SELECT year, country, product, SUM(profit) AS total_profit FROM sales GROUP BY year, country, product ORDER BY year, country;
