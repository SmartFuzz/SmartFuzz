CREATE TABLE courses (course_id INTEGER PRIMARY KEY, course_name VARCHAR, credits INTEGER CHECK (credits > 0));
CREATE TABLE enrollments (enrollment_id INTEGER PRIMARY KEY, student_id INTEGER, course_id INTEGER, grade INTEGER, FOREIGN KEY (student_id) REFERENCES students(id), FOREIGN KEY (course_id) REFERENCES courses(course_id));
