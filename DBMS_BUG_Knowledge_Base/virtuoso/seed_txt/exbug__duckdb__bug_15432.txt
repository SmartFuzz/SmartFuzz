CREATE TABLE test_data (id INT, values VARCHAR[]);
INSERT INTO test_data (id, values) VALUES (1, ['1', '2', '3']), (2, ['4', '5', NULL]), (3, ['6', '7', '8']);
SELECT id, SUM(CAST(value AS FLOAT)) AS total FROM test_data VALUES UNNEST(values) AS value GROUP BY id;
CREATE OR REPLACE FUNCTION custom_sum(lst VARCHAR[]) AS (SELECT SUM(CAST(value AS FLOAT)) FROM VALUES UNNEST(lst) AS value);
SELECT id, custom_sum(values) AS total FROM test_data WHERE values IS NOT NULL GROUP BY id;
