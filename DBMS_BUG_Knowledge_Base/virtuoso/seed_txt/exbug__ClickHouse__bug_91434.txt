CREATE TABLE default.json_test_v2 (json BLOB, col_b INT, col_c VARCHAR);
INSERT INTO default.json_test_v2 SELECT number AS a, number * 2 AS b, CAST(number * 3 AS VARCHAR) AS c FROM (SELECT ROW_NUMBER() OVER () AS number FROM dbms_random.value(1, 10000000));
SELECT col_b, col_c FROM default.json_test_v2 WHERE col_b IS NOT NULL;
SELECT json_get(json, 'b') AS direct_b, json_get(json, 'c') AS direct_c FROM default.json_test_v2 WHERE json_get(json, 'b') > 5000000;
SELECT col_b, COUNT(*) FROM default.json_test_v2 GROUP BY col_b HAVING col_b < 100;
