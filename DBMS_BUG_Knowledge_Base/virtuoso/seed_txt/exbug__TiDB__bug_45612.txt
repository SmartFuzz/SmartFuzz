CREATE TABLE sales (year INTEGER, country VARCHAR(20), product VARCHAR(32), profit INTEGER, quantity INTEGER);
INSERT INTO sales (year, country, product, profit, quantity) VALUES (2001, 'USA', 'Widget', 100, 10), (2002, 'Canada', 'Gadget', 150, 20), (2003, 'USA', 'Widget', 200, 30);
SELECT year + 1 AS adjusted_year, SUM(profit) AS total_profit FROM sales GROUP BY adjusted_year HAVING adjusted_year > 2002;
SELECT year + 2 AS adjusted_year, SUM(profit) AS profit FROM sales GROUP BY adjusted_year HAVING adjusted_year > 2002;
SELECT country, SUM(profit) AS total_profit FROM sales GROUP BY country HAVING country IS NOT NULL;
