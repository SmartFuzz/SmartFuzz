CREATE TABLE points (id INTEGER PRIMARY KEY, value FLOAT NOT NULL, description VARCHAR(255));

CREATE TABLE relationships (source_id INTEGER, target_id INTEGER, FOREIGN KEY (source_id) REFERENCES points(id), FOREIGN KEY (target_id) REFERENCES points(id));
