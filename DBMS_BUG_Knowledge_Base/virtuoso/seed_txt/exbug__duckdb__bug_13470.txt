CREATE TABLE testgroup2 (id int, isB2b int, priceGross decimal(10, 2), priceNet decimal(10, 2), discount decimal(10, 2));
INSERT INTO testgroup2 VALUES (1, 1, 100.00, 80.00, 10.00), (2, 0, 150.00, 120.00, 15.00);
SELECT SUM(CASE WHEN isB2b = 1 THEN priceGross - discount ELSE priceNet END) AS total_price FROM testgroup2;
SELECT SUM(CASE WHEN isB2b = 1 THEN priceGross ELSE priceNet + discount END) AS conditional_total FROM testgroup2;
SELECT COALESCE(SUM(CASE WHEN isB2b = 1 THEN priceGross ELSE NULL END), AVG(priceNet)) AS final_value FROM testgroup2;
