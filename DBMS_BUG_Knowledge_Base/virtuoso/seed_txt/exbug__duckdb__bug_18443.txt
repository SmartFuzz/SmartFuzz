DROP TABLE IF EXISTS products;
CREATE TABLE products (product_id INTEGER, name VARCHAR(255) NOT NULL, price DECIMAL(10,2) CHECK(price >= 0), created_at TIMESTAMP DEFAULT NOW());
INSERT INTO products (product_id, name, price, created_at) VALUES (1, 'Widget', 19.99, TIMESTAMP '2023-01-01 10:00:00'), (2, 'Gadget', 29.99, TIMESTAMP '2023-02-01 11:00:00'), (3, 'Thingamajig', 39.99, TIMESTAMP '2023-03-01 12:00:00');
SELECT p.product_id, p.name, p.price FROM products p WHERE p.price > (SELECT AVG(price) FROM products) AND p.created_at < NOW();
WITH recent_products AS (SELECT * FROM products WHERE created_at >= TIMESTAMP '2023-01-01 00:00:00') SELECT rp.product_id FROM recent_products rp WHERE EXISTS (SELECT 1 FROM recent_products WHERE price < rp.price);
