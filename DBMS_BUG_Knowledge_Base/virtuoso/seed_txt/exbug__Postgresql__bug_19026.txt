CREATE OR REPLACE FUNCTION fuzz_part_hashint4_noop(value INT, seed BIGINT) RETURNS BIGINT AS BEGIN SELECT value + seed;
RETURN value + seed;
END;
CREATE OPERATOR CLASS fuzz_part_int4_ops FOR TYPE INT USING hash AS FUNCTION 2 fuzz_part_hashint4_noop(INT, BIGINT);
CREATE TABLE fuzz_pt(i INT) PARTITION BY HASH (i);
CREATE TABLE fuzz_p1 PARTITION OF fuzz_pt FOR VALUES WITH (modulus 4, remainder 0);
