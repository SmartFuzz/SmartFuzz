CREATE TABLE test_json_1 AS SELECT JSON '{"is_active": false, "is_true": true}' AS data;
CREATE TABLE test_json_2 AS SELECT JSON '{"is_active": null, "is_true": false}' AS data;
SELECT * FROM test_json_1 WHERE NOT (data -> 'is_active');
SELECT * FROM test_json_2 WHERE NOT (data -> 'is_true');
SELECT * FROM test_json_1 WHERE NOT (data -> 'is_active') AND (data -> 'is_true') IS NOT NULL;
