SELECT category, STRAGG(DISTINCT name) AS a_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
SELECT category, STRAGG(name) AS b_list FROM test_data WHERE id IS NOT NULL GROUP BY category;
