CREATE DATABASE test_db;
CREATE TABLE test_distr (num INT, date TIMESTAMP, value FLOAT DEFAULT 0.0);
INSERT INTO test_distr SELECT number AS num, NOW() AS date, number * 1.5 AS value FROM DB.DUAL WHERE ROWNUM <= 100;
CREATE TABLE replicated_table (num BIGINT, date DATE, metric FLOAT) AS SELECT num, date, NULL AS metric FROM test_distr WHERE num IS NOT NULL;
INSERT INTO replicated_table SELECT num, date, (SELECT date FROM test_distr WHERE num = replicated_table.num) AS metric FROM replicated_table;
