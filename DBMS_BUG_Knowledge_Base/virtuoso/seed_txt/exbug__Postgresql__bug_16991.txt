CREATE TABLE test_table (id INTEGER PRIMARY KEY, name NVARCHAR(255) NOT NULL, value INTEGER CHECK (value >= 0));
CREATE INDEX CaseSensitiveIndex ON test_table (name, value);
CREATE INDEX not_case_sensitive ON test_table (value, id);
SELECT DB.DBA.get_relation_size('not_case_sensitive') AS size FROM DB.DBA.indexes WHERE index_name = 'not_case_sensitive';
SELECT DB.DBA.get_relation_size('CaseSensitiveIndex') AS size FROM DB.DBA.indexes WHERE index_name = 'CaseSensitiveIndex';
