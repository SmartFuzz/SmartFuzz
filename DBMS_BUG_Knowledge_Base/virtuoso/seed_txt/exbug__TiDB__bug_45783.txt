CREATE TABLE comments (id INTEGER NOT NULL PRIMARY KEY, post_id INTEGER, content LONG VARCHAR NOT NULL, created_at TIMESTAMP DEFAULT NOW(), FOREIGN KEY (post_id) REFERENCES posts(id));
