CREATE TABLE dst (e VARCHAR(10), f INT);
CREATE TABLE src (e VARCHAR(10), g INT NULL);
CREATE VIEW mv AS SELECT e, g FROM src;
INSERT INTO src (e, g) SELECT CASE WHEN number % 1 = 0 THEN 'a' ELSE NULL END AS e, number % 10 AS g FROM generate_series(1, 10);
INSERT INTO dst (e, f) SELECT e, g FROM mv WHERE g IS NOT NULL;
