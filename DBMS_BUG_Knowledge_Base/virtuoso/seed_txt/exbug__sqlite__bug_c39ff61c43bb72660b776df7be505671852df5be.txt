CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR(255) NOT NULL, age INTEGER CHECK(age >= 0));
CREATE TABLE comments (id INTEGER PRIMARY KEY, user_id INTEGER, content VARCHAR(255) NOT NULL, FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE);
INSERT INTO users (name, age) VALUES ('Alice', 30);
