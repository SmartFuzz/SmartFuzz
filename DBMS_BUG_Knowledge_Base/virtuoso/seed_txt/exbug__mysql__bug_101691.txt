CREATE TABLE tt3 (id1 INTEGER, id2 INTEGER, value1 INTEGER CHECK (value1 >= 0), value2 INTEGER DEFAULT 0, name VARCHAR(32), PRIMARY KEY (id1, id2));
INSERT INTO tt3 (id1, id2, value1, value2, name) VALUES (1, 1, 10, 100, 'A'), (1, 2, 20, 90, 'B'), (2, 1, 30, 80, 'C'), (2, 2, 40, 70, 'D'), (3, 1, 50, 60, 'E');
SELECT id1, id2, SUM(value1) AS total_value, MAX(value2) AS max_value, SUBSTR(GROUP_CONCAT(name ORDER BY value2), 1, INSTR(GROUP_CONCAT(name ORDER BY value2), ',') - 1) AS first_name, DENSE_RANK() OVER (ORDER BY SUM(value1) DESC, MAX(value2) ASC) AS rank FROM tt3 GROUP BY id1, id2 ORDER BY total_value DESC, max_value ASC;
