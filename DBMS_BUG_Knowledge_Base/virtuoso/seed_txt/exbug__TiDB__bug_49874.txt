CREATE TABLE c(PK int, value varchar(10));
CREATE TABLE d(PK int, value varchar(10) DEFAULT 'default');
INSERT INTO c VALUES (0, 'zero'), (1, 'one'), (2, 'two');
INSERT INTO d VALUES (0, 'none'), (3, 'three'), (4, 'four');
SELECT * FROM c WHERE PK IS NULL UNION ALL SELECT * FROM (SELECT * FROM d WHERE PK = 0 UNION ALL SELECT * FROM d WHERE PK != 0 ORDER BY PK LIMIT 1) AS subquery ORDER BY PK LIMIT 2;
