CREATE TABLE employees (id integer PRIMARY KEY, name varchar, salary decimal(10, 2), department_id integer);
CREATE TABLE departments (id integer PRIMARY KEY, name varchar NOT NULL);

SELECT * FROM departments d JOIN (SELECT id, name, salary, department_id, RANK() OVER (ORDER BY salary) AS salary_rank FROM employees) AS ranked_employees ON d.id = ranked_employees.department_id;
