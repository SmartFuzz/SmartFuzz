CREATE TABLE flights (tailnum VARCHAR(255), arr_delay INTEGER, flight_date DATE);
LOAD DATA LOCAL INFILE 'c:\Users\anthonyd\Desktop\flights.csv' INTO TABLE flights FIELDS TERMINATED BY ',' OPTIONALLY ENCLOSED BY '"' LINES TERMINATED BY '\n' IGNORE 2 LINES;
SELECT tailnum, AVG(arr_delay) AS avg_delay FROM flights WHERE arr_delay IS NOT NULL GROUP BY tailnum;
SELECT tailnum, AVG(arr_delay) AS avg_delay FROM flights GROUP BY tailnum HAVING AVG(arr_delay) IS NOT NULL;
SELECT * FROM (SELECT tailnum, AVG(arr_delay) AS avg_delay FROM flights GROUP BY tailnum) AS avg_flights WHERE avg_delay > 0;
