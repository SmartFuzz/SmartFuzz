CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR(255) NOT NULL, age INTEGER CHECK(age >= 0), created_at TIMESTAMP DEFAULT NOW());
CREATE TABLE orders (id INTEGER PRIMARY KEY, user_id INTEGER, amount FLOAT NOT NULL, order_date TIMESTAMP DEFAULT NOW(), FOREIGN KEY(user_id) REFERENCES users(id));
CREATE PROCEDURE update_order_timestamp() AS BEGIN UPDATE orders SET order_date = NOW() WHERE id = ?;
END;
