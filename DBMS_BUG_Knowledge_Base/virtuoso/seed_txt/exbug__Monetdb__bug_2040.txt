CREATE TABLE large_data (value DECIMAL(15,5));
INSERT INTO large_data SELECT ROUND(RAND() * 100, 2) FROM (SELECT ROW_NUMBER() OVER() AS num FROM rdf_data) AS temp LIMIT 51000000;
SELECT AVG(SQRT(value)) FROM large_data WHERE value IS NOT NULL;
CREATE TABLE temp_data (num FLOAT);
INSERT INTO temp_data SELECT SQRT(value) FROM large_data WHERE value > 0;
