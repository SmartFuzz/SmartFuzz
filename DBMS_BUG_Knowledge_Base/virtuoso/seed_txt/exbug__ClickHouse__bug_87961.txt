CREATE TABLE user_activity (activity_id INTEGER, user_id INTEGER, activity_time TIMESTAMP, activity_type VARCHAR(10), details VARCHAR(255));
INSERT INTO user_activity (activity_id, user_id, activity_time, activity_type, details) SELECT number AS activity_id, RAND() % 1000 AS user_id, NOW() + number * INTERVAL '5' SECOND AS activity_time, CASE WHEN number % 3 = 0 THEN 'login' WHEN number % 3 = 1 THEN 'logout' ELSE 'purchase' END AS activity_type, CONCAT('Activity details for ', CAST(number AS VARCHAR)) AS details FROM (SELECT ROW_NUMBER() OVER () AS number FROM DUAL LIMIT 100);
CREATE TABLE product_sales (sale_id INTEGER, product_id INTEGER, sale_time TIMESTAMP, quantity SMALLINT, price FLOAT);
INSERT INTO product_sales (sale_id, product_id, sale_time, quantity, price) SELECT number AS sale_id, RAND() % 500 AS product_id, NOW() + number * INTERVAL '2' SECOND AS sale_time, (RAND() % 10 + 1) AS quantity, (RAND() % 100 + 1) * 1.5 AS price FROM (SELECT ROW_NUMBER() OVER () AS number FROM DUAL LIMIT 50);
SELECT ua.user_id, COUNT(*) AS activity_count, SUM(ps.quantity) AS total_quantity FROM user_activity ua LEFT JOIN product_sales ps ON ua.user_id = ps.product_id WHERE ua.activity_time >= NOW() - INTERVAL '1' DAY GROUP BY ua.user_id HAVING total_quantity > 5 ORDER BY activity_count DESC;
