CREATE TABLE images (imageid INTEGER, ds_id INTEGER, bmaj FLOAT, PRIMARY KEY (imageid));
CREATE TABLE xtrsrc (xtrsrcid INTEGER, image_id INTEGER, zone INTEGER, ra FLOAT, decl FLOAT, ra_err FLOAT, decl_err FLOAT, x FLOAT, y FLOAT, z FLOAT, PRIMARY KEY (xtrsrcid), FOREIGN KEY (image_id) REFERENCES images (imageid));
