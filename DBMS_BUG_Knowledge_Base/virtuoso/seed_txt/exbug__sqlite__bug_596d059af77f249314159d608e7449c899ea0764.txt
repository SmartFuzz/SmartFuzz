CREATE TABLE users(id INTEGER PRIMARY KEY, name VARCHAR(255) NOT NULL, age INTEGER CHECK(age >= 0));
CREATE TABLE orders(order_id INTEGER PRIMARY KEY, user_id INTEGER, amount FLOAT, FOREIGN KEY(user_id) REFERENCES users(id));
