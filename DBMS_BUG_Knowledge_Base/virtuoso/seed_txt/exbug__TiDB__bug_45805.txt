CREATE TABLE signed_estimate_volume ( id bigint NOT NULL, estimate_id bigint NOT NULL DEFAULT 0, dealer_id varchar(40) NOT NULL DEFAULT '', broker_id varchar(40) NOT NULL DEFAULT '', real_t0_volume bigint NOT NULL DEFAULT 0, real_t1_volume bigint NOT NULL DEFAULT 0, manager_yid varchar(32) NOT NULL DEFAULT '', estimate_volume bigint NOT NULL DEFAULT 0, estimate_year_month varchar(8) NOT NULL DEFAULT '', is_dropped integer NOT NULL DEFAULT 0, PRIMARY KEY (id), UNIQUE (estimate_id), UNIQUE (estimate_year_month, manager_yid, broker_id, dealer_id, is_dropped) );
CREATE TABLE unsigned_estimate_volume ( id bigint NOT NULL, estimate_id bigint NOT NULL DEFAULT 0, estimate_year_month varchar(8) NOT NULL DEFAULT '', business_id bigint NOT NULL DEFAULT 0, broker_id varchar(40) NOT NULL DEFAULT '', estimate_volume bigint NOT NULL DEFAULT 0, manager_yid varchar(32) NOT NULL DEFAULT '', PRIMARY KEY (id), UNIQUE (estimate_id), UNIQUE (estimate_year_month, manager_yid, broker_id, business_id) );
INSERT INTO signed_estimate_volume (id, estimate_volume, real_t1_volume) VALUES (1, 100, 200);
SELECT estimate_year_month, a_manager_yid AS manager_yid, real_t1_volume, real_t0_volume, a_estimate_volume + COALESCE(b_estimate_volume, 0) AS estimate_volume, COALESCE(a.increase_sum, 0) + COALESCE(b.increase_sum, 0) AS increase_sum, COALESCE(a.decrease_sum, 0) + COALESCE(b.decrease_sum, 0) AS decrease_sum FROM ( SELECT estimate_year_month, manager_yid AS a_manager_yid, SUM(real_t1_volume) AS real_t1_volume, SUM(real_t0_volume) AS real_t0_volume, SUM(estimate_volume) AS a_estimate_volume, SUM(CASE WHEN estimate_volume > real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS increase_sum, SUM(CASE WHEN estimate_volume < real_t1_volume THEN estimate_volume - real_t1_volume ELSE 0 END) AS decrease_sum FROM signed_estimate_volume GROUP BY manager_yid ) AS a LEFT JOIN ( SELECT manager_yid AS b_manager_yid, SUM(estimate_volume) AS b_estimate_volume, SUM(estimate_volume) AS increase_sum, 0 AS decrease_sum FROM unsigned_estimate_volume GROUP BY manager_yid ) AS b ON a.a_manager_yid = b.b_manager_yid;
