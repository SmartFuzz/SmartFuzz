DROP TABLE IF EXISTS agg_test;
CREATE TABLE agg_test (id INTEGER, accepted_at TIMESTAMP, delivery_report_at BLOB);
INSERT INTO agg_test (id, accepted_at, delivery_report_at) SELECT number AS id, TIMESTAMP '2025-09-24 00:00:00' + INTERVAL number SECOND AS accepted_at, argMaxState(CAST(number AS VARCHAR), TIMESTAMP '2025-09-24 00:00:00' + INTERVAL number SECOND) FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM rdf::get_graph()) WHERE number < 100;
SELECT id, finalizeAggregation(delivery_report_at) AS delivered_at FROM agg_test ORDER BY accepted_at ASC LIMIT 5;
SELECT id, finalizeAggregation(delivery_report_at) AS delivered_at FROM agg_test ORDER BY accepted_at DESC OFFSET 5 LIMIT 10;
