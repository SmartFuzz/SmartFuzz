CREATE TABLE users (id INTEGER PRIMARY KEY, name VARCHAR(255) NOT NULL, age INTEGER);
CREATE TABLE posts (id INTEGER PRIMARY KEY, user_id INTEGER, content VARCHAR(255), FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id INTEGER PRIMARY KEY, post_id INTEGER, user_id INTEGER, comment VARCHAR(255), FOREIGN KEY(post_id) REFERENCES posts(id), FOREIGN KEY(user_id) REFERENCES users(id));
