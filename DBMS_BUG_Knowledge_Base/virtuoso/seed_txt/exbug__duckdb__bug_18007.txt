CREATE TABLE cols (c VARCHAR);
INSERT INTO cols (SELECT 'c'||i FROM 1 TO 21);
CREATE TABLE matrix (col1 VARCHAR, col2 VARCHAR, val UTINYINT NOT NULL CHECK (val < 250));
INSERT INTO matrix (SELECT c1.c, c2.c, CAST(RAND()*250 AS UTINYINT) FROM cols c1 JOIN cols c2 ON c1.c != c2);
SELECT col1, SUM(val) AS total_val FROM matrix GROUP BY col1 ORDER BY col1 LIMIT 2;
