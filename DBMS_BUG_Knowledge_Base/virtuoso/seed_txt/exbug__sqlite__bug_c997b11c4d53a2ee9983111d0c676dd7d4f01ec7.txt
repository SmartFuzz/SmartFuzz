CREATE TABLE users (id INTEGER IDENTITY PRIMARY KEY, name NVARCHAR(255) NOT NULL, age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE posts (id INTEGER IDENTITY PRIMARY KEY, user_id INTEGER, content NVARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(user_id) REFERENCES users(id));
CREATE TABLE comments (id INTEGER IDENTITY PRIMARY KEY, post_id INTEGER, user_id INTEGER, comment NVARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY(post_id) REFERENCES posts(id), FOREIGN KEY(user_id) REFERENCES users(id));
