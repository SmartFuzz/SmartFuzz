CREATE TABLE bug (time TIMESTAMP, val INTEGER, description VARCHAR(255) NOT NULL);
INSERT INTO bug (time, val, description) VALUES (CURRENT_TIMESTAMP, 1, 'First entry');
INSERT INTO bug (time, val, description) VALUES (CURRENT_TIMESTAMP + INTERVAL '2' SECOND, 2, 'Second entry');
INSERT INTO bug (time, val, description) VALUES (CURRENT_TIMESTAMP - INTERVAL '3' SECOND, 3, 'Third entry');
SELECT t1.time, COUNT(t2.val) FROM bug AS t1 JOIN bug AS t2 ON t2.time BETWEEN (t1.time - INTERVAL '1' SECOND) AND (t1.time + INTERVAL '1' SECOND) GROUP BY t1.time ORDER BY t1.time;
