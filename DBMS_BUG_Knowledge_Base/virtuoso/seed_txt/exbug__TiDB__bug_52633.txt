DROP TABLE IF EXISTS p1;
CREATE TABLE p1 ( id bigint NOT NULL, code varchar(64) NOT NULL DEFAULT '', third_user_name varchar(64) NOT NULL DEFAULT '', PRIMARY KEY (id, code), INDEX idx_third_user_name (third_user_name));
DROP TABLE IF EXISTS t1;
CREATE TABLE t1 ( id bigint NOT NULL, code varchar(64) NOT NULL DEFAULT '', third_user_name varchar(64) NOT NULL DEFAULT '', PRIMARY KEY (id, code), INDEX idx_third_user_name (third_user_name));
DELETE FROM t1 WHERE third_user_name IS NULL;
