CREATE TABLE test_blob(COL1 BLOB, COL2 VARCHAR(255));
INSERT INTO test_blob (COL1, COL2) VALUES (CAST(0xFF AS BLOB), NULL), (CAST(0x7F AS BLOB), NULL), (CAST(0xF9 AS BLOB), NULL), (CAST(0xFA AS BLOB), NULL), (NULL, NULL);
SELECT COL1, CAST(COL1 AS VARCHAR) AS lcase_col1 FROM test_blob WHERE CAST(COL1 AS VARCHAR) NOT LIKE CAST(0xFF AS VARCHAR);
