CREATE TABLE IF NOT EXISTS users (id xsd:integer, name xsd:string, created_at xsd:dateTime);
CREATE TABLE IF NOT EXISTS user_logs (user_id xsd:integer, log_message xsd:string, log_time xsd:dateTime);
INSERT DATA { (1, 'Alice', now()) } INTO users;
INSERT { user_id, log_message, log_time } { SELECT id, 'User logged in', now() FROM users WHERE id = 1 };
CREATE TABLE user_summary AS SELECT user_id, COUNT(*) AS log_count FROM user_logs GROUP BY user_id;
