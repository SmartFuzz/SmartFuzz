CREATE TABLE c (a xsd:integer, b xsd:integer) AS SELECT ? AS a, ? AS b WHERE ? IS NULL;
INSERT INTO c SELECT number, CASE WHEN number % 2 = 0 THEN number ELSE NULL END FROM (SELECT ? AS number WHERE ? < 10);
CREATE TABLE d (m xsd:integer, n xsd:integer) AS SELECT ? AS m, ? AS n WHERE ? IS NULL;
INSERT INTO d SELECT number, number + 1 FROM (SELECT ? AS number WHERE ? < 5);
SELECT c.a, d.n FROM c LEFT OUTER JOIN d ON c.a > d.m;
