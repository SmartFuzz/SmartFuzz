CREATE TABLE test_data (id INTEGER PRIMARY KEY, value_real DOUBLE, value_int INTEGER, value_text VARCHAR(255));
INSERT INTO test_data (value_real, value_int, value_text) VALUES (1.5, 1, 'one'), (2.5, 2, 'two'), (3.0, NULL, 'three'), (NULL, 4, 'four');
SELECT value_real, value_int, COUNT(*) AS count FROM test_data GROUP BY value_real, value_int;
SELECT value_real, SUM(value_int) AS total FROM test_data GROUP BY value_real HAVING total > 0;
SELECT value_text, COUNT(*) AS count FROM test_data WHERE value_int IS NULL GROUP BY value_text;
