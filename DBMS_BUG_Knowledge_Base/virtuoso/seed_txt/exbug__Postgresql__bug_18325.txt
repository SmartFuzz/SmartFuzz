DROP TABLE IF EXISTS xyz;
CREATE TABLE xyz (id INTEGER PRIMARY KEY, name VARCHAR(255), age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO xyz (name, age) VALUES ('Alice', 30), ('Bob', NULL);
CREATE OR REPLACE FUNCTION get_age() RETURNS INTEGER { DECLARE user_record xyz;
FOR user_record IN SELECT * FROM xyz WHERE age IS NOT NULL { RETURN user_record.age;
} RETURN NULL;
