DROP DATABASE IF EXISTS my_db;
CREATE DATABASE my_db;
CREATE TABLE my_db.my_table (id xsd:integer, name xsd:string, created_at xsd:dateTime DEFAULT now());
CREATE VIEW my_db.my_view1 AS SELECT * FROM my_db.my_table WHERE id = ?;
CREATE VIEW my_db.my_view2 AS SELECT * FROM my_db.my_view1 WHERE id = CAST(NULL AS xsd:integer);
