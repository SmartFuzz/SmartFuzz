CREATE TABLE extreme_dates(COL1 TIMESTAMP, COL2 INTEGER NOT NULL);

SELECT COL1, COL1 - INTERVAL '10' MONTH AS result FROM extreme_dates WHERE COL1 IS NOT NULL;
