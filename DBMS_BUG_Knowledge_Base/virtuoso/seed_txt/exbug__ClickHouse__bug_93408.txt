CREATE TABLE throttle_test (id INTEGER, payload VARCHAR(1000), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
INSERT INTO throttle_test (id, payload) SELECT number, SUBSTRING(CONVERT(NEWID(), VARCHAR(1000)), 1, 1000) FROM (SELECT ROW_NUMBER() OVER() AS number FROM sysibm.sysdummy1 FETCH FIRST 100 ROWS ONLY);
CREATE TABLE throttle_test_backup AS SELECT * FROM throttle_test;
SELECT id, payload FROM throttle_test WHERE created_at IS NOT NULL AND MOD(id, 2) = 0;
SELECT COUNT(*) FROM throttle_test_backup WHERE payload LIKE '%test%' AND id IN (SELECT id FROM throttle_test WHERE id < 50);
