CREATE TABLE T1 (K INTEGER, C0 VARCHAR, C1 VARCHAR, C2 FLOAT);
INSERT INTO T1 (K, C0, C1, C2) SELECT number, ARRAY_AGG(CAST(number * i AS VARCHAR)) FROM (SELECT number, i FROM numbers(100) CROSS JOIN numbers(10) AS i) GROUP BY number;
CREATE TABLE T2 (K INTEGER, C0 VARCHAR, C1 INTEGER);
INSERT INTO T2 (K, C0, C1) SELECT K, C0, CAST(K AS INTEGER) FROM T1 WHERE C1 IS NOT NULL;
SELECT * FROM T1 WHERE C1 = 'even';
