CREATE TABLE t1 (id INTEGER, name VARCHAR(50), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE t2 (id INTEGER, value FLOAT CHECK (value >= 0));
CREATE VIEW v1 AS SELECT * FROM t1 UNION SELECT id, CAST(value AS VARCHAR(50)) AS name, NULL AS created_at FROM t2;
DROP TABLE t1;
SELECT * FROM v1;
