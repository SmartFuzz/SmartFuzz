CREATE TABLE EMPLOYEES (EMPNUM INTEGER PRIMARY KEY, NAME VARCHAR(50), SALARY DECIMAL(10, 2));
CREATE TABLE DEPARTMENTS (DEPTID INTEGER PRIMARY KEY, DEPTNAME VARCHAR(50));
CREATE VIEW EMP_DEPT AS SELECT E.EMPNUM, E.NAME, D.DEPTNAME FROM EMPLOYEES E LEFT JOIN DEPARTMENTS D ON E.EMPNUM = D.DEPTID;
