CREATE TABLE users (id INTEGER, name VARCHAR(255) NOT NULL, age INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (id));

CREATE TABLE orders (id INTEGER, user_id INTEGER, amount FLOAT, FOREIGN KEY(user_id) REFERENCES users(id), PRIMARY KEY (id));
