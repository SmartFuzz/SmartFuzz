CREATE TABLE test_strings (id INTEGER PRIMARY KEY, name VARCHAR(3000) NOT NULL);
INSERT INTO test_strings SELECT i, STR_REPEAT('abc', 1000) FROM (SELECT ROW_NUMBER() OVER () AS i FROM rdf_data LIMIT 100);
CREATE TABLE test_numbers (num INTEGER, name VARCHAR(3000));
INSERT INTO test_numbers SELECT i, STR_REPEAT('xyz', 200) FROM (SELECT ROW_NUMBER() OVER () AS i FROM rdf_data LIMIT 50);
SELECT t1.id, t2.name FROM test_strings t1 JOIN test_numbers t2 ON t1.id = t2.num WHERE t1.name IS NOT NULL;
