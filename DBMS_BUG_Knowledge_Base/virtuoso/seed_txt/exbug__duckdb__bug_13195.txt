CREATE TABLE courses (course_id INTEGER, course_name VARCHAR NOT NULL, credits INTEGER CHECK (credits > 0), PRIMARY KEY (course_id));

CREATE TABLE enrollments (enrollment_id INTEGER, sid INTEGER, course_id INTEGER, PRIMARY KEY (enrollment_id), FOREIGN KEY (sid) REFERENCES students(sid), FOREIGN KEY (course_id) REFERENCES courses(course_id));
