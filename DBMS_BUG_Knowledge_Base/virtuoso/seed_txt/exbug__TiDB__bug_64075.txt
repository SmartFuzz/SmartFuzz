CREATE TABLE t3 (id INTEGER NOT NULL, value DECIMAL(10,2) NOT NULL, created_at TIMESTAMP DEFAULT NOW(), PRIMARY KEY (id));
INSERT INTO t3 (id, value) VALUES (1, 1.23), (2, 2.34), (3, 3.45), (4, 4.56), (5, 5.67);
SELECT VARIANCE(value) AS variance_value, SQRT(VARIANCE(value)) AS stddev_value FROM t3 WHERE value > 0;
SELECT AVG(value) AS average_value FROM t3 WHERE created_at >= TIMESTAMP '2025-01-01 00:00:00';
