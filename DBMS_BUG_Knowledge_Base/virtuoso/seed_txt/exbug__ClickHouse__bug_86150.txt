DROP TABLE IF EXISTS test_final_partitions;
CREATE TABLE test_final_partitions (organization_id INTEGER, keyword_id INTEGER, search_date TIMESTAMP, position_on_serp INTEGER, version INTEGER, value VARCHAR) INDEX BY (organization_id, keyword_id, search_date, position_on_serp);
INSERT INTO test_final_partitions (organization_id, keyword_id, search_date, position_on_serp, version, value) SELECT RAND() % 1000, RAND() % 100000, TIMESTAMP '2023-01-01 00:00:00' + (RAND() % (86400*180)), RAND() % 50, ROW_NUMBER() OVER (), CAST(ROW_NUMBER() OVER () AS VARCHAR) FROM (SELECT ROW_NUMBER() OVER () AS rn FROM DB.DUAL CONNECT BY LEVEL <= 50000000);
SELECT COUNT(*) FROM test_final_partitions WHERE organization_id IS NOT NULL GROUP BY search_date;
SELECT COUNT(*) FROM test_final_partitions WHERE position_on_serp > 25 AND version IS NULL GROUP BY search_date;
