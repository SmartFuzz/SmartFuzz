CREATE TABLE temp.try_dict_table_c2 (col_2 Nullable(Integer), col_2_value Nullable(VARCHAR)) AS SELECT MOD(number, 7) AS col_2, CONCAT('value_s_', CAST(MOD(number, 7) AS VARCHAR)) AS col_2_value FROM DB.DBA.system_numbers WHERE number <= 6;
CREATE TABLE temp.try_facts_table (data_type Nullable(VARCHAR), data_date Nullable(Date), col_1 Nullable(Integer), col_2 Nullable(Integer), col_3 Nullable(Integer), col_4 Nullable(Integer), json_string Nullable(VARCHAR));
INSERT INTO temp.try_facts_table SELECT CONCAT('data_type', CAST(MOD(number, 10) AS VARCHAR)) AS data_type, DATEADD(DAY, MOD(number, 100), DATE('2025-01-01')) AS data_date, MOD(number, 5) AS col_1, MOD(number, 7) AS col_2, MOD(number, 11) AS col_3, MOD(number, 15) AS col_4, CONCAT('{"add_col1_join_key": "', CAST(col_1 AS VARCHAR), '"}') AS json_string FROM DB.DBA.system_numbers WHERE number <= 10000000;
SELECT * FROM temp.try_facts_table AS f JOIN temp.try_dict_table_c2 AS d ON f.col_2 = d.col_2 WHERE d.col_2_value IS NOT NULL;
