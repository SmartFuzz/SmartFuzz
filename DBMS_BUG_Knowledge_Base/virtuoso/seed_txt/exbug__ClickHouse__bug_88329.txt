CREATE DATABASE test2;
CREATE TABLE test2.raw_source_local (id INT, time TIMESTAMP, value FLOAT DEFAULT 0.0);
INSERT INTO test2.raw_source_local (id, time, value) SELECT number AS id, TIMESTAMP '2024-01-01 00:00:00' + INTERVAL number SECOND AS time, RAND() * 100 AS value FROM (SELECT ROW_NUMBER() OVER () AS number FROM DUAL LIMIT 1000);
CREATE TABLE test2.table_for_mv (sum_value FLOAT, date DATE, count_id INT);
CREATE MATERIALIZED VIEW test2.mv_from_raw_source_local_to_table AS SELECT SUM(value) AS sum_value, CAST(time AS DATE) AS date, COUNT(*) AS count_id FROM test2.raw_source_local GROUP BY CAST(time AS DATE);
