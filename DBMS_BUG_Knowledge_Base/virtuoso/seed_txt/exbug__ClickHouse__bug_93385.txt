CREATE TABLE user_events (event_id VARCHAR(36), user_id INTEGER, event_type VARCHAR(10), event_time TIMESTAMP);
INSERT INTO user_events (event_id, user_id, event_type, event_time) SELECT UUID(), CAST(rand() * 10000 AS INTEGER), CASE CAST(rand() * 3 AS INTEGER) WHEN 0 THEN 'click' WHEN 1 THEN 'view' ELSE 'purchase' END, NOW() - (CAST(rand() * 86400 AS INTEGER) * INTERVAL '1' SECOND) FROM (SELECT ROW_NUMBER() OVER () AS n FROM sysibm.sysdummy1) AS numbers WHERE n <= 100000;
CREATE TABLE user_summary (user_id INTEGER, event_count INTEGER);
INSERT INTO user_summary (user_id, event_count) SELECT user_id, COUNT(*) FROM user_events GROUP BY user_id;
SELECT user_id, event_count FROM user_summary WHERE event_count > 10 ORDER BY event_count DESC;
