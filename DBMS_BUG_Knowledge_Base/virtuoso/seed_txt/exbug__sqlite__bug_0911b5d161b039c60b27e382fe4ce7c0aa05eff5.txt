SET CHARACTER SET utf16;
CREATE TABLE t1 (id INTEGER PRIMARY KEY, c1 NVARCHAR(255) NOT NULL, c2 VARBINARY);
INSERT INTO t1(c1, c2) VALUES (CAST(0x00 AS NVARCHAR(255)), CAST(0x01 AS VARBINARY));
CREATE VIRTUAL TABLE t2 USING fts5(c3);
INSERT INTO t2(c3) VALUES (CAST((SELECT c1 FROM t1 WHERE id = 1) AS NVARCHAR(255)));
SELECT * FROM t2 WHERE CAST(SUBSTR(c3, 1, 1) AS NVARCHAR(255)) > '0';
