DROP DATABASE IF EXISTS test_db;
CREATE DATABASE test_db;
CREATE TABLE test_db.sample_table (id INTEGER, name VARCHAR, created_at TIMESTAMP, score FLOAT);
CREATE VIEW test_db.sample_view AS SELECT id, name, created_at FROM test_db.sample_table WHERE score > 50.0;
INSERT INTO test_db.sample_table SELECT number, CONCAT('Name', CAST(number AS VARCHAR)), NOW(), number % 100 FROM (SELECT ROW_NUMBER() OVER () AS number FROM rdf::get_data(100));
