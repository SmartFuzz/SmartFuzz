CREATE TABLE test_table ( id INTEGER NOT NULL, name VARCHAR(50) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, status VARCHAR(10) NOT NULL, PRIMARY KEY (id));

CREATE TABLE related_table ( ref_id INTEGER NOT NULL, description VARCHAR(255), FOREIGN KEY (ref_id) REFERENCES test_table(id));
