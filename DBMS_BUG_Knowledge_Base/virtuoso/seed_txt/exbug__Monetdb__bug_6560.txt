CREATE TABLE test_table(id INTEGER PRIMARY KEY, value1 INTEGER, value2 FLOAT, value3 VARCHAR(255));
INSERT INTO test_table (id, value1, value2, value3) VALUES(1, 10, 20.5, 'A'), (2, 15, NULL, 'B'), (3, 20, 30.0, 'C');
SELECT value1, SUM(value2) AS total_value2 FROM test_table GROUP BY value1 HAVING total_value2 > 15;
SELECT value3, COUNT(*) AS count_value3 FROM test_table WHERE value2 IS NOT NULL GROUP BY value3 HAVING count_value3 < 3;
SELECT value1, AVG(value2) AS avg_value2 FROM test_table GROUP BY value1 HAVING avg_value2 IN (SELECT AVG(value2) FROM test_table WHERE value1 > 1);
