CREATE TABLE db.test_data (id INTEGER, download INTEGER, upload INTEGER, app VARCHAR(255), key VARCHAR(44));
INSERT INTO db.test_data SELECT id, (RAND() % 3) AS download, (RAND() % 3) AS upload, 'App_' || (RAND() % 100) AS app, SUBSTR(md5(RAND()::VARCHAR), 1, 44) AS key FROM (SELECT ROW_NUMBER() OVER () AS id FROM db.dummy_table LIMIT 100000);
SELECT app, key, COUNT(*) AS count_download, COUNT(CASE WHEN download = 0 THEN 1 END) AS low_count, COUNT(CASE WHEN download = 1 THEN 1 END) AS average_count, COUNT(CASE WHEN download = 2 THEN 1 END) AS high_count FROM db.test_data GROUP BY app, key;
WITH download_summary AS (SELECT app, key, COUNT(CASE WHEN download = 0 THEN 1 END) AS low_count, COUNT(CASE WHEN download = 1 THEN 1 END) AS average_count, COUNT(CASE WHEN download = 2 THEN 1 END) AS high_count FROM db.test_data GROUP BY app, key) SELECT app, key, CASE WHEN low_count = GREATEST(low_count, average_count, high_count) THEN 0 WHEN average_count = GREATEST(low_count, average_count, high_count) THEN 1 ELSE 2 END AS mode_download FROM download_summary;
SELECT app, key, COALESCE(NULLIF(MAX(download), 0), 1) AS max_download FROM db.test_data GROUP BY app, key HAVING COUNT(*) > 10;
