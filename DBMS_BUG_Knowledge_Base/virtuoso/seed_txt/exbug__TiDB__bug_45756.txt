CREATE TABLE sales ( year int NOT NULL, country varchar(20) NOT NULL, product varchar(32) NOT NULL, profit int DEFAULT 0, PRIMARY KEY (year, country, product) );
INSERT INTO sales (year, country, product, profit) VALUES (2020, 'USA', 'Gadget', 1500), (2020, 'USA', 'Widget', 2000), (2021, 'Canada', 'Gadget', 2500);
SELECT year, country, product, CASE WHEN year IS NOT NULL AND country IS NOT NULL AND product IS NOT NULL THEN 0 ELSE 1 END AS grouping FROM sales GROUP BY year, country, product HAVING grouping = 0;
CREATE TABLE products ( product_id int NOT NULL, product_name varchar(50) NOT NULL, PRIMARY KEY (product_id) );
