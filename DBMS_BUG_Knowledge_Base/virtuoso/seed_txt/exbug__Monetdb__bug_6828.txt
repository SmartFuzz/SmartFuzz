CREATE TABLE departments (id INTEGER PRIMARY KEY, name VARCHAR(50) NOT NULL);
INSERT INTO departments (id, name) VALUES (1, 'Management'), (2, 'Development'), (3, 'Sales');
CREATE TABLE employees (dep_id INTEGER, name VARCHAR(20), salary DOUBLE, hire_date DATE, FOREIGN KEY (dep_id) REFERENCES departments(id));
INSERT INTO employees (dep_id, name, salary, hire_date) VALUES (1, 'Smith', 81, '2019-08-10'), (2, 'Jones', 55, '2019-08-09'), (3, 'Williams', 55, '2019-07-14');
SELECT dep_id, name, salary, SUM(salary) OVER (ORDER BY YEAR(hire_date)) AS salary_sum FROM employees;
