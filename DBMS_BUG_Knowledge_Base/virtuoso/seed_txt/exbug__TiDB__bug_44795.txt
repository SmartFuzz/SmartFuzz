CREATE TABLE customer (c_custkey INTEGER PRIMARY KEY, c_name VARCHAR(50), c_address VARCHAR(100), c_comment VARCHAR(200));
CREATE TABLE orders (o_orderkey INTEGER PRIMARY KEY, o_custkey INTEGER, o_comment VARCHAR(200), FOREIGN KEY (o_custkey) REFERENCES customer(c_custkey));

SELECT c_count, COUNT(*) AS custdist FROM (SELECT c_custkey, COUNT(o_orderkey) AS c_count FROM customer LEFT JOIN orders ON c_custkey = o_custkey AND NOT CONTAINS(o_comment, 'special requests') GROUP BY c_custkey) c_orders GROUP BY c_count ORDER BY custdist DESC, c_count DESC;
