CREATE SCHEMA test_schema;
CREATE TABLE test_schema.test_table (id INTEGER PRIMARY KEY, path VARCHAR(255) NOT NULL, description VARCHAR(255));
CREATE FUNCTION test_schema.trigger_function() RETURNS trigger AS { BEGIN RETURN NEW;
END;
} LANGUAGE plpgsql;
CREATE TRIGGER path_change_trigger AFTER UPDATE ON test_schema.test_table FOR EACH ROW WHEN (CAST(NEW.path AS VARCHAR) != CAST(OLD.path AS VARCHAR)) EXECUTE PROCEDURE test_schema.trigger_function();
