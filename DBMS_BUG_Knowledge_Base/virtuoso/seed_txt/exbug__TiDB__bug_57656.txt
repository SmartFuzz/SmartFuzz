DROP TABLE IF EXISTS employee;
CREATE TABLE employee (id INTEGER NOT NULL PRIMARY KEY, name VARCHAR(255) NOT NULL, salary DECIMAL(10, 2) NOT NULL, department_id INTEGER, birth_date DATE);
