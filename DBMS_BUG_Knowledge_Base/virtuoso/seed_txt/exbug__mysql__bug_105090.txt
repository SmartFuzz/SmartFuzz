DROP DATABASE IF EXISTS sandbox;
CREATE DATABASE sandbox;
USE sandbox;
CREATE TABLE t_ref (id bigint NOT NULL, PRIMARY KEY (id));
CREATE TABLE t_main (id int NOT NULL, unique_id varchar(32) AS ('1'), t_ref_id bigint NOT NULL, PRIMARY KEY (id), INDEX (unique_id), FOREIGN KEY (t_ref_id) REFERENCES t_ref (id));
SELECT * FROM t_main;
