DROP TABLE IF EXISTS t1;
CREATE TABLE t1(c0 INT, c1 VARCHAR(255), c2 DECIMAL(10,2));
INSERT INTO t1(c0, c1, c2) VALUES (1, 'A', 10.50), (2, 'B', NULL), (3, 'C', 20.00);
SELECT COUNT(*) FROM t1 WHERE c1 >= CAST(c0 AS VARCHAR) OR c1 LIKE 'A%';
SELECT SUM(CASE WHEN c1 >= CAST(c0 AS VARCHAR) OR c1 LIKE 'B%' THEN 1 ELSE 0 END) FROM t1;
