DROP TABLE IF EXISTS t1;
CREATE TABLE t1(c1 BLOB NOT NULL, c2 INTEGER DEFAULT 0);
INSERT INTO t1 VALUES (CAST('true' AS BLOB)), (CAST('false' AS BLOB)), (CAST('1xx' AS BLOB)), (NULL);
SELECT CASE WHEN CAST(t1.c1 AS VARCHAR) IN ('true', '1') THEN 1 ELSE 0 END AS is_true FROM t1 WHERE c2 > 0;
SELECT CASE WHEN CAST(t1.c1 AS VARCHAR) IS NOT NULL THEN 1 ELSE 0 END AS is_not_null FROM t1 WHERE c2 IS NULL;
