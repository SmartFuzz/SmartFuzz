DROP TABLE IF EXISTS test_lwu;
CREATE TABLE test_lwu (a INTEGER, b VARCHAR, c FLOAT DEFAULT 0.0);
INSERT INTO test_lwu (a, b, c) SELECT number, CASE WHEN number % 5 = 0 THEN NULL ELSE CAST(number % 5 AS VARCHAR) END, number / 2.0 FROM (SELECT ROW_NUMBER() OVER () - 1 AS number FROM rdf::get_graph()) LIMIT 100;
DELETE FROM test_lwu WHERE (RAND() % 10) = 0;
SELECT COUNT(*) FROM test_lwu;
