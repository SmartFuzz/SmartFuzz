CREATE TABLE test_table (id INTEGER, value VARCHAR, flag BOOLEAN);

WITH cte AS (SELECT id, value FROM test_table WHERE flag IS NOT NULL) SELECT STRING_AGG(value, ', ') AS aggregated_values FROM cte;
SELECT id, CASE WHEN value IS NULL THEN 'NULL' ELSE value END AS value_check FROM test_table;
