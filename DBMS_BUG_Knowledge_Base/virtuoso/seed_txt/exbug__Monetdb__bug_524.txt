CREATE TABLE users (id INTEGER IDENTITY PRIMARY KEY, name VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE events (event_id INTEGER IDENTITY PRIMARY KEY, event_time TIMESTAMP NOT NULL, user_id INTEGER REFERENCES users(id));
