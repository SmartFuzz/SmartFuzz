CREATE TABLE Users (id INTEGER, name NVARCHAR(255) NOT NULL, email NVARCHAR(255) UNIQUE, PRIMARY KEY (id));
CREATE TABLE Comments (id INTEGER, user_id INTEGER, content NVARCHAR(255), PRIMARY KEY (id), FOREIGN KEY (user_id) REFERENCES Users(id) ON DELETE CASCADE);
