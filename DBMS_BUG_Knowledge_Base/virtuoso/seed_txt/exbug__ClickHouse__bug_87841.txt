CREATE TABLE table_src (order_id INTEGER, random_date DATE, amount FLOAT);
CREATE TABLE table_dst (order_id INTEGER, random_date DATE, amount FLOAT);
INSERT INTO table_src SELECT number AS order_id, CURRENT_DATE - (rand() % 3650) AS random_date, rand() * 100 AS amount FROM (SELECT ROW_NUMBER() OVER () AS number FROM rdf::table(1, 1000000)) AS numbers;
INSERT INTO table_dst SELECT order_id, random_date, amount FROM table_src WHERE amount > 50.0;
INSERT INTO table_dst SELECT order_id, random_date, amount FROM table_src ORDER BY random_date;
