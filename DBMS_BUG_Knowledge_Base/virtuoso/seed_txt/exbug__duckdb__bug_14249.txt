CREATE TABLE test_corr (a DOUBLE, b DOUBLE);
INSERT INTO test_corr (a, b) VALUES (1.0, 1.0), (1.0, 1.0);
SELECT AVG(a) AS avg_a, AVG(b) AS avg_b, (SUM((a - AVG(a)) * (b - AVG(b))) / NULLIF(SQRT(SUM((a - AVG(a)) * (a - AVG(a)))), 0) * NULLIF(SQRT(SUM((b - AVG(b)) * (b - AVG(b)))), 0)) AS corr_ab FROM test_corr;
