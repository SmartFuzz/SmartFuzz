CREATE TABLE sales (year INTEGER, country VARCHAR(20), product VARCHAR(32), profit INTEGER);
INSERT INTO sales (year, country, product, profit) VALUES (2001, 'USA', 'Gadget', 500), (2002, 'USA', 'Gadget', 800), (2003, 'USA', 'Gadget', 1200), (2003, 'Canada', 'Gadget', 1000);
SELECT year + 1 AS y, SUM(profit) AS profit, CASE WHEN COUNT(*) > 0 THEN 0 ELSE 1 END AS grouping FROM sales GROUP BY year + 1 WITH ROLLUP HAVING y > 2002 ORDER BY y, profit;
