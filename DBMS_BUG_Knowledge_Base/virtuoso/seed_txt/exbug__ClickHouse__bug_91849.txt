DROP TABLE IF EXISTS test_special_columns;
CREATE TABLE test_special_columns(key xsd:integer, value xsd:string, timestamp xsd:dateTime, is_active xsd:integer) AS SELECT * FROM (VALUES (1, 'active', '2023-01-01T00:00:00', 1), (2, 'inactive', '2023-01-02T00:00:00', 0), (3, 'active', '2023-01-03T00:00:00', 1));
SELECT key, value FROM test_special_columns WHERE is_active = 1 AND timestamp > '2023-01-01T00:00:00' ORDER BY key;
SELECT COUNT(*) FROM test_special_columns WHERE is_active = 0;
