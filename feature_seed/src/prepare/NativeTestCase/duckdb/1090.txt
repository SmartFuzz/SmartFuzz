begin transaction;
create table unsupported_key as select val from ( 	select MAP { 		'hello': '01010101000'::BIT, 		'HELLO': NULL::BIT 	} val ), range(3);
create table unsupported_value as select val from ( 	select MAP { 		'01010101000'::BIT: 'hello', 		'11110111101'::BIT: 'world' 	} val ), range(3);
select * from unsupported_key;
select * from unsupported_value;
EXPORT DATABASE '__TEST_DIR__/export_test' (FORMAT PARQUET);
ROLLBACK;
IMPORT DATABASE '__TEST_DIR__/export_test';
select * from unsupported_key;
select * from unsupported_value;
