select     field1,     field2,     min(datum) as firstseen,     max(datum) as lastseen from read_csv('{DATA_DIR}/csv/hebere.csv.gz',         delim='\t',         header=0,         columns={field1: varchar, field2: varchar, datum: varchar},         compression='gzip',         ignore_errors=0,         auto_detect = false) group by field1, field2 order by all limit 10;
