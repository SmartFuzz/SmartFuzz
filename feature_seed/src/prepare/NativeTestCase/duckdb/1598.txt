CREATE TABLE probes AS 	SELECT key, ts 	FROM range(1,3) k(key) CROSS JOIN range(0,10) t(ts);
SELECT p.key, p.ts, e.value FROM  	probes p JOIN ( 	SELECT key, value, begin,  		LEAD(begin, 1, 'infinity'::DOUBLE) OVER (PARTITION BY key ORDER BY begin ASC) AS end 	FROM events ) e ON p.key = e.key AND p.ts >= e.begin AND p.ts < e.end ORDER BY 1, 2 ASC;
