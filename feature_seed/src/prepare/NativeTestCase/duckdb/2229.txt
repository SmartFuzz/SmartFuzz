CREATE OR REPLACE TABLE tbl AS SELECT 	{ 		'a': i, 		'b': NULL::VARCHAR 	} col FROM range(5000) t(i) union all select 	{ 		'a': 10000, 		'b': 'hello' 	} FROM range(2);
