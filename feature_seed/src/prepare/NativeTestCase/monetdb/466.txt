set optimizer='sequential_pipe';
start transaction;
explain physical select name,func from functions where name like '%optimizers%';
explain physical select name,func from functions where name not like '%optimizers%';
explain physical select name,func from functions where name ilike '%optimizers%';
explain physical select name,func from functions where name not ilike '%optimizers%';
create function mycontains(str string, substr string) returns boolean begin return str like '%'||substr||'%'; end;
create function not_mycontains(str string, substr string) returns boolean begin return str not like '%'||substr||'%'; end;
create function imycontains(str string, substr string) returns boolean begin return str ilike '%'||substr||'%'; end;
create function not_imycontains(str string, substr string) returns boolean begin return str not ilike '%'||substr||'%'; end;
explain physical select name,func from functions where mycontains(name, 'optimizers');
explain physical select name,func from functions where not_mycontains(name, 'optimizers');
explain physical select name,func from functions where imycontains(name, 'optimizers');
explain physical select name,func from functions where not_imycontains(name, 'optimizers');
rollback;
set optimizer='default_pipe';
