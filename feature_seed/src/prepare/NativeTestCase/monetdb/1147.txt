CREATE AGGREGATE jit_sum(input INTEGER) RETURNS BIGINT LANGUAGE C { // initialize one aggregate per group result->initialize(result, aggr_group.count); // zero initialize the sums memset(result->data, 0, result->count * sizeof(result->null_value)); // gather the sums for each of the groups for(size_t i = 0; i < input.count; i++) { result->data[aggr_group.data[i]] += input.data[i]; } };
CREATE TABLE tab(x INTEGER, y INTEGER);
insert into tab values (1,1),(2,2),(3,3),(1,10),(2,50),(3,6),(1,100000);
SELECT jit_sum(y) FROM tab;
SELECT sum(y), jit_sum(y) FROM tab;
