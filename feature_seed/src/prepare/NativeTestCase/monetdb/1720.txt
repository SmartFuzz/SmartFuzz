CREATE ROLE my_role;
CREATE SCHEMA my_schema AUTHORIZATION my_role;
CREATE USER usr1 WITH PASSWORD 'p1' NAME 'usr1' SCHEMA my_schema;
CREATE USER usr2 WITH PASSWORD 'p2' NAME 'usr2' SCHEMA my_schema;
GRANT my_role to usr1;
GRANT my_role to usr2;
CREATE SCHEMA your_schema;
SET SCHEMA your_schema;
CREATE TABLE person (name VARCHAR(10), birthday DATE, ssn CHAR(9));
INSERT INTO person VALUES ('alice', '1980-01-01', 'AAAAAAAAA'), ('bob', '1970-01-01', '000000000');
CREATE TABLE employee (name VARCHAR(10), salary DECIMAL(10,2));
INSERT INTO employee VALUES ('alice', 888.42), ('bob', 444.42);
CREATE VIEW v1 AS SELECT p.name, EXTRACT(YEAR FROM birthday) as yr, '********'||substring(ssn,9,9) as ssn, ifthenelse((salary > 500), 'high', 'low') as salary FROM person p, employee e WHERE p.name = e.name;
SELECT * FROM v1;
CREATE VIEW v2 AS SELECT v1.name, v1.ssn, e.salary FROM v1, employee e WHERE v1.name = e.name;
SELECT * FROM v2;
CREATE VIEW v3 AS SELECT v1.name, v1.yr, v2.salary FROM v1, v2 WHERE v1.name = v2.name;
SELECT * FROM v3;
CREATE VIEW v4 AS SELECT yr, ifthenelse((salary > 500), 'high', 'low') as salary FROM v3;
SELECT * FROM v4;
SET SCHEMA sys;
