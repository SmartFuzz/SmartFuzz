SELECT CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA, CONSTRAINT_NAME, TABLE_CATALOG, TABLE_SCHEMA, TABLE_NAME, CONSTRAINT_TYPE, IS_DEFERRABLE, INITIALLY_DEFERRED, ENFORCED FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE CONSTRAINT_SCHEMA = '' OR CONSTRAINT_NAME = '';
SELECT CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA, CONSTRAINT_NAME, TABLE_CATALOG, TABLE_SCHEMA, TABLE_NAME, CONSTRAINT_TYPE, IS_DEFERRABLE, INITIALLY_DEFERRED, ENFORCED, schema_id, table_id, key_id, key_type, is_system FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE CONSTRAINT_SCHEMA IS NULL OR CONSTRAINT_SCHEMA = '' OR CONSTRAINT_NAME IS NULL OR CONSTRAINT_NAME = '' OR TABLE_SCHEMA IS NULL OR TABLE_SCHEMA = '' OR TABLE_NAME IS NULL OR TABLE_NAME = '' OR CONSTRAINT_TYPE IS NULL OR CONSTRAINT_TYPE = '' OR schema_id IS NULL OR table_id IS NULL OR key_id IS NULL OR key_type IS NULL OR is_system IS NULL;
SELECT COUNT(*) AS duplicates, CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS GROUP BY CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA, CONSTRAINT_NAME HAVING COUNT(*) > 1;
SELECT COUNT(*) AS duplicates, CONSTRAINT_SCHEMA, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS GROUP BY CONSTRAINT_SCHEMA, CONSTRAINT_NAME HAVING COUNT(*) > 1;
SELECT COUNT(*) AS duplicates, schema_id, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS GROUP BY schema_id, CONSTRAINT_NAME HAVING COUNT(*) > 1;
SELECT COUNT(*) AS duplicates, key_id FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS GROUP BY key_id HAVING COUNT(*) > 1;
SELECT CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (CONSTRAINT_CATALOG, CONSTRAINT_SCHEMA) NOT IN (SELECT CATALOG_NAME, SCHEMA_NAME FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (CONSTRAINT_SCHEMA) NOT IN (SELECT SCHEMA_NAME FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT schema_id, CONSTRAINT_SCHEMA, CONSTRAINT_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (schema_id) NOT IN (SELECT schema_id FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT TABLE_CATALOG, TABLE_SCHEMA, TABLE_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (TABLE_CATALOG, TABLE_SCHEMA) NOT IN (SELECT CATALOG_NAME, SCHEMA_NAME FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT TABLE_SCHEMA, TABLE_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (TABLE_SCHEMA) NOT IN (SELECT SCHEMA_NAME FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT schema_id, TABLE_SCHEMA, TABLE_NAME FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (schema_id) NOT IN (SELECT schema_id FROM INFORMATION_SCHEMA.SCHEMATA);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, schema_id FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (schema_id) NOT IN (SELECT id FROM sys.schemas);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, table_id FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (table_id) NOT IN (SELECT id FROM sys.tables);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, key_id FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (key_id) NOT IN (SELECT id FROM sys.keys UNION ALL SELECT id FROM tmp.keys);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, key_type FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (key_type) NOT IN (SELECT key_type_id FROM sys.key_types);
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, CONSTRAINT_TYPE FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (CONSTRAINT_TYPE) NOT IN ('PRIMARY KEY', 'UNIQUE', 'FOREIGN KEY', 'UNIQUE NULLS NOT DISTINCT', 'CHECK');
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, IS_DEFERRABLE FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (IS_DEFERRABLE) NOT IN ('NO', 'YES');
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, INITIALLY_DEFERRED FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (INITIALLY_DEFERRED) NOT IN ('NO', 'YES');
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, ENFORCED FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (ENFORCED) NOT IN ('NO', 'YES');
SELECT CONSTRAINT_SCHEMA, CONSTRAINT_NAME, is_system FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE (is_system) NOT IN (FALSE, TRUE);
