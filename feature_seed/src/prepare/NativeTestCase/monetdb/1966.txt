set time zone interval '+01:00' hour to minute;
create table isp (n int, s varchar(30), p varchar(30));
insert into isp values (1, '2021-01-01 00:00:12', '%Y-%m-%d %H:%M:%S'), (2, '2022-01-01 00:01:23', '%Y-%d-%m %H:%M:%S'), (3, '2023-01-01 00:12:34', '%F %T');
select n, str_to_timestamp(s,'%Y-%m-%d %H:%M:%S') from isp;
select n, case when n < 2 then str_to_timestamp(s,'%Y-%m-%d %H:%M:%S') else str_to_timestamp(s,'%Y-%m-%d %H:%M:%S') end from isp;
select n, str_to_timestamp('2023-01-11 00:12:34', p) from isp;
select n, case when n < 2 then str_to_timestamp('2023-01-11 00:12:34', p) else str_to_timestamp('2023-01-11 00:12:34', p) end from isp;
