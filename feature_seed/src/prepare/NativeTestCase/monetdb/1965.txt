set time zone interval '+01:00' hour to minute;
create table itt (n int, t varchar(10), p varchar(10));
insert into itt values (1, '00:01:02', '%H:%M:%S'), (2, '00:10:02', '%H:%S:%M'), (3, '00:20:02', '%T');
select n, str_to_time(t,'%H:%M:%S') from itt;
select n, case when n < 2 then str_to_time(t,'%H:%M:%S') else str_to_time(t,'%H:%M:%S') end from itt;
select n, str_to_time('00:12:34', p) from itt;
select n, case when n < 2 then str_to_time('00:12:34',p) else str_to_time('00:12:34',p) end from itt;
