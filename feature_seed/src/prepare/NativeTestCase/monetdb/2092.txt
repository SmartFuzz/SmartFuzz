CREATE TABLE Test (k1 int, k2 int, k3 int, v int);
INSERT INTO Test SELECT value % 10 as k1, value % 100 as k2, value % 1000 as k3, value as v FROM generate_series(1, 100000);
set optimizer='sequential_pipe';
explain physical select * from test where (k1, k2) not in ( select k1, k2 from test where k1 = 1 );
explain physical select * from test where k2 not in ( select k2 from test where k1 = 1 );
set optimizer='default_pipe';
