create table itt (n int, t timetz, p text);
insert into itt values (1, timetz '00:00:12', '%I hour %M:%S %p'), (2, timetz '00:01:23', '%I hour %S:%M %p'), (3, timetz '00:12:34', '%T %p');
select n, time_to_str(t, '%I hour %M:%S %p') from itt;
select n, case when n < 2 then time_to_str(t, '%I hour %M:%S %p') else time_to_str(t, '%I hour %M:%S %p') end from itt;
select n, time_to_str(timetz '00:12:34', p) from itt;
select n, case when n < 2 then time_to_str(timetz '00:12:34', p) else time_to_str(timetz '00:12:34', p) end from itt;
